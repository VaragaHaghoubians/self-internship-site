# ==============================================================
# WEEK 5: Pandas & NumPy — The Core of Data Science
# FILE: pandas_basics.py
#
# TASK: Load a CSV file, explore it, filter it, and calculate
#       basic statistics using Pandas and NumPy.
#
# WHY:  Pandas is THE tool for working with tabular data (like Excel
#       but with code). Every data science job requires it.
#       NumPy provides fast math operations on arrays/numbers.
#
# BEFORE RUNNING: Make sure your ds-env is activated and you have
#   pandas and numpy installed:
#   pip install pandas numpy
#
# HOW TO RUN: python pandas_basics.py
# ==============================================================

import pandas as pd    # 'pd' is the standard shortcut for pandas
import numpy as np     # 'np' is the standard shortcut for numpy

print("=" * 60)
print("WEEK 5: Pandas & NumPy Basics")
print("=" * 60)


# ==============================================================
# PART 1: Create a DataFrame from scratch
# ==============================================================

# A DataFrame is like a spreadsheet/table with rows and columns.
# We create one by giving it a dictionary where:
#   - Keys   = column names
#   - Values = lists of data (one item per row)

print("\n--- PART 1: Create a DataFrame ---")

data = {
    "day":         ["Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"],
    "temperature": [22.1,  23.5,  21.8,  24.0,  22.7,  23.1,  21.5],
    "humidity":    [55,     60,    58,    52,    57,    61,    59],
    "sensor_ok":   [True,  True,  True,  False, True,  True,  True]
}

# pd.DataFrame() turns our dictionary into a table
df = pd.DataFrame(data)

# Print the full table
print(df)

# HINT: The numbers on the left (0, 1, 2...) are the INDEX — the row numbers.


# ==============================================================
# PART 2: Exploring a DataFrame
# ==============================================================

print("\n--- PART 2: Exploring the DataFrame ---")

# .shape tells you (number of rows, number of columns)
print("Shape (rows, columns):", df.shape)

# .dtypes tells you the data type of each column
print("\nColumn data types:")
print(df.dtypes)

# .head(n) shows the first n rows (default is 5)
print("\nFirst 3 rows:")
print(df.head(3))

# .describe() gives you a statistical summary of number columns
print("\nStatistical summary:")
print(df.describe())

# HINT: describe() shows: count, mean, std (standard deviation),
# min, 25%, 50% (median), 75%, max


# ==============================================================
# PART 3: Selecting Columns
# ==============================================================

print("\n--- PART 3: Selecting Columns ---")

# Select ONE column — returns a Series (like a list with labels)
temp_column = df["temperature"]
print("Temperature column:")
print(temp_column)

# Select MULTIPLE columns — returns a new DataFrame
temp_and_humidity = df[["temperature", "humidity"]]
print("\nTemperature and humidity together:")
print(temp_and_humidity)

# HINT: Single brackets df["col"] → Series
#        Double brackets df[["col1", "col2"]] → DataFrame


# ==============================================================
# PART 4: Filtering Rows (This is VERY important!)
# ==============================================================

print("\n--- PART 4: Filtering Rows ---")

# Get rows where temperature is above 23 degrees
hot_days = df[df["temperature"] > 23]
print("Days where temperature > 23:")
print(hot_days)

# Get rows where sensor was working (sensor_ok == True)
working_sensor = df[df["sensor_ok"] == True]
print("\nDays where sensor was working:")
print(working_sensor)

# Combine two conditions with & (AND) or | (OR)
hot_and_working = df[(df["temperature"] > 23) & (df["sensor_ok"] == True)]
print("\nHot days WITH working sensor:")
print(hot_and_working)

# HINT: df[condition] returns only rows where condition is True
# The condition inside df[] must use column names like df["col"]


# ==============================================================
# PART 5: Basic Statistics
# ==============================================================

print("\n--- PART 5: Statistics ---")

temps = df["temperature"]

print("Average temperature:", temps.mean())
print("Maximum temperature:", temps.max())
print("Minimum temperature:", temps.min())
print("Median temperature:", temps.median())
print("Standard deviation:", temps.std())

# NumPy also works on the same data
temps_array = np.array(df["temperature"])
print("\nUsing NumPy:")
print("Sum of all temperatures:", np.sum(temps_array))
print("Variance:", np.var(temps_array))


# ==============================================================
# PART 6: Adding a New Column
# ==============================================================

print("\n--- PART 6: Adding a New Column ---")

# Convert Celsius to Fahrenheit: F = (C × 9/5) + 32
# Pandas applies this formula to EVERY row automatically!
df["temp_fahrenheit"] = (df["temperature"] * 9/5) + 32

print("DataFrame with new Fahrenheit column:")
print(df[["day", "temperature", "temp_fahrenheit"]])


# ==============================================================
# PART 7: Loading a Real CSV File
# ==============================================================

print("\n--- PART 7: Loading a CSV File ---")

# First, let's create and save a CSV so we have one to load
df.to_csv("week5_temperature_data.csv", index=False)
print("Saved data to 'week5_temperature_data.csv'")

# Now load it back — this is how you'll load ANY dataset
loaded_df = pd.read_csv("week5_temperature_data.csv")
print("\nLoaded CSV (first 3 rows):")
print(loaded_df.head(3))

# HINT: pd.read_csv("filename.csv") is how you load almost every dataset
# The file can be a local path OR a URL from the internet!


# ==============================================================
# PART 8: Sorting and Value Counts
# ==============================================================

print("\n--- PART 8: Sorting and Value Counts ---")

# Build a slightly richer dataset to work with
sales_data = {
    "city":     ["NYC", "LA", "NYC", "Chicago", "LA", "NYC", "Chicago", "LA"],
    "product":  ["A",   "B",  "B",   "A",       "A",  "C",   "B",       "C"],
    "revenue":  [1200,  850,  940,   1100,      760,  1350,  890,       920],
    "units":    [12,    9,    10,    11,        8,    14,    9,         10],
}
sales = pd.DataFrame(sales_data)
print("Sales data:")
print(sales)

# sort_values() — sort by one or more columns
print("\nSorted by revenue (highest first):")
print(sales.sort_values("revenue", ascending=False))

# Sort by multiple columns: first city (A→Z), then revenue within each city (high→low)
print("\nSorted by city, then revenue descending:")
print(sales.sort_values(["city", "revenue"], ascending=[True, False]))

# value_counts() — how many times each value appears in a column
print("\nHow many rows per city:")
print(sales["city"].value_counts())
# HINT: value_counts() always sorts from most common to least common

print("\nHow many rows per product:")
print(sales["product"].value_counts())


# ==============================================================
# PART 9: GroupBy — The Most Powerful Pandas Tool
# ==============================================================

print("\n--- PART 9: GroupBy ---")

# GroupBy splits the DataFrame into groups, applies a function, then combines results.
# Think of it like: "For each city, calculate the total revenue."

# Total revenue per city
revenue_by_city = sales.groupby("city")["revenue"].sum()
print("Total revenue per city:")
print(revenue_by_city)

# Average revenue per product
avg_by_product = sales.groupby("product")["revenue"].mean()
print("\nAverage revenue per product:")
print(avg_by_product)

# Multiple aggregations at once using .agg()
summary = sales.groupby("city").agg(
    total_revenue=("revenue", "sum"),
    avg_revenue=("revenue", "mean"),
    total_units=("units", "sum"),
    num_orders=("revenue", "count")
)
print("\nFull summary by city:")
print(summary)

# HINT: .agg() lets you apply different functions to different columns in one step.
# The format is: new_col_name=("original_column", "function")

# Reset index to turn the grouped result back into a regular DataFrame
summary = summary.reset_index()
print("\nAfter reset_index (city is now a regular column):")
print(summary)


# ==============================================================
# PART 10: Merge and Concat — Joining Tables
# ==============================================================

print("\n--- PART 10: Merge and Concat ---")

# Real datasets are often split across multiple tables, just like in a database.
# Pandas merge = SQL JOIN

# Table 1: customer information
customers = pd.DataFrame({
    "customer_id": [1, 2, 3, 4],
    "name":        ["Alice", "Bob", "Carol", "Diana"],
    "city":        ["NYC", "LA", "Chicago", "NYC"],
})

# Table 2: orders
orders = pd.DataFrame({
    "order_id":    [101, 102, 103, 104, 105],
    "customer_id": [1,   2,   1,   3,   2],
    "amount":      [250, 180, 310, 95,  420],
})

print("Customers table:")
print(customers)
print("\nOrders table:")
print(orders)

# INNER JOIN — only rows that have a matching customer_id in BOTH tables
merged = pd.merge(customers, orders, on="customer_id", how="inner")
print("\nInner join result (customers with their orders):")
print(merged)

# LEFT JOIN — all customers, even those with no orders
left_merged = pd.merge(customers, orders, on="customer_id", how="left")
print("\nLeft join (all customers, NaN if no orders):")
print(left_merged)

# HINT: how="inner" → only matching rows
#        how="left"  → all rows from the LEFT table, NaN where no match
#        how="right" → all rows from the RIGHT table
#        how="outer" → all rows from BOTH tables

# pd.concat() — stack DataFrames vertically (like appending rows)
week1 = pd.DataFrame({"day": ["Mon", "Tue"], "temp": [22.1, 23.5]})
week2 = pd.DataFrame({"day": ["Wed", "Thu"], "temp": [21.8, 24.0]})
combined = pd.concat([week1, week2], ignore_index=True)
print("\nConcatenated (stacked) result:")
print(combined)
# HINT: ignore_index=True resets the row numbers (0, 1, 2, 3...)
# Without it, you'd see the original indexes (0, 1, 0, 1)


# ==============================================================
# PART 11: Working with Datetime Columns
# ==============================================================

print("\n--- PART 11: Datetime Columns ---")

# Dates look like strings when you first load data.
# You must convert them to datetime64 type to use date math.

sensor_log = pd.DataFrame({
    "timestamp":   ["2026-01-15 08:00", "2026-01-15 12:30",
                    "2026-02-03 09:15", "2026-02-03 17:45",
                    "2026-03-20 11:00", "2026-03-21 14:20"],
    "temperature": [22.1, 23.8, 21.5, 24.1, 20.9, 22.7],
    "device":      ["esp01", "esp01", "esp02", "esp01", "esp02", "esp01"],
})

# Convert the string column to a real datetime type
sensor_log["timestamp"] = pd.to_datetime(sensor_log["timestamp"])
print("After pd.to_datetime():")
print(sensor_log.dtypes)

# Extract useful parts from the datetime
sensor_log["year"]  = sensor_log["timestamp"].dt.year
sensor_log["month"] = sensor_log["timestamp"].dt.month
sensor_log["day"]   = sensor_log["timestamp"].dt.day
sensor_log["hour"]  = sensor_log["timestamp"].dt.hour

print("\nWith extracted date parts:")
print(sensor_log)

# Calculate time differences
sensor_log_sorted = sensor_log.sort_values("timestamp")
sensor_log_sorted["time_since_previous"] = sensor_log_sorted["timestamp"].diff()
print("\nTime between consecutive readings:")
print(sensor_log_sorted[["timestamp", "device", "temperature", "time_since_previous"]])

# Filter by date range
jan_readings = sensor_log[sensor_log["month"] == 1]
print("\nJanuary readings only:")
print(jan_readings)

# Group by month and get average temperature
monthly_avg = sensor_log.groupby("month")["temperature"].mean()
print("\nAverage temperature by month:")
print(monthly_avg)

# HINT: .dt accessor unlocks all datetime operations.
# Common ones: .dt.year .dt.month .dt.day .dt.hour .dt.weekday .dt.date
# .diff() calculates the difference between consecutive rows.
