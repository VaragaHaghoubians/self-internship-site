# ==============================================================
# WEEK 5 (Part 3): Statistics for Data Scientists
# FILE: statistics_basics.py
#
# TASK: Learn the core statistical concepts every data science
#       interview will test you on. Run real examples in Python.
#
# WHY:  Statistics is the LANGUAGE of data science.
#       Without it, you can describe data but not understand it.
#       "Is this difference real, or just random noise?" — statistics answers that.
#
# INSTALL: pip install pandas numpy scipy matplotlib seaborn
# HOW TO RUN: python statistics_basics.py
# ==============================================================

import numpy as np
import pandas as pd
from scipy import stats
import matplotlib.pyplot as plt
import seaborn as sns

sns.set_theme(style="whitegrid")

print("=" * 60)
print("WEEK 5 (Part 3): Statistics for Data Scientists")
print("=" * 60)


# ==============================================================
# PART 1: DESCRIPTIVE STATISTICS
# ==============================================================
print("\n--- PART 1: Descriptive Statistics ---")

# Measures of CENTRAL TENDENCY — where is the "middle" of the data?
scores = [55, 62, 68, 70, 71, 72, 73, 74, 75, 78, 80, 82, 88, 90, 95]

print(f"Data:   {scores}")
print(f"Count:  {len(scores)}")
print(f"Mean:   {np.mean(scores):.2f}   ← average — sensitive to outliers")
print(f"Median: {np.median(scores):.2f} ← middle value — robust to outliers")
print(f"Mode:   {stats.mode(scores, keepdims=True).mode[0]}  ← most common value")

# Measures of SPREAD — how wide is the data?
print(f"\nMin:    {np.min(scores)}")
print(f"Max:    {np.max(scores)}")
print(f"Range:  {np.max(scores) - np.min(scores)}  (max - min)")

variance = np.var(scores, ddof=1)   # ddof=1 for sample variance
std_dev  = np.std(scores, ddof=1)
print(f"\nVariance:         {variance:.2f}")
print(f"Std Deviation:    {std_dev:.2f}  ← same units as original data")
print("  → 68% of data falls within 1 std dev of the mean (for normal distributions)")
print("  → 95% of data falls within 2 std dev of the mean")

# Percentiles
q1, q2, q3 = np.percentile(scores, [25, 50, 75])
iqr = q3 - q1
print(f"\nQ1 (25th percentile): {q1}")
print(f"Q2 (50th = median):   {q2}")
print(f"Q3 (75th percentile): {q3}")
print(f"IQR (Q3 - Q1):        {iqr}  ← middle 50% of data")
print(f"Outlier fence:        < {q1 - 1.5*iqr:.1f}  or  > {q3 + 1.5*iqr:.1f}")


# ==============================================================
# PART 2: DISTRIBUTIONS
# ==============================================================
print("\n--- PART 2: The Normal Distribution ---")

# The Normal (Gaussian) distribution is the most important in statistics.
# It's shaped like a bell curve. Many real things follow it roughly:
# heights, exam scores, measurement errors.

np.random.seed(42)
normal_data = np.random.normal(loc=70, scale=10, size=1000)
# loc   = mean (center of bell)
# scale = standard deviation (how wide the bell is)

print(f"Generated 1,000 samples from Normal(mean=70, std=10)")
print(f"Sample mean: {normal_data.mean():.2f}  (expected: 70)")
print(f"Sample std:  {normal_data.std():.2f}   (expected: 10)")

# The Empirical Rule (68-95-99.7 rule)
within_1 = np.sum(np.abs(normal_data - 70) < 10)  / 1000 * 100
within_2 = np.sum(np.abs(normal_data - 70) < 20)  / 1000 * 100
within_3 = np.sum(np.abs(normal_data - 70) < 30)  / 1000 * 100
print(f"\nEmpirical Rule check:")
print(f"  Within 1 std dev: {within_1:.1f}%  (expected ~68%)")
print(f"  Within 2 std dev: {within_2:.1f}%  (expected ~95%)")
print(f"  Within 3 std dev: {within_3:.1f}%  (expected ~99.7%)")

# Skewness and Kurtosis
print(f"\nSkewness: {stats.skew(normal_data):.2f}")
print("  0 = symmetric, >0 = right skew (tail right), <0 = left skew (tail left)")
print(f"Kurtosis: {stats.kurtosis(normal_data):.2f}")
print("  0 = normal, >0 = heavier tails (more outliers), <0 = lighter tails")

# Visualize
fig, axes = plt.subplots(1, 2, figsize=(12, 4))
axes[0].hist(normal_data, bins=40, color="steelblue", edgecolor="white", density=True)
axes[0].set_title("Normal Distribution (histogram)")
axes[0].set_xlabel("Value"); axes[0].set_ylabel("Density")

# Compare Normal vs Skewed
skewed = np.random.exponential(scale=10, size=1000)
axes[1].hist(normal_data, bins=40, alpha=0.6, label="Normal", color="steelblue", density=True)
axes[1].hist(skewed,      bins=40, alpha=0.6, label="Skewed (exponential)", color="coral",    density=True)
axes[1].set_title("Normal vs Skewed Distribution")
axes[1].legend()
plt.tight_layout()
plt.savefig("plot_stats_01_distributions.png", dpi=150)
plt.close()
print("Saved: plot_stats_01_distributions.png")


# ==============================================================
# PART 3: HYPOTHESIS TESTING — "Is this difference real?"
# ==============================================================
print("\n--- PART 3: Hypothesis Testing ---")

# The core idea:
#   H₀ (Null Hypothesis):      "There is NO real difference — it's just random chance"
#   H₁ (Alternative Hypothesis): "There IS a real difference"
#
# The p-value = probability of seeing this result IF the null hypothesis were true.
#   p < 0.05 → statistically significant → reject H₀ (the difference is real)
#   p ≥ 0.05 → not significant → fail to reject H₀ (could just be chance)
#
# 0.05 is the conventional threshold (called "alpha" or significance level)

# Example scenario:
# Two study methods were tested. Did Method B actually improve scores?
np.random.seed(1)
group_A = np.random.normal(72, 10, 40)   # Method A: 40 students
group_B = np.random.normal(77, 10, 40)   # Method B: 40 students (5 points higher on average)

print(f"Group A (Method A): mean = {group_A.mean():.2f}, std = {group_A.std():.2f}")
print(f"Group B (Method B): mean = {group_B.mean():.2f}, std = {group_B.std():.2f}")
print(f"Observed difference: {group_B.mean() - group_A.mean():.2f} points")

# INDEPENDENT SAMPLES T-TEST
# Use when: comparing means of TWO independent groups
t_stat, p_value = stats.ttest_ind(group_A, group_B)
print(f"\nIndependent Samples t-test:")
print(f"  t-statistic: {t_stat:.3f}")
print(f"  p-value:     {p_value:.4f}")
if p_value < 0.05:
    print(f"  → p < 0.05: SIGNIFICANT — Method B really does improve scores!")
else:
    print(f"  → p ≥ 0.05: NOT significant — difference could be random chance")


# ONE-SAMPLE T-TEST
# Use when: testing if a sample mean differs from a known/target value
# Example: Did this class perform differently from the national average of 70?
class_scores = np.random.normal(73, 11, 35)
t_stat2, p_val2 = stats.ttest_1samp(class_scores, popmean=70)
print(f"\nOne-Sample t-test (vs national average = 70):")
print(f"  Class mean:  {class_scores.mean():.2f}")
print(f"  p-value:     {p_val2:.4f}")
if p_val2 < 0.05:
    print(f"  → Significant: this class differs from national average")
else:
    print(f"  → Not significant: class performance is consistent with average")


# ==============================================================
# PART 4: CHI-SQUARE TEST — For categorical variables
# ==============================================================
print("\n--- PART 4: Chi-Square Test (Categorical Data) ---")

# Use when: testing if two CATEGORICAL variables are related
# Example: Is pass/fail rate different between two majors?

# Observed counts (contingency table)
#              Pass  Fail
# Data Science   35    15
# CS             28    22
observed = np.array([[35, 15],
                     [28, 22]])

chi2, p_chi, dof, expected = stats.chi2_contingency(observed)
print(f"Contingency table:")
print(f"              Pass  Fail")
print(f"  Data Science  35    15")
print(f"  CS            28    22")
print(f"\nChi-square statistic: {chi2:.3f}")
print(f"p-value:              {p_chi:.4f}")
print(f"Degrees of freedom:   {dof}")
if p_chi < 0.05:
    print("→ Significant: pass rate IS different between majors")
else:
    print("→ Not significant: pass rates are similar between majors")


# ==============================================================
# PART 5: CORRELATION
# ==============================================================
print("\n--- PART 5: Correlation ---")

# Pearson correlation: measures LINEAR relationship between two numeric variables
# Range: -1 (perfect negative) to 0 (none) to +1 (perfect positive)

np.random.seed(7)
hours   = np.random.uniform(1, 12, 100)
scores2 = hours * 4 + np.random.normal(50, 8, 100)   # score ≈ 4*hours + noise

r, p = stats.pearsonr(hours, scores2)
print(f"Pearson r (hours vs score):  {r:.3f}  (p = {p:.4f})")
print(f"  → r = {r:.2f}: {'strong' if abs(r)>0.7 else 'moderate' if abs(r)>0.4 else 'weak'} positive correlation")
print("  → Rule of thumb: |r| > 0.7 = strong, 0.4-0.7 = moderate, < 0.4 = weak")

# Spearman correlation: works on RANKED data — not affected by outliers
rho, p_spear = stats.spearmanr(hours, scores2)
print(f"\nSpearman ρ (hours vs score): {rho:.3f}  (p = {p_spear:.4f})")
print("  → Use Spearman when data is not normally distributed or has outliers")

# Correlation matrix for multiple variables
df = pd.DataFrame({
    "hours":      hours,
    "score":      scores2,
    "attendance": np.random.uniform(50, 100, 100)
})
print(f"\nCorrelation matrix:")
print(df.corr().round(2))


# ==============================================================
# PART 6: CONFIDENCE INTERVALS
# ==============================================================
print("\n--- PART 6: Confidence Intervals ---")

# A 95% confidence interval means:
# "We are 95% confident the true population mean falls within this range."
# It does NOT mean "95% of data points are in this range" (common mistake!)

sample = np.random.normal(70, 12, 50)
mean   = sample.mean()
se     = stats.sem(sample)   # Standard Error = std / sqrt(n)
ci     = stats.t.interval(confidence=0.95, df=len(sample)-1, loc=mean, scale=se)

print(f"Sample mean:   {mean:.2f}")
print(f"Standard error: {se:.2f}")
print(f"95% CI:        ({ci[0]:.2f}, {ci[1]:.2f})")
print(f"  → We are 95% confident the true mean is between {ci[0]:.1f} and {ci[1]:.1f}")
print(f"  → Wider CI = less precise (small sample or high variance)")
print(f"  → Narrower CI = more precise (large sample or low variance)")


# ==============================================================
# SUMMARY TABLE — When to Use What
# ==============================================================
print("\n" + "=" * 60)
print("QUICK REFERENCE: Which Test to Use?")
print("=" * 60)
print("""
SCENARIO                                    TEST
─────────────────────────────────────────────────────────────
Compare means of 2 independent groups     Independent t-test
Compare sample mean vs known value        One-sample t-test
Compare means before/after (same people)  Paired t-test
Test relationship between 2 categorical   Chi-square test
Measure linear relationship (numeric)     Pearson correlation
Measure ranked relationship / outliers    Spearman correlation
Range for the true mean                   Confidence interval
""")
