# ==============================================================
# WEEK 6: Data Cleaning & SQL Basics
# FILE: data_cleaning.py
#
# TASK: Clean a messy dataset and query a SQLite database with SQL.
#
# WHY:  Real-world data is NEVER clean. Missing values, wrong data
#       types, duplicates — these cause wrong analysis results.
#       Data cleaning is 70-80% of a data scientist's actual work!
#       SQL is asked in almost every data job interview.
#
# INSTALL: pip install pandas sqlite3
#          (sqlite3 is built-in to Python — no install needed)
#
# HOW TO RUN: python data_cleaning.py
# ==============================================================

import pandas as pd
import numpy as np
import sqlite3    # sqlite3 is Python's built-in database library

print("=" * 60)
print("WEEK 6: Data Cleaning & SQL Basics")
print("=" * 60)


# ==============================================================
# PART 1: Create a Messy Dataset (like real-world data)
# ==============================================================

print("\n--- PART 1: The Messy Data ---")

# This simulates what real data often looks like when you receive it
messy_data = {
    "name":        ["Alice", "Bob", None,    "Diana", "Eve",  "Bob",   "Grace"],
    "age":         [25,       30,   22,       None,    28,     30,      -5],
    "city":        ["NYC",   "LA",  "NYC",   "NYC",   "  LA ", "LA",   "Chicago"],
    "temperature": [98.6,    99.1,  None,    98.2,    101.3,  99.1,    98.8],
    "score":       [85,      92,    78,      91,      None,   92,      88]
}

df = pd.DataFrame(messy_data)
print("Original messy data:")
print(df)
print("\nShape:", df.shape)


# ==============================================================
# PART 2: Find the Problems
# ==============================================================

print("\n--- PART 2: Detecting Problems ---")

# Check for missing values (NaN = "Not a Number" = missing)
print("Missing values per column:")
print(df.isnull().sum())
# HINT: isnull() returns True/False for each cell
# .sum() counts the True values (= number of missing values)

# Check for duplicates
print("\nNumber of duplicate rows:", df.duplicated().sum())

# Look at the data types
print("\nData types:")
print(df.dtypes)


# ==============================================================
# PART 3: Fix the Problems
# ==============================================================

print("\n--- PART 3: Cleaning the Data ---")

# 3a. Remove duplicate rows
df_clean = df.drop_duplicates()
print(f"After removing duplicates: {df.shape[0]} rows → {df_clean.shape[0]} rows")

# 3b. Fix the city column — strip extra whitespace and make consistent
# "  LA " → "LA" (the strip() function removes leading/trailing spaces)
df_clean["city"] = df_clean["city"].str.strip()
print("\nCity values after strip:", df_clean["city"].unique())

# 3c. Handle missing names — drop rows where name is missing
# (we can't have an anonymous record)
df_clean = df_clean.dropna(subset=["name"])
print(f"\nAfter removing rows with no name: {df_clean.shape[0]} rows")

# 3d. Handle missing age — fill with the MEDIAN age
# (we use median instead of mean because it's less affected by outliers)
median_age = df_clean["age"].median()
df_clean["age"] = df_clean["age"].fillna(median_age)
print(f"\nFilled missing age with median: {median_age}")

# 3e. Remove invalid age (age cannot be negative)
df_clean = df_clean[df_clean["age"] > 0]
print(f"After removing invalid ages: {df_clean.shape[0]} rows")

# 3f. Handle missing temperature — fill with median
df_clean["temperature"] = df_clean["temperature"].fillna(df_clean["temperature"].median())

# 3g. Handle missing score — fill with median
df_clean["score"] = df_clean["score"].fillna(df_clean["score"].median())

print("\nCleaned data:")
print(df_clean)
print("\nMissing values after cleaning:")
print(df_clean.isnull().sum())

# Save the clean data
df_clean.to_csv("clean_data.csv", index=False)
print("\nSaved clean data to 'clean_data.csv'")


# ==============================================================
# PART 4: SQL Basics with SQLite
# ==============================================================

print("\n--- PART 4: SQL with SQLite ---")

# SQLite is a database that lives in a single file on your computer
# Perfect for learning SQL without installing a database server

# Create a connection to a database file
# (it creates the file automatically if it doesn't exist)
conn = sqlite3.connect("practice.db")

# A 'cursor' is what you use to send SQL commands
cursor = conn.cursor()

# Create a table (like creating a spreadsheet with column headers)
cursor.execute("""
    CREATE TABLE IF NOT EXISTS students (
        id INTEGER PRIMARY KEY,
        name TEXT NOT NULL,
        age INTEGER,
        city TEXT,
        score REAL
    )
""")
# HINT: IF NOT EXISTS means "only create if the table doesn't already exist"
# SQL column types: INTEGER (whole numbers), TEXT (strings), REAL (decimals)

# Insert data into the table
# We'll use the clean data we prepared above
for _, row in df_clean.iterrows():
    cursor.execute("""
        INSERT OR IGNORE INTO students (name, age, city, score)
        VALUES (?, ?, ?, ?)
    """, (row["name"], int(row["age"]), row["city"], row["score"]))
    # HINT: We use ? placeholders (never put variables directly in SQL strings!)
    # This prevents SQL injection — a security vulnerability

conn.commit()  # Save (commit) the changes to the database


# ==============================================================
# PART 5: Writing SQL Queries
# ==============================================================

print("\n--- PART 5: SQL Queries ---")

# SELECT — get data from a table
print("All students:")
cursor.execute("SELECT * FROM students")
rows = cursor.fetchall()
for row in rows:
    print(" ", row)

# WHERE — filter rows
print("\nStudents from NYC:")
cursor.execute("SELECT name, city, score FROM students WHERE city = 'NYC'")
for row in cursor.fetchall():
    print(" ", row)

# ORDER BY — sort results
print("\nStudents sorted by score (highest first):")
cursor.execute("SELECT name, score FROM students ORDER BY score DESC")
for row in cursor.fetchall():
    print(" ", row)

# Aggregate functions — calculations on groups of rows
print("\nAverage score:")
cursor.execute("SELECT AVG(score) FROM students")
print(" ", cursor.fetchone()[0])

# GROUP BY — calculate stats for each group
print("\nAverage score per city:")
cursor.execute("""
    SELECT city, AVG(score) as avg_score, COUNT(*) as num_students
    FROM students
    GROUP BY city
""")
for row in cursor.fetchall():
    print(f"  {row[0]}: avg score = {row[1]:.1f}, students = {row[2]}")

# Close the connection when done
conn.close()
