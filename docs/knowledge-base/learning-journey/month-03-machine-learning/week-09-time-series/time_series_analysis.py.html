# ==============================================================
# PART 1: What Makes Time Series Data Special?
# ==============================================================

print("\n--- PART 1: What is Time Series Data? ---")
print("""
Regular tabular data: each row is INDEPENDENT.
  Knowing row 5 tells you nothing about row 6.

Time series data: each row DEPENDS on previous rows.
  Today's temperature is related to yesterday's.
  This week's sales are related to last week's.

That dependency is both the main challenge AND the main
source of signal in time series analysis.

Key concepts:
  Trend:       the long-term direction (going up, going down, flat)
  Seasonality: repeating patterns at fixed intervals
               (higher sales every December, lower temps every winter)
  Noise:       random fluctuations with no pattern
  Stationarity: a series is stationary if its statistical properties
               (mean, variance) don't change over time.
               Most forecasting models REQUIRE stationarity.
""")


# ==============================================================
# PART 2: Creating a Realistic Time Series
# ==============================================================

print("\n--- PART 2: Synthetic Sensor Dataset ---")

# Build 2 years of daily temperature + humidity readings
# (same concept as your Arduino/ESP32 data, just at daily scale)
rng = np.random.default_rng(seed=42)
dates = pd.date_range(start="2024-01-01", end="2025-12-31", freq="D")
n = len(dates)

# Trend component: slight warming over 2 years (+2°C total)
trend = np.linspace(0, 2, n)

# Yearly seasonality: colder in winter, hotter in summer
# sin wave with period = 365 days
yearly_season = 8 * np.sin(2 * np.pi * np.arange(n) / 365 - np.pi / 2)

# Weekly seasonality: slightly higher readings on weekdays (heating on)
weekly_season = 1.5 * np.sin(2 * np.pi * np.arange(n) / 7)

# Random noise
noise = rng.normal(0, 1.2, n)

# Combine all components
temperature = 18 + trend + yearly_season + weekly_season + noise
humidity    = 60 - 0.3 * temperature + rng.normal(0, 3, n)
humidity    = np.clip(humidity, 20, 95)

df = pd.DataFrame({
    "temperature": temperature.round(2),
    "humidity":    humidity.round(1),
}, index=dates)
df.index.name = "date"

print(f"Dataset: {len(df)} daily readings from {df.index[0].date()} to {df.index[-1].date()}")
print(df.describe().round(2))


# ==============================================================
# PART 3: Indexing and Slicing Time Series
# ==============================================================

print("\n--- PART 3: DatetimeIndex Slicing ---")

# Pandas DatetimeIndex supports very convenient slicing
print("January 2024 only:")
jan_2024 = df["2024-01"]
print(f"  {len(jan_2024)} rows  |  mean temp: {jan_2024['temperature'].mean():.2f}°C")

print("\nAll of 2024:")
year_2024 = df["2024"]
print(f"  {len(year_2024)} rows  |  mean temp: {year_2024['temperature'].mean():.2f}°C")

print("\nSummer 2024 (June–August):")
summer = df["2024-06":"2024-08"]
print(f"  {len(summer)} rows  |  mean temp: {summer['temperature'].mean():.2f}°C")

# Shift: compare today's value with yesterday's
df["temp_yesterday"]    = df["temperature"].shift(1)      # lag 1
df["temp_change"]       = df["temperature"] - df["temp_yesterday"]
df["temp_7d_ago"]       = df["temperature"].shift(7)      # lag 7 (same weekday last week)

print("\nFirst 5 rows with lag features:")
print(df[["temperature", "temp_yesterday", "temp_change"]].head())
print("\nHINT: 'shift(n)' moves values DOWN n rows. Row 0 gets NaN because there is no row -1.")

# Drop lag columns (we added them for demonstration only)
df = df.drop(columns=["temp_yesterday", "temp_change", "temp_7d_ago"])


# ==============================================================
# PART 4: Resampling and Rolling Windows
# ==============================================================

print("\n--- PART 4: Resampling and Rolling Windows ---")

# Resampling: change the time frequency of the data
# D=daily, W=weekly, ME=month-end, QE=quarter-end, YE=year-end
monthly = df["temperature"].resample("ME").agg(["mean", "min", "max"])
monthly.columns = ["monthly_mean", "monthly_min", "monthly_max"]
print("Monthly temperature summary:")
print(monthly.head(6).round(2))

# Rolling window: compute statistics over a sliding window of the last N observations
# This is the core tool for smoothing noisy sensor data
df["rolling_7d_mean"]  = df["temperature"].rolling(window=7,  center=False).mean()
df["rolling_30d_mean"] = df["temperature"].rolling(window=30, center=False).mean()

# center=False (default): window uses the PAST N rows (no future data — realistic)
# center=True:            window is centred (uses future data — only for retrospective analysis)

print("\nRolling windows (first 35 rows):")
print(df[["temperature", "rolling_7d_mean", "rolling_30d_mean"]].head(35).tail(5).round(2))
print("HINT: The first N-1 rows are NaN because there aren't enough past values yet.")

# Rolling std — spot when volatility is increasing
df["rolling_7d_std"] = df["temperature"].rolling(7).std()

# Exponentially Weighted Moving Average (EWMA)
# Gives more weight to recent values — reacts faster than simple rolling mean
df["ewm_14d"] = df["temperature"].ewm(span=14, adjust=False).mean()

print("\nEWM vs. rolling mean (first 40 rows, last 5):")
print(df[["temperature", "rolling_7d_mean", "ewm_14d"]].head(40).tail(5).round(2))

# PART 5: Visualising Time Series
# ==============================================================

print("\n--- PART 5: Visualisations ---")

fig, axes = plt.subplots(3, 1, figsize=(14, 12))

# Plot 1: Raw data + rolling means
ax = axes[0]
ax.plot(df.index, df["temperature"],     alpha=0.4, color="steelblue", label="Daily temp")
ax.plot(df.index, df["rolling_7d_mean"], color="orange",    linewidth=1.5, label="7-day rolling mean")
ax.plot(df.index, df["rolling_30d_mean"],color="red",       linewidth=2,   label="30-day rolling mean")
ax.set_title("Temperature: Raw vs. Smoothed", fontsize=13)
ax.set_ylabel("°C")
ax.legend()
ax.grid(True, alpha=0.3)
ax.xaxis.set_major_formatter(mdates.DateFormatter("%b %Y"))

# Plot 2: Monthly aggregation
ax = axes[1]
ax.fill_between(monthly.index, monthly["monthly_min"], monthly["monthly_max"],
                alpha=0.3, color="steelblue", label="Min–Max range")
ax.plot(monthly.index, monthly["monthly_mean"], marker="o", color="steelblue",
        linewidth=2, label="Monthly mean")
ax.set_title("Monthly Temperature Summary", fontsize=13)
ax.set_ylabel("°C")
ax.legend()
ax.grid(True, alpha=0.3)
ax.xaxis.set_major_formatter(mdates.DateFormatter("%b %Y"))

# Plot 3: Rolling volatility (std)
ax = axes[2]
ax.plot(df.index, df["rolling_7d_std"], color="purple", linewidth=1.2)
ax.set_title("Rolling 7-Day Standard Deviation (Volatility)", fontsize=13)
ax.set_ylabel("Std Dev (°C)")
ax.grid(True, alpha=0.3)
ax.xaxis.set_major_formatter(mdates.DateFormatter("%b %Y"))

plt.tight_layout()
plt.savefig("ts_overview.png", bbox_inches="tight", dpi=100)
plt.show()
plt.close()
print("Saved: ts_overview.png")


# ==============================================================
# PART 6: Time Series Decomposition
# ==============================================================

print("\n--- PART 6: Decomposition (Trend + Seasonality + Residual) ---")

from statsmodels.tsa.seasonal import seasonal_decompose

# Decompose the monthly mean (less noisy than daily for illustration)
monthly_mean = df["temperature"].resample("ME").mean()

# model='additive': series = trend + seasonality + residual
# model='multiplicative': series = trend × seasonality × residual
# Use additive when the amplitude of seasonality is CONSTANT over time
# Use multiplicative when seasonality grows proportionally with the trend

result = seasonal_decompose(monthly_mean, model="additive", period=12)

fig, axes = plt.subplots(4, 1, figsize=(14, 10), sharex=True)
titles = ["Observed", "Trend", "Seasonality", "Residual (Noise)"]
components = [result.observed, result.trend, result.seasonal, result.resid]

for ax, title, component in zip(axes, titles, components):
    ax.plot(monthly_mean.index, component, color="steelblue")
    ax.set_title(title, fontsize=12)
    ax.grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig("ts_decomposition.png", bbox_inches="tight", dpi=100)
plt.show()
plt.close()
print("Saved: ts_decomposition.png")
print("""
Decomposition reveals:
  Trend:      the underlying direction without seasonality or noise
  Seasonality: the repeating pattern (period=12 for monthly data)
  Residual:   whatever is left — pure noise, or unexplained patterns

If the residual shows patterns, your model is missing something important.
""")


# ==============================================================
# PART 7: Stationarity — The Core Requirement for Forecasting
# ==============================================================

print("\n--- PART 7: Stationarity Test (Augmented Dickey-Fuller) ---")

from statsmodels.tsa.stattools import adfuller

def adf_test(series: pd.Series, name: str = ""):
    """
    Augmented Dickey-Fuller test.
    H0: the series has a unit root (non-stationary)
    H1: the series is stationary

    If p-value < 0.05: reject H0 → series IS stationary ✓
    If p-value ≥ 0.05: fail to reject H0 → series is NOT stationary ✗
    """
    result = adfuller(series.dropna(), autolag="AIC")
    label = name or "Series"
    print(f"\nADF test — {label}")
    print(f"  ADF Statistic : {result[0]:.4f}")
    print(f"  p-value       : {result[1]:.4f}")
    print(f"  Result        : {'STATIONARY ✓' if result[1] < 0.05 else 'NON-STATIONARY ✗'}")

adf_test(monthly_mean, "Monthly temperature (original)")

# Differencing: subtract each value from the previous one
# This removes trend and often achieves stationarity
monthly_diff = monthly_mean.diff().dropna()
adf_test(monthly_diff, "Monthly temperature (1st difference)")

print("""
Why does stationarity matter?
  ARIMA and similar models assume the series has constant statistical properties.
  A trending series (e.g., steadily rising sales) violates this.
  Differencing (diff()) removes the trend by looking at CHANGES rather than levels.
  If one diff isn't enough, apply it again: series.diff().diff()
""")


# ==============================================================
# PART 8: ACF and PACF Plots — How Much Does the Past Matter?
# ==============================================================

print("\n--- PART 8: ACF and PACF ---")

from statsmodels.graphics.tsaplots import plot_acf, plot_pacf

fig, axes = plt.subplots(2, 1, figsize=(12, 7))

# ACF (Autocorrelation Function):
# Shows correlation of the series with itself at lag k.
# Helps identify MA (moving average) order.
# If ACF cuts off after lag q → MA(q) component.
plot_acf(monthly_diff, lags=24, ax=axes[0], color="steelblue")
axes[0].set_title("ACF — Autocorrelation (how correlated is today with k days ago?)")
axes[0].grid(True, alpha=0.3)

# PACF (Partial Autocorrelation Function):
# Shows correlation at lag k after removing the effect of shorter lags.
# Helps identify AR (autoregressive) order.
# If PACF cuts off after lag p → AR(p) component.
plot_pacf(monthly_diff, lags=24, ax=axes[1], color="tomato", method="ywm")
axes[1].set_title("PACF — Partial Autocorrelation (effect of lag k, controlling for shorter lags)")
axes[1].grid(True, alpha=0.3)

plt.tight_layout()
plt.savefig("ts_acf_pacf.png", bbox_inches="tight", dpi=100)
plt.show()
plt.close()
print("Saved: ts_acf_pacf.png")
print("""
Reading ACF/PACF plots:
  Blue shaded area = 95% confidence interval
  Bars inside the band = not statistically significant (ignore them)
  Bars outside the band = significant correlation at that lag

  AR(p): PACF cuts off sharply after lag p; ACF decays slowly
  MA(q): ACF cuts off sharply after lag q; PACF decays slowly
  ARMA:  both decay slowly → need both p and q terms
""")


# ==============================================================
# PART 9: Forecasting — Three Models
# ==============================================================

print("\n--- PART 9: Forecasting ---")

# Use the last 24 months of monthly data
# Train: first 20 months | Test: last 4 months
train = monthly_mean.iloc[:-4]
test  = monthly_mean.iloc[-4:]
horizon = len(test)

print(f"Train: {len(train)} months | Test: {len(test)} months")

# ---- MODEL 1: Baseline — Naïve Forecast (last known value) ----
# The simplest possible forecast: predict the last observed value for all future steps.
# Any model worth using must BEAT this baseline.
naive_forecast = pd.Series(
    [train.iloc[-1]] * horizon,
    index=test.index
)

# ---- MODEL 2: Simple Exponential Smoothing ----
# Weighted average of ALL past values, with exponentially decaying weights.
# α (alpha) controls how fast old observations are forgotten.
# α close to 1 → very responsive, mostly just last observation
# α close to 0 → very smooth, mostly the long-run average
from statsmodels.tsa.holtwinters import SimpleExpSmoothing, ExponentialSmoothing

ses_model = SimpleExpSmoothing(train, initialization_method="estimated").fit()
ses_forecast = ses_model.forecast(horizon)

# ---- MODEL 3: Holt-Winters Exponential Smoothing ----
# Extends SES to handle TREND and SEASONALITY.
# trend='add':  linear trend (additive)
# seasonal='add': additive seasonality
# seasonal_periods=12: one cycle = 12 months
hw_model = ExponentialSmoothing(
    train,
    trend="add",
    seasonal="add",
    seasonal_periods=12,
    initialization_method="estimated"
).fit()
hw_forecast = hw_model.forecast(horizon)

# ---- Evaluate: Mean Absolute Error ----
def mae(actual, predicted):
    return np.abs(actual.values - predicted.values).mean()

print(f"\nForecast accuracy (MAE — lower is better):")
print(f"  Naïve (baseline):           {mae(test, naive_forecast):.3f}°C")
print(f"  Simple Exponential Smoothing: {mae(test, ses_forecast):.3f}°C")
print(f"  Holt-Winters:               {mae(test, hw_forecast):.3f}°C")

# ---- Plot ----
plt.figure(figsize=(14, 5))
plt.plot(train.index, train, label="Train", color="steelblue", linewidth=1.5)
plt.plot(test.index,  test,  label="Actual (test)", color="black",
         linewidth=2, marker="o")
plt.plot(test.index, naive_forecast, label="Naïve",          color="grey",
         linestyle="--", marker="s")
plt.plot(test.index, ses_forecast,   label="Simple Exp Smoothing", color="orange",
         linestyle="--", marker="^")
plt.plot(test.index, hw_forecast,    label="Holt-Winters",    color="tomato",
         linestyle="--", marker="D")
plt.title("Monthly Temperature Forecast — 3 Models", fontsize=13)
plt.ylabel("°C")
plt.legend()
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.savefig("ts_forecast.png", bbox_inches="tight", dpi=100)
plt.show()
plt.close()
print("Saved: ts_forecast.png")


# ==============================================================
# PART 10: ARIMA — The Classic Forecasting Model
# ==============================================================

print("\n--- PART 10: ARIMA ---")
print("""
ARIMA stands for:
  AR(p)  — AutoRegressive: today's value is a linear combination
             of the past p values
  I(d)   — Integrated: the series was differenced d times to achieve stationarity
  MA(q)  — Moving Average: today's value depends on past q forecast ERRORS

Choosing p, d, q:
  d: how many times you had to difference to pass the ADF test (usually 0 or 1)
  p: look at PACF — how many lags are significant?
  q: look at ACF  — how many lags are significant?

SARIMA adds S for Seasonal: SARIMA(p,d,q)(P,D,Q,m)
  P, D, Q are the seasonal AR, differencing, MA orders
  m is the seasonal period (12 for monthly, 7 for daily-with-weekly-cycle)

For this dataset:
  d=1  (one differencing achieved stationarity)
  p=1  (one significant PACF spike)
  q=1  (one significant ACF spike)
  Seasonal: P=1, D=1, Q=1, m=12
""")

from statsmodels.tsa.statespace.sarimax import SARIMAX

sarima_model = SARIMAX(
    train,
    order=(1, 1, 1),                  # (p, d, q)
    seasonal_order=(1, 1, 1, 12),     # (P, D, Q, m)
    enforce_stationarity=False,
    enforce_invertibility=False
).fit(disp=False)                     # disp=False silences the fitting log

sarima_forecast = sarima_model.forecast(steps=horizon)
print(f"\nSARIMA MAE: {mae(test, sarima_forecast):.3f}°C")

# Confidence intervals
forecast_result = sarima_model.get_forecast(steps=horizon)
conf_int = forecast_result.conf_int(alpha=0.05)   # 95% confidence interval

plt.figure(figsize=(14, 5))
plt.plot(train.index, train,          label="Train",        color="steelblue", linewidth=1.5)
plt.plot(test.index,  test,           label="Actual (test)",color="black",     linewidth=2, marker="o")
plt.plot(test.index,  sarima_forecast,label="SARIMA forecast", color="tomato", linewidth=2, marker="D")
plt.fill_between(test.index,
                 conf_int.iloc[:, 0],
                 conf_int.iloc[:, 1],
                 color="tomato", alpha=0.2, label="95% confidence interval")
plt.title("SARIMA(1,1,1)(1,1,1,12) Forecast with Confidence Interval", fontsize=13)
plt.ylabel("°C")
plt.legend()
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.savefig("ts_sarima.png", bbox_inches="tight", dpi=100)
plt.show()
plt.close()
print("Saved: ts_sarima.png")
print("""
Confidence interval: the shaded region shows the 95% prediction interval.
The further out you forecast, the wider it gets — honest uncertainty.
Always show this to stakeholders, not just a single line!
""")


# ==============================================================
# PART 11: Lag Features for ML Models (Bridge to sklearn/XGBoost)
# ==============================================================

print("\n--- PART 11: Turning Time Series into ML Features ---")
print("""
ARIMA is great for classical forecasting, but XGBoost/sklearn don't
understand time. The trick: convert temporal patterns into plain features.

Instead of: "predict tomorrow from the history"
We do:      "predict tomorrow from engineered features of the past"

This approach lets you use ANY regression model for forecasting,
and it's often more accurate than ARIMA on large datasets.
""")

# Use daily data, create lag features and rolling statistics
ml_df = df[["temperature"]].copy()

# Lag features: the value N steps ago
for lag in [1, 2, 3, 7, 14, 30]:
    ml_df[f"lag_{lag}d"] = ml_df["temperature"].shift(lag)

# Rolling statistics
ml_df["roll_7d_mean"]  = ml_df["temperature"].shift(1).rolling(7).mean()
ml_df["roll_7d_std"]   = ml_df["temperature"].shift(1).rolling(7).std()
ml_df["roll_30d_mean"] = ml_df["temperature"].shift(1).rolling(30).mean()

# Calendar features (extracted from the DatetimeIndex)
ml_df["month"]       = ml_df.index.month
ml_df["day_of_week"] = ml_df.index.dayofweek  # 0=Monday, 6=Sunday
ml_df["day_of_year"] = ml_df.index.dayofyear
ml_df["week"]        = ml_df.index.isocalendar().week.astype(int)

# Target: next day's temperature
ml_df["target"] = ml_df["temperature"].shift(-1)

# Drop rows with NaN (from lag/rolling creation)
ml_df = ml_df.dropna()
print(f"Feature matrix shape: {ml_df.shape}")
print("\nFeatures used:")
feature_cols = [c for c in ml_df.columns if c not in ["temperature", "target"]]
for f in feature_cols:
    print(f"  {f}")

# Train/test split — ALWAYS split chronologically, NEVER randomly!
# Random splitting would let the model see "future" data during training.
cutoff = int(len(ml_df) * 0.8)
X_train = ml_df[feature_cols].iloc[:cutoff]
X_test  = ml_df[feature_cols].iloc[cutoff:]
y_train = ml_df["target"].iloc[:cutoff]
y_test  = ml_df["target"].iloc[cutoff:]

from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error

rf = RandomForestRegressor(n_estimators=100, random_state=42, n_jobs=-1)
rf.fit(X_train, y_train)
preds = rf.predict(X_test)
print(f"\nRandom Forest next-day temperature forecast MAE: {mean_absolute_error(y_test, preds):.3f}°C")

# Feature importances
importances = pd.Series(rf.feature_importances_, index=feature_cols).sort_values(ascending=False)
print("\nTop feature importances:")
print(importances.head(8).round(4))
print("""
HINT: lag_1d (yesterday's temperature) is almost always the most important
feature for next-day forecasting. This is the "persistence" baseline.
Rolling features capture medium-term trends.
Calendar features capture seasonality.
""")

# PART 13: Cheatsheet
# ==============================================================

print("""
=================================================================
TIME SERIES CHEATSHEET
=================================================================

INDEXING / SLICING:
  df["2024"]              → full year
  df["2024-06":"2024-08"] → date range
  df.resample("ME").mean()  → monthly mean
  df.resample("W").agg({"temp": "mean", "humidity": "max"})

ROLLING / SMOOTHING:
  df.rolling(7).mean()    → 7-period simple moving average
  df.ewm(span=14).mean()  → exponentially weighted moving average
  df.shift(1)             → lag by 1 (yesterday's value)
  df.diff(1)              → 1st difference (change vs. yesterday)

DECOMPOSITION:
  from statsmodels.tsa.seasonal import seasonal_decompose
  result = seasonal_decompose(series, model="additive", period=12)
  result.trend / result.seasonal / result.resid

STATIONARITY:
  from statsmodels.tsa.stattools import adfuller
  adfuller(series)        → p < 0.05 means stationary
  series.diff()           → make non-stationary series stationary

FORECASTING:
  Holt-Winters            → handles trend + seasonality (fast, interpretable)
  SARIMA(p,d,q)(P,D,Q,m) → classical statistical model
  Lag features + XGBoost  → ML approach (often best on large datasets)

TRAIN/TEST SPLIT:
  NEVER use train_test_split(shuffle=True) on time series!
  Always split chronologically:
  train = df.iloc[:cutoff]
  test  = df.iloc[cutoff:]

EVALUATION METRICS:
  MAE  = mean(|actual - predicted|)       → interpretable in original units
  RMSE = sqrt(mean((actual - predicted)²)) → penalises large errors more
  MAPE = mean(|actual - predicted| / |actual|) × 100  → % error (avoid if actuals ≈ 0)
=================================================================
""")


# ==============================================================
# PART 14: Prophet — Modern Forecasting for Supply Chain
# ==============================================================

print("\n--- PART 14: Prophet (Meta's Forecasting Library) ---")
print("""
Prophet vs. SARIMA — when to use each:

  SARIMA:
    + Best when you have <2 years of data
    + Best when seasonality is clean and regular
    + Requires manual selection of (p, d, q) from ACF/PACF plots
    - Breaks with missing dates; needs manual imputation
    - Slow to fit on long series

  Prophet:
    + Handles multiple seasonalities simultaneously (daily + weekly + yearly)
    + Handles missing data and outliers automatically
    + Handles public holidays with one parameter (add_country_holidays)
    + No manual parameter selection — just fit and forecast
    - Requires pip install prophet (heavier install than statsmodels)
    - Less interpretable than ARIMA for statisticians
    - Can overfit on short series (<1 year)

For retail demand forecasting with weekly + yearly patterns: use Prophet first.

""")

    prophet_train = prophet_df.iloc[:-4]
    prophet_test  = prophet_df.iloc[-4:]

    m = Prophet(
        seasonality_mode="additive",    # additive: seasonality is constant in amplitude
                                        # multiplicative: seasonality scales with trend
        yearly_seasonality=True,
        weekly_seasonality=False,       # monthly data has no weekly cycle
        daily_seasonality=False,

    m.fit(prophet_train)

    # make_future_dataframe extends the date range by `periods` steps
    future   = m.make_future_dataframe(periods=4, freq="ME")
