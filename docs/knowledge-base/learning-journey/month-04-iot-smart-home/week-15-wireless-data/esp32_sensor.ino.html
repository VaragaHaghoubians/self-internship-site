/*
  ==============================================================
  WEEK 15: ESP32 Wireless Sensor
  FILE: esp32_sensor.ino
  
  TASK: Read temperature from DHT11 sensor and send it wirelessly
        to your Flask server over WiFi every 5 minutes.
  
  HARDWARE WIRING (ESP32 + DHT11):
    DHT11 VCC  → ESP32 3.3V pin (NOT 5V! ESP32 uses 3.3V)
    DHT11 GND  → ESP32 GND pin
    DHT11 DATA → ESP32 GPIO pin 4
    
    Add a 10kΩ resistor between VCC and DATA pins (pull-up resistor)
    (This stabilizes the signal)
  
  LIBRARIES NEEDED (install in Arduino IDE):
    1. DHT sensor library by Adafruit
       → Tools > Manage Libraries > Search "DHT sensor library"
    2. Adafruit Unified Sensor
       → Tools > Manage Libraries > Search "Adafruit Unified Sensor"
  
  BEFORE UPLOADING:
    1. Set your WiFi name and password below
    2. Set your laptop's IP address (run 'ipconfig' on Windows or 'hostname -I' on Linux)
    3. Make sure your Flask server is running!
  ==============================================================
*/

#include <WiFi.h>           // ESP32 WiFi library (built-in)
#include <HTTPClient.h>     // For making HTTP POST requests (built-in)
#include <DHT.h>            // Adafruit DHT sensor library
#include <ArduinoJson.h>    // For creating JSON — install: "ArduinoJson" by Benoit Blanchon

// ==============================================================
// ⚙️  CONFIGURATION — CHANGE THESE!
// ==============================================================

const char* WIFI_SSID     = "YOUR_WIFI_NAME";        // Your WiFi network name
const char* WIFI_PASSWORD = "YOUR_WIFI_PASSWORD";    // Your WiFi password

// Your LAPTOP's IP address (the machine running flask_server.py)
// Run 'ipconfig' on Windows or 'hostname -I' on Linux to find it
const char* SERVER_IP     = "192.168.1.10";          // ← CHANGE THIS!
const int   SERVER_PORT   = 5000;
const char* DEVICE_ID     = "esp32_living_room";     // Give your device a name

// DHT Sensor settings
#define DHT_PIN  4          // GPIO pin where DHT11 DATA wire is connected
#define DHT_TYPE DHT11      // We're using DHT11 (use DHT22 for more accuracy)

// How often to send data (in milliseconds)
// 300000 = 5 minutes. Change to 30000 for every 30 seconds while testing
#define SEND_INTERVAL_MS 300000

// ==============================================================
// GLOBAL OBJECTS
// ==============================================================

DHT dht(DHT_PIN, DHT_TYPE);

unsigned long lastSendTime = 0;   // Tracks when we last sent data


// ==============================================================
// SETUP — runs once when ESP32 powers on
// ==============================================================

void setup() {
    Serial.begin(115200);         // Start serial communication for debugging
    dht.begin();                  // Initialize DHT sensor
    
    Serial.println("\n==============================");
    Serial.println("ESP32 Smart Sensor Starting...");
    Serial.println("==============================");
    
    connectToWiFi();              // Connect to your home WiFi
}


// ==============================================================
// MAIN LOOP — runs forever after setup()
// ==============================================================

void loop() {
    // Check if WiFi disconnected and reconnect if needed
    if (WiFi.status() != WL_CONNECTED) {
        Serial.println("WiFi disconnected! Reconnecting...");
        connectToWiFi();
    }
    
    // Check if it's time to send data
    unsigned long currentTime = millis();   // millis() = ms since device started
    
    if (currentTime - lastSendTime >= SEND_INTERVAL_MS) {
        // Time to take a reading and send it!
        readAndSendData();
        lastSendTime = currentTime;   // Reset the timer
    }
    
    // Small delay to prevent watchdog timer issues
    delay(1000);
}


// ==============================================================
// FUNCTION: Connect to WiFi
// ==============================================================

void connectToWiFi() {
    Serial.print("Connecting to WiFi: ");
    Serial.println(WIFI_SSID);
    
    WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
    
    int attempts = 0;
    while (WiFi.status() != WL_CONNECTED && attempts < 20) {
        delay(500);
        Serial.print(".");
        attempts++;
    }
    
    if (WiFi.status() == WL_CONNECTED) {
        Serial.println("\nConnected!");
        Serial.print("ESP32 IP address: ");
        Serial.println(WiFi.localIP());
        
        // Blink built-in LED 3 times to show successful connection
        for (int i = 0; i < 3; i++) {
            digitalWrite(2, HIGH); delay(200);
            digitalWrite(2, LOW);  delay(200);
        }
    } else {
        Serial.println("\nFailed to connect! Check SSID and password.");
    }
}


// ==============================================================
// FUNCTION: Read sensor and send data to Flask server
// ==============================================================

void readAndSendData() {
    Serial.println("\n--- Taking a reading... ---");
    
    // Read from DHT11 sensor
    float humidity    = dht.readHumidity();
    float temperature = dht.readTemperature();   // Celsius by default
    
    // Validate readings (NaN = "Not a Number" = sensor failed)
    if (isnan(humidity) || isnan(temperature)) {
        Serial.println("ERROR: DHT sensor read failed!");
        Serial.println("Check your wiring (VCC, GND, DATA pin)");
        return;   // Exit function early — don't send bad data
    }
    
    Serial.print("Temperature: ");
    Serial.print(temperature);
    Serial.println("°C");
    
    Serial.print("Humidity: ");
    Serial.print(humidity);
    Serial.println("%");
    
    // Build the server URL
    // Example: "http://192.168.1.10:5000/upload_data"
    String serverURL = "http://";
    serverURL += SERVER_IP;
    serverURL += ":";
    serverURL += SERVER_PORT;
    serverURL += "/upload_data";
    
    // Create JSON payload
    // We use the ArduinoJson library to safely build JSON
    StaticJsonDocument<200> doc;
    doc["temperature"] = temperature;
    doc["humidity"]    = humidity;
    doc["device_id"]   = DEVICE_ID;
    
    String jsonPayload;
    serializeJson(doc, jsonPayload);
    // jsonPayload is now: {"temperature":22.5,"humidity":55.0,"device_id":"esp32_living_room"}
    
    Serial.print("Sending to: ");
    Serial.println(serverURL);
    Serial.print("Payload: ");
    Serial.println(jsonPayload);
    
    // Send HTTP POST request
    HTTPClient http;
    http.begin(serverURL);
    http.addHeader("Content-Type", "application/json");
    
    int responseCode = http.POST(jsonPayload);
    
    if (responseCode > 0) {
        String response = http.getString();
        Serial.print("Server responded (");
        Serial.print(responseCode);
        Serial.print("): ");
        Serial.println(response);
        
        if (responseCode == 201) {
            Serial.println("✓ Data saved successfully!");
        }
    } else {
        Serial.print("ERROR: HTTP request failed. Code: ");
        Serial.println(responseCode);
        Serial.println("Is the Flask server running? Is the IP correct?");
    }
    
    http.end();   // Free resources
}
