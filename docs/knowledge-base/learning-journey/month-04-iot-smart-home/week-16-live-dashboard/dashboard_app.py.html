# ==============================================================
# WEEK 16: Live Dashboard — SQLite + Flask + Auto-Refreshing Chart
# FILE: dashboard_app.py
#
# TASK: Save incoming ESP32 sensor data to a SQLite database,
#       then serve a live HTML chart that auto-refreshes every
#       10 seconds in the browser.
#
# WHY:  This is the final piece of your IoT pipeline:
#         [ESP32] → [Flask POST] → [SQLite] → [Live Chart in Browser]
#       This is the same concept used in factory dashboards,
#       hospital monitors, and smart city systems.
#
# PREREQS:
#   - Week 14's Flask server running (or merged into this file)
#   - ESP32 sending POST requests with temperature data
#   - pip install flask
#
# HOW TO RUN: python dashboard_app.py
# THEN OPEN:  http://localhost:5000
# ==============================================================

from flask import Flask, request, jsonify, render_template_string
import sqlite3
import datetime
import os

app = Flask(__name__)

# ==============================================================
# PART 1: DATABASE SETUP
# ==============================================================

# SQLite stores everything in a single .db file — no server needed.
# Perfect for a Raspberry Pi or laptop project.

DB_PATH = "sensor_data.db"

def init_db():
    """Create the database and table if they don't exist yet."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS readings (
            id          INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp   TEXT    NOT NULL,
            temperature REAL    NOT NULL,
            humidity    REAL,
            device_id   TEXT    DEFAULT 'esp32-01'
        )
    """)
    conn.commit()
    conn.close()
    print(f"[DB] Database ready at: {DB_PATH}")

def save_reading(temperature, humidity=None, device_id="esp32-01"):
    """Insert one sensor reading into the database."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    cursor.execute(
        "INSERT INTO readings (timestamp, temperature, humidity, device_id) VALUES (?, ?, ?, ?)",
        (datetime.datetime.now().isoformat(), temperature, humidity, device_id)
    )
    conn.commit()
    conn.close()

def get_recent_readings(limit=50):
    """Fetch the most recent N readings from the database."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    cursor.execute(
        "SELECT timestamp, temperature, humidity FROM readings ORDER BY id DESC LIMIT ?",
        (limit,)
    )
    rows = cursor.fetchall()
    conn.close()
    # Reverse so oldest is first (for left-to-right chart)
    return list(reversed(rows))

def get_stats():
    """Calculate summary statistics across all readings."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    cursor.execute("""
        SELECT
            COUNT(*)         AS total,
            ROUND(AVG(temperature), 2) AS avg_temp,
            ROUND(MIN(temperature), 2) AS min_temp,
            ROUND(MAX(temperature), 2) AS max_temp
        FROM readings
    """)
    row = cursor.fetchone()
    conn.close()
    return {"total": row[0], "avg": row[1], "min": row[2], "max": row[3]}


# ==============================================================
# PART 2: FLASK ROUTES
# ==============================================================

@app.route("/data", methods=["POST"])
def receive_data():
    """
    Endpoint the ESP32 POSTs to.
    Expected JSON body: {"temperature": 24.5, "humidity": 60.1}
    """
    payload = request.get_json(silent=True)
    if not payload or "temperature" not in payload:
        return jsonify({"error": "Missing temperature field"}), 400

    temp = float(payload["temperature"])
    hum  = float(payload.get("humidity", 0)) or None
    save_reading(temp, hum)

    print(f"[DATA] Received: {temp}°C  humidity={hum}%")
    return jsonify({"status": "ok", "temperature": temp}), 201


@app.route("/api/readings")
def api_readings():
    """Return the last 50 readings as JSON (used by the chart's JavaScript)."""
    rows = get_recent_readings(50)
    data = [
        {"timestamp": r[0], "temperature": r[1], "humidity": r[2]}
        for r in rows
    ]
    return jsonify(data)


@app.route("/api/stats")
def api_stats():
    """Return summary statistics as JSON."""
    return jsonify(get_stats())


@app.route("/")
def dashboard():
    """
    Serve the main dashboard page.
    The page uses Chart.js (loaded from CDN) to draw the temperature graph.
    JavaScript auto-refreshes every 10 seconds without reloading the page.
    """
    html = """
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Smart Home Dashboard</title>
    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>
    <style>
        body  { font-family: Arial, sans-serif; background: #1a1a2e; color: #eee; margin: 0; padding: 20px; }
        h1    { color: #e94560; text-align: center; }
        .stats { display: flex; justify-content: center; gap: 30px; margin: 20px 0; }
        .stat-box { background: #16213e; padding: 15px 25px; border-radius: 8px; text-align: center; }
        .stat-box .value { font-size: 2em; color: #0f3460; color: #e94560; }
        .stat-box .label { font-size: 0.9em; color: #aaa; }
        canvas { max-width: 900px; display: block; margin: 0 auto; }
        #status { text-align: center; color: #aaa; font-size: 0.85em; margin-top: 10px; }
    </style>
</head>
<body>
    <h1>🌡️ Smart Home Live Dashboard</h1>

    <div class="stats" id="stats-box">
        <div class="stat-box"><div class="value" id="stat-total">—</div><div class="label">Total Readings</div></div>
        <div class="stat-box"><div class="value" id="stat-avg">—</div><div class="label">Avg Temp (°C)</div></div>
        <div class="stat-box"><div class="value" id="stat-min">—</div><div class="label">Min Temp (°C)</div></div>
        <div class="stat-box"><div class="value" id="stat-max">—</div><div class="label">Max Temp (°C)</div></div>
    </div>

    <canvas id="tempChart" height="100"></canvas>
    <div id="status">Auto-refreshes every 10 seconds</div>

    <script>
        // Initialize Chart.js with empty data — we'll fill it from the API
        const ctx = document.getElementById('tempChart').getContext('2d');
        const chart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: [],
                datasets: [{
                    label: 'Temperature (°C)',
                    data: [],
                    borderColor: '#e94560',
                    backgroundColor: 'rgba(233, 69, 96, 0.1)',
                    borderWidth: 2,
                    pointRadius: 3,
                    tension: 0.3,
                    fill: true
                }]
            },
            options: {
                responsive: true,
                scales: {
                    x: { ticks: { color: '#aaa' }, grid: { color: '#333' } },
                    y: { ticks: { color: '#aaa' }, grid: { color: '#333' },
                         title: { display: true, text: 'Temperature (°C)', color: '#aaa' } }
                },
                plugins: { legend: { labels: { color: '#eee' } } }
            }
        });

        async function refresh() {
            // 1. Fetch latest readings and update the chart
            const res  = await fetch('/api/readings');
            const data = await res.json();

            chart.data.labels = data.map(r => r.timestamp.slice(11, 19)); // HH:MM:SS
            chart.data.datasets[0].data = data.map(r => r.temperature);
            chart.update();

            // 2. Fetch stats and update the stat boxes
            const sres  = await fetch('/api/stats');
            const stats = await sres.json();
            document.getElementById('stat-total').innerText = stats.total;
            document.getElementById('stat-avg').innerText   = stats.avg + '°C';
            document.getElementById('stat-min').innerText   = stats.min + '°C';
            document.getElementById('stat-max').innerText   = stats.max + '°C';

            // 3. Update status timestamp
            document.getElementById('status').innerText =
                'Last updated: ' + new Date().toLocaleTimeString() + ' — auto-refreshes every 10 seconds';
        }

        // Load immediately, then every 10 seconds
        refresh();
        setInterval(refresh, 10000);
    </script>
</body>
</html>
"""
    return render_template_string(html)

# PART 3: TEST DATA GENERATOR (for testing without real ESP32)
# ==============================================================

def seed_test_data(n=30):
    """
    Insert fake readings so you can test the dashboard
    without having an ESP32 connected.
    Run this ONCE to populate your database.
    """
    import random
    print(f"[SEED] Inserting {n} fake readings...")
    for i in range(n):
        fake_temp = round(random.uniform(20.0, 32.0), 1)
        fake_hum  = round(random.uniform(40.0, 80.0), 1)
        save_reading(fake_temp, fake_hum)
    print("[SEED] Done! Open http://localhost:5000 to see the chart.")
