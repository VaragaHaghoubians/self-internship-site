# ==============================================================================
# LESSON 01 — CLASS-BASED VIEWS (CBVs)
# ==============================================================================
#
# TEACHER SAYS:
# Function-Based Views (FBV) are explicit and easy to understand.
# Class-Based Views (CBV) handle the BOILERPLATE automatically.
# - ListView:     query + paginate + render a list template
# - DetailView:   get one object by pk + render a detail template
# - CreateView:   render form, validate, save, redirect
# - UpdateView:   load existing object + form, validate, save, redirect
# - DeleteView:   confirm + delete
#
# COMPARISON — the same "create student" in FBV vs CBV:
#
# FBV (20 lines):
#   def create_student(request):
#       if request.method == 'POST':
#           form = StudentForm(request.POST)
#           if form.is_valid():
#               form.save()
#               return redirect('students:list')
#       else:
#           form = StudentForm()
#       return render(request, 'students/form.html', {'form': form})
#
# CBV (5 lines):
#   class StudentCreateView(LoginRequiredMixin, CreateView):
#       model = Student
#       fields = ['first_name', 'last_name', 'email', 'gpa']
#       success_url = reverse_lazy('students:list')
#
# The CBV does ALL of that FBV logic automatically. Less repetition = fewer bugs.

# LESSON 02 — CELERY (Async Tasks — Critical for ML Job Queuing)
# ==============================================================================
#
# TEACHER SAYS:
# Scenario: User uploads a CSV, you need to train a model on it.
# If you train SYNCHRONOUSLY in the view: the HTTP request hangs for 10 minutes.
# The browser times out. The user is furious.
#
# Celery solves this:
# 1. Request comes in → view immediately returns "your job is queued!"
# 2. A CELERY WORKER (separate process) picks up the job
# 3. Worker trains the model in the background
# 4. When done: saves result, sends email, updates status
#
# COMPONENTS:
#   Celery worker  → the process that executes tasks (run separately from Django)
#   Broker         → message queue: tasks go IN here (Redis is most common)
#   Backend        → stores results (Redis or database)
#
# INSTALL:
#   pip install celery redis
#   pip install django-celery-results   (optional — store results in Django DB)

from students.tasks import train_model_task, send_welcome_email

# Kick off async task — returns IMMEDIATELY:
def upload_dataset(request):
    # ... save uploaded file ...
    job = train_model_task.delay(student_id=1, dataset_path='/data/upload.csv')
    return JsonResponse({
        'status': 'queued',
        'job_id': job.id,   # Use this to check progress
    })

# Check task status:
def job_status(request, job_id):
    from celery.result import AsyncResult
    result = AsyncResult(job_id)
    return JsonResponse({
        'state':  result.state,      # PENDING, STARTED, SUCCESS, FAILURE
        'result': result.result if result.successful() else None,
    })
'''

SEED_COMMAND_EXAMPLE = '''
# students/management/commands/seed_data.py
from django.core.management.base import BaseCommand, CommandError
from django.db import transaction
from students.models import Student, Major, Course, Enrollment
import random


class Command(BaseCommand):
    help = "Seed the database with realistic fake data for development/testing"

    def add_arguments(self, parser):
        """Define command-line arguments: python manage.py seed_data --students 50"""
        parser.add_argument('--students', type=int, default=20, help='Number of students to create')
        parser.add_argument('--clear',    action='store_true',   help='Clear all data before seeding')

    def handle(self, *args, **options):
        n_students = options['students']

        if options['clear']:
            self.stdout.write("Clearing existing data...")
            Enrollment.objects.all().delete()
            Student.objects.all().delete()
            Major.objects.all().delete()
            Course.objects.all().delete()

        self.stdout.write("Seeding data...")

        # Create majors:
        majors = []
        for name in ['Data Science', 'Computer Science', 'Statistics', 'Mathematics']:
            major, _ = Major.objects.get_or_create(name=name, defaults={'department': 'Engineering'})
            majors.append(major)

        # Create students:
        first_names = ['Alice', 'Bob', 'Carol', 'David', 'Eve', 'Frank', 'Grace', 'Henry']
        last_names  = ['Smith', 'Jones', 'Brown', 'Davis', 'Wilson', 'Moore', 'Taylor']

        with transaction.atomic():   # All-or-nothing — rollback if any insert fails
            for i in range(n_students):
                first = random.choice(first_names)
                last  = random.choice(last_names)
                Student.objects.get_or_create(
                    email=f"{first.lower()}.{last.lower()}{i}@uni.edu",
                    defaults={
                        'first_name': first,
                        'last_name':  last,
                        'age':        random.randint(20, 30),
                        'gpa':        round(random.uniform(2.5, 4.0), 2),
                        'major':      random.choice(majors),
                    }
                )

        count = Student.objects.count()
        self.stdout.write(self.style.SUCCESS(f"✓ Created {count} students"))

        # Print a warning:
        if options['clear']:
            self.stdout.write(self.style.WARNING("WARNING: Production data was cleared!"))
'''

# CSV Import view:
CSV_IMPORT_VIEW = '''
import csv
import io
from django.shortcuts import render, redirect
from django.contrib import messages

def import_students_csv(request):
    """Accept a CSV upload and bulk-create students."""
    if request.method == 'POST' and request.FILES.get('csv_file'):
        csv_file = request.FILES['csv_file']

        # Validate file type:
        if not csv_file.name.endswith('.csv'):
            messages.error(request, 'Please upload a CSV file.')
            return redirect('students:import')

        # Read and parse:
        decoded = csv_file.read().decode('utf-8')
        reader = csv.DictReader(io.StringIO(decoded))

        created = 0
        errors  = []
        for i, row in enumerate(reader):
            try:
                Student.objects.create(
                    first_name = row['first_name'].strip(),
                    last_name  = row['last_name'].strip(),
                    email      = row['email'].strip(),
                    age        = int(row['age']),
                    gpa        = float(row['gpa']),
                )
                created += 1
            except (KeyError, ValueError) as e:
                errors.append(f"Row {i+2}: {e}")

        if errors:
            messages.warning(request, f"Created {created} students. {len(errors)} errors: {errors[:3]}")
        else:
            messages.success(request, f"Successfully imported {created} students!")

        return redirect('students:list')

    return render(request, 'students/import.html')
'''
