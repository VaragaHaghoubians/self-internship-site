# DAY 03 — MODELS (Database Tables as Python Classes)
# ==============================================================================
#
# TEACHER SAYS:
# A MODEL is a Python class that maps to a database table.
# Each class attribute = one database column.
# Django uses its ORM (Object-Relational Mapper) to translate Python to SQL.
# You NEVER write SQL manually in Django (unless you want to!).
#
# COMMON FIELD TYPES:
#   CharField(max_length=n)  → VARCHAR  — for short text
#   TextField()              → TEXT     — for long text
#   IntegerField()           → INT
#   FloatField()             → FLOAT
#   DecimalField(max_digits, decimal_places) → DECIMAL
#   BooleanField()           → BOOLEAN
#   DateField()              → DATE
#   DateTimeField()          → DATETIME
#   EmailField()             → VARCHAR with email validation
#   ForeignKey(Model, on_delete=...) → link to another table
#   ManyToManyField(Model)   → many-to-many relationship
#
# FIELD OPTIONS (constraints):
#   null=True          → allows NULL in database
#   blank=True         → allows empty in forms
#   default=value      → default value
#   unique=True        → must be unique
#   choices=LIST       → dropdown list of options
#
# Open students/models.py and replace with:

from django.db import models
from django.core.validators import MinValueValidator, MaxValueValidator


class Major(models.Model):
    name        = models.CharField(max_length=100, unique=True)
    department  = models.CharField(max_length=100)
    description = models.TextField(blank=True)

    class Meta:
        ordering = ['name']         # Default sort order

    def __str__(self):
        return self.name


class Student(models.Model):
    YEAR_CHOICES = [
        (1, 'Freshman'),
        (2, 'Sophomore'),
        (3, 'Junior'),
        (4, 'Senior'),
        (5, 'Graduate'),
    ]

    first_name   = models.CharField(max_length=50)
    last_name    = models.CharField(max_length=50)
    email        = models.EmailField(unique=True)
    age          = models.IntegerField(
                        validators=[MinValueValidator(16), MaxValueValidator(60)]
                   )
    gpa          = models.DecimalField(
                        max_digits=3,
                        decimal_places=2,
                        validators=[MinValueValidator(0.0), MaxValueValidator(4.0)],
                        default=0.00
                   )
    major        = models.ForeignKey(
                        Major,
                        on_delete=models.SET_NULL,  # If major deleted, set to NULL
                        null=True,
                        blank=True,
                        related_name='students'     # major.students.all() to get students
                   )
    year         = models.IntegerField(choices=YEAR_CHOICES, default=1)
    is_active    = models.BooleanField(default=True)
    enrolled_on  = models.DateField(auto_now_add=True)  # Set automatically on creation
    updated_at   = models.DateTimeField(auto_now=True)  # Updated on every save

    class Meta:
        ordering = ['last_name', 'first_name']

    def __str__(self):
        return f"{self.first_name} {self.last_name} ({self.email})"

    def get_full_name(self):
        return f"{self.first_name} {self.last_name}"

    @property
    def status(self):
        return "Good Standing" if self.gpa >= 3.0 else "Probation"


class Course(models.Model):
    code         = models.CharField(max_length=10, unique=True)
    name         = models.CharField(max_length=200)
    credits      = models.IntegerField(
                        validators=[MinValueValidator(1), MaxValueValidator(6)]
                   )
    description  = models.TextField(blank=True)
    professor    = models.CharField(max_length=100, blank=True)
    students     = models.ManyToManyField(
                        Student,
                        through='Enrollment',       # Explicit junction table
                        related_name='courses'
                   )

    class Meta:
        ordering = ['code']

    def __str__(self):
        return f"{self.code}: {self.name}"


class Enrollment(models.Model):
    student      = models.ForeignKey(Student, on_delete=models.CASCADE)
    course       = models.ForeignKey(Course, on_delete=models.CASCADE)
    grade        = models.DecimalField(
                        max_digits=5,
                        decimal_places=2,
                        null=True,
                        blank=True
                   )
    enrolled_on  = models.DateField(auto_now_add=True)

    class Meta:
        unique_together = ['student', 'course']     # Each student enrolls once per course

    def __str__(self):
        return f"{self.student} → {self.course}"

from django import forms

class StudentForm(forms.ModelForm):
    """Form for creating and editing students."""

    class Meta:
        # model  = Student
        fields = ['first_name', 'last_name', 'email', 'age', 'gpa', 'major', 'year']
        widgets = {
            'first_name': forms.TextInput(attrs={'class': 'form-control', 'placeholder': 'First Name'}),
            'last_name':  forms.TextInput(attrs={'class': 'form-control'}),
            'email':      forms.EmailInput(attrs={'class': 'form-control'}),
            'age':        forms.NumberInput(attrs={'class': 'form-control', 'min': 16, 'max': 60}),
            'gpa':        forms.NumberInput(attrs={'class': 'form-control', 'step': '0.01'}),
        }

    def clean_gpa(self):
        """Custom validation for GPA field."""
        gpa = self.cleaned_data.get('gpa')
        if gpa is not None and (gpa < 0 or gpa > 4.0):
            raise forms.ValidationError("GPA must be between 0.0 and 4.0")
        return gpa

# DAY 08 — DJANGO ORM: ADVANCED QUERIES
# ==============================================================================
#
# TEACHER SAYS:
# Django's ORM is incredibly powerful — you can do almost any SQL query in Python.
# The queryset is LAZY — it doesn't hit the database until you actually need the data.
#
# QUERYSET METHODS:
#   Model.objects.all()                → SELECT * FROM table
#   Model.objects.filter(field=value)  → WHERE field = value
#   Model.objects.exclude(...)         → WHERE NOT ...
#   Model.objects.get(pk=1)            → SELECT ... WHERE pk=1 (raises error if none/many)
#   Model.objects.order_by('field')    → ORDER BY field ASC
#   Model.objects.order_by('-field')   → ORDER BY field DESC
#   queryset.count()                   → COUNT(*)
#   queryset.first() / .last()         → first/last result
#   queryset.values('field1', ...)     → returns dicts instead of objects
#   queryset.distinct()                → SELECT DISTINCT
#   queryset[:5]                       → LIMIT 5
#
# FIELD LOOKUPS (use __ double underscore):
#   filter(gpa__gte=3.5)   → gpa >= 3.5   (gte, lte, gt, lt)
#   filter(name__contains='Ali')  → LIKE '%Ali%'
#   filter(name__icontains='ali') → case-insensitive
#   filter(name__startswith='A')
#   filter(major__name='Data Science')   → follow FK!
#
# EXAMPLE QUERIES (run in Django shell: python manage.py shell):

ORM_EXAMPLES = '''
# Run these in: python manage.py shell

from students.models import Student, Course, Enrollment

# Basic queries:
all_students   = Student.objects.all()
ds_students    = Student.objects.filter(major__name='Data Science')
high_gpa       = Student.objects.filter(gpa__gte=3.7).order_by('-gpa')
student_alice  = Student.objects.get(email='alice@uni.edu')

# Aggregation:
from django.db.models import Avg, Max, Min, Count
Student.objects.aggregate(avg_gpa=Avg('gpa'), max_gpa=Max('gpa'))

# Group by (annotate):
from students.models import Major
Major.objects.annotate(
    student_count=Count('students'),
    avg_gpa=Avg('students__gpa')
).values('name', 'student_count', 'avg_gpa')

# select_related — avoids N+1 query problem for ForeignKey:
students = Student.objects.select_related('major').all()
# Now accessing student.major.name doesn't hit the database again!

# prefetch_related — for ManyToMany and reverse FK:
students = Student.objects.prefetch_related('courses').all()

# Q objects — complex OR/AND conditions:
from django.db.models import Q
Student.objects.filter(Q(gpa__gte=3.8) | Q(year=5))   # GPA >= 3.8 OR Graduate student
'''

# students/serializers.py
from rest_framework import serializers
from .models import Student, Course

class CourseSerializer(serializers.ModelSerializer):
    class Meta:
        model  = Course
        fields = ['id', 'code', 'name', 'credits', 'professor']

class StudentSerializer(serializers.ModelSerializer):
    full_name = serializers.SerializerMethodField()
    courses   = CourseSerializer(many=True, read_only=True)
