"""Train a churn model and save it — run ONCE before building the Docker image."""
import pickle
from sklearn.ensemble import RandomForestClassifier
from sklearn.preprocessing import StandardScaler
from sklearn.pipeline import Pipeline
import numpy as np

# Simulate training data (in real life: load from CSV/database)
np.random.seed(42)
n = 500
X = np.column_stack([
    np.random.randint(0, 100, n),    # days_since_last_order
    np.random.randint(0, 50, n),     # total_orders
    np.random.uniform(0, 10000, n),  # lifetime_value
    np.random.randint(0, 5, n),      # support_tickets
])
y = ((X[:, 0] > 60) & (X[:, 1] < 10)).astype(int)  # fake churn label

# Build sklearn Pipeline (scaler + model in one object)
pipeline = Pipeline([
    ('scaler', StandardScaler()),
    ('model',  RandomForestClassifier(n_estimators=100, random_state=42)),
])
pipeline.fit(X, y)

# Save the pipeline
with open('models/churn_pipeline.pkl', 'wb') as f:
    pickle.dump(pipeline, f)

print("Model saved to models/churn_pipeline.pkl")
print(f"Training accuracy: {pipeline.score(X, y):.3f}")
TRAIN_EOF

# STEP 2: Create the FastAPI application:
cat > src/app.py << 'APP_EOF'
"""FastAPI app that loads the trained model and serves predictions."""
import pickle
import os
from pathlib import Path
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel, Field
import numpy as np

app = FastAPI(title="Churn Prediction API", version="1.0.0")

# Load model at startup — NOT per request (expensive operation!)
MODEL_PATH = os.getenv("MODEL_PATH", "/app/models/churn_pipeline.pkl")

def load_model():
    try:
        with open(MODEL_PATH, 'rb') as f:
            return pickle.load(f)
    except FileNotFoundError:
        raise RuntimeError(f"Model not found at {MODEL_PATH}. Train first.")

model = load_model()


class CustomerFeatures(BaseModel):
    days_since_last_order: int = Field(ge=0, le=3650, description="Days since last order")
    total_orders:          int = Field(ge=0, description="Total historical orders")
    lifetime_value:      float = Field(ge=0, description="Total spend in USD")
    support_tickets:       int = Field(ge=0, description="Number of support tickets")


class PredictionResponse(BaseModel):
    customer_id:    str
    churn_prob:   float
    churn_label:    str
    confidence:   float


@app.get("/health")
def health():
    return {"status": "ok", "model_loaded": model is not None}


@app.get("/info")
def info():
    return {
        "model": "RandomForestClassifier",
        "version": "1.0.0",
        "features": ["days_since_last_order", "total_orders", "lifetime_value", "support_tickets"],
    }


@app.post("/predict", response_model=PredictionResponse)
def predict(features: CustomerFeatures, customer_id: str = "unknown"):
    X = np.array([[
        features.days_since_last_order,
        features.total_orders,
        features.lifetime_value,
        features.support_tickets,
    ]])
    prob = float(model.predict_proba(X)[0][1])
    label = "CHURN" if prob >= 0.5 else "RETAIN"
    return PredictionResponse(
        customer_id=customer_id,
        churn_prob=round(prob, 4),
        churn_label=label,
        confidence=round(max(prob, 1 - prob), 4),
    )


@app.post("/predict/batch")
def predict_batch(batch: list[CustomerFeatures]):
    if len(batch) > 500:
        raise HTTPException(status_code=400, detail="Max batch size is 500")
    X = np.array([[
        f.days_since_last_order, f.total_orders, f.lifetime_value, f.support_tickets
    ] for f in batch])
    probs = model.predict_proba(X)[:, 1]
    return [
        {"index": i, "churn_prob": round(float(p), 4), "churn_label": "CHURN" if p >= 0.5 else "RETAIN"}
        for i, p in enumerate(probs)
    ]
APP_EOF

# Stage 1: install dependencies
FROM python:3.11-slim AS builder

WORKDIR /app

# System deps for sklearn (needed for compilation)
RUN apt-get update && apt-get install -y --no-install-recommends \
    gcc \
    && rm -rf /var/lib/apt/lists/*

COPY requirements.txt .
RUN pip install --user --no-cache-dir -r requirements.txt

# Stage 2: runtime image (no build tools, smaller!)
FROM python:3.11-slim AS runtime

# Non-root user for security
RUN useradd --create-home --shell /bin/bash mluser
WORKDIR /app

# Copy installed packages from builder
COPY --from=builder /root/.local /home/mluser/.local

# Copy app code and pre-trained model
COPY --chown=mluser:mluser src/ ./src/
COPY --chown=mluser:mluser models/ ./models/

USER mluser
ENV PATH=/home/mluser/.local/bin:$PATH
ENV PYTHONUNBUFFERED=1
ENV MODEL_PATH=/app/models/churn_pipeline.pkl

EXPOSE 8000

# Health check — Docker will restart container if this fails repeatedly
HEALTHCHECK --interval=30s --timeout=5s --start-period=10s --retries=3 \
    CMD python3 -c "import urllib.request; urllib.request.urlopen('http://localhost:8000/health')" || exit 1

CMD ["uvicorn", "src.app:app", "--host", "0.0.0.0", "--port", "8000"]
DOCKER_EOF

# ==============================================================================
# LESSON 02 — DOCKER NETWORKING (Containers Talking to Each Other)
# ==============================================================================
#
# TEACHER SAYS:
# Containers are ISOLATED by default — they can't reach each other.
# Docker networks let you connect them while keeping them isolated from everything else.
#
# THREE NETWORK TYPES:
#   bridge   → default. Containers on same bridge network can communicate by name!
#   host     → container shares host's network (no isolation — use carefully)
#   none     → no networking at all
#
# REAL SCENARIO: FastAPI app needs to reach PostgreSQL container.
# WRONG: hardcode 127.0.0.1 (that's the container's localhost, not the DB container)
# RIGHT: use the CONTAINER NAME as hostname on a shared bridge network

# Create a custom network:
docker network create ml-network
docker network ls                        # List all networks
docker network inspect ml-network        # See which containers are connected

# Run PostgreSQL on the ml-network:
docker run -d \
    --name ml-postgres \
    --network ml-network \
    -e POSTGRES_DB=churndb \
    -e POSTGRES_USER=mluser \
    -e POSTGRES_PASSWORD=mlpassword \
    postgres:15

# Run your FastAPI app on the SAME network:
docker run -d \
    --name churn-api \
    --network ml-network \
    -p 8000:8000 \
    -e DATABASE_URL=postgresql://mluser:mlpassword@ml-postgres:5432/churndb \
    churn-api:v1
# Note: ml-postgres is the container NAME — Docker resolves it to the right IP
# This is "Docker DNS" — works automatically on custom networks!

# Verify they can communicate:
docker exec churn-api ping ml-postgres -c 3
docker exec churn-api curl -s http://localhost:8000/health

# Network cleanup:
docker stop churn-api ml-postgres
docker rm churn-api ml-postgres
docker network rm ml-network


# ==============================================================================
# LESSON 03 — DOCKER COMPOSE FOR THE FULL ML STACK
# ==============================================================================
#
# Networks in Docker Compose are automatic — services in the same compose file
# can reach each other by their service name. No manual network creation needed!

cat > docker-compose.ml.yml << 'COMPOSE_EOF'
version: '3.9'

services:

  # Feature store / operational database
  postgres:
    image: postgres:15
    container_name: churn_postgres
    environment:
      POSTGRES_DB:       churndb
      POSTGRES_USER:     mluser
      POSTGRES_PASSWORD: mlpassword
    volumes:
      - pgdata:/var/lib/postgresql/data
    ports:
      - "5432:5432"   # Expose to host for debugging (remove in pure production)
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U mluser -d churndb"]
      interval: 10s
      retries: 5

  # ML Prediction API
  api:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: churn_api
    ports:
      - "8000:8000"
    environment:
      DATABASE_URL: postgresql://mluser:mlpassword@postgres:5432/churndb
      MODEL_PATH: /app/models/churn_pipeline.pkl
      LOG_LEVEL: INFO
    depends_on:
      postgres:
        condition: service_healthy   # Wait for DB to be ready!
    volumes:
      - ./models:/app/models        # Mount models dir — swap model without rebuild!
    restart: unless-stopped         # Auto-restart if container crashes

  # Redis for caching predictions (optional but common in production)
  redis:
    image: redis:7-alpine
    container_name: churn_redis
    ports:
      - "6379:6379"

  # MLflow model tracking server (optional — for experiment tracking)
  mlflow:
    image: ghcr.io/mlflow/mlflow:v2.10.2
    container_name: churn_mlflow
    ports:
      - "5000:5000"
    command: mlflow server --host 0.0.0.0 --port 5000
    volumes:
      - ./mlruns:/mlruns

volumes:
  pgdata:
COMPOSE_EOF

# ==============================================================================
# LESSON 05 — AWS ECR (Elastic Container Registry — Professional)
# ==============================================================================
#
# TEACHER SAYS:
# Docker Hub is public. In companies, you push to a PRIVATE registry.
# AWS ECR is the standard at Amazon (and most AWS shops).
# The workflow is the same as Docker Hub — just different login/tag commands.
#
# PREREQUISITE: AWS CLI installed and configured (aws configure)
# Install: pip install awscli OR brew install awscli
# Setup:   aws configure   (enter your AWS Access Key ID and Secret)

AWS_ACCOUNT_ID="123456789012"     # Your AWS account ID
AWS_REGION="us-east-1"            # Your AWS region
REPO_NAME="churn-api"

# Step 1: Create ECR repository (one-time):
aws ecr create-repository \
    --repository-name $REPO_NAME \
    --region $AWS_REGION

# Step 2: Login to ECR (token expires every 12h — refresh before push):
aws ecr get-login-password --region $AWS_REGION | \
    docker login --username AWS --password-stdin \
    $AWS_ACCOUNT_ID.dkr.ecr.$AWS_REGION.amazonaws.com

# Step 3: Tag your image for ECR:
ECR_URI="$AWS_ACCOUNT_ID.dkr.ecr.$AWS_REGION.amazonaws.com/$REPO_NAME"
docker tag churn-api:v1 $ECR_URI:v1
docker tag churn-api:v1 $ECR_URI:latest

# Step 4: Push:
docker push $ECR_URI:v1
docker push $ECR_URI:latest

# Step 5: Pull anywhere (on your EC2, ECS, EKS):
docker pull $ECR_URI:v1


# ==============================================================================
# LESSON 06 — GITHUB ACTIONS CI/CD FOR DOCKER (Automated Build & Push)
# ==============================================================================
#
# TEACHER SAYS:
# Manual build → tag → push is fine for learning. In production, it's automated.
# Every time you push code to main:
#   1. GitHub Actions triggers automatically
#   2. It builds the Docker image
#   3. It runs tests
#   4. It pushes to Docker Hub / ECR
#   5. It deploys to your server (optional)
# You never manually build or push again.

# Create the GitHub Actions workflow file:
mkdir -p .github/workflows

cat > .github/workflows/docker-ci.yml << 'CI_EOF'
name: Build, Test & Push Docker Image

on:
  push:
    branches: [main]
    tags: ['v*']            # Also trigger on version tags like v1.2.0
  pull_request:
    branches: [main]        # Test PRs but don't push

env:
  REGISTRY: docker.io       # docker.io for Docker Hub; use ECR URI for AWS
  IMAGE_NAME: ${{ github.repository_owner }}/churn-api

jobs:
  build-and-push:
    name: Build & Push
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python (for training the model before build)
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Train and save model
        run: |
          pip install -r requirements.txt
          python src/train_and_save.py

      - name: Run API tests
        run: |
          pip install pytest httpx
          # pytest tests/ -v --tb=short   ← uncomment when you have tests

      - name: Set up Docker Buildx (advanced build features)
        uses: docker/setup-buildx-action@v3

      - name: Login to Docker Hub
        if: github.event_name != 'pull_request'   # Don't push on PRs
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKERHUB_USERNAME }}
          password: ${{ secrets.DOCKERHUB_TOKEN }}
        # Add DOCKERHUB_USERNAME and DOCKERHUB_TOKEN in:
        # GitHub → Settings → Secrets and variables → Actions

      - name: Extract metadata (tags & labels)
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: ${{ env.IMAGE_NAME }}
          tags: |
            type=ref,event=branch
            type=semver,pattern={{version}}
            type=sha,prefix=sha-,length=7

      - name: Build and push Docker image
        uses: docker/build-push-action@v5
        with:
          context: .
          push: ${{ github.event_name != 'pull_request' }}
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=gha        # GitHub Actions cache (faster builds!)
          cache-to: type=gha,mode=max

      - name: Notify on success
        if: success()
        run: echo "Image pushed: ${{ steps.meta.outputs.tags }}"
CI_EOF

# ==============================================================================
# LESSON 07 — RESOURCE LIMITS FOR ML TRAINING CONTAINERS
# ==============================================================================
#
# TEACHER SAYS:
# ML training can consume ALL available resources and starve other containers.
# Docker lets you set LIMITS: max RAM, max CPU, GPU access.
# In production (Kubernetes/ECS) these limits are REQUIRED — scheduler needs to know.

# Limit memory and CPU:
docker run \
    --memory="4g" \
    --memory-swap="4g" \         # Same as memory → no swap (swap = slow)
    --cpus="2" \                 # Use at most 2 CPU cores
    --name training_job \
    churn-api:v1

# Check resource usage of running containers:
docker stats                     # Live stats for all running containers
docker stats churn-api           # Stats for one container
docker stats --no-stream         # One snapshot (don't follow)

# GPU access (NVIDIA — requires nvidia-container-toolkit):
# Install: https://docs.nvidia.com/datacenter/cloud-native/container-toolkit/install-guide.html
docker run \
    --gpus all \                 # Give ALL GPUs to this container
    --memory="32g" \
    --name gpu_training \
    pytorch/pytorch:2.1.2-cuda12.1-cudnn8-runtime \
    python train_gpu.py

# Give specific GPU (multi-GPU server, run on GPU #1 only):
docker run \
    --gpus '"device=1"' \
    pytorch/pytorch:2.1.2-cuda12.1-cudnn8-runtime \
    python train_gpu.py

# Verify GPU is visible inside the container:
docker run --gpus all pytorch/pytorch:latest nvidia-smi


# ==============================================================================
# LESSON 08 — ENTRYPOINT vs CMD (The Distinction That Trips Everyone Up)
# ==============================================================================
#
# TEACHER SAYS:
# Both ENTRYPOINT and CMD specify what runs when the container starts.
# The difference:
#   CMD     → the default command. Can be completely OVERRIDDEN at runtime.
#   ENTRYPOINT → the fixed executable. Arguments are APPENDED (not replaced).
#
# PATTERN: ENTRYPOINT = the program, CMD = the default arguments
#   (Using both together is the professional standard)

# Example in Dockerfile:
cat << 'ENTRYPOINT_EOF'
# Pattern 1: CMD only (most common for web apps)
CMD ["uvicorn", "src.app:app", "--host", "0.0.0.0", "--port", "8000"]
# docker run churn-api                     → runs uvicorn
# docker run churn-api bash                → overrides CMD, runs bash instead

# Pattern 2: ENTRYPOINT + CMD (for CLIs and tools)
ENTRYPOINT ["python", "train.py"]
CMD ["--epochs", "10", "--lr", "0.001"]   # Default args
# docker run training-image                → python train.py --epochs 10 --lr 0.001
# docker run training-image --epochs 50   → python train.py --epochs 50 (CMD overridden)

# Pattern 3: Shell script as ENTRYPOINT (for startup logic)
COPY entrypoint.sh /entrypoint.sh
RUN chmod +x /entrypoint.sh
ENTRYPOINT ["/entrypoint.sh"]
CMD ["uvicorn", "src.app:app", "--host", "0.0.0.0", "--port", "8000"]
ENTRYPOINT_EOF

# Example entrypoint.sh (runs setup THEN the CMD):
cat > entrypoint.sh << 'EP_EOF'
#!/bin/bash
set -e

echo "Starting container..."
echo "Python: $(python --version)"
echo "Model path: $MODEL_PATH"

# Wait for database to be ready (useful when depends_on isn't enough):
if [ -n "$DATABASE_URL" ]; then
    echo "Waiting for database..."
    while ! python3 -c "
import psycopg2, os, sys
try:
    psycopg2.connect(os.environ['DATABASE_URL'])
    sys.exit(0)
except: sys.exit(1)
" 2>/dev/null; do
        sleep 1
    done
    echo "Database ready!"
fi

# Run database migrations if this is a web app (Django):
# python manage.py migrate --noinput

exec "$@"   # Run the CMD passed to entrypoint (e.g., uvicorn ...)
EP_EOF
