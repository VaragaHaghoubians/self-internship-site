# ==============================================================================
# DAY 01 — WHAT IS DOCKER? WHY DOES IT MATTER?
# ==============================================================================
#
# TEACHER SAYS:
# "It works on my machine!" — the most famous problem in software development.
# Docker SOLVES this by packaging your app + ALL its dependencies into one unit.
#
# KEY CONCEPTS:
#   IMAGE     → a snapshot/blueprint (like a class in OOP)
#               Built from a Dockerfile. Stored in Docker Hub (like GitHub but for images).
#   CONTAINER → a running instance of an image (like an object created from a class)
#               Isolated, lightweight, runs the same on ANY machine.
#   DOCKERFILE → a text file with instructions to build an image
#   DOCKER HUB → public registry of images: hub.docker.com
#   VOLUME     → persistent storage attached to a container
#   NETWORK    → communication between containers
#
# ANALOGY:
#   Dockerfile  = recipe
#   Image       = meal photo (frozen, reusable)
#   Container   = actual meal on your plate (running, eating it = using it)
#
# WHY DATA SCIENTISTS NEED DOCKER:
#   - Share Jupyter notebooks with EXACT Python version and packages
#   - Deploy ML models as APIs that work anywhere
#   - Run databases (PostgreSQL, MySQL) locally without installing them
#   - Reproduce experiments exactly
#
# FIRST COMMANDS:

docker --version                            # Check Docker is installed
docker run hello-world                      # Pull and run your first container!
docker run -it ubuntu bash                  # Run Ubuntu Linux interactively!
# -i = interactive, -t = terminal, ubuntu = image name, bash = command to run
# Inside the container: ls, pwd, apt install python3, etc.
# Type 'exit' to leave the container


# ==============================================================================
# YOUR TURN — DAY 01:
# 1. Install Docker Desktop
# 2. Run: docker run hello-world   — read the output carefully
# 3. Run: docker run -it ubuntu bash
#    - Inside: run 'ls', 'pwd', 'whoami', 'cat /etc/os-release'
#    - Type 'exit' to leave
# 4. Run: docker ps       → see running containers
# 5. Run: docker ps -a    → see ALL containers (including stopped ones)
# ==============================================================================


# ==============================================================================
# DAY 02 — DOCKER IMAGES & THE DOCKER HUB
# ==============================================================================
#
# TEACHER SAYS:
# Docker Hub has thousands of pre-built images — you don't build everything from scratch.
# Popular data science images: python, jupyter/scipy-notebook, postgres, mysql, redis
#
# IMAGE NAMING: name:tag
#   python:3.11         → Python 3.11
#   python:3.11-slim    → Python 3.11, minimal install (smaller)
#   python:latest       → latest version (avoid in production — unpredictable!)
#   postgres:15         → PostgreSQL 15
#   nginx:alpine        → Nginx with Alpine Linux (very small)
#
# KEY COMMANDS:

docker pull python:3.11-slim                # Download image without running
docker images                               # List all downloaded images
docker image ls                             # Same as above

# Run Python without installing it locally!
docker run python:3.11-slim python --version
docker run -it python:3.11-slim python      # Interactive Python REPL

# Run a Jupyter Lab in Docker (your whole data science environment!):
docker run -p 8888:8888 jupyter/scipy-notebook
# -p 8888:8888 = map port 8888 on your machine to port 8888 in the container
# Open browser: http://127.0.0.1:8888

# Run PostgreSQL database:
docker run \
    --name my_postgres \
    -e POSTGRES_PASSWORD=mysecretpassword \
    -e POSTGRES_DB=university_db \
    -p 5432:5432 \
    -d \
    postgres:15
# -e = environment variable, -d = detached (run in background), --name = container name

# Connect to the PostgreSQL container:
docker exec -it my_postgres psql -U postgres -d university_db
# docker exec = run a command in a RUNNING container

# Manage containers:
docker stop my_postgres                     # Stop a running container
docker start my_postgres                    # Start a stopped container
docker restart my_postgres                  # Restart
docker rm my_postgres                       # Remove container (must be stopped first)
docker rm -f my_postgres                    # Force remove (even if running)
docker rmi python:3.11-slim                 # Remove an image

# ==============================================================================
# DAY 03 — WRITING YOUR FIRST DOCKERFILE
# ==============================================================================
#
# TEACHER SAYS:
# A Dockerfile is a series of instructions that BUILD an image layer by layer.
# Each instruction creates a new LAYER — Docker caches layers (fast rebuilds).
# Order matters: put LEAST changing instructions first (dependencies before code).
#
# KEY DOCKERFILE INSTRUCTIONS:
#   FROM image:tag      → base image to start from (always first!)
#   WORKDIR /path       → set working directory inside container
#   COPY src dst        → copy files from host INTO the image
#   RUN command         → run a shell command DURING BUILD
#   ENV KEY=VALUE       → set environment variables
#   EXPOSE port         → document which port the container uses
#   CMD ["cmd", "arg"]  → default command to run when container STARTS
#   ENTRYPOINT [...]    → similar to CMD but harder to override
#
# BUILD: docker build -t image-name:tag .
# RUN:   docker run image-name
#
# EXAMPLE — Dockerize a Python Data Science script:
# Create a file named 'Dockerfile' (no extension):

cat > Dockerfile.python_example << 'EOF'
# Start from official Python image (slim = smaller size):
FROM python:3.11-slim

# Set working directory inside the container:
WORKDIR /app

# Copy requirements first (Docker caches this layer if unchanged):
COPY requirements.txt .

# Install Python dependencies:
RUN pip install --no-cache-dir -r requirements.txt

# Copy the rest of the application code:
COPY . .

# Environment variable (can override at runtime):
ENV DATA_PATH=/app/data

# What to run when the container starts:
CMD ["python", "analysis.py"]
EOF

# Create requirements.txt:
cat > requirements.example.txt << 'EOF'
numpy==1.26.0
pandas==2.1.0
matplotlib==3.8.0
scikit-learn==1.3.0
EOF

# Create a sample analysis script:
cat > analysis.example.py << 'EOF'
import numpy as np
import pandas as pd
print("Libraries loaded successfully!")
data = np.random.randn(100, 3)
df = pd.DataFrame(data, columns=['A', 'B', 'C'])
print(df.describe())
print("Analysis complete!")
EOF

# BUILD the image:
# docker build -t my-analysis:v1 .
# -t = tag (name:version)
# .  = build context (current directory, where Dockerfile is)

# RUN the container:
# docker run my-analysis:v1

# ==============================================================================
# DAY 04 — VOLUMES: PERSISTENT DATA
# ==============================================================================
#
# TEACHER SAYS:
# By default, when a container is REMOVED, all its data is GONE.
# VOLUMES solve this — they mount a folder from your HOST into the container.
# Changes in the volume persist even after the container is stopped/removed.
#
# USE CASES:
#   - Database data (PostgreSQL, MySQL)
#   - Input data files for analysis
#   - Output results and models
#   - Development: live code changes without rebuild
#
# TWO TYPES:
#   Named volumes   → Docker manages the storage location (for databases)
#   Bind mounts     → You specify the host folder (for development)
#
# SYNTAX: -v host_path:container_path
#         -v volume_name:container_path
#
# EXAMPLE:

# Bind mount — share your current folder with the container:
docker run \
    -v $(pwd)/data:/app/data \
    -v $(pwd)/output:/app/output \
    my-analysis:v1

# Now the container reads from ./data and writes to ./output on YOUR machine!

# Named volume for PostgreSQL (data persists across container restarts):
docker volume create postgres_data
docker run \
    --name my_postgres \
    -e POSTGRES_PASSWORD=mysecretpassword \
    -v postgres_data:/var/lib/postgresql/data \
    -p 5432:5432 \
    -d \
    postgres:15
# Even if you 'docker rm my_postgres', data is in the volume!

# Volume commands:
docker volume ls                            # List all volumes
docker volume inspect postgres_data         # Details about a volume
docker volume rm postgres_data              # Remove volume (only if not in use)

# Development workflow — live code reload without rebuild:
docker run \
    -v $(pwd):/app \
    -it \
    python:3.11-slim \
    bash
# Now changes to files on your host instantly appear inside the container!

# ==============================================================================
# DAY 05 — DOCKER COMPOSE: MULTI-CONTAINER APPS
# ==============================================================================
#
# TEACHER SAYS:
# Real apps need MULTIPLE containers: web app + database + cache + worker.
# docker compose manages them all with a single YAML file.
# docker-compose.yml defines all services, their settings, and how they connect.
#
# STRUCTURE:
#   services:    → containers
#   volumes:     → named volumes
#   networks:    → custom networks
#
# COMMANDS:
#   docker compose up            → start all services
#   docker compose up -d         → start in background (detached)
#   docker compose down          → stop and remove containers
#   docker compose down -v       → also remove volumes (careful!)
#   docker compose logs service  → see logs
#   docker compose exec service cmd → run command in running service
#   docker compose build         → rebuild images
#
# EXAMPLE — Django + PostgreSQL + Redis:

cat > docker-compose.example.yml << 'EOF'
version: '3.9'

services:

  # PostgreSQL Database
  db:
    image: postgres:15
    container_name: university_db
    environment:
      POSTGRES_DB:       university
      POSTGRES_USER:     django_user
      POSTGRES_PASSWORD: secure_password_here
    volumes:
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"     # host:container
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U django_user -d university"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis Cache (optional — for sessions and caching)
  redis:
    image: redis:7-alpine
    container_name: university_redis
    ports:
      - "6379:6379"

  # Django Web Application
  web:
    build: .                # Build from Dockerfile in current directory
    container_name: university_web
    command: python manage.py runserver 0.0.0.0:8000
    volumes:
      - .:/app              # Bind mount for live development
    ports:
      - "8000:8000"
    environment:
      DEBUG:          "True"
      DATABASE_URL:   postgres://django_user:secure_password_here@db:5432/university
      REDIS_URL:      redis://redis:6379/0
      SECRET_KEY:     your-secret-key-here
    depends_on:
      db:
        condition: service_healthy    # Wait for DB to be ready!

volumes:
  postgres_data:
EOF

# Django Dockerfile for use with docker-compose:
cat > Dockerfile.django << 'EOF'
FROM python:3.11-slim

WORKDIR /app

RUN apt-get update && apt-get install -y \
    libpq-dev gcc \
    && rm -rf /var/lib/apt/lists/*

COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY . .

EXPOSE 8000
EOF

# Django requirements.txt for PostgreSQL:
cat > requirements.django.txt << 'EOF'
django>=4.2
psycopg2-binary     # PostgreSQL adapter
django-environ      # Manage environment variables
redis
EOF

# ==============================================================================
# DAY 07 — DEPLOYING WITH DOCKER: PRODUCTION BEST PRACTICES
# ==============================================================================
#
# TEACHER SAYS:
# Development Docker ≠ Production Docker.
# In production, you need: smaller images, no debug mode, proper secrets management,
# health checks, and a proper web server (not Django's dev server!).
#
# BEST PRACTICES:
#   1. Use slim/alpine base images (smaller attack surface & faster)
#   2. Run as NON-ROOT user inside container
#   3. Multi-stage builds (build stage vs runtime stage)
#   4. Never put secrets in Dockerfile or images — use environment variables
#   5. Use .dockerignore (like .gitignore but for Docker)
#   6. Add health checks
#   7. Use Gunicorn (not Django dev server) for production
#
# EXAMPLE — Production Dockerfile with multi-stage build:

cat > Dockerfile.production << 'EOF'
# === BUILD STAGE ===
FROM python:3.11-slim AS builder

WORKDIR /app
COPY requirements.txt .
RUN pip install --user --no-cache-dir -r requirements.txt

# === RUNTIME STAGE ===
FROM python:3.11-slim AS runtime

# Create non-root user:
RUN useradd --create-home appuser
WORKDIR /home/appuser/app

# Copy installed packages from build stage:
COPY --from=builder /root/.local /home/appuser/.local

# Copy application code:
COPY --chown=appuser:appuser . .

# Switch to non-root user:
USER appuser

ENV PATH=/home/appuser/.local/bin:$PATH
ENV PYTHONUNBUFFERED=1
ENV PYTHONDONTWRITEBYTECODE=1

# Health check:
HEALTHCHECK --interval=30s --timeout=3s --retries=3 \
    CMD python -c "import urllib.request; urllib.request.urlopen('http://localhost:8000/health/')" || exit 1

EXPOSE 8000

# Use Gunicorn for production:
CMD ["gunicorn", "university_portal.wsgi:application", "--bind", "0.0.0.0:8000", "--workers", "3"]
EOF

# .dockerignore file (don't copy these into the image):
cat > .dockerignore << 'EOF'
.git
.gitignore
__pycache__
*.pyc
*.pyo
.env
.venv
venv/
*.egg-info
dist/
build/
docs/
tests/
*.log
data/raw/
*.sqlite3
EOF
