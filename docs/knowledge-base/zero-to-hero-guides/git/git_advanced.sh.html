# ==============================================================================
# LESSON 01 — GIT STASH: Temporary Save Without Committing
# ==============================================================================
#
# TEACHER SAYS:
# Scenario: You're halfway through a feature. Urgent bug report comes in.
# You can't commit unfinished code. But you can't lose your work.
# Solution: git stash → temporarily saves your work, cleans the working dir.
# Later: git stash pop → brings your work back.
#
# Think of stash as a CLIPBOARD for your uncommitted changes.
#
# COMMANDS:
#   git stash                     → stash all uncommitted changes
#   git stash push -m "message"   → stash with a description (recommended)
#   git stash list                → see all stashes
#   git stash pop                 → apply latest stash AND delete it
#   git stash apply stash@{1}     → apply specific stash but KEEP it in list
#   git stash drop stash@{0}      → delete a specific stash
#   git stash clear               → delete ALL stashes (careful!)
#
# EXAMPLE:

echo "=== LESSON 01: git stash ==="

# You're working on a feature:
# (pretend you've edited model.py halfway through)
echo "Scenario: halfway through editing model.py when urgent bug comes in"

# git stash push -m "WIP: adding dropout layer to model"
# → Your working directory is now CLEAN
# → You can switch branches safely

# git checkout main
# git checkout -b hotfix/fix-null-pointer
# ... fix the bug, commit, merge ...

# git checkout feature/dropout-layer
# git stash pop
# → Your half-finished work is BACK, exactly as you left it

cat << 'EOF'

LAB: Try this yourself:
  mkdir stash-lab && cd stash-lab && git init
  echo "original content" > feature.py && git add . && git commit -m "init"
  echo "WORK IN PROGRESS - half done" >> feature.py
  git status                  ← shows modified: feature.py
  git stash push -m "WIP: adding feature X"
  git status                  ← CLEAN! Working dir empty
  git stash list              ← shows: stash@{0}: WIP: adding feature X
  git stash pop               ← brings feature.py changes back
  git status                  ← shows modified again

Advanced: stash ONLY specific files:
  git stash push -m "only model.py" -- model.py

Stash INCLUDING untracked new files:
  git stash push -u -m "include new files"

EOF

# LESSON 02 — GIT LOG ADVANCED: Reading History Like a Pro
# ==============================================================================
#
# TEACHER SAYS:
# git log is more than just a history list. It's a SEARCH and ANALYSIS tool.
# These flags are used every day in professional teams.

echo ""
echo "=== LESSON 02: Advanced git log ==="

cat << 'EOF'

USEFUL GIT LOG FLAGS:

git log --oneline                    # Compact: one line per commit
git log --oneline --graph --all      # ASCII branch/merge diagram (VERY useful!)
git log --oneline -10                # Last 10 commits only

git log --author="Alice"             # All commits by Alice
git log --since="2026-01-01"         # Commits after a date
git log --until="2026-04-01"         # Commits before a date
git log --since="1 week ago"         # Relative dates work too!

git log --grep="fix"                 # Commits with "fix" in message
git log -- model.py                  # Commits that touched model.py

git log -p                           # Show FULL diff for each commit (verbose)
git log -p -- data_loader.py         # Show all changes to data_loader.py over time
git log --stat                       # Files changed + lines added/deleted per commit

# See commits on feature-branch but NOT on main:
git log main..feature-branch

# The MOST USEFUL combination for understanding a repo:
git log --oneline --graph --all --decorate

EXAMPLE OUTPUT:
* a1b2c3d (HEAD -> main) feat: add batch scoring endpoint
* f4e5d6a feat: resolve conflict — combine Alice + Bob tuning
|\
| * c7b8e9f (feature/bob-tuning) feat: increase epochs (Bob)
* | d0e1f2a (feature/alice-tuning) feat: tune lr and batch (Alice)
|/
* 1234567 feat: initial model configuration

EOF


# ==============================================================================
# LESSON 03 — GIT DIFF: Comparing Anything with Anything
# ==============================================================================

echo ""
echo "=== LESSON 03: git diff ==="

cat << 'EOF'

git diff                        # Unstaged changes (working dir vs last commit)
git diff --staged               # Staged changes (what will be in next commit)
git diff HEAD                   # ALL changes (staged + unstaged) vs last commit
git diff HEAD~1                 # Changes since 2 commits ago
git diff HEAD~3..HEAD           # Changes over last 3 commits

git diff main..feature-branch   # What's different between branches?
git diff a1b2c3d f4e5d6a        # Compare two specific commits by hash

git diff -- model.py            # Changes in ONE specific file only
git diff --stat                 # Summary: which files changed, how many lines

PRACTICAL USE (find what changed before a deployment):
git diff origin/main..HEAD      # What am I about to push to remote?

EOF


# ==============================================================================
# LESSON 04 — GIT BLAME: Find Who Changed What
# ==============================================================================
#
# TEACHER SAYS:
# git blame shows who last edited EACH LINE of a file, and in which commit.
# NOT for blaming people — for UNDERSTANDING context:
#   "Why was this line written this way? Who knows? Let me find the author and ask."
#   "This line changed in commit abc123 — what was the PR that introduced it?"

echo ""
echo "=== LESSON 04: git blame ==="

cat << 'EOF'

git blame filename.py           # Show all lines with author + commit + date
git blame -L 10,25 filename.py  # Only lines 10-25
git blame --since="1 month ago" filename.py  # Only recent changes

EXAMPLE OUTPUT:
a1b2c3d4 (Alice  2026-04-15 10:23:11 +0000 1) LEARNING_RATE = 0.001
f9e8d7c6 (Bob    2026-04-14 14:05:32 +0000 2) BATCH_SIZE = 64
1a2b3c4d (Carol  2026-04-10 09:11:45 +0000 3) EPOCHS = 50

Reading the output:
  a1b2c3d4     = commit hash (short)
  (Alice ...)  = author, date
  1)           = line number
  LEARNING_RATE = 0.001  = the line content

WORKFLOW: You find a weird line in the code.
  1. git blame filename.py                    → find the commit hash (e.g., a1b2c3d4)
  2. git show a1b2c3d4                        → see the full commit — what changed and why
  3. Read the commit message for context
  4. If needed, contact the author

VS Code: Right-click a line → "Git Blame" shows this inline (GitLens extension)

EOF


# ==============================================================================
# LESSON 05 — GIT BISECT: Binary Search for a Bug
# ==============================================================================
#
# TEACHER SAYS:
# Scenario: "The model was accurate last month. Now it's wrong. Which commit broke it?"
# With 200 commits between "good" and "now" — you can't test them all manually.
# git bisect does a BINARY SEARCH: it cuts the commits in half each time.
# 200 commits → only ~8 tests needed!

echo ""
echo "=== LESSON 05: git bisect — Binary Search for Bugs ==="

cat << 'EOF'

WORKFLOW:
  git bisect start                    # Start the bisect session
  git bisect bad                      # Mark CURRENT commit as bad
  git bisect good abc1234             # Mark a known GOOD commit (e.g., last month)

  → Git checks out the MIDDLE commit automatically.
  → You test: does the bug exist here?

  git bisect good                     # If bug NOT here → bug is NEWER
  git bisect bad                      # If bug IS here → bug is in first half

  → Git checks out the next midpoint. You test again.
  → Repeat until Git tells you: "abc1234 is the first bad commit"

  git bisect reset                    # IMPORTANT: always reset when done!

AUTOMATE IT (when you have a test script):
  git bisect start
  git bisect bad
  git bisect good abc1234
  git bisect run python test_accuracy.py
  # Git auto-runs your script at each step!
  # 0 = good, non-zero = bad
  git bisect reset

EXAMPLE:
  You know the model was accurate on 2026-03-01:
  git log --oneline --before="2026-03-01" -1
  # → gets hash of a good commit

  git bisect start
  git bisect bad HEAD
  git bisect good <that hash>
  # → git binary-searches and finds the exact bad commit in ~8 steps

EOF

# ==============================================================================
# LESSON 06 — GIT REFLOG: Your Safety Net
# ==============================================================================
#
# TEACHER SAYS:
# git reflog is your UNDO for undos. It records EVERY position HEAD has been at.
# Even if you deleted a branch or did git reset --hard, reflog lets you recover.
# NEVER panic after a bad git command — check reflog first!

echo ""
echo "=== LESSON 06: git reflog — Recovery Tool ==="

cat << 'EOF'

git reflog                      # See every position HEAD has moved to
git reflog --all                # Include all branches (not just HEAD)

EXAMPLE OUTPUT:
a1b2c3d (HEAD -> main) HEAD@{0}: merge feature/bob: Fast-forward
f9e8d7c HEAD@{1}: checkout: moving from feature/bob to main
c8b7a6f HEAD@{2}: commit: feat: add epoch tuning
d5e4f3a HEAD@{3}: checkout: moving from main to feature/bob
1234abc HEAD@{4}: commit (amend): fix: correct typo

RECOVERY SCENARIO — accidentally reset --hard:
  git reset --hard HEAD~3      ← OH NO, lost 3 commits!
  git reflog                   ← find the commit BEFORE the reset: HEAD@{1}
  git reset --hard HEAD@{1}   ← restore to that point — commits recovered!

RECOVERY — accidentally deleted a branch:
  git branch -D my-feature     ← deleted! Oh no!
  git reflog                   ← find the last commit on that branch
  git checkout -b my-feature <hash>  ← recreate the branch

RULE: reflog is LOCAL only — it's not pushed to GitHub.
      It only helps for local mistakes. Pushed+deleted branches need GitHub restore.

IMPORTANT: reflog entries expire after 90 days by default.
           Recover ASAP if you need it!

EOF


# ==============================================================================
# LESSON 07 — GIT REBASE: Rewrite History Cleanly
# ==============================================================================
#
# TEACHER SAYS:
# MERGE creates a "merge commit" — the graph gets forked.
# REBASE replays your commits ON TOP of another branch — history stays LINEAR.
#
# When to use:
#   ✅ rebase: your feature branch, before opening a PR (keeps history clean)
#   ✅ merge:  when accepting a PR into main (preserves full history)
#   ❌ NEVER rebase commits already pushed to shared branches!
#
#  BEFORE rebase:          AFTER rebase:
#  main: A-B-C             main: A-B-C
#  feat:   D-E             feat:       D'-E'  (replayed on top of C)
#
#  After merge(rebase):    A-B-C-D'-E'  (linear!)
#  After merge(no-rebase): A-B-C-M  (merge commit M = messy graph)

echo ""
echo "=== LESSON 07: git rebase ==="

cat << 'EOF'

BASIC REBASE (update your feature branch with latest main):
  git checkout feature/my-feature
  git fetch origin
  git rebase origin/main         # Replay feature commits on top of latest main

If conflicts arise during rebase:
  1. Resolve the conflict in the file
  2. git add <resolved-file>
  3. git rebase --continue        # Move to next commit
  (OR: git rebase --abort        # Cancel the entire rebase)

AFTER rebase, if you previously pushed the branch:
  git push --force-with-lease    # Force push (safe version — checks no one else pushed)

INTERACTIVE REBASE (git squash + reorder):
  git rebase -i HEAD~5           # Open editor for last 5 commits

  Editor shows:
  pick a1b2c3d feat: add model training loop
  pick f4e5d6a fix: correct variable name
  pick c7b8e9f fix: another typo
  pick d0e1f2a feat: add evaluation metrics
  pick e1f2a3b docs: update readme

  COMMANDS you can use (change "pick" to):
    p  pick    = keep commit as-is
    r  reword  = change commit message
    e  edit    = pause and amend
    s  squash  = merge into PREVIOUS commit (keeps all messages)
    f  fixup   = merge into previous (discards this commit's message)
    d  drop    = delete this commit entirely

  SQUASH the 2 typo fixes into the first commit:
  pick a1b2c3d feat: add model training loop
  squash f4e5d6a fix: correct variable name
  squash c7b8e9f fix: another typo
  pick d0e1f2a feat: add evaluation metrics
  f    e1f2a3b docs: update readme

  → Git squashes fixes into the feat commit, drops readme commit.
  → Result: 2 clean commits instead of 5 messy ones.
  → MUCH better for code review!

GOLDEN RULE: Only rebase LOCAL / unpushed commits.
             Rebasing pushed commits = rewriting shared history = team conflict!

EOF

# ==============================================================================
# LESSON 08 — GIT CHERRY-PICK: Apply One Commit from Another Branch
# ==============================================================================

echo ""
echo "=== LESSON 08: git cherry-pick ==="

cat << 'EOF'

Scenario: You committed a critical bug fix on a feature branch.
          Main is BROKEN and needs ONLY that fix — not the whole feature.

git cherry-pick <commit-hash>   # Apply that ONE commit to current branch

Example:
  git log feature/new-model --oneline
  a1b2c3d feat: add attention mechanism
  f4e5d6a fix: critical null pointer in scorer   ← we only want THIS
  c7b8e9f feat: experimental layer

  git checkout main
  git cherry-pick f4e5d6a       # Apply ONLY the bug fix commit
  git push

Cherry-pick creates a NEW commit with the SAME changes (different hash).
The original commit stays on the feature branch.

Conflicts can occur during cherry-pick — resolve them the same way as merge conflicts:
  git add <resolved-file>
  git cherry-pick --continue
  (or: git cherry-pick --abort)

EOF


# ==============================================================================
# LESSON 09 — GIT TAG: Versioning Releases
# ==============================================================================

echo ""
echo "=== LESSON 09: git tag — Versioning Releases ==="

cat << 'EOF'

Tags mark SPECIFIC commits as important versions (releases).
Think: "git log tells you commit history; tags tell you which commits are releases."

git tag                         # List all tags
git tag v1.0.0                  # Create lightweight tag on current commit
git tag -a v1.0.0 -m "First stable release"  # Annotated tag (preferred — has metadata)
git tag -a v1.0.0 abc1234 -m "Tagging v1.0"  # Tag a specific past commit

git push origin v1.0.0          # Push a specific tag to GitHub
git push origin --tags          # Push ALL tags to GitHub

git checkout v1.0.0             # Check out code at that exact version (detached HEAD)

git tag -d v1.0.0               # Delete local tag
git push origin --delete v1.0.0 # Delete remote tag

SEMANTIC VERSIONING (the standard: MAJOR.MINOR.PATCH):
  v1.0.0 → initial release
  v1.1.0 → new feature added (backward compatible)
  v1.1.1 → bug fix
  v2.0.0 → breaking changes (API changed)

ML USAGE:
  Tag model releases:
    v1.0.0  → first production model
    v1.1.0  → retrained with more data
    v2.0.0  → completely new architecture
  Then: git checkout v1.0.0 → reproduce ANY previous model exactly.

EOF


# ==============================================================================
# LESSON 10 — GIT FLOW: Professional Branching Strategy
# ==============================================================================

echo ""
echo "=== LESSON 10: Git Flow (Team Branching Strategy) ==="

cat << 'EOF'

SIMPLE GITHUB FLOW (small teams, continuous deployment):
  main ← always deployable
  feature/X → PR → main

GIT FLOW (larger teams, release cycles):
  main     → production-ready only (tagged releases)
  develop  → integration branch (all features merged here first)
  feature/* → one branch per feature, merged into develop
  release/* → prepare a release (bugfixes only), merged to main + develop
  hotfix/*  → emergency fixes to main (bypass develop)

WORKFLOW:
  1. git checkout -b feature/add-churn-model develop
  2. Work → commit → push
  3. PR: feature/add-churn-model → develop
  4. When ready for release:
     git checkout -b release/v2.0.0 develop
  5. Bug fixes on release branch
  6. Merge release/v2.0.0 → main AND → develop
  7. Tag main: git tag -a v2.0.0 -m "Release v2.0.0"

WHICH TO USE?
  GitHub Flow → startups, SaaS, continuous deploy → SIMPLER, RECOMMENDED
  Git Flow    → enterprise, release cycles, multiple versions in prod

For your self-internship projects: use GitHub Flow.
For a company like Amazon/Uber: likely some variant of Git Flow.

EOF

cat << 'EOF'

STASH
  git stash push -m "msg"          Save work temporarily
  git stash list                   See all stashes
  git stash pop                    Restore latest stash (delete it)
  git stash apply stash@{1}        Restore specific stash (keep it)
  git stash drop stash@{0}         Delete a stash

LOG
  git log --oneline --graph --all  Full visual history
  git log --author="Name"          Filter by author
  git log --since="1 week ago"     Filter by date
  git log -- filename.py           Commits touching a file
  git log -p -- filename.py        Full diffs for a file

DIFF
  git diff main..feature           What's different between branches
  git diff HEAD~3..HEAD            Changes over last 3 commits
  git diff --stat                  Summary of what changed

BLAME / INVESTIGATE
  git blame filename.py            Who wrote each line?
  git blame -L 10,25 filename.py   Specific lines only
  git show <hash>                  Full details of a commit
  git bisect start/bad/good        Binary search for bug

RECOVERY
  git reflog                       See all HEAD positions
  git reset --hard HEAD@{2}        Jump back to earlier position
  git cherry-pick <hash>           Apply one commit to current branch

REBASE / SQUASH
  git rebase origin/main           Update feature branch with latest main
  git rebase -i HEAD~5             Interactive: squash/reorder commits
  git push --force-with-lease      Safe force push after rebase

TAGS
  git tag -a v1.0.0 -m "Release"  Create annotated tag
  git push origin --tags           Push all tags
  git checkout v1.0.0              Check out a specific version

EOF
