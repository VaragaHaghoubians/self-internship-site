#!/bin/bash
# ==============================================================================
#  GIT — CONFLICT RESOLUTION (Step-by-Step)
#  FILE: git_conflict_resolution.sh
#
#  TASK: Learn to resolve merge conflicts confidently.
#        This is the #1 Git skill that beginners fear and professionals master.
#
#  WHY:  Every team eventually has two people editing the same file.
#        Git can't decide which version wins — that's YOUR job.
#        Knowing how to resolve conflicts is the difference between
#        "I broke the repo" and "I fixed it in 2 minutes."
#
#  PREREQ: git_commands.sh (you know add, commit, branch, merge)
#
#  HOW TO USE:
#    Run each command block in your terminal.
#    READ all comments before running — understand WHAT you're doing and WHY.
# ==============================================================================


# ==============================================================================
# LESSON 01 — WHAT IS A MERGE CONFLICT AND WHY DOES IT HAPPEN?
# ==============================================================================
#
# TEACHER SAYS:
# A MERGE CONFLICT happens when:
#   1. Branch A edits line 10 of file.py
#   2. Branch B ALSO edits line 10 of file.py
#   3. You try to merge them → Git doesn't know which version to keep
#
# Git marks the conflict INSIDE the file with special markers:
#
#   <<<<<<< HEAD               ← start of YOUR version (current branch)
#   your code here
#   =======                    ← separator
#   their code here
#   >>>>>>> feature-branch     ← start of THEIR version (incoming branch)
#
# Your job: edit the file to keep the RIGHT version, remove the markers.
#
# THREE OUTCOMES when you see a conflict:
#   1. Keep YOUR version (delete their block)
#   2. Keep THEIR version (delete your block)
#   3. Keep BOTH (combine them intelligently)
#   4. Write something completely new
#


# ==============================================================================
# LESSON 02 — CREATE A CONFLICT FROM SCRATCH (Lab Setup)
# ==============================================================================
#
# We'll create a controlled conflict so you can practice resolving it.
# Follow each step exactly.

echo "=== LESSON 02: Creating a Lab Environment ==="

# 1. Create a fresh repo:
mkdir git-conflict-lab
cd git-conflict-lab
git init
git config user.name  "Student"
git config user.email "student@example.com"

# 2. Create the initial shared file (both branches will start from this):
cat > model_config.py << 'EOF'
# Model Configuration

LEARNING_RATE = 0.01
BATCH_SIZE = 32
EPOCHS = 10
MODEL_NAME = "churn_v1"

def get_config():
    return {
        "lr":    LEARNING_RATE,
        "batch": BATCH_SIZE,
        "epochs": EPOCHS
    }
EOF

git add model_config.py
git commit -m "feat: initial model configuration"

echo "✓ Initial commit created"
git log --oneline


# ==============================================================================
# LESSON 03 — CREATING THE CONFLICT
# ==============================================================================

echo ""
echo "=== LESSON 03: Creating the Conflict ==="

# 3. Create branch-A (Alice's changes):
git checkout -b feature/alice-tuning

# Alice changes learning rate and batch size:
cat > model_config.py << 'EOF'
# Model Configuration — Alice's tuning

LEARNING_RATE = 0.001          # Alice: reduced for stability
BATCH_SIZE = 64                # Alice: increased batch size
EPOCHS = 10
MODEL_NAME = "churn_v1"

def get_config():
    return {
        "lr":    LEARNING_RATE,
        "batch": BATCH_SIZE,
        "epochs": EPOCHS
    }
EOF

git add model_config.py
git commit -m "feat: tune learning rate and batch size (Alice)"

echo "✓ Alice's branch committed"

# 4. Go back to main, create branch-B (Bob's changes):
git checkout main
git checkout -b feature/bob-tuning

# Bob ALSO changes learning rate (different value!) and epochs:
cat > model_config.py << 'EOF'
# Model Configuration — Bob's tuning

LEARNING_RATE = 0.005          # Bob: tried a different rate
BATCH_SIZE = 32
EPOCHS = 50                    # Bob: more epochs for better convergence
MODEL_NAME = "churn_v1"

def get_config():
    return {
        "lr":    LEARNING_RATE,
        "batch": BATCH_SIZE,
        "epochs": EPOCHS
    }
EOF

git add model_config.py
git commit -m "feat: increase epochs and adjust learning rate (Bob)"

echo "✓ Bob's branch committed"

# 5. Merge Alice's branch first (no conflict — main was clean):
git checkout main
git merge feature/alice-tuning
echo "✓ Alice's branch merged into main (no conflict)"

# 6. Now try to merge Bob's branch — CONFLICT!
git merge feature/bob-tuning
echo "→ Expected: CONFLICT! Git cannot auto-merge model_config.py"


# ==============================================================================
# LESSON 04 — READING THE CONFLICT MARKERS
# ==============================================================================

echo ""
echo "=== LESSON 04: Reading the Conflict File ==="
echo ""
echo "The conflict looks like this in model_config.py:"
echo ""
cat << 'EOF'
# Model Configuration

<<<<<<< HEAD
# Model Configuration — Alice's tuning

LEARNING_RATE = 0.001          # Alice: reduced for stability
BATCH_SIZE = 64                # Alice: increased batch size
EPOCHS = 10
=======
# Model Configuration — Bob's tuning

LEARNING_RATE = 0.005          # Bob: tried a different rate
BATCH_SIZE = 32
EPOCHS = 50                    # Bob: more epochs for better convergence
>>>>>>> feature/bob-tuning
MODEL_NAME = "churn_v1"

def get_config():
    return {
        "lr":    LEARNING_RATE,
        "batch": BATCH_SIZE,
        "epochs": EPOCHS
    }
EOF

echo ""
echo "Git status shows:"
git status

echo ""
echo "EXPLAIN THE MARKERS:"
echo "  <<<<<<< HEAD            = Start of YOUR version (Alice's — already in main)"
echo "  =======                 = Separator between the two versions"
echo "  >>>>>>> feature/bob-tuning = Start of THEIR version (Bob's — incoming)"


# ==============================================================================
# LESSON 05 — RESOLVING THE CONFLICT
# ==============================================================================

echo ""
echo "=== LESSON 05: Resolving the Conflict ==="
echo ""
echo "STEP 1: Open the file and decide what to keep."
echo "        Option A: keep Alice's values (0.001 lr, 64 batch)"
echo "        Option B: keep Bob's values  (0.005 lr, 50 epochs)"
echo "        Option C: COMBINE BOTH (best! take the best from each person)"
echo ""
echo "In a real team, you'd TALK to Alice and Bob about which values to use."
echo "For this exercise, we combine: Alice's lr + Bob's epochs + Alice's batch."
echo ""
echo "STEP 2: Edit the file to the FINAL version (remove all conflict markers):"

# Write the RESOLVED version (no conflict markers):
cat > model_config.py << 'EOF'
# Model Configuration — Combined tuning (Alice + Bob)

LEARNING_RATE = 0.001          # Alice: stable low rate (accepted)
BATCH_SIZE = 64                # Alice: larger batch (accepted)
EPOCHS = 50                    # Bob: more epochs (accepted)
MODEL_NAME = "churn_v1"

def get_config():
    return {
        "lr":    LEARNING_RATE,
        "batch": BATCH_SIZE,
        "epochs": EPOCHS
    }
EOF

echo "✓ Conflict resolved: combined best of both contributions"
echo ""
cat model_config.py

# STEP 3: Stage the resolved file:
git add model_config.py
echo ""
echo "STEP 3: Stage the resolved file: git add model_config.py"

# STEP 4: Complete the merge commit:
git commit -m "merge: resolve conflict — combine Alice's lr/batch + Bob's epochs"
echo ""
echo "STEP 4: Complete merge with commit"

git log --oneline --graph --all


# ==============================================================================
# LESSON 06 — GIT STATUS DURING A CONFLICT
# ==============================================================================

echo ""
echo "=== LESSON 06: What git status shows during a conflict ==="
echo ""
echo "During a conflict, 'git status' shows:"
echo ""
echo "  On branch main"
echo "  You have unmerged paths."
echo "    (fix conflicts and run 'git commit')"
echo ""
echo "  Unmerged paths:"
echo "    (use 'git add <file>...' to mark resolution)"
echo "          both modified:   model_config.py"
echo ""
echo "KEY RULE: Until you git add the resolved file, the merge is NOT complete."
echo "  git add <file>   →  marks conflict as resolved"
echo "  git commit       →  finalizes the merge"


# ==============================================================================
# LESSON 07 — ABORTING A MERGE CONFLICT
# ==============================================================================

echo ""
echo "=== LESSON 07: How to ABORT a merge ==="
echo ""
echo "Sometimes you're mid-conflict and realize:"
echo "  'I need to go back! I'll merge this later.'"
echo ""
echo "Safe escape hatch:"
echo ""
echo "  git merge --abort"
echo ""
echo "This CANCELS the merge entirely and returns your branch to its state"
echo "BEFORE you ran git merge. No damage. All conflict markers removed."
echo ""
echo "After aborting: your repo is clean, both branches still exist separately."
echo "You can re-attempt the merge anytime."


# ==============================================================================
# LESSON 08 — USING VS CODE TO RESOLVE CONFLICTS (Modern Way)
# ==============================================================================

echo ""
echo "=== LESSON 08: VS Code Conflict UI ==="
echo ""
cat << 'EOF'
VS Code has a built-in conflict editor that shows 4 buttons above each conflict:

  [ Accept Current Change ]   → keep YOUR version (HEAD)
  [ Accept Incoming Change ]  → keep THEIR version (feature branch)
  [ Accept Both Changes ]     → keep BOTH (concatenated)
  [ Compare Changes ]         → see a side-by-side diff

HOW TO USE:
  1. When a conflict occurs, open the file in VS Code.
  2. You'll see colored highlights:
       GREEN  = current branch (HEAD) version
       BLUE   = incoming branch version
  3. Click the button that matches your decision.
  4. Save the file.
  5. git add <file>
  6. git commit

This is FAR faster than editing conflict markers by hand.
Most developers use this approach for all but the simplest conflicts.
EOF


# ==============================================================================
# LESSON 09 — MULTIPLE FILES IN CONFLICT
# ==============================================================================

echo ""
echo "=== LESSON 09: Handling Multiple Conflicted Files ==="
echo ""
echo "When multiple files conflict:"
echo ""
echo "  git status               ← lists ALL conflicted files"
echo "  git diff                 ← shows ALL conflict markers across all files"
echo ""
echo "Resolve ONE file at a time:"
echo "  1. Open the first file"
echo "  2. Resolve its conflicts"
echo "  3. git add that-file"
echo "  4. Move to the next file"
echo "  5. Repeat until all files are staged"
echo "  6. git commit (finalizes the merge)"
echo ""
echo "TIP: use git mergetool to open a visual 3-way merge editor:"
echo "  git config --global merge.tool vscode"
echo "  git config --global mergetool.vscode.cmd 'code --wait \$MERGED'"
echo "  git mergetool"


# ==============================================================================
# LESSON 10 — PREVENTING CONFLICTS (BEST PRACTICES)
# ==============================================================================

echo ""
echo "=== LESSON 10: How to Prevent Conflicts ==="
echo ""
cat << 'EOF'
BEST PRACTICES to minimize merge conflicts:

1. PULL OFTEN:
   git pull origin main   ← Do this every morning before you start coding.
   This keeps your local branch close to main — fewer divergences.

2. SMALL, FOCUSED COMMITS:
   Don't work for 5 days without committing.
   Commit small, meaningful chunks: one feature, one fix.
   Smaller commits = less divergence = fewer conflicts.

3. SHORT-LIVED FEATURE BRANCHES:
   Merge feature branches into main within 1-3 days.
   Long-lived branches = more drift = more conflicts.

4. COMMUNICATE WITH YOUR TEAM:
   "Hey, I'm refactoring data_loader.py today."
   If two people know they're touching the same file, they coordinate.

5. USE A CONSISTENT STYLE (via tools):
   Black (Python), Prettier (JS), Google Java Format...
   If everyone's code is formatted the same, no formatting conflicts.

6. DIVIDE FILES BY RESPONSIBILITY:
   Alice works on model.py, Bob works on data.py.
   Agree on file ownership to minimize overlap.

7. REBASE BEFORE MERGING:
   git rebase origin/main   ← replay YOUR commits on top of latest main.
   Resolves conflicts one at a time, in order — easier to manage.
   (Covered in git_advanced.sh)
EOF


# ==============================================================================
# 🎯 YOUR TASKS
# ==============================================================================

echo ""
echo "=================================================================="
echo "YOUR TURN — Conflict Resolution Practice"
echo "=================================================================="
cat << 'EOF'

TASK 1 — REPRODUCE THE LAB:
    Re-create the conflict lab from LESSON 02-05 from memory.
    (Don't look at the script — try to remember the steps!)
    Verify you can:
    a) Create a conflict
    b) See the conflict markers in the file
    c) Resolve it and complete the merge

TASK 2 — MULTI-FILE CONFLICT:
    From your conflict lab repo:
    a) Create branch 'feature/alice-v2'
    b) Modify BOTH model_config.py AND create requirements.txt:
         numpy==1.26.0
         pandas==2.1.0
    c) Create branch 'feature/bob-v2' from main
    d) Bob ALSO modifies model_config.py (different values) AND creates requirements.txt:
         numpy==2.0.0
         scikit-learn==1.4.0
    e) Merge feature/alice-v2 into main (no conflict)
    f) Merge feature/bob-v2 — both files conflict!
    g) Resolve BOTH files and complete the merge.
    Hints:
    - For requirements.txt: keep BOTH libraries (combine the lists)
    - For model_config.py: choose values collaboratively (document WHY)

TASK 3 — ABORT AND RETRY:
    a) Start merging a conflicting branch
    b) ABORT the merge with: git merge --abort
    c) Verify git status is clean
    d) Re-attempt the merge and resolve it properly

TASK 4 — VS CODE CONFLICT UI:
    a) Create a small conflict (modify the same line in two branches)
    b) Open the conflicted file in VS Code
    c) Use the "Accept Current Change" / "Accept Incoming Change" buttons
    d) Complete the merge
    Which button did you use and why?

TASK 5 — REAL WORLD (SIMULATE TEAM):
    a) Create a file: pipeline.py with 20 lines of fake ML pipeline code
    b) Create branch 'alice/feature-preprocessing'
    c) Alice modifies lines 5-8 (data cleaning section)
    d) Create branch 'bob/feature-scaling' from main
    e) Bob modifies lines 5-8 (same section — different approach)
    f) Merge Alice → main, then Bob → main
    g) RESOLVE: write a combined version that uses Alice's cleaning
       AND Bob's scaling. Document in a comment WHY you made this decision.

Write your experience and any lessons learned in a comment below:
EOF


echo ""
echo "git_conflict_resolution.sh complete!"
echo "Next: git_advanced.sh (rebase, squash, stash, blame, bisect)"
