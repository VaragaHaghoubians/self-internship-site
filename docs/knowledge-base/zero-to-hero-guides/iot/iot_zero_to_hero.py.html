# ==============================================================================
# DAY 03 — READING SENSORS (DHT11/DHT22 Temperature & Humidity)
# ==============================================================================
#
# TEACHER SAYS:
# DHT11 = cheap, slower, less accurate (±2°C, ±5%RH, reads every 1s)
# DHT22 = better, faster, more accurate (±0.5°C, ±2%RH, reads every 0.5s)
# Both are digital sensors — they do the analog-to-digital conversion internally.
# Library: pip install adafruit-circuitpython-dht (newer) or Adafruit_DHT (older)
#
# WIRING (3.3V logic):
#   Pin 1 → 3.3V
#   Pin 2 → GPIO 4 (data) + 10kΩ pull-up resistor to 3.3V
#   Pin 3 → Not connected
#   Pin 4 → GND
#
# ==============================================================================

import random

class DHTSimulator:
    """Simulate a DHT11 sensor for testing without hardware."""

    def __init__(self, base_temp=22.0, base_humidity=55.0):
        self.base_temp     = base_temp
        self.base_humidity = base_humidity

    def read(self):
        """Simulate a sensor reading with slight variation."""
        temperature = round(self.base_temp + random.uniform(-1.5, 1.5), 1)
        humidity    = round(self.base_humidity + random.uniform(-3.0, 3.0), 1)
        return temperature, humidity


def read_dht_sensor(pin=4, sensor_type='DHT22', simulator=None):
    """
    Read temperature and humidity from DHT sensor.
    Falls back to simulator if hardware not available.

    Returns:
        (temperature_celsius, humidity_percent) or (None, None) on error
    """
    if GPIO_AVAILABLE:
        try:
            import adafruit_dht
            import board

            if sensor_type == 'DHT22':
                dht = adafruit_dht.DHT22(board.D4)
            else:
                dht = adafruit_dht.DHT11(board.D4)

            temperature = dht.temperature
            humidity    = dht.humidity
            return temperature, humidity

        except RuntimeError as e:
            print(f"Sensor read error (try again): {e}")
            return None, None
    else:
        # Use simulator
        if simulator is None:
            simulator = DHTSimulator()
        return simulator.read()


def monitor_climate(readings=10, interval_seconds=2):
    """
    Continuously read the DHT sensor and display readings.
    Simulates a real IoT monitoring loop.
    """
    simulator = DHTSimulator(base_temp=23.0, base_humidity=60.0)
    data = []

    print(f"\n{'Time':^8} | {'Temp (°C)':^10} | {'Humidity (%)':^14} | Status")
    print("-" * 50)

    for i in range(readings):
        temperature, humidity = read_dht_sensor(simulator=simulator)

        if temperature is not None:
            timestamp = time.strftime("%H:%M:%S")

            # Determine status
            if temperature > 30:
                status = "TOO HOT!"
            elif temperature < 15:
                status = "TOO COLD!"
            elif humidity > 80:
                status = "TOO HUMID!"
            else:
                status = "OK"

            print(f"{timestamp:^8} | {temperature:^10.1f} | {humidity:^14.1f} | {status}")
            data.append({'time': timestamp, 'temperature': temperature, 'humidity': humidity})

        time.sleep(0.1)     # Short sleep for simulation (use real interval_seconds on Pi)

    return data

# Run simulation:
readings = monitor_climate(readings=8)
print(f"\nCollected {len(readings)} readings.")

# ==============================================================================
# DAY 04 — MQTT: THE IoT MESSAGING PROTOCOL
# ==============================================================================
#
# TEACHER SAYS:
# MQTT = Message Queuing Telemetry Transport
# It's the STANDARD protocol for IoT communication. Lightweight, reliable,
# works even on slow/unreliable connections (like 3G or WiFi with interference).
#
# MQTT CONCEPTS:
#   BROKER    → Server that routes messages (like a post office)
#               Free public broker: test.mosquitto.org
#               Self-host: Mosquitto (pip install paho-mqtt)
#   PUBLISHER → Device that SENDS data (your sensor)
#   SUBSCRIBER→ Device or app that RECEIVES data (your dashboard)
#   TOPIC     → Address for messages, like file paths:
#               "university/room101/temperature"
#               "factory/machine5/vibration"
#   QoS       → Quality of Service: 0=at most once, 1=at least once, 2=exactly once
#
# INSTALL: pip install paho-mqtt

import json
import threading

try:
    import paho.mqtt.client as mqtt
    MQTT_AVAILABLE = True
except ImportError:
    MQTT_AVAILABLE = False
    print("paho-mqtt not installed. Run: pip install paho-mqtt")


def publish_sensor_data_example():
    """
    PUBLISHER: Reads sensor and publishes to MQTT broker.
    This would run on your Raspberry Pi.
    """
    if not MQTT_AVAILABLE:
        print("[MQTT NOT INSTALLED] Would publish sensor data here.")
        return

    BROKER   = "test.mosquitto.org"     # Free public test broker
    PORT     = 1883
    TOPIC    = "university/lab/climate"
    DEVICE_ID = "raspi-lab-001"

    simulator = DHTSimulator(base_temp=22.0, base_humidity=58.0)

    def on_connect(client, userdata, flags, rc):
        if rc == 0:
            print(f"Connected to MQTT broker: {BROKER}")
        else:
            print(f"Connection failed, code: {rc}")

    client = mqtt.Client(client_id=DEVICE_ID)
    client.on_connect = on_connect
    client.connect(BROKER, PORT, keepalive=60)
    client.loop_start()

    print(f"Publishing to topic: {TOPIC}")

    for i in range(5):
        temperature, humidity = read_dht_sensor(simulator=simulator)

        if temperature is not None:
            payload = json.dumps({
                "device_id":   DEVICE_ID,
                "timestamp":   time.strftime("%Y-%m-%dT%H:%M:%S"),
                "temperature": temperature,
                "humidity":    humidity,
            })
            result = client.publish(TOPIC, payload, qos=1)
            print(f"Published: {payload}")
        time.sleep(2)

    client.loop_stop()
    client.disconnect()


def subscribe_sensor_data_example():
    """
    SUBSCRIBER: Listens for sensor data from MQTT broker.
    This would run on your laptop/server/dashboard.
    """
    if not MQTT_AVAILABLE:
        print("[MQTT NOT INSTALLED] Would subscribe to sensor data here.")
        return

    BROKER = "test.mosquitto.org"
    TOPIC  = "university/lab/climate"

    received_data = []

    def on_connect(client, userdata, flags, rc):
        print(f"Subscriber connected, subscribing to: {TOPIC}")
        client.subscribe(TOPIC, qos=1)

    def on_message(client, userdata, msg):
        payload = json.loads(msg.payload.decode())
        print(f"\nReceived on '{msg.topic}':")
        print(f"  Temperature: {payload['temperature']}°C")
        print(f"  Humidity:    {payload['humidity']}%")
        print(f"  Time:        {payload['timestamp']}")
        received_data.append(payload)

        # Trigger action on threshold:
        if payload['temperature'] > 28:
            print("  ⚠ ALERT: Temperature too high!")

    client = mqtt.Client(client_id="dashboard-subscriber-001")
    client.on_connect = on_connect
    client.on_message = on_message
    client.connect(BROKER, 1883, keepalive=60)
    client.loop_forever()   # Runs forever, listening

import csv

# ---- CSV Logging ----

def log_to_csv(temperature: float, humidity: float, csv_file: str = "iot_data/climate.csv"):
    """Append a sensor reading to a CSV file."""
    file_path = Path(csv_file)
    file_exists = file_path.exists()

    with open(file_path, 'a', newline='', encoding='utf-8') as f:
        writer = csv.DictWriter(f, fieldnames=['timestamp', 'temperature', 'humidity'])
        if not file_exists:
            writer.writeheader()    # Write header only for new file
        writer.writerow({
            'timestamp':   datetime.now().isoformat(),
            'temperature': temperature,
            'humidity':    humidity,
        })


# ---- SQLite Logging ----

def setup_database(db_path: str = "iot_data/sensors.db") -> sqlite3.Connection:
    """Create the database and tables if they don't exist."""
    conn = sqlite3.connect(db_path)
    conn.execute("""
        CREATE TABLE IF NOT EXISTS climate_readings (
            id          INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp   TEXT    NOT NULL DEFAULT (datetime('now')),
            device_id   TEXT    NOT NULL DEFAULT 'raspi-001',
            temperature REAL    NOT NULL,
            humidity    REAL    NOT NULL
        )
    """)
    conn.execute("""
        CREATE INDEX IF NOT EXISTS idx_timestamp ON climate_readings(timestamp)
    """)
    conn.commit()
    return conn


def log_to_sqlite(conn: sqlite3.Connection, temperature: float, humidity: float,
                  device_id: str = "raspi-001"):
    """Insert a sensor reading into SQLite."""
    conn.execute(
        "INSERT INTO climate_readings (device_id, temperature, humidity) VALUES (?, ?, ?)",
        (device_id, temperature, humidity)
    )
    conn.commit()


def analyze_recent_readings(conn: sqlite3.Connection, hours: int = 24):
    """Query and analyze recent sensor data."""
    cursor = conn.execute("""
        SELECT
            COUNT(*)                AS total_readings,
            AVG(temperature)        AS avg_temp,
            MIN(temperature)        AS min_temp,
            MAX(temperature)        AS max_temp,
            AVG(humidity)           AS avg_humidity,
            SUM(CASE WHEN temperature > 28 THEN 1 ELSE 0 END) AS hot_readings
        FROM climate_readings
        WHERE timestamp >= datetime('now', ?)
    """, (f"-{hours} hours",))
    return cursor.fetchone()


# Demonstrate logging:
def run_logging_demo(num_readings: int = 20):
    """Simulate collecting and logging sensor data."""
    simulator = DHTSimulator(base_temp=23.0, base_humidity=62.0)
    conn = setup_database()

    print(f"\nLogging {num_readings} simulated readings...")
    for _ in range(num_readings):
        temp, hum = read_dht_sensor(simulator=simulator)
        if temp is not None:
            log_to_csv(temp, hum)
            log_to_sqlite(conn, temp, hum)

    # Analyze:
    stats = analyze_recent_readings(conn)
    print("\n--- Sensor Statistics (last 24 hours) ---")
    print(f"Total readings:   {stats[0]}")
    print(f"Avg temperature:  {stats[1]:.1f}°C")
    print(f"Min temperature:  {stats[2]:.1f}°C")
    print(f"Max temperature:  {stats[3]:.1f}°C")
    print(f"Avg humidity:     {stats[4]:.1f}%")
    print(f"Hot readings >28°C: {stats[5]}")

    conn.close()
    print("\nData saved to iot_data/climate.csv and iot_data/sensors.db")

FLASK_API_EXAMPLE = '''
# Save this as iot_api.py and run: python iot_api.py

from flask import Flask, jsonify, request
import sqlite3
from datetime import datetime

app = Flask(__name__)
DB_PATH = "iot_data/sensors.db"

def get_db():
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row     # Returns rows as dicts
    return conn

@app.route('/api/readings', methods=['GET'])
def get_readings():
    """Get recent sensor readings."""
    limit = request.args.get('limit', 100, type=int)
    device = request.args.get('device', None)

    query = "SELECT * FROM climate_readings"
    params = []

    if device:
        query += " WHERE device_id = ?"
        params.append(device)

    query += " ORDER BY timestamp DESC LIMIT ?"
    params.append(min(limit, 1000))    # Cap at 1000

    conn = get_db()
    rows = conn.execute(query, params).fetchall()
    conn.close()

    return jsonify([dict(row) for row in rows])


@app.route('/api/stats', methods=['GET'])
def get_stats():
    """Get summary statistics."""
    conn = get_db()
    stats = conn.execute("""
        SELECT
            COUNT(*)         AS total,
            AVG(temperature) AS avg_temp,
            MAX(temperature) AS max_temp,
            MIN(temperature) AS min_temp,
            AVG(humidity)    AS avg_humidity
        FROM climate_readings
    """).fetchone()
    conn.close()
    return jsonify(dict(stats))


@app.route('/api/readings', methods=['POST'])
def add_reading():
    """Accept a new reading from a sensor device."""
    data = request.get_json()

    required = ['temperature', 'humidity']
    if not all(k in data for k in required):
        return jsonify({'error': 'Missing required fields'}), 400

    conn = get_db()
    conn.execute(
        "INSERT INTO climate_readings (device_id, temperature, humidity) VALUES (?, ?, ?)",
        (data.get('device_id', 'unknown'), data['temperature'], data['humidity'])
    )
    conn.commit()
    conn.close()

    return jsonify({'status': 'ok', 'message': 'Reading saved'}), 201


if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000, debug=True)
    # Visit: http://localhost:5000/api/readings
    # Visit: http://localhost:5000/api/stats
'''

import matplotlib.pyplot as plt
import matplotlib.animation as animation
from collections import deque

def live_dashboard(max_points: int = 50):
    """
    Create a live-updating dashboard for sensor data.
    Uses simulated data here; replace read_dht_sensor with real hardware.
    """
    simulator    = DHTSimulator(base_temp=22.0, base_humidity=60.0)
    temperatures = deque(maxlen=max_points)
    humidities   = deque(maxlen=max_points)
    timestamps   = deque(maxlen=max_points)

    fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(10, 6))
    fig.suptitle('IoT Climate Monitor — Live', fontsize=14, fontweight='bold')

    def update(frame):
        temp, hum = read_dht_sensor(simulator=simulator)
        if temp is None:
            return

        now = datetime.now().strftime("%H:%M:%S")
        temperatures.append(temp)
        humidities.append(hum)
        timestamps.append(now)

        # Update temperature plot:
        ax1.clear()
        ax1.plot(list(temperatures), color='red', linewidth=2, label='Temperature (°C)')
        ax1.axhline(y=28, color='orange', linestyle='--', label='Warning (28°C)')
        ax1.axhline(y=15, color='blue', linestyle='--', label='Warning (15°C)')
        ax1.fill_between(range(len(temperatures)), list(temperatures),
                         alpha=0.3, color='red')
        ax1.set_ylabel('Temperature (°C)')
        ax1.set_ylim(10, 40)
        ax1.legend(loc='upper right')
        ax1.grid(True, alpha=0.3)

        # Update humidity plot:
        ax2.clear()
        ax2.plot(list(humidities), color='blue', linewidth=2, label='Humidity (%)')
        ax2.fill_between(range(len(humidities)), list(humidities),
                         alpha=0.3, color='blue')
        ax2.set_ylabel('Humidity (%)')
        ax2.set_xlabel('Reading Number')
        ax2.set_ylim(0, 100)
        ax2.legend(loc='upper right')
        ax2.grid(True, alpha=0.3)

        plt.tight_layout()

    # Update every 500ms:
    ani = animation.FuncAnimation(fig, update, interval=500, cache_frame_data=False)
    plt.show()
