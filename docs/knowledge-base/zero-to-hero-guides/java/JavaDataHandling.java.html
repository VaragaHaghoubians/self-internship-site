// ==============================================================================
//  JAVA DATA HANDLING — CSV, JSON, File I/O for Data Work
//  FILE: JavaDataHandling.java
//
//  TASK: Learn to read, write, and process data files in Java —
//        the foundation of any data pipeline or ML preprocessing step.
//
//  WHY:  Every data engineering job requires:
//          → Reading CSV exports from databases/systems
//          → Parsing JSON from REST APIs
//          → Writing processed results back to files
//          → Computing descriptive statistics
//        In Python you'd use pandas. In Java, you do it with:
//          → java.io and java.nio (built-in — no install needed)
//          → org.json or Jackson for JSON (popular libraries)
//
//  PREREQ: JavaZeroToHero.java, JavaStreamsLambda.java
//
//  NOTE:  Parts 1-4 use ONLY the standard library (no external dependencies).
//         Part 5 shows Jackson usage with the import pattern — add Jackson
//         to your Maven/Gradle project to run that section.
//
//  COMPILE: javac JavaDataHandling.java
//  RUN:     java JavaDataHandling
// ==============================================================================

import java.io.*;
import java.nio.file.*;
import java.util.*;
import java.util.stream.*;

public class JavaDataHandling {

    public static void main(String[] args) throws IOException {

        // ==============================================================================
        // LESSON 01 — WRITING FILES: Creating Data for Subsequent Lessons
        // ==============================================================================
        //
        // TEACHER SAYS:
        // java.nio.file.Files (Java 7+) is the modern, clean file API.
        // Prefer it over the old java.io.FileWriter for most tasks.
        //
        // Key classes:
        //   Path          → represents a file/directory path
        //   Paths.get()   → create a Path from string
        //   Files.write() → write bytes or lines
        //   Files.read()  → read all bytes
        //   Files.lines() → lazy stream of lines (memory-efficient for large files)
        //
        // PrintWriter    → formatted text writing (like printf to a file)
        //
        // EXAMPLE:

        System.out.println("=== LESSON 01: Writing Files ===");

        // Write a CSV file using PrintWriter (formatted):
        Path csvPath = Paths.get("employees.csv");
        try (PrintWriter pw = new PrintWriter(Files.newBufferedWriter(csvPath))) {
            pw.println("employee_id,name,department,salary,years_exp,city");
            pw.printf("%d,%s,%s,%.2f,%d,%s%n", 1,  "Alice",  "DS",      85000.00, 7, "New York");
            pw.printf("%d,%s,%s,%.2f,%d,%s%n", 2,  "Bob",    "CS",      72000.00, 3, "London");
            pw.printf("%d,%s,%s,%.2f,%d,%s%n", 3,  "Carol",  "DS",      91000.00, 10,"Paris");
            pw.printf("%d,%s,%s,%.2f,%d,%s%n", 4,  "David",  "Finance", 68000.00, 2, "Berlin");
            pw.printf("%d,%s,%s,%.2f,%d,%s%n", 5,  "Eve",    "CS",      79000.00, 5, "Tokyo");
            pw.printf("%d,%s,%s,%.2f,%d,%s%n", 6,  "Frank",  "DS",      88000.00, 8, "New York");
            pw.printf("%d,%s,%s,%.2f,%d,%s%n", 7,  "Grace",  "Finance", 75000.00, 4, "London");
            pw.printf("%d,%s,%s,%.2f,%d,%s%n", 8,  "Henry",  "CS",      81000.00, 6, "Paris");
            pw.printf("%d,%s,%s,%.2f,%d,%s%n", 9,  "Iris",   "DS",      95000.00, 12,"New York");
            pw.printf("%d,%s,%s,%.2f,%d,%s%n", 10, "Jack",   "Finance", 70000.00, 3, "Berlin");
        }
        System.out.println("Written: " + csvPath + "  (" + Files.size(csvPath) + " bytes)");

        // Write a plain text report using Files.writeString (Java 11+):
        String report = "=== Daily Report ===\nDate: 2026-04-21\nStatus: All systems OK\n";
        Files.writeString(Paths.get("report.txt"), report);
        System.out.println("Written: report.txt");


        // ==============================================================================
        // LESSON 02 — READING CSV FILES: The Core Data Skill
        // ==============================================================================
        //
        // TEACHER SAYS:
        // Java has no built-in CSV library (unlike Python's csv module or pandas).
        // You process it manually: read line → split by comma → parse each field.
        //
        // For production code, use OpenCSV or Apache Commons CSV.
        // Here we learn the manual approach first — it teaches you EXACTLY what
        // those libraries do under the hood.
        //
        // try-with-resources ensures the file is ALWAYS closed, even if an exception occurs.

        System.out.println("\n=== LESSON 02: Reading CSV Files ===");

        // Method 1: BufferedReader line by line (memory-efficient, works for huge files):
        System.out.println("--- Method 1: BufferedReader ---");
        List<Employee> employees = new ArrayList<>();

        try (BufferedReader br = Files.newBufferedReader(csvPath)) {
            String header = br.readLine();  // Skip header line
            System.out.println("Columns: " + header);

            String line;
            while ((line = br.readLine()) != null) {
                String[] fields = line.split(",");
                try {
                    Employee e = new Employee(
                            Integer.parseInt(fields[0].trim()),
                            fields[1].trim(),
                            fields[2].trim(),
                            Double.parseDouble(fields[3].trim()),
                            Integer.parseInt(fields[4].trim()),
                            fields[5].trim()
                    );
                    employees.add(e);
                } catch (NumberFormatException ex) {
                    System.out.println("Skipping malformed row: " + line);
                }
            }
        }

        System.out.println("Loaded " + employees.size() + " employees");
        employees.forEach(System.out::println);

        // Method 2: Files.lines() → Stream — ideal for Streams pipeline:
        System.out.println("\n--- Method 2: Files.lines() + Stream ---");
        long dsCount = Files.lines(csvPath)
                .skip(1)                                  // Skip header
                .filter(l -> l.split(",")[2].equals("DS"))
                .count();
        System.out.println("DS employees: " + dsCount);

        // ==============================================================================
        // YOUR TURN — LESSON 02:
        // 1. Read employees.csv and print only employees from New York.
        // 2. Read employees.csv and find the employee with the highest salary.
        //    HINT: Files.lines().skip(1).map(l -> parse).max(Comparator...).get()
        // 3. Count how many employees are in each department.
        //    HINT: Files.lines().skip(1).collect(Collectors.groupingBy(dept, counting()))
        // ==============================================================================


        // ==============================================================================
        // LESSON 03 — DATA PROCESSING & STATISTICS (Standard Library Only)
        // ==============================================================================
        //
        // TEACHER SAYS:
        // Java doesn't have numpy/pandas. For the Data Analyst / Data Scientist role,
        // you implement statistical computations manually — OR use Apache Commons Math.
        // Here we implement from scratch so you understand what the library does.

        System.out.println("\n=== LESSON 03: Statistics from Scratch ===");

        double[] salaries = employees.stream()
                .mapToDouble(Employee::getSalary)
                .toArray();

        System.out.printf("Count:    %d%n", salaries.length);
        System.out.printf("Sum:      $%,.0f%n", StatUtils.sum(salaries));
        System.out.printf("Mean:     $%,.0f%n", StatUtils.mean(salaries));
        System.out.printf("Median:   $%,.0f%n", StatUtils.median(salaries));
        System.out.printf("Std Dev:  $%,.0f%n", StatUtils.stdDev(salaries));
        System.out.printf("Min:      $%,.0f%n", StatUtils.min(salaries));
        System.out.printf("Max:      $%,.0f%n", StatUtils.max(salaries));

        // Percentiles:
        System.out.printf("25th pct: $%,.0f%n", StatUtils.percentile(salaries, 25));
        System.out.printf("75th pct: $%,.0f%n", StatUtils.percentile(salaries, 75));

        // Group by department → average salary:
        System.out.println("\nAverage salary by department:");
        employees.stream()
                .collect(Collectors.groupingBy(
                        Employee::getDepartment,
                        Collectors.averagingDouble(Employee::getSalary)
                ))
                .entrySet().stream()
                .sorted(Map.Entry.<String, Double>comparingByValue().reversed())
                .forEach(e -> System.out.printf("  %-10s → $%,.0f%n", e.getKey(), e.getValue()));

        // Correlation between salary and years_exp:
        double[] yearsExp = employees.stream().mapToDouble(Employee::getYearsExp).toArray();
        double   r        = StatUtils.pearsonCorrelation(salaries, yearsExp);
        System.out.printf("%nPearson correlation (salary vs years_exp): %.3f%n", r);
        System.out.println("  → " + (r > 0.7 ? "Strong" : r > 0.4 ? "Moderate" : "Weak")
                + " positive correlation");

        // ==============================================================================
        // YOUR TURN — LESSON 03:
        // 1. Compute the IQR (Q3 - Q1) for salaries. Identify outliers using
        //    the rule: value < Q1 - 1.5*IQR  OR  value > Q3 + 1.5*IQR.
        //    Are there any salary outliers in this dataset?

        // 2. Normalize salaries to [0, 1]:
        //    normalized = (salary - min) / (max - min)
        //    Collect normalized values into a list and print them.
        //
        // 3. Compute correlation between salary and years_exp per department.
        //    Use employees.stream().filter(dept).collect() to get per-dept arrays.
        // ==============================================================================


        // ==============================================================================
        // LESSON 04 — WRITING STRUCTURED REPORTS
        // ==============================================================================

        System.out.println("\n=== LESSON 04: Writing Reports ===");

        // Write a formatted CSV report:
        Path outputPath = Paths.get("salary_report.csv");
        try (PrintWriter pw = new PrintWriter(Files.newBufferedWriter(outputPath))) {
            pw.println("department,count,avg_salary,max_salary,min_salary");

            employees.stream()
                    .collect(Collectors.groupingBy(Employee::getDepartment))
                    .forEach((dept, list) -> {
                        DoubleSummaryStatistics s = list.stream()
                                .mapToDouble(Employee::getSalary)
                                .summaryStatistics();
                        pw.printf("%s,%d,%.2f,%.2f,%.2f%n",
                                dept, s.getCount(), s.getAverage(), s.getMax(), s.getMin());
                    });
        }
        System.out.println("Salary report written: " + outputPath);

        // Read it back and print:
        Files.lines(outputPath).forEach(System.out::println);

        // Append to an existing file (append=true):
        try (PrintWriter pw = new PrintWriter(
                new BufferedWriter(new FileWriter("report.txt", true)))) {
            pw.println("Employee count: " + employees.size());
            pw.printf("Average salary: $%,.0f%n", StatUtils.mean(salaries));

    // ===========================================================================
    // EMPLOYEE RECORD CLASS
    // ===========================================================================

    static class Employee {
        private final int    id;
        private final String name;
        private final String department;
        private final double salary;
        private final int    yearsExp;
        private final String city;

        public Employee(int id, String name, String department,
                        double salary, int yearsExp, String city) {
            this.id         = id;
            this.name       = name;
            this.department = department;
            this.salary     = salary;
            this.yearsExp   = yearsExp;
            this.city       = city;
        }

        public int    getId()         { return id; }
        public String getName()       { return name; }
        public String getDepartment() { return department; }
        public double getSalary()     { return salary; }
        public int    getYearsExp()   { return yearsExp; }
        public String getCity()       { return city; }

        @Override
        public String toString() {
            return String.format("Employee{id=%d, name=%-6s, dept=%-8s, salary=$%,.0f, exp=%d, city=%s}",
                    id, name, department, salary, yearsExp, city);
        }
    }


    // ===========================================================================
    // STATISTICS UTILITY (standard library only — no external dependencies)
    // ===========================================================================

    static class StatUtils {

        public static double sum(double[] data) {
            double total = 0;
            for (double d : data) total += d;
            return total;
        }

        public static double mean(double[] data) {
            return data.length == 0 ? 0 : sum(data) / data.length;
        }

        public static double median(double[] data) {
            double[] sorted = Arrays.copyOf(data, data.length);
            Arrays.sort(sorted);
            int n = sorted.length;
            return n % 2 == 0
                    ? (sorted[n / 2 - 1] + sorted[n / 2]) / 2.0
                    : sorted[n / 2];
        }

        public static double variance(double[] data) {
            double m = mean(data);
            double sum = 0;
            for (double d : data) sum += (d - m) * (d - m);
            return data.length > 1 ? sum / (data.length - 1) : 0;  // sample variance
        }

        public static double stdDev(double[] data) {
            return Math.sqrt(variance(data));
        }

        public static double min(double[] data) {
            return Arrays.stream(data).min().orElse(0);
        }

        public static double max(double[] data) {
            return Arrays.stream(data).max().orElse(0);
        }

        public static double percentile(double[] data, double pct) {
            double[] sorted = Arrays.copyOf(data, data.length);
            Arrays.sort(sorted);
            double index = (pct / 100.0) * (sorted.length - 1);
            int    lower = (int) Math.floor(index);
            int    upper = (int) Math.ceil(index);
            if (lower == upper) return sorted[lower];
            double weight = index - lower;
            return sorted[lower] * (1 - weight) + sorted[upper] * weight;
        }

        public static double pearsonCorrelation(double[] x, double[] y) {
            double mx = mean(x), my = mean(y);
            double num = 0, dx = 0, dy = 0;
            for (int i = 0; i < x.length; i++) {
                num += (x[i] - mx) * (y[i] - my);
                dx  += (x[i] - mx) * (x[i] - mx);
                dy  += (y[i] - my) * (y[i] - my);
            }
            return (dx == 0 || dy == 0) ? 0 : num / Math.sqrt(dx * dy);
        }
    }
