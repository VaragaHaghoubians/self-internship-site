// ==============================================================================
//  JAVA MULTITHREADING & CONCURRENCY
//  FILE: JavaMultithreading.java
//
//  TASK: Learn to run code in parallel — process large datasets faster
//        and build systems that handle multiple requests simultaneously.
//
//  WHY:  Operations systems (Amazon routing, Uber dispatch) process
//        thousands of records per second. One thread is not enough.
//        As an ML Engineer, you need to:
//          → Process ML predictions for many requests simultaneously
//          → Run background jobs while serving live traffic
//          → Avoid blocking your API while waiting for I/O
//
//        This is CORE to ML Engineer and AI Engineer roles.
//        Interviews at Amazon/Google/Uber test multithreading knowledge.
//
//  PREREQ: JavaZeroToHero.java (OOP), JavaStreamsLambda.java (lambdas)
//
//  COMPILE: javac JavaMultithreading.java
//  RUN:     java JavaMultithreading
// ==============================================================================

import java.util.*;
import java.util.concurrent.*;
import java.util.concurrent.atomic.*;
import java.util.stream.*;

public class JavaMultithreading {

    public static void main(String[] args) throws InterruptedException, ExecutionException {

        // ==============================================================================
        // LESSON 01 — THREADS: The Basics
        // ==============================================================================
        //
        // TEACHER SAYS:
        // A THREAD is a separate path of execution within the same program.
        // Your JVM starts with 1 thread (the main thread). You can create more.
        //
        // TWO ways to create a thread:
        //   1. Extend Thread class
        //   2. Implement Runnable (PREFERRED — separates task from thread mechanics)
        //
        // Key methods:
        //   thread.start()           → starts the thread (calls run() in a new thread)
        //   thread.join()            → wait for thread to FINISH before continuing
        //   Thread.sleep(ms)         → pause current thread (throws InterruptedException)
        //   Thread.currentThread()   → reference to the running thread
        //
        // WARNING: thread.run() does NOT start a new thread — it runs on CURRENT thread!
        //          Always use thread.start() to launch a new thread.

        System.out.println("=== LESSON 01: Thread Basics ===");
        System.out.println("Main thread: " + Thread.currentThread().getName());

        // Way 1: Implement Runnable (recommended):
        Runnable task1 = () -> {
            System.out.println("Task1 running on: " + Thread.currentThread().getName());
            try {
                Thread.sleep(100);      // Simulate work
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
            }
            System.out.println("Task1 done");
        };

        Runnable task2 = () -> {
            System.out.println("Task2 running on: " + Thread.currentThread().getName());
            System.out.println("Task2 done");
        };

        Thread t1 = new Thread(task1, "Worker-1");
        Thread t2 = new Thread(task2, "Worker-2");

        t1.start();     // Launch t1 in a NEW thread — returns immediately
        t2.start();     // Launch t2 in another NEW thread — returns immediately

        // Main thread continues here while t1 and t2 run concurrently!
        System.out.println("Main thread continues...");

        t1.join();      // Wait for t1 to finish
        t2.join();      // Wait for t2 to finish
        System.out.println("Both threads completed.\n");

        // ==============================================================================
        // YOUR TURN — LESSON 01:
        // 1. Create 5 threads, each printing its thread name and a number 1-5.
        //    Start them all, then join all. Use a for loop.
        //    Observe: the order of output may vary each run — that's concurrency!
        // 2. Simulate a download: create 3 threads, each "downloads" (Thread.sleep)
        //    for a random 200-500ms, then prints "Download X complete".
        //    After all 3, print total elapsed time.
        //    HINT: long start = System.currentTimeMillis(); ... System.currentTimeMillis() - start
        // ==============================================================================


        // ==============================================================================
        // LESSON 02 — EXECUTORSERVICE: Professional Thread Management
        // ==============================================================================
        //
        // TEACHER SAYS:
        // Creating threads manually is messy for large-scale work.
        // ExecutorService manages a POOL of reusable threads — much better!
        //
        // Types of thread pools:
        //   Executors.newFixedThreadPool(n)     → exactly n threads, queue extra tasks
        //   Executors.newCachedThreadPool()     → creates threads as needed, reuses idle ones
        //   Executors.newSingleThreadExecutor() → 1 thread, tasks run one by one
        //
        // submit(Runnable)   → submit task, no return value (returns Future<?>)
        // submit(Callable)   → submit task WITH return value (returns Future<T>)
        // future.get()       → block until result is ready, then return it
        // shutdown()         → stop accepting new tasks, finish existing ones
        // awaitTermination() → wait until all tasks finish

        System.out.println("=== LESSON 02: ExecutorService ===");

        // Fixed thread pool — 4 workers, simulate ML batch predictions:
        ExecutorService pool = Executors.newFixedThreadPool(4);

        List<String> customerIds = Arrays.asList(
                "CUST001", "CUST002", "CUST003", "CUST004",
                "CUST005", "CUST006", "CUST007", "CUST008"
        );

        System.out.println("Submitting " + customerIds.size() + " prediction jobs to 4 workers...");
        long startTime = System.currentTimeMillis();

        for (String customerId : customerIds) {
            pool.submit(() -> {
                // Simulate ML inference (50-100ms per prediction):
                try {
                    Thread.sleep(50 + (int)(Math.random() * 50));
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
                System.out.printf("  Prediction done for %-8s on %s%n",
                        customerId, Thread.currentThread().getName());
            });
        }

        pool.shutdown();                              // No more new tasks
        pool.awaitTermination(30, TimeUnit.SECONDS);  // Wait for all to finish

        long elapsed = System.currentTimeMillis() - startTime;
        System.out.printf("All 8 predictions done in %d ms (4 parallel workers)%n%n", elapsed);

        // ==============================================================================
        // YOUR TURN — LESSON 02:
        // 1. Create a fixed thread pool with 3 workers.
        //    Submit 10 tasks: each prints "Processing batch [i]" and sleeps 100ms.
        //    Measure and print total wall-clock time.
        //    Compare: ~300-400ms (parallel) vs ~1000ms (sequential estimate).
        // 2. Why did 10 tasks × 100ms NOT take 1000ms? Explain in a comment.
        // ==============================================================================


        // ==============================================================================
        // LESSON 03 — CALLABLE & FUTURE: Getting Results Back
        // ==============================================================================
        //
        // TEACHER SAYS:
        // Runnable = "do this" — no return value.
        // Callable<T> = "compute this and give me a T back" — has return value.
        // submit(callable) returns a Future<T> — a PROMISE of a future result.
        //
        // Future<T> methods:
        //   future.get()                         → wait and get result (blocking)
        //   future.get(timeout, TimeUnit.SECONDS) → wait with timeout
        //   future.isDone()                      → true if computation finished
        //   future.cancel(true)                  → attempt to cancel
        //
        // Pattern: submit all → do other work → collect results later.

        System.out.println("=== LESSON 03: Callable and Future ===");

        ExecutorService executor = Executors.newFixedThreadPool(3);

        // Simulate: compute churn score for 5 customers in parallel
        List<Callable<Double>> scoringJobs = List.of(
            () -> computeChurnScore("CUST001", 0.85),
            () -> computeChurnScore("CUST002", 0.32),
            () -> computeChurnScore("CUST003", 0.71),
            () -> computeChurnScore("CUST004", 0.50),
            () -> computeChurnScore("CUST005", 0.95)
        );

        // Submit all and collect Futures:
        List<Future<Double>> futures = scoringJobs.stream()
                .map(executor::submit)
                .collect(Collectors.toList());

        System.out.println("All 5 scoring jobs submitted. Waiting for results...");

        // Collect results (future.get() blocks until each is ready):
        List<Double> scores = new ArrayList<>();
        for (Future<Double> future : futures) {
            scores.add(future.get());       // Waits if not done yet
        }

        System.out.println("Churn scores: " + scores);
        double highRiskCount = scores.stream().filter(s -> s > 0.7).count();
        System.out.printf("High-risk customers: %.0f / %d%n", highRiskCount, scores.size());

        executor.shutdown();
        System.out.println();

        // ==============================================================================
        // YOUR TURN — LESSON 03:
        // 1. Create a Callable that computes the sum of squares from 1 to N.
        //    Submit it for N = [100, 1000, 10000, 100000] simultaneously.
        //    Collect and print each result.
        // 2. Add a timeout to future.get(2, TimeUnit.SECONDS).
        //    What happens if a task takes too long? Catch TimeoutException.
        // ==============================================================================


        // ==============================================================================
        // LESSON 04 — THE RACE CONDITION PROBLEM & synchronized
        // ==============================================================================
        //
        // TEACHER SAYS:
        // When two threads read and write the SAME variable at the same time,
        // results become UNPREDICTABLE. This is a RACE CONDITION — a critical bug.
        //
        // Example: counter++ looks atomic but is actually 3 steps:
        //   1. Read counter from memory
        //   2. Add 1
        //   3. Write back to memory
        //   If another thread jumps between steps 1 and 3, you LOSE an update!
        //
        // SOLUTIONS:
        //   1. synchronized keyword      → lock the object, only one thread at a time
        //   2. AtomicInteger             → hardware-level atomic operations (FASTER)
        //   3. ConcurrentHashMap         → thread-safe HashMap
        //   4. Collections.synchronizedList() → thread-safe wrapper

        System.out.println("=== LESSON 04: Race Conditions & Solutions ===");

        // PROBLEM — unsafe counter:
        int[] unsafeCount = {0};
        ExecutorService racePool = Executors.newFixedThreadPool(10);
        for (int i = 0; i < 1000; i++) {
            racePool.submit(() -> unsafeCount[0]++);    // RACE CONDITION!
        }
        racePool.shutdown();
        racePool.awaitTermination(5, TimeUnit.SECONDS);
        System.out.println("Unsafe counter (should be 1000): " + unsafeCount[0]);
        // You'll likely see < 1000 — lost updates due to race condition!

        // SOLUTION — AtomicInteger (preferred for simple counters):
        AtomicInteger safeCount = new AtomicInteger(0);
        ExecutorService safePool = Executors.newFixedThreadPool(10);
        for (int i = 0; i < 1000; i++) {
            safePool.submit(() -> safeCount.incrementAndGet());   // Atomic!
        }
        safePool.shutdown();
        safePool.awaitTermination(5, TimeUnit.SECONDS);
        System.out.println("Safe counter   (should be 1000): " + safeCount.get());

        // SOLUTION — synchronized method (for complex shared state):
        SafeAggregator aggregator = new SafeAggregator();
        ExecutorService aggPool   = Executors.newFixedThreadPool(5);

        for (int i = 1; i <= 100; i++) {
            final int value = i;
            aggPool.submit(() -> aggregator.add(value));
        }
        aggPool.shutdown();
        aggPool.awaitTermination(5, TimeUnit.SECONDS);
        System.out.println("Synchronized sum(1..100) = " + aggregator.getTotal()
                + " (expected: 5050)");

        // ConcurrentHashMap — thread-safe Map:
        ConcurrentHashMap<String, Integer> wordCount = new ConcurrentHashMap<>();
        String[] words = {"java", "stream", "lambda", "java", "thread", "lambda", "java"};
        Arrays.stream(words)
              .parallel()   // Process in parallel!
              .forEach(word -> wordCount.merge(word, 1, Integer::sum));
        System.out.println("Word count (concurrent): " + wordCount);
        System.out.println();

        // ==============================================================================
        // YOUR TURN — LESSON 04:
        // 1. Without fixing it, run the "unsafe counter" 5 times. Do you always get 1000?
        //    (Uncomment to run multiple times)
        // 2. Replace AtomicInteger with a synchronized method — does it still give 1000?
        // 3. Create a ConcurrentHashMap to count character frequency in a String.
        //    Use parallel stream to populate it. Verify results match a sequential count.
        // ==============================================================================


        // ==============================================================================
        // LESSON 05 — PARALLEL STREAMS: Easy Parallelism for Data Processing
        // ==============================================================================
        //
        // TEACHER SAYS:
        // The EASIEST way to parallelize a data pipeline in Java is:
        // change .stream() to .parallelStream()
        //
        // parallelStream() splits the data across multiple CPU cores automatically.
        // Great for: CPU-bound operations on large collections (ML batch scoring).
        //
        // WHEN TO USE:
        //   ✅ Large collections (>10,000 elements)
        //   ✅ CPU-heavy operations (scoring, calculations)
        //   ❌ Small collections (overhead > gain)
        //   ❌ Operations that must stay in ORDER (use sequential stream)
        //   ❌ Shared mutable state (use Atomic classes if needed)

        System.out.println("=== LESSON 05: Parallel Streams ===");

        List<Integer> bigList = IntStream.rangeClosed(1, 2_000_000)
                .boxed()
                .collect(Collectors.toList());

        // Sequential:
        long seqStart = System.currentTimeMillis();
        long seqSum   = bigList.stream()
                               .mapToLong(Integer::longValue)
                               .sum();
        long seqTime  = System.currentTimeMillis() - seqStart;

        // Parallel:
        long parStart = System.currentTimeMillis();
        long parSum   = bigList.parallelStream()
                               .mapToLong(Integer::longValue)
                               .sum();
        long parTime  = System.currentTimeMillis() - parStart;

        System.out.printf("Sequential sum: %,d in %d ms%n", seqSum, seqTime);
        System.out.printf("Parallel sum:   %,d in %d ms%n", parSum, parTime);
        System.out.printf("Speedup: %.1fx (on %d CPU cores)%n%n",
                (double) seqTime / Math.max(parTime, 1),
                Runtime.getRuntime().availableProcessors());

        // Parallel batch ML simulation:
        List<String> customers = IntStream.rangeClosed(1, 10_000)
                .mapToObj(i -> "CUST" + String.format("%05d", i))
                .collect(Collectors.toList());

        long mlStart = System.currentTimeMillis();
        long highRisk = customers.parallelStream()
                .mapToDouble(id -> simulatePrediction(id))
                .filter(score -> score > 0.7)
                .count();
        long mlTime = System.currentTimeMillis() - mlStart;
        System.out.printf("Parallel ML: %,d high-risk customers out of %,d in %d ms%n%n",
                highRisk, customers.size(), mlTime);

        // ==============================================================================
        // YOUR TURN — LESSON 05:
        // 1. Compute the sum of squares (n²) for n from 1 to 1,000,000:
        //    - Using sequential stream
        //    - Using parallel stream
        //    Print both results and times. Are they always equal?
        //
        // 2. Filter all prime numbers between 1 and 100,000 using:
        //    - Stream filter with isPrime()
        //    - Parallel stream filter with isPrime()
        //    Which is faster? How many primes are there?
        //
        // 3. DANGER exercise: try to collect results with a non-thread-safe List:
        //    List<Integer> results = new ArrayList<>();
        //    bigList.parallelStream().forEach(results::add);
        //    Is results.size() always equal to bigList.size()? Why not?
        //    Fix: use .collect(Collectors.toList()) instead of forEach + add.
        // ==============================================================================


        // ==============================================================================
        // LESSON 06 — CAPSTONE: Parallel Batch Scoring Engine
        // ==============================================================================
        //
        // TEACHER SAYS:
        // A real ML deployment scenario:
        //   - You have a list of customers
        //   - You need to compute churn scores for ALL of them
        //   - Results must be stored with customer ID
        //   - Compare sequential vs parallel performance
        //
        // This is EXACTLY what a batch scoring job does at Amazon/Uber.

        System.out.println("=== LESSON 06: Capstone — Parallel Batch Scoring ===");

        List<String> batchCustomers = IntStream.rangeClosed(1, 500)
                .mapToObj(i -> "C" + i)
                .collect(Collectors.toList());

        // Sequential scoring:
        long seqBatch = System.currentTimeMillis();
        Map<String, Double> seqResults = new HashMap<>();
        for (String cid : batchCustomers) {
            seqResults.put(cid, simulatePrediction(cid));
        }
        System.out.printf("Sequential batch: 500 customers in %d ms%n",
                System.currentTimeMillis() - seqBatch);

        // Parallel scoring with CompletableFuture (production pattern):
        ExecutorService batchPool  = Executors.newFixedThreadPool(
                Runtime.getRuntime().availableProcessors());

        long parBatch = System.currentTimeMillis();
        List<CompletableFuture<Map.Entry<String, Double>>> cfFutures = batchCustomers.stream()
                .map(cid -> CompletableFuture.supplyAsync(
                        () -> Map.entry(cid, simulatePrediction(cid)), batchPool))
                .collect(Collectors.toList());

        Map<String, Double> parResults = cfFutures.stream()
                .map(CompletableFuture::join)
                .collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue));

        System.out.printf("Parallel batch:   500 customers in %d ms%n",
                System.currentTimeMillis() - parBatch);

        // Summarize results:
        long highRiskPar = parResults.values().stream().filter(s -> s > 0.7).count();
        System.out.printf("High-risk (>0.7): %d / 500%n", highRiskPar);
        System.out.println("Both results match: " + (seqResults.size() == parResults.size()));

        batchPool.shutdown();

        System.out.println("\nMultithreading module complete!");

        // ==============================================================================
        // YOUR TURN — LESSON 06 (Full Challenge):
        // Build a parallel batch scoring system with these requirements:
        //
        // TASK 1: Create a ProductCatalog of 1000 products:
        //   Product(id, name, price, category)

        //   demand = (random) * price / 100

        //   Measure time. What's the speedup?
        //
        // TASK 4: Aggregate results:
        //   - Average demand score per category
        //   - Top 10 products by demand score
        //   - Count products above/below mean demand

        // ==============================================================================

    } // end main()


    // ===========================================================================
    // HELPER METHODS
    // ===========================================================================

    /** Simulates ML model inference with artificial delay. */
    static double computeChurnScore(String customerId, double score) {
        try {
            Thread.sleep(80 + (int)(Math.random() * 40));   // 80-120ms "inference"
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
        System.out.printf("  Scored %-8s → %.2f  [%s]%n",
                customerId, score, Thread.currentThread().getName());
        return score;
    }

    /** Simulates a fast ML prediction (no sleep — for large-scale parallel demos). */
    static double simulatePrediction(String customerId) {
        double hash = Math.abs(customerId.hashCode() % 100) / 100.0;
        return 0.1 + 0.8 * hash + Math.random() * 0.1;
    }

    static boolean isPrime(int n) {
        if (n < 2) return false;
        for (int i = 2; i <= Math.sqrt(n); i++) if (n % i == 0) return false;
        return true;
    }


    // ===========================================================================
    // THREAD-SAFE AGGREGATOR (used in Lesson 04)
    // ===========================================================================

    static class SafeAggregator {
        private long total = 0;

        public synchronized void add(long value) {
            total += value;          // synchronized: only one thread at a time
        }

        public synchronized long getTotal() {
            return total;
        }
    }

} // end class
