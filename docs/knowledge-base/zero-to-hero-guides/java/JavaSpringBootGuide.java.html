package com.internship.churnapi.service;

import com.internship.churnapi.model.CustomerRequest;
import com.internship.churnapi.model.PredictionResponse;
import org.springframework.stereotype.Service;

import java.util.*;

// @Service marks this as a Spring "bean" — Spring creates ONE instance (singleton)
// and injects it wherever needed automatically.
@Service
public class ChurnPredictionService {

    // In a real system, you would:
    //   1. Load a serialized model file (e.g., via ONNX or a Python subprocess call)
    //   2. Or call a Python FastAPI ML service via HTTP
    // Here we simulate with a rule-based score for learning purposes.

    public PredictionResponse predict(CustomerRequest customer) {
        double score = computeChurnScore(customer);
        return new PredictionResponse(
                customer.getCustomerId(),
                score >= 0.5,
                score
        );
    }

    public List<PredictionResponse> predictBatch(List<CustomerRequest> customers) {
        return customers.stream()
                .map(this::predict)
                .toList();
    }

    // Simulate a model: rule-based scoring (replace with real model in production)
    private double computeChurnScore(CustomerRequest c) {
        double score = 0.0;

        // High support calls → high churn risk:
        if (c.getSupportCalls() > 8) score += 0.40;
        else if (c.getSupportCalls() > 5) score += 0.20;

        // Low spend → high churn risk:
        if (c.getMonthlySpend() < 50)  score += 0.30;
        else if (c.getMonthlySpend() < 100) score += 0.15;

        // Short contract → high churn risk:
        if (c.getContractMonths() < 6)  score += 0.20;
        else if (c.getContractMonths() < 12) score += 0.10;

        // Basic plan → slightly higher risk:
        if ("Basic".equals(c.getPlan())) score += 0.10;

        return Math.min(score, 0.99);   // Cap at 99%
    }
}

// The CONTROLLER handles HTTP requests and delegates to services.
// It knows about HTTP (status codes, headers) but NOT business logic.
//
// Key Spring MVC annotations:
//   @RestController           → mark class as REST controller (returns JSON)

//   @GetMapping("/path")      → handle GET requests
//   @PostMapping("/path")     → handle POST requests
//   @PathVariable             → extract {id} from URL path
//   @RequestBody              → parse JSON body into a Java object
//   @RequestParam             → parse query param ?key=value
//   @Valid                    → trigger validation on @RequestBody
//   ResponseEntity<T>         → control HTTP status code AND body
//
// File: src/main/java/com/internship/churnapi/controller/ChurnController.java

// @ControllerAdvice makes this class handle exceptions across ALL controllers
@ControllerAdvice
public class GlobalExceptionHandler {

    // Handle validation errors (@Valid failures):
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<Map<String, Object>> handleValidationError(
            MethodArgumentNotValidException ex) {

        List<String> errors = ex.getBindingResult()
                .getFieldErrors()
                .stream()
                .map(fe -> fe.getField() + ": " + fe.getDefaultMessage())
                .toList();

        return ResponseEntity
                .status(HttpStatus.BAD_REQUEST)
                .body(Map.of(
                    "error",   "Validation failed",
                    "details", errors
                ));
    }

    // Handle unexpected errors:
    @ExceptionHandler(Exception.class)
    public ResponseEntity<Map<String, String>> handleGeneral(Exception ex) {
        return ResponseEntity
                .status(HttpStatus.INTERNAL_SERVER_ERROR)
                .body(Map.of(
                    "error",   "Internal server error",

// PART 7 — SPRING BOOT QUICK REFERENCE
// ==============================================================================
/*
ANNOTATION        WHAT IT DOES
──────────────────────────────────────────────────────────────────
@SpringBootApplication   Entry point — auto-configures everything
@RestController          Class handles REST requests, returns JSON
@RequestMapping("/path") Base URL prefix for all methods in class
@GetMapping("/path")     Handle GET /path
@PostMapping("/path")    Handle POST /path
@PutMapping("/path")     Handle PUT /path
@DeleteMapping("/path")  Handle DELETE /path
@PathVariable            Bind URL path segment: /users/{id}
@RequestParam            Bind query string: /search?term=java
@RequestBody             Parse JSON request body to Java object
@Valid                   Validate @RequestBody using constraints
@Service                 Business logic bean (Spring manages lifecycle)
@Repository              Data access bean (DB layer)
@Component               Generic Spring-managed bean
@Autowired               Inject dependency (prefer constructor injection)
ResponseEntity<T>        Control HTTP status + body

HTTP STATUS CODES (always return the right one):
  200 OK          → successful GET or POST
  201 CREATED     → resource created (POST that creates something)
  400 BAD REQUEST → bad input, validation error
  401 UNAUTHORIZED → not authenticated
  403 FORBIDDEN   → authenticated but no permission
  404 NOT FOUND   → resource doesn't exist
  422 UNPROCESSABLE → valid JSON but semantically wrong
  500 SERVER ERROR → bug in your code
*/
