// ==============================================================================
//  JAVA STREAMS & LAMBDA EXPRESSIONS
//  FILE: JavaStreamsLambda.java
//
//  TASK: Master the most important feature of modern Java (Java 8+):
//        functional-style data processing with lambdas and Streams.
//
//  WHY:  Before Java 8, processing a list required verbose for-loops.
//        Now you can filter, transform, sort, and aggregate data
//        in ONE clean pipeline — just like pandas or SQL.
//        Every Java data/ML job uses Streams constantly.
//        Interviews ALWAYS test this.
//
//  PREREQ: JavaZeroToHero.java (collections, OOP basics)
//
//  COMPILE: javac JavaStreamsLambda.java
//  RUN:     java JavaStreamsLambda
// ==============================================================================

import java.util.*;
import java.util.stream.*;
import java.util.function.*;

public class JavaStreamsLambda {

    public static void main(String[] args) {

        // ==============================================================================
        // LESSON 01 — LAMBDA EXPRESSIONS: Anonymous Functions
        // ==============================================================================
        //
        // TEACHER SAYS:
        // A LAMBDA is a short, nameless function — like Python's lambda.
        // Syntax:  (parameters) -> expression
        //          (parameters) -> { multiple; statements; }
        //
        // Before Java 8, you needed a full anonymous class. Now: one line.
        //
        // FUNCTIONAL INTERFACES — interfaces with exactly ONE abstract method:
        //   Runnable        → () -> void            (no args, no return)
        //   Comparator<T>   → (T,T) -> int          (compare two values)
        //   Function<T,R>   → (T) -> R              (transform T to R)
        //   Predicate<T>    → (T) -> boolean        (test a condition)
        //   Consumer<T>     → (T) -> void           (process, no return)
        //   Supplier<T>     → () -> T               (produce a value)
        //   BiFunction<T,U,R> → (T,U) -> R         (two inputs)
        //
        // EXAMPLE:

        System.out.println("=== LESSON 01: Lambda Expressions ===");

        // Old way (anonymous class):
        Comparator<String> oldWay = new Comparator<String>() {
            @Override
            public int compare(String a, String b) {
                return a.compareTo(b);
            }
        };

        // New way (lambda):
        Comparator<String> byName   = (a, b) -> a.compareTo(b);
        Comparator<String> byLength = (a, b) -> Integer.compare(a.length(), b.length());

        List<String> names = new ArrayList<>(Arrays.asList("Charlie", "Alice", "Bob", "Diana"));
        names.sort(oldWay);   // old anonymous class works identically to lambda
        System.out.println("Sorted (old way):  " + names);
        names.sort(byName);
        System.out.println("Sorted by name:   " + names);
        names.sort(byLength);
        System.out.println("Sorted by length: " + names);

        // Predicate — test a condition:
        Predicate<Integer> isEven    = n -> n % 2 == 0;
        Predicate<Integer> isPositive = n -> n > 0;
        Predicate<Integer> isEvenAndPositive = isEven.and(isPositive);  // Combine predicates!

        System.out.println("4 isEven: "           + isEven.test(4));
        System.out.println("3 isEven: "           + isEven.test(3));
        System.out.println("4 isEvenAndPositive: " + isEvenAndPositive.test(4));

        // Function — transform one type to another:
        Function<String, Integer> lengthOf  = s -> s.length();
        Function<Integer, String> intToStr  = i -> "Number: " + i;
        Function<String, String>  chained   = lengthOf.andThen(intToStr);  // Compose functions

        System.out.println(chained.apply("Hello"));     // "Number: 5"

        // Consumer — process without returning:
        Consumer<String> printer = s -> System.out.println("→ " + s);
        printer.accept("Lambda works!");

        // Supplier — produce a value with no input:
        Supplier<List<String>> listFactory = () -> new ArrayList<>();
        List<String> newList = listFactory.get();
        System.out.println("Empty list from Supplier: " + newList);

        // BiFunction — two inputs:
        BiFunction<String, Integer, String> repeat = (s, n) -> s.repeat(n);
        System.out.println(repeat.apply("Java! ", 3));

        // ==============================================================================
        // YOUR TURN — LESSON 01:
        // 1. Write a Predicate<String> that returns true if a String starts with 'A'
        // 2. Write a Function<Double, String> that formats a double to 2 decimal places
        //    HINT: String.format("%.2f", d)
        // 3. Write a Comparator<Integer> that sorts in DESCENDING order
        //    Use it to sort: [5, 2, 8, 1, 9, 3]
        // ==============================================================================


        // ==============================================================================
        // LESSON 02 — METHOD REFERENCES: Even Shorter Lambdas
        // ==============================================================================
        //
        // TEACHER SAYS:
        // Method references are shorthand for lambdas that just call one existing method.
        // 4 forms:
        //   ClassName::staticMethod    → Math::sqrt       (lambda: x -> Math.sqrt(x))
        //   object::instanceMethod     → printer::println (lambda: s -> printer.println(s))
        //   ClassName::instanceMethod  → String::toUpperCase (lambda: s -> s.toUpperCase())
        //   ClassName::new             → ArrayList::new   (lambda: () -> new ArrayList<>())

        System.out.println("\n=== LESSON 02: Method References ===");

        List<String> cities = Arrays.asList("london", "paris", "berlin", "tokyo", "nyc");

        // Lambda:       s -> s.toUpperCase()
        // Method ref:   String::toUpperCase
        cities.stream()
              .map(String::toUpperCase)
              .forEach(System.out::println);     // System.out::println = s -> System.out.println(s)

        // Static method reference:
        List<Double> rawScores = Arrays.asList(1.7, 4.2, 2.1, 9.0, 5.5);
        rawScores.stream()
                 .map(Math::sqrt)               // x -> Math.sqrt(x)
                 .forEach(v -> System.out.printf("%.2f  ", v));
        System.out.println();

        // ==============================================================================
        // YOUR TURN — LESSON 02:
        // 1. Given: List<String> words = Arrays.asList("hello", "world", "java")
        //    Use method references to: trim each word, then print each word
        // 2. Use Integer::parseInt with a Stream to convert
        //    List<String> numStrings = Arrays.asList("10", "20", "30", "40")
        //    to a List<Integer> and print their sum
        // ==============================================================================


        // ==============================================================================
        // LESSON 03 — THE STREAMS API: Data Pipelines in Java
        // ==============================================================================
        //
        // TEACHER SAYS:
        // A STREAM is a pipeline of operations applied to a data source.
        // THINK of it like: pandas chain, or SQL query.
        //
        // stream() → create stream from collection
        // INTERMEDIATE ops (return a Stream, can chain):
        //   filter(Predicate)  → keep only matching elements
        //   map(Function)      → transform each element
        //   sorted()           → sort (natural order or with Comparator)
        //   distinct()         → remove duplicates
        //   limit(n)           → take only first n elements
        //   skip(n)            → skip first n elements
        //   flatMap(Function)  → flatten nested lists
        //   peek(Consumer)     → debug without consuming (print and pass through)
        //
        // TERMINAL ops (consume the stream — can only call ONE per stream):
        //   forEach(Consumer)           → do something for each element
        //   collect(Collector)          → collect into List/Set/Map
        //   count()                     → count elements
        //   findFirst()                 → first element (returns Optional<T>)
        //   anyMatch(Predicate)         → true if ANY element matches
        //   allMatch(Predicate)         → true if ALL elements match
        //   noneMatch(Predicate)        → true if NO element matches
        //   min(Comparator)             → minimum (returns Optional<T>)
        //   max(Comparator)             → maximum (returns Optional<T>)
        //   reduce(identity, accumulator) → fold into one value (like sum, product)
        //   toList()                    → Java 16+ shorthand for collect(Collectors.toList())
        //
        // IMPORTANT: Streams are LAZY — intermediate ops don't run until a terminal op is called.
        //            After a terminal op, the stream is CONSUMED — you can't reuse it!

        System.out.println("\n=== LESSON 03: Streams API ===");

        List<Integer> numbers = Arrays.asList(1, 5, 3, 9, 2, 8, 4, 7, 6, 10, 5, 3);

        // --- filter + collect ---
        List<Integer> evens = numbers.stream()
                .filter(n -> n % 2 == 0)
                .collect(Collectors.toList());
        System.out.println("Even numbers: " + evens);

        // --- map + sorted + collect ---
        List<Integer> squaredSorted = numbers.stream()
                .distinct()                          // remove duplicates: [1,5,3,9,2,8,4,7,6,10]
                .map(n -> n * n)                     // square each: [1,25,9,81,4,64,16,49,36,100]
                .sorted()                            // sort ascending
                .collect(Collectors.toList());
        System.out.println("Squared and sorted: " + squaredSorted);

        // --- filter + count ---
        long bigNumbers = numbers.stream()
                .filter(n -> n > 5)
                .count();
        System.out.println("Numbers > 5: " + bigNumbers);

        // --- reduce: sum ---
        int total = numbers.stream()
                .reduce(0, Integer::sum);            // 0 = identity, Integer::sum = accumulator
        System.out.println("Sum: " + total);

        // --- min and max ---
        Optional<Integer> maximum = numbers.stream().max(Comparator.naturalOrder());
        Optional<Integer> minimum = numbers.stream().min(Comparator.naturalOrder());
        System.out.println("Max: " + maximum.get() + ", Min: " + minimum.get());

        // --- anyMatch, allMatch, noneMatch ---
        System.out.println("Any > 8: "  + numbers.stream().anyMatch(n -> n > 8));
        System.out.println("All > 0: "  + numbers.stream().allMatch(n -> n > 0));
        System.out.println("None < 0: " + numbers.stream().noneMatch(n -> n < 0));

        // --- Streams on Objects: Employee example ---
        List<Employee> employees = Arrays.asList(
            new Employee("Alice", "DS",      85000, 7),
            new Employee("Bob",   "CS",      72000, 3),
            new Employee("Carol", "DS",      91000, 10),
            new Employee("David", "Finance", 68000, 2),
            new Employee("Eve",   "CS",      79000, 5),
            new Employee("Frank", "DS",      88000, 8)
        );

        // Get all DS employees sorted by salary descending:
        System.out.println("\nDS employees by salary (desc):");
        employees.stream()
                 .filter(e -> e.getDept().equals("DS"))
                 .sorted(Comparator.comparingDouble(Employee::getSalary).reversed())
                 .forEach(e -> System.out.printf("  %-8s $%,.0f%n", e.getName(), e.getSalary()));

        // Average salary by department:
        System.out.println("\nAverage salary per department:");
        Map<String, Double> avgSalaryByDept = employees.stream()
                .collect(Collectors.groupingBy(
                        Employee::getDept,
                        Collectors.averagingDouble(Employee::getSalary)
                ));
        avgSalaryByDept.forEach((dept, avg) ->
                System.out.printf("  %-10s → $%,.0f%n", dept, avg));

        // Count per department:
        Map<String, Long> countByDept = employees.stream()
                .collect(Collectors.groupingBy(Employee::getDept, Collectors.counting()));
        System.out.println("\nHead count per dept: " + countByDept);

        // Names as a comma-separated String:
        String allNames = employees.stream()
                .map(Employee::getName)
                .collect(Collectors.joining(", "));
        System.out.println("All names: " + allNames);

        // Extract just names into a list:
        List<String> seniorEmployees = employees.stream()
                .filter(e -> e.getYearsExp() >= 5)
                .map(Employee::getName)
                .sorted()
                .collect(Collectors.toList());
        System.out.println("Senior employees (5+ years): " + seniorEmployees);

        // ==============================================================================
        // YOUR TURN — LESSON 03:
        // Use the employees list above for all tasks.
        //
        // TASK 1: Find the highest-paid employee in the CS department.
        //         Print their name and salary.
        //         HINT: filter dept="CS", then max(Comparator.comparingDouble(...))
        //
        // TASK 2: Calculate the total salary budget (sum of all salaries).
        //         HINT: mapToDouble(Employee::getSalary).sum()

        // TASK 3: Find all employees earning more than $80,000.
        //         Collect their names into a Set<String>.
        //         HINT: collect(Collectors.toSet())
        //
        // TASK 4: Group employees into two groups: "Senior" (yearsExp >= 5) and "Junior".
        //         HINT: Collectors.partitioningBy(e -> e.getYearsExp() >= 5)
        //
        // TASK 5: Create a Map<String, String> mapping employee name → department.
        //         HINT: Collectors.toMap(Employee::getName, Employee::getDept)
        // ==============================================================================


        // ==============================================================================
        // LESSON 04 — NUMERIC STREAMS: IntStream, DoubleStream, LongStream
        // ==============================================================================
        //
        // TEACHER SAYS:
        // Working with primitives in a Stream<Integer> boxes/unboxes every value → slow.
        // IntStream, DoubleStream, LongStream work directly with primitives — FAST.
        // They also have statistics methods not on Stream<T>: sum(), average(), min(), max()
        //
        // EXAMPLE:

        System.out.println("\n=== LESSON 04: Numeric Streams ===");

        // IntStream.range: like Python range()
        IntStream.range(0, 5).forEach(i -> System.out.print(i + " "));
        System.out.println();

        IntStream.rangeClosed(1, 5).forEach(i -> System.out.print(i + " "));  // includes 5
        System.out.println();

        // Statistics on numeric data:
        int[] scores = {78, 85, 92, 67, 88, 79, 95, 82};
        IntSummaryStatistics stats = Arrays.stream(scores).summaryStatistics();

        System.out.println("Count:   " + stats.getCount());
        System.out.println("Sum:     " + stats.getSum());
        System.out.printf ("Average: %.2f%n", stats.getAverage());
        System.out.println("Max:     " + stats.getMax());
        System.out.println("Min:     " + stats.getMin());

        // Map objects to double stream:
        DoubleSummaryStatistics salaryStats = employees.stream()
                .mapToDouble(Employee::getSalary)
                .summaryStatistics();
        System.out.printf("Salary → avg=$%,.0f, max=$%,.0f%n",
                salaryStats.getAverage(), salaryStats.getMax());

        // ==============================================================================
        // YOUR TURN — LESSON 04:
        // 1. Generate all perfect squares from 1 to 100 using IntStream.rangeClosed()
        //    and filter(n -> (int)Math.sqrt(n) * (int)Math.sqrt(n) == n)
        // 2. Using employees, compute DoubleStream of salaries:
        //    - sum, average, max, min in one pass using summaryStatistics()
        // 3. Compute the sum of all integers from 1 to 1,000,000 using IntStream
        //    Compare using: rangeClosed(1,1_000_000).sum()
        // ==============================================================================


        // ==============================================================================
        // LESSON 05 — OPTIONAL<T>: Eliminating NullPointerException
        // ==============================================================================
        //
        // TEACHER SAYS:
        // NullPointerException (NPE) is Java's #1 runtime error.
        // Optional<T> wraps a value that MIGHT be null — forces you to handle it.
        // Never call .get() directly! Use: .orElse(), .orElseGet(), .map(), .ifPresent()
        //
        // EXAMPLE:

        System.out.println("\n=== LESSON 05: Optional<T> ===");

        Optional<String> present = Optional.of("Data Science");
        Optional<String> empty   = Optional.empty();

        // Safe ways to use Optional:
        System.out.println(present.isPresent());                    // true
        System.out.println(empty.isEmpty());                        // true
        System.out.println(present.get());                          // "Data Science"
        System.out.println(empty.orElse("default value"));          // "default value"
        System.out.println(empty.orElseGet(() -> "computed"));      // lazy computation

        // map() — transform if present:
        Optional<Integer> length = present.map(String::length);
        System.out.println("Length if present: " + length.orElse(0));

        // Filter within Optional:
        Optional<String> longField = present.filter(s -> s.length() > 5);
        System.out.println("Long field: " + longField.orElse("too short"));

        // ifPresent — do something if value exists:
        present.ifPresent(s -> System.out.println("Found: " + s));

        // Real use: stream + findFirst returns Optional:
        Optional<Employee> firstDS = employees.stream()
                .filter(e -> e.getDept().equals("DS"))
                .findFirst();

        firstDS.ifPresent(e -> System.out.println("First DS employee: " + e.getName()));

        String deptName = firstDS
                .map(Employee::getDept)
                .orElse("No department");
        System.out.println("Dept: " + deptName);

        // ==============================================================================
        // YOUR TURN — LESSON 05:

        // 3. Chain: find the first employee earning > $90,000,

        // ==============================================================================
        // LESSON 06 — CAPSTONE: Streams Data Pipeline
        // ==============================================================================
        //
        // TEACHER SAYS:

        // real enterprise data systems.
        //
        // Scenario: You have raw sales records. You need to:
        //   1. Clean invalid entries
        //   2. Enrich with computed fields
        //   3. Aggregate by region
        //   4. Find top performers
        //   5. Format the report

        System.out.println("\n=== LESSON 06: Capstone — Sales Pipeline ===");

        List<SalesRecord> sales = Arrays.asList(
            new SalesRecord("Alice",  "North", 1500, true),
            new SalesRecord("Bob",    "South", 900,  false),   // invalid
            new SalesRecord("Carol",  "North", 2300, true),
            new SalesRecord("David",  "East",  1800, true),
            new SalesRecord("Eve",    "West",  3100, true),
            new SalesRecord("Frank",  "North", 700,  false),   // invalid
            new SalesRecord("Grace",  "South", 2100, true),
            new SalesRecord("Henry",  "East",  1600, true),
            new SalesRecord("Iris",   "West",  2800, true),
            new SalesRecord("Jack",   "South", 1900, true)
        );

        // Step 1: Filter valid records only
        // Step 2: Group by region and sum sales
        // Step 3: Find top region
        Map<String, Integer> salesByRegion = sales.stream()
                .filter(SalesRecord::isValid)
                .collect(Collectors.groupingBy(
                        SalesRecord::getRegion,
                        Collectors.summingInt(SalesRecord::getAmount)
                ));

        System.out.println("Sales by region:");
        salesByRegion.entrySet().stream()
                .sorted(Map.Entry.<String, Integer>comparingByValue().reversed())
                .forEach(e -> System.out.printf("  %-8s → $%,d%n", e.getKey(), e.getValue()));

        Optional<Map.Entry<String, Integer>> topRegion = salesByRegion.entrySet()
                .stream()
                .max(Map.Entry.comparingByValue());
        topRegion.ifPresent(e ->
                System.out.printf("Top region: %s ($%,d)%n", e.getKey(), e.getValue()));

        // Average sale amount (valid only):
        OptionalDouble avgSale = sales.stream()
                .filter(SalesRecord::isValid)
                .mapToInt(SalesRecord::getAmount)
                .average();
        System.out.printf("Average sale: $%,.0f%n", avgSale.orElse(0));

    // ============================================================================
    // HELPER CLASSES (used in the examples above)
    // ============================================================================

    static class Employee {
        private final String name;
        private final String dept;
        private final double salary;
        private final int    yearsExp;

        public Employee(String name, String dept, double salary, int yearsExp) {
            this.name     = name;
            this.dept     = dept;
            this.salary   = salary;
            this.yearsExp = yearsExp;
        }

        public String getName()    { return name; }
        public String getDept()    { return dept; }
        public double getSalary()  { return salary; }
        public int    getYearsExp(){ return yearsExp; }

        @Override
        public String toString() {
            return String.format("Employee{%s, %s, $%.0f, %d yrs}", name, dept, salary, yearsExp);
        }
    }

    static class SalesRecord {
        private final String name;
        private final String region;
        private final int    amount;
        private final boolean valid;

        public SalesRecord(String name, String region, int amount, boolean valid) {
            this.name   = name;
            this.region = region;
            this.amount = amount;
            this.valid  = valid;
        }

        public String  getName()   { return name; }
        public String  getRegion() { return region; }
        public int     getAmount() { return amount; }
        public boolean isValid()   { return valid; }
    }

} // end class
