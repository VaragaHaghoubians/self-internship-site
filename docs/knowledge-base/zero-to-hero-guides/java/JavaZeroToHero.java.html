    /**
     * Returns a greeting message for the given name.
     * @param name the person's name
     * @return formatted greeting string
     */
    public static String greet(String name) {
        return "Hello, " + name + "! Welcome to Java.";
    }

    /**
     * Calculates the letter grade for a score out of a total.
     * @param score  the student's score
     * @param total  the maximum possible score
     * @return letter grade A, B, C, D, or F
     */
    public static String calculateGrade(int score, int total) {
        double percentage = (double) score / total * 100;
        if      (percentage >= 90) return "A";
        else if (percentage >= 80) return "B";
        else if (percentage >= 70) return "C";
        else if (percentage >= 60) return "D";
        else                       return "F";
    }

    /**
     * Checks whether a number is prime.
     * @param n the number to check
     * @return true if n is prime, false otherwise
     */
    public static boolean isPrime(int n) {
        if (n < 2) return false;
        for (int i = 2; i <= Math.sqrt(n); i++) {
            if (n % i == 0) return false;   // Found a divisor → not prime
        }
        return true;
    }

    /**
     * Prints basic statistics for an integer array.
     * @param data array of integers
     */
    public static void printStats(int[] data) {
        if (data == null || data.length == 0) {
            System.out.println("Empty data!");
            return;
        }
        int min = data[0], max = data[0], sum = 0;
        for (int val : data) {
            if (val < min) min = val;
            if (val > max) max = val;
            sum += val;
        }
        double avg = (double) sum / data.length;
        System.out.printf("Count: %d, Sum: %d, Min: %d, Max: %d, Avg: %.2f%n",
                data.length, sum, min, max, avg);
    }

    // Method overloading — same name, different signatures:
    public static int    add(int a, int b)            { return a + b; }
    public static double add(double a, double b)      { return a + b; }
    public static int    add(int a, int b, int c)     { return a + b + c; }

} // END OF JavaZeroToHero class


// ==============================================================================
// DAY 09 — COLLECTIONS: ArrayList, HashMap, HashSet
// ==============================================================================
//
// TEACHER SAYS:
// Java has a COLLECTIONS FRAMEWORK with dynamic data structures.
// Unlike arrays, these grow and shrink automatically.
//
//   ArrayList<T>   → dynamic array — ordered, allows duplicates
//   LinkedList<T>  → fast insert/delete at any position
//   HashMap<K,V>   → key-value pairs (like Python dict)
//   HashSet<T>     → unique values only (like Python set)
//
// The <T> is a GENERIC TYPE — you specify what type goes inside.
// You cannot use primitives: use Integer, Double, Boolean (wrapper classes).
//
// EXAMPLE (put this in a new class or inside main):

class CollectionsExample {
    public static void main(String[] args) {
        // ArrayList — dynamic list:
        java.util.ArrayList<String> subjects = new java.util.ArrayList<>();
        subjects.add("Python");
        subjects.add("SQL");
        subjects.add("Machine Learning");
        subjects.add("Statistics");

        System.out.println(subjects);           // [Python, SQL, Machine Learning, Statistics]
        System.out.println(subjects.get(0));    // Python
        System.out.println(subjects.size());    // 4
        subjects.remove("SQL");                 // Remove by value
        subjects.set(0, "Java");               // Replace at index

        for (String s : subjects) {
            System.out.println(s);
        }

        // Sort:
        java.util.Collections.sort(subjects);
        System.out.println(subjects);

        // HashMap — key-value store:
        java.util.HashMap<String, Double> studentGPA = new java.util.HashMap<>();
        studentGPA.put("Alice", 3.8);
        studentGPA.put("Bob",   3.5);
        studentGPA.put("Carol", 3.9);

        System.out.println(studentGPA.get("Alice"));       // 3.8
        System.out.println(studentGPA.containsKey("Bob")); // true
        studentGPA.remove("Bob");

// ==============================================================================
// DAY 10 — OOP: CLASSES & OBJECTS
// ==============================================================================
//
// TEACHER SAYS:
// Java is a PURE OOP language — everything is a class!
// A CLASS defines the BLUEPRINT: what data (fields) and what behavior (methods).
// An OBJECT is an INSTANCE created from a class blueprint.
//
// ENCAPSULATION — hide internal data, expose controlled access:
//   private fields → only accessible within the class
//   public getters → read field value: getAge()
//   public setters → write field value with validation: setAge(int age)
//
// CONSTRUCTOR — special method called when object is created with 'new'
//   Same name as class, no return type.
//   'this' keyword refers to the current object's fields/methods.
//
// EXAMPLE:

class Student {
    // Fields (attributes) — private for encapsulation:
    private String name;
    private int    age;
    private double gpa;
    private java.util.ArrayList<String> courses;

    // Class constant:
    public static final int MAX_COURSES = 6;

    /**
     * Constructor — called when you write: new Student("Alice", 25, 3.8)
     */
    public Student(String name, int age, double gpa) {
        this.name    = name;    // 'this.name' = field, 'name' = parameter
        this.age     = age;
        this.gpa     = gpa;
        this.courses = new java.util.ArrayList<>();
    }

    // Getters:
    public String getName()  { return name; }
    public int    getAge()   { return age;  }
    public double getGpa()   { return gpa;  }

    // Setter with validation:
    public void setGpa(double gpa) {
        if (gpa >= 0.0 && gpa <= 4.0) {
            this.gpa = gpa;
        } else {
            System.out.println("Invalid GPA: must be 0.0 to 4.0");
        }
    }

    // Method:
    public void enroll(String course) {
        if (courses.size() < MAX_COURSES) {
            courses.add(course);
            System.out.println(name + " enrolled in " + course);
        } else {
            System.out.println("Cannot enroll — maximum courses reached!");
        }
    }

    public String getStatus() {
        return gpa >= 3.0 ? "Good Standing" : "Academic Probation";
    }

    // toString — called when you print the object:
    @Override
    public String toString() {
        return String.format("Student{name='%s', age=%d, gpa=%.2f, courses=%s}",
                name, age, gpa, courses);
    }
}

// ==============================================================================
// DAY 11 — OOP: INHERITANCE & INTERFACES
// ==============================================================================
//
// TEACHER SAYS:
// INHERITANCE: a class can EXTEND another — gets all its fields and methods.
//   Use 'extends' keyword. Java only allows ONE parent class (single inheritance).
//   super() calls the parent's constructor.
//   @Override annotation marks a method that overrides the parent's version.
//
// INTERFACE: a CONTRACT — defines what a class CAN DO (methods without body).
//   Use 'implements' keyword. A class CAN implement MULTIPLE interfaces.
//   All interface methods are implicitly public and abstract.
//   From Java 8: interfaces can have 'default' method implementations.
//
// ABSTRACT CLASS: between regular class and interface — can have both
//   concrete methods and abstract methods (no body, must be implemented by subclass).
//
// EXAMPLE:

interface Gradable {
    String getLetterGrade();    // Every Gradable thing must implement this
    default boolean isPassing() { return true; }  // default implementation
}

abstract class Person {
    protected String name;
    protected int    age;

    public Person(String name, int age) {
        this.name = name;
        this.age  = age;
    }

    public abstract String getRole();   // Subclasses MUST implement this

    public String introduce() {
        return String.format("Hi, I'm %s (%d years old), I'm a %s.",
                name, age, getRole());
    }
}

class GraduateStudent extends Person implements Gradable {
    private double finalScore;
    private String thesis;

    public GraduateStudent(String name, int age, double finalScore, String thesis) {
        super(name, age);           // Call Person's constructor
        this.finalScore = finalScore;
        this.thesis     = thesis;
    }

    @Override
    public String getRole() { return "Graduate Student"; }

    @Override
    public String getLetterGrade() {
        if (finalScore >= 90) return "A";
        if (finalScore >= 80) return "B";
        if (finalScore >= 70) return "C";
        return "F";
    }

    @Override
    public String toString() {
        return String.format("%s | Thesis: %s | Grade: %s",
                introduce(), thesis, getLetterGrade());
    }
}

// ==============================================================================
// DAY 12 — EXCEPTION HANDLING
// ==============================================================================
//
// TEACHER SAYS:
// Java has TWO types of exceptions:
//   CHECKED   → must handle with try/catch or declare with 'throws'
//               e.g., IOException, SQLException
//   UNCHECKED → runtime exceptions, don't have to handle (but should!)
//               e.g., NullPointerException, ArrayIndexOutOfBoundsException
//
// try-catch-finally works like Python, with additions:
//   try-with-resources → automatically closes resources (files, DB connections)
//   multi-catch → catch multiple exception types in one catch block
//
// Always catch SPECIFIC exceptions — never just 'catch (Exception e)' lazily!
// Create CUSTOM exceptions for your domain problems.
//
// EXAMPLE:

class DataProcessor {
    public static double parseScore(String input) {
        try {
            double score = Double.parseDouble(input);
            if (score < 0 || score > 100) {
                throw new IllegalArgumentException("Score must be 0-100, got: " + score);
            }
            return score;
        } catch (NumberFormatException e) {
            System.out.println("Invalid number format: " + input);
            return -1;
        } catch (IllegalArgumentException e) {
            System.out.println("Invalid value: " + e.getMessage());
            return -1;
        } finally {
            System.out.println("parseScore() called with: " + input);
        }
    }

    // try-with-resources — file automatically closed:
    public static void readFile(String filename) {
        try (java.io.BufferedReader reader = new java.io.BufferedReader(
                new java.io.FileReader(filename))) {
            String line;
            while ((line = reader.readLine()) != null) {
                System.out.println(line);
            }
        } catch (java.io.FileNotFoundException e) {
            System.out.println("File not found: " + filename);
        } catch (java.io.IOException e) {
            System.out.println("Error reading file: " + e.getMessage());
        }
    }
}
