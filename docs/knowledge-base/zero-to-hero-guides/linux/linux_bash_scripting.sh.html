# ==============================================================================
# LESSON 01 — BASH VARIABLES
# ==============================================================================
#
# TEACHER SAYS:
# Variables store values. Rules:
#   - NO spaces around = when assigning: name="Alice" (not name = "Alice"!)
#   - Use $name or ${name} to READ the variable
#   - ${name} is safer in complex strings: "${name}_backup" (prevents ambiguity)
#   - UPPER_CASE for constants / config; lower_case for local script variables
#
# SPECIAL VARIABLES (built-in):
#   $0        → name of the script itself
#   $1 $2 $3  → positional arguments (what the user passes when calling the script)
#   $@        → all arguments as separate words
#   $#        → number of arguments
#   $?        → exit code of LAST command (0 = success, non-zero = error)
#   $$        → PID (process ID) of current shell
#   $!        → PID of last background job

#!/bin/bash

# Simple variable assignment:
name="Alice"
age=25
project_dir="/home/alice/ml_project"

# Read variables:
echo "Name: $name"
echo "Age: $age"
echo "Project: $project_dir"

# String concatenation:
model_name="churn"
version="v3"
full_name="${model_name}_${version}"        # churn_v3
backup_name="${full_name}_backup"           # churn_v3_backup
echo "Model: $full_name"

# Command substitution — store output of a command in a variable:
today=$(date +%Y-%m-%d)                     # e.g., 2026-04-21
num_files=$(ls *.py | wc -l)               # count .py files
current_user=$(whoami)
disk_usage=$(df -h / | tail -1 | awk '{print $5}')   # e.g., "45%"

echo "Date: $today"
echo "Python files: $num_files"
echo "User: $current_user"
echo "Disk: $disk_usage"

# Arithmetic (use $(( )) for integer math):
x=10
y=3
echo "Sum: $((x + y))"
echo "Product: $((x * y))"
echo "Integer division: $((x / y))"    # 3 (truncates)
echo "Modulo: $((x % y))"              # 1

# For floating point, use bc or Python:
result=$(echo "scale=4; 10 / 3" | bc)   # 3.3333
echo "Float division: $result"

# Check if variable is set (empty string vs not set):
echo ${name:-"Unknown"}           # Use "Unknown" if name is empty/unset
echo ${name:="Default"}           # SET name to "Default" if empty/unset
echo ${name:?"name is required"}  # Print error and exit if name is empty/unset


# ==============================================================================
# LESSON 02 — CONDITIONALS (if/elif/else)
# ==============================================================================
#
# TEACHER SAYS:
# [ ] (test command) evaluates conditions. Note: spaces inside brackets are REQUIRED!
# [[ ]] is the modern bash extended version — prefer this.
#
# STRING comparisons: == (equal), != (not equal), -z (empty), -n (not empty)
# NUMBER comparisons: -eq, -ne, -lt, -le, -gt, -ge
# FILE checks:        -f (file exists), -d (directory exists), -e (exists), -r (readable)

#!/bin/bash

gpa=3.7
name="Alice"
data_file="data/train.csv"

# String condition:
if [[ "$name" == "Alice" ]]; then
    echo "Hello Alice!"
elif [[ "$name" == "Bob" ]]; then
    echo "Hello Bob!"
else
    echo "Hello stranger!"
fi

# Number condition:
if [[ $gpa -ge 370 ]]; then          # Note: bash doesn't do floats — multiply by 100
    echo "High performer"
fi

# Better: use Python or bc for float comparisons:
if (( $(echo "$gpa >= 3.5" | bc -l) )); then
    echo "GPA is above 3.5"
fi

# File/directory checks (CRITICAL in scripts — don't assume files exist!):
if [[ -f "$data_file" ]]; then
    echo "✓ Data file found: $data_file"
    wc -l "$data_file"
else
    echo "✗ Data file NOT found: $data_file"
    echo "Please download the data first."
    exit 1           # Exit the script with error code 1
fi

if [[ -d "models/" ]]; then
    echo "✓ models/ directory exists"
else
    mkdir -p models/
    echo "✓ Created models/ directory"
fi

# Check if a command exists (before trying to run it):
if command -v jq &> /dev/null; then
    echo "✓ jq is installed"
else
    echo "✗ jq not found. Install with: sudo apt install jq"
fi

# Check exit code ($?) of previous command:
python3 -c "import pandas"
if [[ $? -eq 0 ]]; then
    echo "✓ pandas is installed"
else
    echo "✗ pandas not found. Run: pip install pandas"
fi

# ==============================================================================
# LESSON 04 — FUNCTIONS
# ==============================================================================
#
# TEACHER SAYS:
# Functions in bash work like functions in Python — they encapsulate reusable logic.
# Key rules:
#   - Define BEFORE you call them
#   - No types for parameters — just use $1, $2 inside the function
#   - Use 'local' for variables inside functions (prevents polluting global scope)
#   - Return values via echo (not 'return' — return only sets exit code 0-255)

#!/bin/bash

# Define functions:
log_info() {
    local message="$1"
    echo "[$(date +%H:%M:%S)] INFO: $message"
}

log_error() {
    local message="$1"
    echo "[$(date +%H:%M:%S)] ERROR: $message" >&2   # Print to stderr
}

check_file() {
    local filepath="$1"
    if [[ -f "$filepath" ]]; then
        log_info "File found: $filepath ($(wc -l < "$filepath") lines)"
        return 0    # success
    else
        log_error "File not found: $filepath"
        return 1    # failure
    fi
}

# Function that RETURNS a value (via echo):
get_file_size() {
    local filepath="$1"
    du -sh "$filepath" | awk '{print $1}'
}

create_experiment_dir() {
    local experiment_name="$1"
    local base_dir="${2:-./experiments}"    # Default to ./experiments if not given
    local exp_dir="${base_dir}/${experiment_name}_$(date +%Y%m%d_%H%M%S)"

    mkdir -p "$exp_dir"/{logs,models,plots}
    log_info "Created experiment directory: $exp_dir"
    echo "$exp_dir"    # Return the path so caller can use it
}

# Call functions:
log_info "Starting experiment"
check_file "data/train.csv" || exit 1   # Exit if file check fails

size=$(get_file_size "data/train.csv")
log_info "Training data size: $size"

exp_dir=$(create_experiment_dir "churn_rf_v3")
log_info "Running in: $exp_dir"

# Function to send notification (Slack webhook example):
notify_slack() {
    local message="$1"
    local webhook_url="${SLACK_WEBHOOK_URL:-}"     # Read from env var

    if [[ -z "$webhook_url" ]]; then
        log_error "SLACK_WEBHOOK_URL not set — skipping notification"
        return 1
    fi

    curl -s -X POST "$webhook_url" \
        -H 'Content-type: application/json' \
        -d "{\"text\": \"$message\"}" > /dev/null
    log_info "Slack notification sent"
}

# ==============================================================================
#
# PATH tells the shell WHERE to look for executables.

echo $PATH          # Shows all directories separated by :
                    # e.g. /usr/local/bin:/usr/bin:/usr/local/sbin:/bin

# ==============================================================================
# LESSON 06 — WRITING ROBUST SCRIPTS (Error Handling, Logging)
# ==============================================================================
#
# TEACHER SAYS:
# A production script needs 3 things that beginners skip:
#   1. Exit on error (set -e) — don't silently continue after a failure!
#   2. Validation — check arguments and environment before doing anything
#   3. Logging — write timestamped records so you can debug failures at 3 AM

#!/bin/bash
# ==== ROBUST SCRIPT TEMPLATE ====

set -euo pipefail
# set -e    → exit immediately if any command fails (CRITICAL!)
# set -u    → treat unset variables as errors (catches typos!)
# set -o pipefail → catch errors in pipes (not just the last command)
# ALWAYS put this at the top of production scripts!

# Logging function with levels:
LOG_FILE="./pipeline.log"

log() {
    local level="$1"
    local message="$2"
    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    echo "[$timestamp] [$level] $message" | tee -a "$LOG_FILE"
}

log_info()  { log "INFO " "$1"; }
log_warn()  { log "WARN " "$1"; }
log_error() { log "ERROR" "$1" >&2; }

# Trap: run cleanup even if the script fails (like Python's try/finally):
cleanup() {
    local exit_code=$?
    log_warn "Script exited with code: $exit_code"
    # Add cleanup here: remove temp files, send failure notification, etc.
    rm -f /tmp/pipeline_lock.tmp 2>/dev/null || true
}
trap cleanup EXIT   # cleanup() will run even on Ctrl+C or set -e exit!

# Validate required arguments:
usage() {
    echo "Usage: $0 <model_name> <data_path>"
    echo "  model_name: Name for this training run (e.g. churn_rf_v3)"
    echo "  data_path:  Path to training CSV file"
    exit 1
}

if [[ $# -lt 2 ]]; then
    log_error "Missing required arguments."
    usage
fi

MODEL_NAME="$1"
DATA_PATH="$2"

# Validate required environment variables:
REQUIRED_VARS=("DATABASE_URL" "MLFLOW_TRACKING_URI")
for var in "${REQUIRED_VARS[@]}"; do
    if [[ -z "${!var:-}" ]]; then
        log_error "Required environment variable not set: $var"
        log_error "Check your .env file."
        exit 1
    fi
done

# Validate required files:
if [[ ! -f "$DATA_PATH" ]]; then
    log_error "Data file not found: $DATA_PATH"
    exit 1
fi

log_info "Starting training pipeline"
log_info "  Model: $MODEL_NAME"
log_info "  Data:  $DATA_PATH ($(wc -l < "$DATA_PATH") rows)"
log_info "  User:  $USER on $HOSTNAME"

# Create output directory with timestamp:
OUTPUT_DIR="./experiments/${MODEL_NAME}_$(date +%Y%m%d_%H%M%S)"
mkdir -p "$OUTPUT_DIR"
log_info "Output directory: $OUTPUT_DIR"

# ==== END TEMPLATE ====

# LESSON 08 — CRON JOBS (Schedule Your Pipelines)
# ==============================================================================
#
# TEACHER SAYS:
# cron is Linux's job scheduler. It runs commands on a schedule automatically.
# "Nightly ETL at 2 AM" → cron.
# "Retrain model every Sunday at midnight" → cron.
# "Clean up disk space every day at 3 AM" → cron.
#
# CRON SYNTAX: 5 time fields then the command
#   ┌─── minute (0-59)
#   │ ┌─── hour (0-23)
#   │ │ ┌─── day of month (1-31)
#   │ │ │ ┌─── month (1-12)
#   │ │ │ │ ┌─── day of week (0=Sun, 1=Mon, ..., 6=Sat)
#   │ │ │ │ │
#   * * * * *  command_to_run
#
# * = every possible value (wildcard)

# EXAMPLES (memorize these patterns):
#   0 2 * * *        → 2:00 AM every day
#   0 0 * * 0        → midnight every Sunday
#   */5 * * * *      → every 5 minutes
#   0 9 * * 1-5      → 9:00 AM Monday through Friday (weekdays)
#   0 */4 * * *      → every 4 hours
#   30 23 * * *      → 11:30 PM every day

# How to edit your crontab (cron schedule):
# crontab -e         → opens your crontab in the default editor (nano usually)
# crontab -l         → lists your current cron jobs
# crontab -r         → REMOVES all your cron jobs (careful!)

# After running crontab -e, add lines like:

cat << 'CRONTAB_EXAMPLE'
# Example crontab entries:

# Run nightly ETL at 2 AM every day:
0 2 * * * /home/alice/ml_project/scripts/pipeline.sh >> /home/alice/logs/cron.log 2>&1

# Retrain model every Sunday at midnight:
0 0 * * 0 /home/alice/ml_project/scripts/train_weekly.sh >> /home/alice/logs/weekly.log 2>&1

# Monitor disk every hour and alert if >90%:
0 * * * * /home/alice/scripts/disk_check.sh

# Clear old logs every night at 3:30 AM:
30 3 * * * find /home/alice/logs -name "*.log" -mtime +30 -delete

# Run model inference every 5 minutes (real-time scoring):
*/5 * * * * python3 /home/alice/ml_project/src/score_new_customers.py >> /home/alice/logs/scoring.log 2>&1

CRONTAB_EXAMPLE

# IMPORTANT CRON RULES:
# 1. Use ABSOLUTE paths everywhere (cron doesn't know your PATH)
#    ❌  python train.py     (cron can't find 'python')
#    ✅  /home/alice/venv/bin/python /home/alice/project/train.py
#
# 2. Always redirect output to a log file (cron runs silently):
#    >> /home/alice/logs/cron.log 2>&1
#
# 3. Set environment variables in the crontab file itself:
#    SHELL=/bin/bash
#    HOME=/home/alice

# ==============================================================================
# LESSON 09 — SYSTEMD SERVICES (Always-On Production ML APIs)
# ==============================================================================
#
# TEACHER SAYS:
# Scenario: You've trained your churn model. You've built a FastAPI app to serve it.
# You start it manually: uvicorn app:app --port 8000
# It runs while you're logged in. If the server reboots? It's gone.
# If it crashes? Nobody restarts it. Not production!
#
# systemd is Linux's service manager. When you use:
#   sudo systemctl start nginx
#   sudo systemctl status mysql
# That's systemd managing those services.
#
# You can register YOUR FastAPI app as a systemd service:
#   - Starts automatically when the server boots
#   - Restarts automatically if it crashes
#   - Logs go to journald (viewable with journalctl)
#   - Runs under a specific user account (not root!)
#
# NOTE: systemd requires a real Linux machine (not WSL2 by default)

# ==== STEP-BY-STEP: DEPLOY FASTAPI AS A SERVICE ====

# 1. Your FastAPI app file: /home/ubuntu/ml_project/app.py
#    (Your working FastAPI app — see JavaSpringBootGuide.java for the concept)

# 2. Create the systemd service file:
cat << 'SERVICE_FILE'
# File: /etc/systemd/system/churn-api.service
# (Run: sudo nano /etc/systemd/system/churn-api.service to create it)

[Unit]
Description=Churn Prediction FastAPI Service
After=network.target                        # Wait for network before starting

[Service]
User=ubuntu                                 # Run as this user (NOT root!)
Group=ubuntu
WorkingDirectory=/home/ubuntu/ml_project    # cd here before running
EnvironmentFile=/home/ubuntu/ml_project/.env  # Load environment variables from .env
ExecStart=/home/ubuntu/ml_project/venv/bin/uvicorn app:app --host 0.0.0.0 --port 8000
Restart=always                              # Restart if it crashes (key for production!)
RestartSec=5                               # Wait 5 seconds before restarting
StandardOutput=journal                      # Log stdout to journald
StandardError=journal                       # Log stderr to journald

[Install]
WantedBy=multi-user.target                  # Start at normal boot
SERVICE_FILE

# 3. Enable and start the service:
# sudo systemctl daemon-reload             # Tell systemd to read new files
# sudo systemctl enable churn-api         # Auto-start on boot
# sudo systemctl start churn-api          # Start now

# 4. Management commands:

# When you deploy a service, you need to open the right ports so traffic can reach it.

# Check what services are currently listening on which ports:
ss -tlnp
# -t = TCP, -l = listening, -n = numeric ports, -p = show processes
# Output: port 8000 (your FastAPI), 5432 (postgres), 22 (ssh), 80 (nginx)...

netstat -tlnp 2>/dev/null

# Check a specific port:

# FIREWALL with ufw:
sudo ufw status             # Check current firewall rules
sudo ufw enable             # Turn on the firewall
sudo ufw disable            # Turn off (DON'T do this on a production server!)

# Allow specific ports:
sudo ufw allow 22           # SSH (ALWAYS keep this open or you'll lock yourself out!)
sudo ufw allow 80           # HTTP
sudo ufw allow 443          # HTTPS
sudo ufw allow 8000         # FastAPI development
sudo ufw allow 8888         # Jupyter Notebook

# Allow from specific IP only (more secure):
sudo ufw allow from 192.168.1.100 to any port 5432  # Only allow postgres from one IP

# Deny and delete rules:
sudo ufw deny 23            # Block telnet
