-- SETUP — Logistics/Operations Schema (real-world context)
-- ==============================================================================
--
-- TEACHER SAYS:
-- We switch from university data to a LOGISTICS dataset — this is exactly the
-- kind of data you'll see at Amazon Logistics, Uber Freight, Flexport.
-- Understand the schema before writing any query — that's what senior analysts do.

CREATE TABLE IF NOT EXISTS customers (
    customer_id   INT         PRIMARY KEY,
    name          VARCHAR(100) NOT NULL,
    city          VARCHAR(50),
    signup_date   DATE,
    tier          VARCHAR(20)  DEFAULT 'standard'   -- 'standard', 'premium', 'enterprise'
);

CREATE TABLE IF NOT EXISTS shipments (
    shipment_id      INT     PRIMARY KEY,
    customer_id      INT     REFERENCES customers(customer_id),
    origin_city      VARCHAR(50),
    dest_city        VARCHAR(50),
    weight_kg        DECIMAL(8,2),
    cost_usd         DECIMAL(10,2),
    shipped_date     DATE,
    delivered_date   DATE,
    status           VARCHAR(20)   -- 'delivered', 'in_transit', 'delayed', 'cancelled'
);

CREATE TABLE IF NOT EXISTS support_tickets (
    ticket_id      INT     PRIMARY KEY,
    customer_id    INT     REFERENCES customers(customer_id),
    shipment_id    INT     REFERENCES shipments(shipment_id),
    created_date   DATE,
    resolved_date  DATE,           -- NULL if still open
    category       VARCHAR(50),    -- 'damage', 'delay', 'lost', 'billing'
    priority       VARCHAR(20)     -- 'low', 'medium', 'high', 'critical'
);

-- ==============================================================================
-- LESSON 01 — NULL HANDLING (The #1 Source of Silent Bugs)
-- ==============================================================================
--
-- TEACHER SAYS:
-- NULLs are dangerous because they SPREAD silently:
--   NULL + 5 = NULL
--   NULL = NULL is FALSE (use IS NULL, not = NULL)
--   AVG() ignores NULLs — your average might be wrong without you knowing!
--
-- THREE FUNCTIONS to tame NULLs:
--   COALESCE(a, b, c) → return first non-NULL value
--   NULLIF(a, b)      → return NULL if a = b, else return a
--   IFNULL(a, default) → return default if a is NULL (MySQL only; use COALESCE in Postgres)
--
-- REAL SCENARIO: delivered_date is NULL for unfinished shipments.
-- If you compute transit days without handling this, you get NULL for half your data!

-- ❌ WRONG: NULL where you wanted "in transit" text
SELECT
    shipment_id,
    status,
    delivered_date - shipped_date AS transit_days    -- NULL for undelivered!
FROM shipments;

-- ✅ CORRECT: COALESCE handles missing values
SELECT
    shipment_id,
    status,
    COALESCE(delivered_date, CURRENT_DATE) AS effective_delivery,
    DATEDIFF(
        COALESCE(delivered_date, CURRENT_DATE),
        shipped_date
    ) AS transit_days,
    COALESCE(delivered_date, 'Still in transit') AS delivery_status_text
FROM shipments
ORDER BY transit_days DESC;

-- COALESCE chained: try column A, then column B, then hardcoded default
SELECT
    ticket_id,
    COALESCE(resolved_date, '9999-12-31') AS close_date,   -- unresolved → far future
    COALESCE(shipment_id, 0)              AS shipment_ref   -- no shipment → 0
FROM support_tickets;

-- NULLIF: avoid divide-by-zero (returns NULL instead of error)
SELECT
    customer_id,
    SUM(cost_usd) AS total_revenue,
    COUNT(*)      AS num_shipments,
    SUM(cost_usd) / NULLIF(COUNT(*), 0) AS avg_shipment_value  -- safe division!
FROM shipments
GROUP BY customer_id;

-- COALESCE for NULL in GROUP BY results:
SELECT
    COALESCE(dest_city, 'Unknown') AS destination,
    COUNT(*) AS shipments
FROM shipments
GROUP BY dest_city;


-- ==============================================================================
-- LESSON 02 — JOIN VARIANTS (The Complete Picture)
-- ==============================================================================
--
-- TEACHER SAYS:
-- You know INNER and LEFT. But three more JOIN types solve specific problems:
--   RIGHT JOIN   → all rows from right table (mirror of LEFT JOIN — rare in practice)
--   FULL OUTER   → all rows from BOTH tables (NULLs where no match)
--   SELF JOIN    → join a table to ITSELF — for hierarchies or comparisons
--   CROSS JOIN   → every row × every row — use carefully (can be huge!)
--
-- In real analytics:
--   LEFT JOIN for "show all X even if no Y"  ← most common
--   FULL OUTER for "show everything, flag mismatches"
--   SELF JOIN for "compare rows in the same table"

-- LEFT JOIN — all customers, even those with no shipments:
SELECT
    c.customer_id,
    c.name,
    c.tier,
    COUNT(s.shipment_id)     AS total_shipments,
    SUM(s.cost_usd)          AS total_revenue,
    MIN(s.shipped_date)      AS first_shipment,
    MAX(s.shipped_date)      AS last_shipment
FROM customers c
LEFT JOIN shipments s ON c.customer_id = s.customer_id
GROUP BY c.customer_id, c.name, c.tier
ORDER BY total_revenue DESC NULLS LAST;   -- NULL (no shipments) goes last

-- Find customers with ZERO shipments using LEFT JOIN + IS NULL:
SELECT c.name, c.tier, c.signup_date
FROM customers c
LEFT JOIN shipments s ON c.customer_id = s.customer_id
WHERE s.shipment_id IS NULL;   -- these never shipped anything!

-- SELF JOIN — compare shipment cost to the SAME customer's average:
SELECT
    s1.shipment_id,
    s1.customer_id,
    s1.cost_usd                                   AS this_shipment_cost,
    ROUND(AVG(s2.cost_usd), 2)                    AS customer_avg_cost,
    ROUND(s1.cost_usd - AVG(s2.cost_usd), 2)      AS deviation_from_avg
FROM shipments s1
JOIN shipments s2 ON s1.customer_id = s2.customer_id   -- same customer, any shipment
GROUP BY s1.shipment_id, s1.customer_id, s1.cost_usd
HAVING ABS(s1.cost_usd - AVG(s2.cost_usd)) > 500       -- unusually big/small shipments
ORDER BY deviation_from_avg DESC;

-- FULL OUTER JOIN pattern (SQLite/MySQL don't support it — simulate with UNION):
-- Shows: customers with no shipments + shipments with no customer (orphans)
SELECT c.customer_id, c.name, s.shipment_id, s.cost_usd
FROM customers c
LEFT JOIN shipments s ON c.customer_id = s.customer_id

UNION

SELECT c.customer_id, c.name, s.shipment_id, s.cost_usd
FROM shipments s
LEFT JOIN customers c ON s.customer_id = c.customer_id
WHERE c.customer_id IS NULL;


-- ==============================================================================
-- LESSON 03 — DATE & TIME FUNCTIONS (Analyst Superpower)
-- ==============================================================================
--
-- TEACHER SAYS:
-- Almost every business question involves time:
--   "How many shipments last month?" → you need date filtering
--   "What's our 7-day rolling revenue?" → you need date arithmetic
--   "Which cohort (signup month) retained best?" → you need date extraction
--
-- KEY DATE FUNCTIONS (MySQL/MariaDB syntax; Postgres uses similar but :: for casting):
--   DATEDIFF(end, start)          → days between two dates
--   DATE_ADD(date, INTERVAL n DAY) → add N days
--   DATE_FORMAT(date, '%Y-%m')    → format as YYYY-MM (for month grouping!)
--   YEAR(date), MONTH(date), DAY(date) → extract parts
--   DAYOFWEEK(date)               → 1=Sunday ... 7=Saturday
--   NOW() / CURRENT_DATE          → today's datetime / today's date
--
-- SQLite equivalents:
--   DATE('now')                   → today
--   strftime('%Y-%m', date_col)   → YYYY-MM format
--   julianday(end) - julianday(start) → days between dates

-- Transit time analysis:
SELECT
    shipment_id,
    shipped_date,
    delivered_date,
    DATEDIFF(delivered_date, shipped_date)  AS transit_days,
    CASE
        WHEN DATEDIFF(delivered_date, shipped_date) <= 3  THEN 'Express (<= 3 days)'
        WHEN DATEDIFF(delivered_date, shipped_date) <= 7  THEN 'Standard (4-7 days)'
        ELSE                                                   'Slow (> 7 days)'
    END AS speed_tier
FROM shipments
WHERE delivered_date IS NOT NULL
ORDER BY transit_days;

-- Monthly revenue trend (group by month):
SELECT
    DATE_FORMAT(shipped_date, '%Y-%m')  AS month,      -- '2023-02', '2023-04', etc.
    COUNT(*)                            AS shipments,
    ROUND(SUM(cost_usd), 2)            AS revenue,
    ROUND(AVG(cost_usd), 2)            AS avg_shipment_value
FROM shipments
GROUP BY DATE_FORMAT(shipped_date, '%Y-%m')
ORDER BY month;

-- SQLite version (same logic, different function):
-- SELECT strftime('%Y-%m', shipped_date) AS month, COUNT(*), ROUND(SUM(cost_usd),2)
-- FROM shipments GROUP BY strftime('%Y-%m', shipped_date) ORDER BY month;

-- Day-of-week analysis (are Monday shipments slower?):
SELECT
    DAYOFWEEK(shipped_date)  AS day_num,       -- 1=Sun, 2=Mon, ..., 7=Sat
    DAYNAME(shipped_date)    AS day_name,
    COUNT(*)                 AS shipments,
    ROUND(AVG(DATEDIFF(COALESCE(delivered_date, CURRENT_DATE), shipped_date)), 1) AS avg_transit_days
FROM shipments
GROUP BY DAYOFWEEK(shipped_date), DAYNAME(shipped_date)
ORDER BY day_num;

-- How long are unresolved tickets open?
SELECT
    ticket_id,
    category,
    priority,
    created_date,
    resolved_date,
    CASE
        WHEN resolved_date IS NOT NULL THEN DATEDIFF(resolved_date, created_date)
        ELSE DATEDIFF(CURRENT_DATE, created_date)
    END AS days_open,
    CASE
        WHEN resolved_date IS NOT NULL THEN 'Resolved'
        ELSE 'Open'
    END AS ticket_status
FROM support_tickets
ORDER BY days_open DESC;

-- Customers who signed up in the last 12 months:
SELECT name, signup_date, tier
FROM customers
WHERE signup_date >= DATE_SUB(CURRENT_DATE, INTERVAL 12 MONTH);

-- SQLite version:
-- WHERE signup_date >= DATE('now', '-12 months')


-- ==============================================================================
-- LESSON 04 — BUSINESS KPIs (What You'll Build in Real Jobs)
-- ==============================================================================
--
-- TEACHER SAYS:
-- Business KPIs are the reason SQL exists in analytics.
-- "Revenue", "retention", "churn", "cohort analysis" — these are SQL patterns
-- that every ops data analyst writes in their first week.
--
-- PATTERN 1: REVENUE METRICS
-- PATTERN 2: ON-TIME DELIVERY RATE
-- PATTERN 3: CUSTOMER RETENTION (month-over-month active customers)
-- PATTERN 4: COHORT ANALYSIS (do June signups retain better than May signups?)
-- PATTERN 5: CHURN DETECTION (customers who were active but went silent)

-- PATTERN 1: Revenue by customer tier
SELECT
    c.tier,
    COUNT(DISTINCT c.customer_id)   AS customers,
    COUNT(s.shipment_id)            AS total_shipments,
    ROUND(SUM(s.cost_usd), 2)       AS total_revenue,
    ROUND(AVG(s.cost_usd), 2)       AS avg_per_shipment,
    ROUND(SUM(s.cost_usd) / COUNT(DISTINCT c.customer_id), 2) AS revenue_per_customer
FROM customers c
LEFT JOIN shipments s ON c.customer_id = s.customer_id
GROUP BY c.tier
ORDER BY total_revenue DESC;

-- PATTERN 2: On-Time Delivery Rate (SLA metric)
-- Assume SLA = delivered within 7 days of shipped
SELECT
    COUNT(*)  AS total_delivered,
    SUM(CASE WHEN DATEDIFF(delivered_date, shipped_date) <= 7 THEN 1 ELSE 0 END) AS on_time,
    SUM(CASE WHEN DATEDIFF(delivered_date, shipped_date) > 7  THEN 1 ELSE 0 END) AS late,
    ROUND(
        100.0 * SUM(CASE WHEN DATEDIFF(delivered_date, shipped_date) <= 7 THEN 1 ELSE 0 END)
        / COUNT(*),
        1
    ) AS on_time_pct
FROM shipments
WHERE status = 'delivered';

-- PATTERN 3: Shipment status breakdown (operational dashboard query)
SELECT
    status,
    COUNT(*)                              AS count,
    ROUND(100.0 * COUNT(*) / SUM(COUNT(*)) OVER (), 1) AS pct_of_total,
    ROUND(SUM(cost_usd), 2)              AS revenue_at_stake
FROM shipments
GROUP BY status
ORDER BY count DESC;

-- PATTERN 4: Cohort Analysis — which signup month's customers ship most?
-- Step 1: assign each customer to their signup cohort (YYYY-MM)
WITH customer_cohorts AS (
    SELECT
        c.customer_id,
        c.name,
        DATE_FORMAT(c.signup_date, '%Y-%m')  AS signup_cohort,
        COUNT(s.shipment_id)                 AS total_shipments,
        SUM(s.cost_usd)                      AS total_spend
    FROM customers c
    LEFT JOIN shipments s ON c.customer_id = s.customer_id
    GROUP BY c.customer_id, c.name, DATE_FORMAT(c.signup_date, '%Y-%m')
),
-- Step 2: aggregate by cohort
cohort_summary AS (
    SELECT
        signup_cohort,
        COUNT(customer_id)                   AS cohort_size,
        SUM(total_shipments)                 AS total_shipments,
        ROUND(AVG(total_spend), 2)           AS avg_spend_per_customer,
        SUM(CASE WHEN total_shipments > 0 THEN 1 ELSE 0 END) AS active_customers
    FROM customer_cohorts
    GROUP BY signup_cohort
)
SELECT
    signup_cohort,
    cohort_size,
    active_customers,
    ROUND(100.0 * active_customers / cohort_size, 0) AS activation_rate_pct,
    total_shipments,
    avg_spend_per_customer
FROM cohort_summary
ORDER BY signup_cohort;

-- PATTERN 5: Identify "at-risk" customers — had shipments, but nothing recently
WITH last_shipment AS (
    SELECT
        customer_id,
        MAX(shipped_date)  AS last_ship_date
    FROM shipments
    GROUP BY customer_id
)
SELECT
    c.name,
    c.tier,
    ls.last_ship_date,
    DATEDIFF(CURRENT_DATE, ls.last_ship_date)  AS days_since_last_shipment
FROM customers c
JOIN last_shipment ls ON c.customer_id = ls.customer_id
WHERE DATEDIFF(CURRENT_DATE, ls.last_ship_date) > 180   -- 6 months inactive
ORDER BY days_since_last_shipment DESC;

-- PATTERN 6: Support ticket resolution performance by category
SELECT
    category,
    priority,
    COUNT(*)                                                          AS total_tickets,
    SUM(CASE WHEN resolved_date IS NOT NULL THEN 1 ELSE 0 END)       AS resolved,
    SUM(CASE WHEN resolved_date IS NULL     THEN 1 ELSE 0 END)       AS open,
    ROUND(AVG(CASE WHEN resolved_date IS NOT NULL
                   THEN DATEDIFF(resolved_date, created_date) END), 1) AS avg_resolution_days
FROM support_tickets
GROUP BY category, priority
ORDER BY open DESC, avg_resolution_days DESC;


-- ==============================================================================
-- LESSON 05 — SQL FEATURE ENGINEERING (Data Scientist Path)
-- ==============================================================================
--
-- TEACHER SAYS:
-- Feature engineering in SQL is done BEFORE you pull data into Python.
-- Why? Because SQL can handle billions of rows — pandas can't.
-- Create features in SQL → load clean, ready features into your model.
--
-- KEY TECHNIQUES:
--   1. Binning with CASE WHEN
--   2. Ratio features
--   3. Rolling window aggregations
--   4. Flag features (is_X columns: 0 or 1)
--   5. Lag features (previous value)

-- FEATURE ENGINEERING for a churn prediction model:
-- Each row = one customer, columns = features
WITH customer_features AS (
    SELECT
        c.customer_id,
        c.tier,

        -- Numeric features
        COUNT(s.shipment_id)                                AS total_shipments,
        COALESCE(SUM(s.cost_usd), 0)                       AS lifetime_value,
        COALESCE(AVG(s.cost_usd), 0)                       AS avg_shipment_value,
        COALESCE(AVG(s.weight_kg), 0)                      AS avg_weight_kg,
        DATEDIFF(CURRENT_DATE, MIN(s.shipped_date))        AS customer_age_days,
        DATEDIFF(CURRENT_DATE, MAX(s.shipped_date))        AS days_since_last_shipment,

        -- Ratio features
        COALESCE(
            SUM(s.cost_usd) / NULLIF(COUNT(s.shipment_id), 0),
            0
        ) AS revenue_per_shipment,

        -- Flag features (0/1 binary)
        MAX(CASE WHEN s.status = 'delayed'   THEN 1 ELSE 0 END) AS had_delay,
        MAX(CASE WHEN s.status = 'cancelled' THEN 1 ELSE 0 END) AS had_cancellation,

        -- Support ticket features
        COUNT(DISTINCT t.ticket_id)              AS total_tickets,
        SUM(CASE WHEN t.priority = 'critical'
                 THEN 1 ELSE 0 END)              AS critical_tickets

    FROM customers c
    LEFT JOIN shipments s       ON c.customer_id = s.customer_id
    LEFT JOIN support_tickets t ON c.customer_id = t.customer_id
    GROUP BY c.customer_id, c.tier
),
-- Bin continuous features into categories:
features_with_bins AS (
    SELECT
        customer_id,
        tier,
        total_shipments,
        lifetime_value,
        avg_shipment_value,
        days_since_last_shipment,
        had_delay,
        had_cancellation,
        total_tickets,
        critical_tickets,

        -- Binned features (useful for tree models and human interpretation)
        CASE
            WHEN lifetime_value = 0        THEN 'no_revenue'
            WHEN lifetime_value < 500      THEN 'low'
            WHEN lifetime_value < 2000     THEN 'medium'
            ELSE                                'high'
        END AS revenue_tier,

        CASE
            WHEN days_since_last_shipment IS NULL OR days_since_last_shipment > 365 THEN 'churned'
            WHEN days_since_last_shipment > 180                                     THEN 'at_risk'
            WHEN days_since_last_shipment > 60                                      THEN 'cooling'
            ELSE                                                                         'active'
        END AS recency_label,

        -- Engagement score (simple weighted composite feature):
        ROUND(
            (COALESCE(total_shipments, 0) * 2)
            + (CASE WHEN lifetime_value > 1000 THEN 5 ELSE 0 END)
            - (had_delay * 2)
            - (critical_tickets * 3),
            1
        ) AS engagement_score

    FROM customer_features
)
SELECT * FROM features_with_bins ORDER BY engagement_score DESC;


-- ==============================================================================
-- LESSON 06 — SAMPLING IN SQL (Data Scientist Skill)
-- ==============================================================================
--
-- TEACHER SAYS:
-- When your table has 100M rows but you only need 10K for a quick model,
-- you SAMPLE. Don't load the full table into pandas — sample in SQL first.
--
-- THREE SAMPLING METHODS:
--   1. Random sample with RAND() — simple, slightly biased for large tables
--   2. Deterministic sample with MOD — reproducible (good for train/test split)
--   3. Stratified sample — sample proportionally by group (tier, city, etc.)

-- METHOD 1: Random 50% sample (good for exploration):
SELECT *
FROM shipments
WHERE RAND() < 0.5;   -- approximately 50% of rows, randomly selected

-- METHOD 2: Deterministic sample — same result every run (use customer_id % N):
-- 20% sample: take every 5th customer
SELECT *
FROM customers
WHERE MOD(customer_id, 5) = 0;   -- always same rows — reproducible!

-- Train/Test split (80/20) using MOD:
SELECT *,
    CASE WHEN MOD(customer_id, 10) < 8 THEN 'train' ELSE 'test' END AS split
FROM customers;

-- METHOD 3: Stratified sampling — same proportion from each tier:
-- "I need 2 customers from each tier"
WITH ranked AS (
    SELECT
        customer_id, name, tier,
        ROW_NUMBER() OVER (PARTITION BY tier ORDER BY RAND()) AS rn
    FROM customers
)
SELECT customer_id, name, tier
FROM ranked
WHERE rn <= 2   -- top 2 from each tier (proportional if you use percentage)
ORDER BY tier;

-- Stratified percentage sample (10% from each tier):
WITH tier_counts AS (
    SELECT tier, COUNT(*) AS total FROM customers GROUP BY tier
),
tier_sample_sizes AS (
    SELECT tier, CEIL(total * 0.10) AS sample_size FROM tier_counts
),
ranked AS (
    SELECT
        c.customer_id, c.name, c.tier,
        ROW_NUMBER() OVER (PARTITION BY c.tier ORDER BY RAND()) AS rn,
        ts.sample_size
    FROM customers c
    JOIN tier_sample_sizes ts ON c.tier = ts.tier
)
SELECT customer_id, name, tier
FROM ranked
WHERE rn <= sample_size;


-- ==============================================================================
-- LESSON 07 — ADVANCED CASE WHEN PATTERNS
-- ==============================================================================
--
-- TEACHER SAYS:
-- CASE WHEN is SQL's if/elif/else — but it can do much more than assign labels.
-- Use it inside: SELECT, WHERE (via subquery), GROUP BY, ORDER BY, aggregates.
-- Pivot tables in SQL use CASE WHEN inside SUM() — this is huge!

-- PIVOT TABLE — show shipment count by status × tier in one query:
SELECT
    c.tier,
    COUNT(s.shipment_id)                                              AS total,
    SUM(CASE WHEN s.status = 'delivered'   THEN 1 ELSE 0 END)        AS delivered,
    SUM(CASE WHEN s.status = 'delayed'     THEN 1 ELSE 0 END)        AS delayed,
    SUM(CASE WHEN s.status = 'in_transit'  THEN 1 ELSE 0 END)        AS in_transit,
    SUM(CASE WHEN s.status = 'cancelled'   THEN 1 ELSE 0 END)        AS cancelled,
    ROUND(
        100.0 * SUM(CASE WHEN s.status = 'delivered' THEN 1 ELSE 0 END)
        / NULLIF(COUNT(s.shipment_id), 0),
        1
    ) AS delivery_success_rate
FROM customers c
LEFT JOIN shipments s ON c.customer_id = s.customer_id
GROUP BY c.tier;

-- CONDITIONAL AGGREGATION — revenue split by weight tier:
SELECT
    DATE_FORMAT(shipped_date, '%Y-%m') AS month,
    ROUND(SUM(CASE WHEN weight_kg < 50  THEN cost_usd ELSE 0 END), 2) AS light_revenue,
    ROUND(SUM(CASE WHEN weight_kg < 150 AND weight_kg >= 50  THEN cost_usd ELSE 0 END), 2) AS medium_revenue,
    ROUND(SUM(CASE WHEN weight_kg >= 150 THEN cost_usd ELSE 0 END), 2) AS heavy_revenue
FROM shipments
GROUP BY DATE_FORMAT(shipped_date, '%Y-%m')
ORDER BY month;

-- CASE WHEN in ORDER BY (custom sort):
SELECT name, tier
FROM customers
ORDER BY
    CASE tier
        WHEN 'enterprise' THEN 1
        WHEN 'premium'    THEN 2
        WHEN 'standard'   THEN 3
        ELSE                   4
