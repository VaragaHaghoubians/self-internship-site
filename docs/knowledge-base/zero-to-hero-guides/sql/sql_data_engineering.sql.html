-- ==============================================================================
-- LESSON 01 — QUERY OPTIMIZATION (Why Your Query Is Slow)
-- ==============================================================================
--
-- TEACHER SAYS:
-- A query that takes 1 second on 10K rows takes 100+ SECONDS on 100M rows
-- if you don't think about indexes and query structure.
-- Understanding HOW a database executes your query is the most valuable
-- optimization skill — everything else follows from this.
--
-- THE QUERY EXECUTION ORDER (not what you write — what the DB actually runs):
--   1. FROM / JOIN     → identify which tables, link rows
--   2. WHERE           → filter rows
--   3. GROUP BY        → group remaining rows
--   4. HAVING          → filter groups
--   5. SELECT          → compute final columns
--   6. ORDER BY        → sort output
--   7. LIMIT           → trim to N rows
--
-- WHY THIS MATTERS:
--   - You CANNOT use a SELECT alias in WHERE (WHERE runs before SELECT!)
--   - You CAN use a SELECT alias in ORDER BY (ORDER runs after SELECT)
--   - Put filters in WHERE, not HAVING — WHERE removes rows BEFORE grouping (faster!)

-- ❌ SLOW: filtering in HAVING (computes all groups first, THEN filters):
SELECT major, COUNT(*) AS n
FROM students
GROUP BY major
HAVING major = 'Data Science';   -- runs after GROUP BY — wasteful

-- ✅ FAST: filtering in WHERE (eliminates rows before any grouping):
SELECT major, COUNT(*) AS n
FROM students
WHERE major = 'Data Science'     -- runs before GROUP BY — only groups matching rows
GROUP BY major;


-- ==============================================================================
-- USING EXPLAIN — Reading the Query Execution Plan
-- ==============================================================================
--
-- TEACHER SAYS:
-- EXPLAIN shows you WHAT the database will do, without running the query.
-- It's your debugger for slow queries.
--
-- Read the output columns:
--   type          → access method: 'ALL' = full table scan (❌ slow), 'ref' = index used (✅)
--   rows          → estimated rows examined — if this is 1M+, you need an index
--   key           → which index was used (NULL = no index)
--   Extra         → 'Using filesort' = slow sort, 'Using index' = index-only scan (fast!)
--
-- HOW TO USE:

EXPLAIN SELECT * FROM students WHERE major = 'Data Science';
-- If 'key' is NULL → no index on 'major' → full table scan

-- Add an index and see the plan change:
CREATE INDEX idx_students_major ON students(major);

EXPLAIN SELECT * FROM students WHERE major = 'Data Science';
-- Now 'key' = idx_students_major → index scan → much faster!

-- EXPLAIN ANALYZE runs the query and reports actual times (MySQL 8+, PostgreSQL):
-- EXPLAIN ANALYZE SELECT * FROM students WHERE major = 'Data Science';


-- ==============================================================================
-- INDEX STRATEGY (When to Add, When to Avoid)
-- ==============================================================================
--
-- TEACHER SAYS:
-- Indexes speed up reads but SLOW DOWN writes (INSERT/UPDATE/DELETE must update the index).
-- Indexes also use disk space — sometimes a lot!
--
-- ADD AN INDEX WHEN:
--   - Column is frequently in WHERE clauses
--   - Column is frequently in JOIN conditions (foreign keys!)
--   - Column is in ORDER BY on large result sets
--   - Column has HIGH cardinality (many unique values: email, user_id)
--
-- DON'T ADD AN INDEX WHEN:
--   - Column has LOW cardinality (status: 'active'/'inactive' — only 2 values!)
--   - Table is small (< 10K rows — full scan is fine)
--   - Column is rarely queried
--   - You write more than you read (real-time log tables)
--
-- TYPES OF INDEXES:

-- 1. Single-column index (most common):
CREATE INDEX idx_shipments_customer  ON shipments(customer_id);
CREATE INDEX idx_shipments_status    ON shipments(status);
CREATE INDEX idx_shipments_date      ON shipments(shipped_date);

-- 2. Composite index (multi-column — order matters!):
--    Best for queries that filter on BOTH columns together.
--    Rule: most selective column first, then the next filter column.
CREATE INDEX idx_shipments_status_date ON shipments(status, shipped_date);
-- This helps: WHERE status = 'delayed' AND shipped_date > '2023-01-01'
-- This does NOT help: WHERE shipped_date > '2023-01-01' (status is first in index!)

-- 3. Covering index — includes all columns the query needs so it never reads the table:
CREATE INDEX idx_ship_cover ON shipments(customer_id, status, cost_usd);
-- Query: SELECT cost_usd FROM shipments WHERE customer_id = 1 AND status = 'delivered'
-- → DB reads ONLY the index, never touches the main table row!

-- Check which indexes exist on a table:
SHOW INDEX FROM shipments;   -- MySQL
-- SELECT * FROM pg_indexes WHERE tablename = 'shipments';  -- PostgreSQL

-- Drop an index you don't need:
DROP INDEX idx_shipments_status ON shipments;   -- MySQL syntax
-- DROP INDEX idx_shipments_status;              -- PostgreSQL syntax


-- ==============================================================================
-- QUERY PATTERNS TO AVOID (Anti-Patterns)
-- ==============================================================================
--
-- These patterns cause full table scans even when an index exists!

-- ❌ Functions on indexed columns break the index:
SELECT * FROM shipments WHERE YEAR(shipped_date) = 2023;   -- can't use date index!
-- ✅ Use a range instead:
SELECT * FROM shipments WHERE shipped_date BETWEEN '2023-01-01' AND '2023-12-31';

-- ❌ Leading wildcard disables LIKE index:
SELECT * FROM customers WHERE name LIKE '%Tech%';  -- full scan!
-- ✅ Trailing wildcard uses index:
SELECT * FROM customers WHERE name LIKE 'Tech%';   -- index works (if name is indexed)

-- ❌ OR across columns disables indexes (use UNION instead):
SELECT * FROM shipments WHERE status = 'delayed' OR dest_city = 'Chicago';
-- ✅ UNION: each branch uses its own index
SELECT * FROM shipments WHERE status = 'delayed'
UNION
SELECT * FROM shipments WHERE dest_city = 'Chicago';

-- ❌ SELECT * fetches all columns (even ones you don't need):
SELECT * FROM shipments WHERE customer_id = 1;
-- ✅ Select only what you need:
SELECT shipment_id, status, cost_usd FROM shipments WHERE customer_id = 1;

-- ❌ NOT IN with subquery is notoriously slow:
SELECT * FROM customers WHERE customer_id NOT IN (SELECT customer_id FROM shipments);
-- ✅ LEFT JOIN + IS NULL is faster:
SELECT c.* FROM customers c
LEFT JOIN shipments s ON c.customer_id = s.customer_id
WHERE s.shipment_id IS NULL;


-- ==============================================================================
-- LESSON 02 — ETL PATTERNS (Extract, Transform, Load)
-- ==============================================================================
--
-- TEACHER SAYS:
-- ETL is the backbone of every data pipeline:
--   EXTRACT → pull data from source (another DB, API, CSV)
--   TRANSFORM → clean, join, aggregate, compute features
--   LOAD → write results into a destination table
--
-- SQL handles all three phases in real-world data engineering.
--
-- KEY ETL PATTERNS:
--   1. CREATE TABLE AS SELECT (CTAS) — create + populate in one step
--   2. INSERT INTO ... SELECT — append new rows to existing table
--   3. UPSERT (INSERT ... ON DUPLICATE KEY UPDATE) — update if exists, insert if new
--   4. Incremental load — only process new/changed rows (critical for production!)

-- PATTERN 1: CREATE TABLE AS SELECT (CTAS)
-- Create a denormalized "wide table" for analytics / ML feature store:
CREATE TABLE IF NOT EXISTS shipment_features AS
SELECT
    s.shipment_id,
    s.customer_id,
    c.tier                                                AS customer_tier,
    c.city                                                AS customer_city,
    s.origin_city,
    s.dest_city,
    s.weight_kg,
    s.cost_usd,
    s.status,
    s.shipped_date,
    COALESCE(s.delivered_date, CURRENT_DATE)             AS effective_delivery_date,
    DATEDIFF(
        COALESCE(s.delivered_date, CURRENT_DATE),
        s.shipped_date
    )                                                     AS transit_days,
    CASE WHEN s.status = 'delivered'
              AND DATEDIFF(s.delivered_date, s.shipped_date) <= 7
         THEN 1 ELSE 0 END                               AS on_time_flag,
    s.cost_usd / NULLIF(s.weight_kg, 0)                  AS cost_per_kg,
    COUNT(t.ticket_id)                                    AS num_tickets
FROM shipments s
JOIN customers c ON s.customer_id = c.customer_id
LEFT JOIN support_tickets t ON s.shipment_id = t.shipment_id
GROUP BY
    s.shipment_id, s.customer_id, c.tier, c.city,
    s.origin_city, s.dest_city, s.weight_kg, s.cost_usd,
    s.status, s.shipped_date, s.delivered_date;

-- Verify:
SELECT * FROM shipment_features LIMIT 5;

-- PATTERN 2: INSERT INTO ... SELECT
-- Append new shipments that arrived after the last ETL run:
CREATE TABLE IF NOT EXISTS shipment_features_staging (LIKE shipment_features);

-- Incremental: only load rows not yet in the feature table
INSERT INTO shipment_features
SELECT sf.*
FROM shipment_features_staging sf
WHERE sf.shipment_id NOT IN (SELECT shipment_id FROM shipment_features);

-- PATTERN 3: UPSERT (MySQL — INSERT ... ON DUPLICATE KEY UPDATE)
-- If shipment_id already exists → update status; otherwise → insert new row:
INSERT INTO shipment_features (shipment_id, customer_id, status, cost_usd)
VALUES (999, 1, 'delivered', 950.00)
ON DUPLICATE KEY UPDATE
    status   = VALUES(status),
    cost_usd = VALUES(cost_usd);

-- PostgreSQL equivalent (INSERT ... ON CONFLICT):
-- INSERT INTO shipment_features (shipment_id, customer_id, status, cost_usd)
-- VALUES (999, 1, 'delivered', 950.00)
-- ON CONFLICT (shipment_id) DO UPDATE SET
--     status   = EXCLUDED.status,
--     cost_usd = EXCLUDED.cost_usd;

-- PATTERN 4: Incremental load using a "last_updated" watermark
-- Only process rows modified since the last run:
-- (Assumes shipments table has a 'last_modified' timestamp column)
--
-- Step 1: read the watermark:
--   SELECT MAX(last_run_time) FROM etl_audit WHERE table_name = 'shipments';
--
-- Step 2: load only new rows:
--   INSERT INTO shipment_features
--   SELECT ... FROM shipments WHERE last_modified > '2024-01-20 00:00:00';
--
-- Step 3: update watermark:
--   INSERT INTO etl_audit (table_name, last_run_time) VALUES ('shipments', NOW())
--   ON DUPLICATE KEY UPDATE last_run_time = NOW();

-- PATTERN 5: Data quality checks INSIDE SQL (before loading)
SELECT 'NULL check'      AS check_name, COUNT(*) AS failures
FROM shipment_features WHERE shipment_id IS NULL
UNION ALL
SELECT 'Negative cost',   COUNT(*) FROM shipment_features WHERE cost_usd < 0
UNION ALL
SELECT 'Future date',     COUNT(*) FROM shipment_features WHERE shipped_date > CURRENT_DATE
UNION ALL
SELECT 'Zero weight',     COUNT(*) FROM shipment_features WHERE weight_kg = 0;
-- If any failures > 0, the ETL pipeline should raise an alert!


-- ==============================================================================
-- LESSON 03 — DATA WAREHOUSING CONCEPTS
-- ==============================================================================
--
-- TEACHER SAYS:
-- Operational databases (OLTP) are optimized for individual row reads/writes.
-- Data warehouses (OLAP) are optimized for analytical queries across millions of rows.
--
-- The STAR SCHEMA is the standard warehouse design:
--
--         [dim_customer]           [dim_date]
--               |                      |
--   [dim_route]—[fact_shipments]—[dim_status]
--               |
--         [dim_product]  (in our case: cargo type)
--
-- FACT TABLE: one row per "event" (one shipment, one transaction)
--   - Has foreign keys to all dimensions
--   - Has measurable numeric columns (cost, weight, transit days)
--   - Usually very large — millions/billions of rows
--   - Never put descriptive text here — put it in dimensions!
--
-- DIMENSION TABLE: describes the WHO/WHAT/WHERE/WHEN
--   - Small, descriptive tables
--   - Queried to add context to facts
--   - Slowly Changing Dimensions (SCD): how to handle when a customer changes tier?

-- ==============================================================================
-- STAR SCHEMA IMPLEMENTATION
-- ==============================================================================

-- DIMENSION: Date (pre-populated — every possible date)
-- WHY: Avoids computing YEAR/MONTH/DAYOFWEEK at query time — they're already stored!
CREATE TABLE IF NOT EXISTS dim_date (
    date_id        INT           PRIMARY KEY,   -- YYYYMMDD format: 20230215
    date_actual    DATE          NOT NULL UNIQUE,
    year           SMALLINT      NOT NULL,
    month          TINYINT       NOT NULL,
    month_name     VARCHAR(10)   NOT NULL,
    day            TINYINT       NOT NULL,
    day_of_week    TINYINT       NOT NULL,       -- 1=Monday, 7=Sunday
    day_name       VARCHAR(10)   NOT NULL,
    week_of_year   TINYINT       NOT NULL,
    quarter        TINYINT       NOT NULL,       -- 1-4
    is_weekend     BOOLEAN       NOT NULL,
    fiscal_year    SMALLINT,                     -- if your company uses fiscal years
    fiscal_quarter TINYINT
);

-- FACT TABLE: one row per shipment, foreign keys to all dimensions
CREATE TABLE IF NOT EXISTS fact_shipments (
    shipment_sk    INT           PRIMARY KEY AUTO_INCREMENT,
    -- Foreign keys to dimensions:
    shipment_id    INT           NOT NULL,    -- source system ID
    customer_sk    INT           REFERENCES dim_customer(customer_sk),
    route_sk       INT           REFERENCES dim_route(route_sk),
    ship_date_id   INT           REFERENCES dim_date(date_id),
    deliver_date_id INT          REFERENCES dim_date(date_id),
    -- Measures (the actual facts — numeric, aggregatable):
    weight_kg      DECIMAL(8,2),
    cost_usd       DECIMAL(10,2),
    transit_days   INT,
    on_time_flag   TINYINT,                  -- 1 = on time, 0 = late
    is_cancelled   TINYINT,
    is_delayed     TINYINT,
    num_tickets    INT           DEFAULT 0
);

-- Populate fact from source tables (this is the ETL "transform + load" step):
INSERT INTO fact_shipments
    (shipment_id, customer_sk, route_sk, ship_date_id, deliver_date_id,
     weight_kg, cost_usd, transit_days, on_time_flag, is_cancelled, is_delayed, num_tickets)
SELECT
    s.shipment_id,
    dc.customer_sk,
    dr.route_sk,
    sd.date_id                   AS ship_date_id,
    dd.date_id                   AS deliver_date_id,
    s.weight_kg,
    s.cost_usd,
    DATEDIFF(COALESCE(s.delivered_date, CURRENT_DATE), s.shipped_date) AS transit_days,
    CASE WHEN s.status = 'delivered'
              AND DATEDIFF(s.delivered_date, s.shipped_date) <= 7
         THEN 1 ELSE 0 END       AS on_time_flag,
    CASE WHEN s.status = 'cancelled' THEN 1 ELSE 0 END AS is_cancelled,
    CASE WHEN s.status = 'delayed'   THEN 1 ELSE 0 END AS is_delayed,
    COUNT(t.ticket_id)           AS num_tickets
FROM shipments s
JOIN dim_customer dc    ON s.customer_id  = dc.customer_id AND dc.is_current = TRUE
JOIN dim_route    dr    ON s.origin_city  = dr.origin_city
                       AND s.dest_city    = dr.dest_city
LEFT JOIN dim_date sd   ON sd.date_actual = s.shipped_date
LEFT JOIN dim_date dd   ON dd.date_actual = s.delivered_date
LEFT JOIN support_tickets t ON s.shipment_id = t.shipment_id
GROUP BY
    s.shipment_id, dc.customer_sk, dr.route_sk, sd.date_id, dd.date_id,
    s.weight_kg, s.cost_usd, s.status, s.shipped_date, s.delivered_date;


-- ==============================================================================
-- QUERYING THE STAR SCHEMA (OLAP Queries)
-- ==============================================================================
--
-- TEACHER SAYS:
-- Once your star schema is loaded, analytics queries become fast and readable.
-- Always JOIN fact to dimensions — never JOIN dim to dim directly.
-- The fact table is the hub; dimensions are the spokes.

-- Q1: Revenue by quarter and tier:
SELECT
    d.year,
    d.quarter,
    c.tier,
    COUNT(f.shipment_id)        AS shipments,
    ROUND(SUM(f.cost_usd), 2)  AS revenue,
    ROUND(AVG(f.transit_days), 1) AS avg_transit_days
FROM fact_shipments f
JOIN dim_date     d ON f.ship_date_id = d.date_id
JOIN dim_customer c ON f.customer_sk  = c.customer_sk
GROUP BY d.year, d.quarter, c.tier
ORDER BY d.year, d.quarter, c.tier;

-- Q2: Best and worst routes by on-time rate:
SELECT
    r.origin_city,
    r.dest_city,
    r.region,
    COUNT(f.shipment_id)         AS total_shipments,
    SUM(f.on_time_flag)          AS on_time,
    ROUND(100.0 * SUM(f.on_time_flag) / NULLIF(COUNT(f.shipment_id), 0), 1) AS on_time_pct,
    ROUND(AVG(f.transit_days), 1) AS avg_transit_days
FROM fact_shipments f
JOIN dim_route r ON f.route_sk = r.route_sk
WHERE f.is_cancelled = 0
GROUP BY r.origin_city, r.dest_city, r.region
ORDER BY on_time_pct ASC;

-- Q3: Monthly revenue trend with MoM % change (window function on warehouse data):
WITH monthly_revenue AS (
    SELECT
        d.year,
        d.month,
        d.month_name,
        ROUND(SUM(f.cost_usd), 2) AS revenue
    FROM fact_shipments f
    JOIN dim_date d ON f.ship_date_id = d.date_id
    GROUP BY d.year, d.month, d.month_name
)
SELECT
    year,
    month_name,
    revenue,
    LAG(revenue) OVER (ORDER BY year, month) AS prev_month_revenue,
    ROUND(
        100.0 * (revenue - LAG(revenue) OVER (ORDER BY year, month))
        / NULLIF(LAG(revenue) OVER (ORDER BY year, month), 0),
        1
    ) AS mom_growth_pct
FROM monthly_revenue
ORDER BY year, month;


-- ==============================================================================
-- LESSON 04 — REAL-TIME vs BATCH SQL PATTERNS
-- ==============================================================================
--
-- TEACHER SAYS:
-- In production ML systems, you need two types of SQL queries:
--   BATCH → runs on a schedule (nightly, hourly), processes large historical data
--   REAL-TIME → runs per event (per API call), must return in < 100ms
--
-- Real-time queries require:
--   1. Indexed lookups (never full table scans)
--   2. Pre-computed aggregates (don't compute on the fly)
--   3. Point lookups (fetch one customer, not all customers)

-- BATCH PATTERN: Nightly job to recompute customer feature table
-- (Run every night at 2 AM via cron + dbt/Airflow)
CREATE TABLE IF NOT EXISTS customer_ml_features (
    customer_id          INT     PRIMARY KEY,
    computed_at          DATETIME DEFAULT CURRENT_TIMESTAMP,
    -- Recency / Frequency / Monetary (RFM) features:
    days_since_last_ship INT,
    total_shipments      INT,
    lifetime_value       DECIMAL(12,2),
    avg_shipment_value   DECIMAL(10,2),
    -- Risk signals:
    on_time_rate         DECIMAL(5,2),
    delay_rate           DECIMAL(5,2),
    open_tickets         INT,
    -- ML prediction output (stored back after model runs):
    churn_score          DECIMAL(5,4),
    churn_label          VARCHAR(20)
);

-- The batch ETL that feeds the above table:
INSERT INTO customer_ml_features
    (customer_id, days_since_last_ship, total_shipments, lifetime_value,
     avg_shipment_value, on_time_rate, delay_rate, open_tickets)
SELECT
    c.customer_id,
    DATEDIFF(CURRENT_DATE, MAX(s.shipped_date))         AS days_since_last_ship,
    COUNT(DISTINCT s.shipment_id)                        AS total_shipments,
    COALESCE(SUM(s.cost_usd), 0)                        AS lifetime_value,
    COALESCE(AVG(s.cost_usd), 0)                        AS avg_shipment_value,
    ROUND(
        100.0 * SUM(CASE WHEN s.status = 'delivered'
                              AND DATEDIFF(s.delivered_date, s.shipped_date) <= 7
                         THEN 1 ELSE 0 END) /
        NULLIF(SUM(CASE WHEN s.status = 'delivered' THEN 1 ELSE 0 END), 0),
        2
    )                                                    AS on_time_rate,
    ROUND(
        100.0 * SUM(CASE WHEN s.status = 'delayed' THEN 1 ELSE 0 END)
        / NULLIF(COUNT(DISTINCT s.shipment_id), 0),
        2
    )                                                    AS delay_rate,
    SUM(CASE WHEN t.resolved_date IS NULL THEN 1 ELSE 0 END) AS open_tickets
FROM customers c
LEFT JOIN shipments       s ON c.customer_id = s.customer_id
LEFT JOIN support_tickets t ON c.customer_id = t.customer_id
GROUP BY c.customer_id
ON DUPLICATE KEY UPDATE
    computed_at           = CURRENT_TIMESTAMP,
    days_since_last_ship  = VALUES(days_since_last_ship),
    total_shipments       = VALUES(total_shipments),
    lifetime_value        = VALUES(lifetime_value),
    avg_shipment_value    = VALUES(avg_shipment_value),

-- LESSON 05 — CLOUD SQL SYNTAX NOTES (BigQuery / Redshift / Snowflake)
-- ==============================================================================
--
-- TEACHER SAYS:
-- At Amazon, Uber, Flexport you won't use MySQL — you'll use:
--   Amazon Redshift  → Amazon's data warehouse (PostgreSQL-based)
--   Google BigQuery  → Google's fully managed warehouse (standard SQL)
--   Snowflake        → multi-cloud warehouse (SQL compatible)
--
-- 90% of what you've learned works the same. These are the KEY DIFFERENCES:

-- BIGQUERY:
--   - No CREATE INDEX (BQ manages this automatically with partition/cluster)
--   - PARTITION BY ingestion time or date column:
--       CREATE TABLE project.dataset.shipments
--       PARTITION BY DATE(shipped_date)  -- dramatically speeds up date-range queries!
--       CLUSTER BY customer_id, status; -- secondary ordering for point lookups
--
--   - Standard SQL: use QUALIFY instead of a subquery for window function filtering:
--       SELECT * FROM shipments
--       QUALIFY ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY shipped_date DESC) = 1;
--       -- Gets the LATEST shipment per customer — no subquery needed!
--
--   - Date functions: DATE_DIFF(end, start, DAY) instead of DATEDIFF(end, start)
--
-- REDSHIFT:
--   - DISTKEY / SORTKEY instead of traditional indexes:
--       CREATE TABLE fact_shipments (
--           shipment_id INT,
--           customer_id INT DISTKEY,    -- distribute rows by this column across nodes
--           ship_date   DATE SORTKEY    -- sort rows by this column for range scans
--       );
--   - Use COPY command for bulk loads (not INSERT INTO ... SELECT for large data)
--
-- SNOWFLAKE:
--   - Automatic clustering — no indexes needed
--   - FLATTEN for JSON arrays (BigQuery uses JSON_VALUE/UNNEST)
--   - TIME TRAVEL: query data as it existed at a past timestamp:
--       SELECT * FROM shipments AT (TIMESTAMP => '2023-06-01 00:00:00');
--
-- COMMON PATTERN THAT WORKS EVERYWHERE (BigQuery, Redshift, Snowflake, MySQL, PostgreSQL):
--   CTEs, window functions, CASE WHEN, JOIN, GROUP BY, HAVING — 100% portable!
