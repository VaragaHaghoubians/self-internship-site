-- ==============================================================================
-- DAY 01 — WHAT IS SQL? YOUR FIRST QUERY
-- ==============================================================================
--
-- TEACHER SAYS:
-- SQL (Structured Query Language) talks to databases.
-- A database holds data in TABLES (like spreadsheets but structured).
-- Every table has COLUMNS (attributes) and ROWS (records).
-- The 5 core SQL commands you MUST know:
--   SELECT  → read data
--   INSERT  → add data
--   UPDATE  → modify data
--   DELETE  → remove data
--   CREATE  → build tables/databases
--
-- A comment in SQL starts with -- (single line) or /* */ (multi-line)
--
-- YOUR FIRST QUERY — selects everything from a table:
-- SELECT * FROM table_name;
-- The * means "all columns". The ; ends the statement.
-- ==============================================================================

-- ==============================================================================
-- DAY 02 — CREATE DATABASE & TABLES
-- ==============================================================================
--
-- TEACHER SAYS:
-- Before storing data you need to CREATE a database and then tables inside it.
-- Each column needs a DATA TYPE:
--   INT          → whole numbers (age, id, count)
--   VARCHAR(n)   → text up to n characters ("Alice", "Python")
--   TEXT         → long text (descriptions, notes)
--   FLOAT/DECIMAL(p,s) → decimal numbers (gpa: DECIMAL(3,2) = 3 digits, 2 after point)
--   BOOLEAN      → TRUE/FALSE
--   DATE         → dates (2024-01-15)
--   DATETIME     → date + time
-- Constraints add rules:
--   PRIMARY KEY  → unique identifier for each row — NEVER NULL
--   NOT NULL     → this column must always have a value
--   UNIQUE       → all values in this column must be different
--   DEFAULT val  → if no value given, use this default
--   FOREIGN KEY  → links this column to a primary key in another table
--
-- EXAMPLE:

CREATE DATABASE IF NOT EXISTS university_db;
USE university_db;

-- Students table:
CREATE TABLE IF NOT EXISTS students (
    student_id   INT          PRIMARY KEY AUTO_INCREMENT,  -- auto-numbers: 1,2,3...
    first_name   VARCHAR(50)  NOT NULL,
    last_name    VARCHAR(50)  NOT NULL,
    email        VARCHAR(100) UNIQUE NOT NULL,
    age          INT          CHECK (age BETWEEN 16 AND 60),
    gpa          DECIMAL(3,2) DEFAULT 0.00,
    major        VARCHAR(100),
    enrolled_on  DATE         DEFAULT (CURRENT_DATE),
    is_active    BOOLEAN      DEFAULT TRUE
);

-- Courses table:
CREATE TABLE IF NOT EXISTS courses (
    course_id    INT         PRIMARY KEY AUTO_INCREMENT,
    course_name  VARCHAR(100) NOT NULL,
    credits      INT          NOT NULL CHECK (credits BETWEEN 1 AND 6),
    department   VARCHAR(50),
    professor    VARCHAR(100)
);

-- Enrollments — a JUNCTION TABLE linking students and courses (many-to-many):
CREATE TABLE IF NOT EXISTS enrollments (
    enrollment_id INT     PRIMARY KEY AUTO_INCREMENT,
    student_id    INT     NOT NULL,
    course_id     INT     NOT NULL,
    grade         DECIMAL(4,2),
    enrolled_date DATE    DEFAULT (CURRENT_DATE),
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    FOREIGN KEY (course_id)  REFERENCES courses(course_id)   ON DELETE CASCADE
);

-- ==============================================================================
-- DAY 04 — SELECT: Reading Data
-- ==============================================================================
--
-- TEACHER SAYS:
-- SELECT is the MOST USED SQL command — you'll write this hundreds of times daily.
-- Always think: WHAT columns do I need? WHERE should I filter?
-- SELECT * fetches ALL columns — avoid it in production (slow, wastes bandwidth)
--
-- Syntax:
--   SELECT col1, col2 FROM table WHERE condition ORDER BY col LIMIT n;
--
-- EXAMPLE:

-- All columns from all rows:
SELECT * FROM students;

-- Specific columns only:
SELECT first_name, last_name, gpa FROM students;

-- Filter with WHERE:
SELECT * FROM students WHERE major = 'Data Science';
SELECT * FROM students WHERE gpa >= 3.5;
SELECT * FROM students WHERE age BETWEEN 23 AND 26;
SELECT * FROM students WHERE major IN ('Data Science', 'Statistics');
SELECT * FROM students WHERE major LIKE '%Science%';   -- contains 'Science'
SELECT * FROM students WHERE email LIKE '%@uni.edu';   -- ends with @uni.edu

-- Multiple conditions:
SELECT * FROM students WHERE major = 'Data Science' AND gpa >= 3.7;
SELECT * FROM students WHERE age < 24 OR gpa > 3.8;

-- NULL checks:
SELECT * FROM students WHERE major IS NULL;
SELECT * FROM students WHERE major IS NOT NULL;

-- ORDER BY:
SELECT first_name, gpa FROM students ORDER BY gpa DESC;    -- highest first
SELECT first_name, age FROM students ORDER BY age ASC;     -- youngest first
SELECT * FROM students ORDER BY major ASC, gpa DESC;       -- sort by 2 columns

-- LIMIT — take only first N rows:
SELECT * FROM students ORDER BY gpa DESC LIMIT 3;          -- Top 3 students

-- DISTINCT — unique values only:
SELECT DISTINCT major FROM students;

-- ==============================================================================
-- DAY 05 — AGGREGATE FUNCTIONS & GROUP BY
-- ==============================================================================
--
-- TEACHER SAYS:
-- Aggregate functions compute a SINGLE value from MANY rows:
--   COUNT(*)        → total number of rows
--   COUNT(col)      → count non-NULL values in a column
--   SUM(col)        → total sum
--   AVG(col)        → average
--   MIN(col)        → minimum value
--   MAX(col)        → maximum value
--
-- GROUP BY → splits rows into groups BEFORE aggregating
-- HAVING   → filters AFTER grouping (WHERE filters BEFORE grouping)
-- Rule: every column in SELECT must be either in GROUP BY or inside an aggregate function!
--
-- EXAMPLE:

-- Count all students:
SELECT COUNT(*) AS total_students FROM students;

-- Average GPA:
SELECT ROUND(AVG(gpa), 2) AS avg_gpa FROM students;

-- Count students per major:
SELECT major, COUNT(*) AS student_count
FROM students
GROUP BY major
ORDER BY student_count DESC;

-- Average GPA per major:
SELECT major, ROUND(AVG(gpa), 2) AS avg_gpa, MAX(gpa) AS top_gpa
FROM students
GROUP BY major
ORDER BY avg_gpa DESC;

-- HAVING — only show majors with more than 2 students:
SELECT major, COUNT(*) AS count
FROM students
GROUP BY major
HAVING count > 2
ORDER BY count DESC;

-- Statistics per major:
SELECT
    major,
    COUNT(*)           AS num_students,
    ROUND(AVG(gpa),2)  AS avg_gpa,
    MIN(gpa)           AS min_gpa,
    MAX(gpa)           AS max_gpa
FROM students
GROUP BY major;

-- ==============================================================================
-- DAY 06 — JOINS (Combining Tables)
-- ==============================================================================
--
-- TEACHER SAYS:
-- JOINS are the heart of SQL — they combine data from multiple tables.
-- Think of JOIN like a Venn diagram:
--
--   INNER JOIN  → only rows that match in BOTH tables
--   LEFT JOIN   → ALL rows from LEFT table + matching rows from right (NULLs if no match)
--   RIGHT JOIN  → ALL rows from RIGHT table + matching rows from left
--   FULL OUTER JOIN → ALL rows from BOTH tables (NULLs where no match)
--
-- Use table aliases (short names): students s, courses c
-- Link on the FOREIGN KEY relationship
--
-- EXAMPLE:

-- INNER JOIN — students with their enrolled courses:
SELECT
    s.first_name,
    s.last_name,
    c.course_name,
    e.grade
FROM enrollments e
INNER JOIN students s ON e.student_id = s.student_id
INNER JOIN courses  c ON e.course_id  = c.course_id
ORDER BY s.last_name, c.course_name;

-- LEFT JOIN — all students, even if not enrolled in any course:
SELECT
    s.first_name,
    s.last_name,
    c.course_name,
    e.grade
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
LEFT JOIN courses     c ON e.course_id  = c.course_id
ORDER BY s.last_name;

-- Aggregate with JOIN — average grade per student:
SELECT
    s.first_name,
    s.last_name,
    s.major,
    COUNT(e.course_id)        AS courses_taken,
    ROUND(AVG(e.grade), 2)    AS avg_grade
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
GROUP BY s.student_id, s.first_name, s.last_name, s.major
ORDER BY avg_grade DESC;

-- ==============================================================================
-- DAY 07 — SUBQUERIES & CTEs
-- ==============================================================================
--
-- TEACHER SAYS:
-- A SUBQUERY is a query INSIDE another query — think nested questions.
-- A CTE (Common Table Expression) uses WITH keyword to create a named, temporary result.
-- CTEs make complex queries readable — ALWAYS prefer CTE over deeply nested subqueries.
--
-- EXAMPLE:

-- Subquery — students with GPA above average:
SELECT first_name, last_name, gpa
FROM students
WHERE gpa > (SELECT AVG(gpa) FROM students)
ORDER BY gpa DESC;

-- Subquery in FROM (inline view):
SELECT major, avg_gpa
FROM (
    SELECT major, ROUND(AVG(gpa), 2) AS avg_gpa
    FROM students
    GROUP BY major
) AS major_averages
WHERE avg_gpa >= 3.6;

-- CTE — same result but MUCH more readable:
WITH major_averages AS (
    SELECT major, ROUND(AVG(gpa), 2) AS avg_gpa
    FROM students
    GROUP BY major
)
SELECT major, avg_gpa
FROM major_averages
WHERE avg_gpa >= 3.6
ORDER BY avg_gpa DESC;

-- Multi-CTE — chain multiple CTEs:
WITH
enrollment_counts AS (
    SELECT student_id, COUNT(*) AS num_courses
    FROM enrollments
    GROUP BY student_id
),
student_grades AS (
    SELECT student_id, ROUND(AVG(grade), 2) AS avg_grade
    FROM enrollments
    GROUP BY student_id
)
SELECT
    s.first_name,
    s.last_name,
    ec.num_courses,
    sg.avg_grade
FROM students s
JOIN enrollment_counts ec ON s.student_id = ec.student_id
JOIN student_grades    sg ON s.student_id = sg.student_id
ORDER BY sg.avg_grade DESC;

-- ==============================================================================
-- DAY 08 — WINDOW FUNCTIONS
-- ==============================================================================
--
-- TEACHER SAYS:
-- Window functions perform calculations ACROSS rows related to the current row,
-- WITHOUT collapsing them into groups like GROUP BY does.
-- They are incredibly powerful for ranking, running totals, and comparisons.
-- Syntax: function() OVER (PARTITION BY col ORDER BY col)
--   PARTITION BY → groups (like GROUP BY but keeps all rows)
--   ORDER BY     → order within the window
--
-- KEY WINDOW FUNCTIONS:
--   ROW_NUMBER()  → 1,2,3... unique sequential number
--   RANK()        → 1,2,2,4 — ties get same rank, gap after
--   DENSE_RANK()  → 1,2,2,3 — ties get same rank, no gap
--   LAG(col, n)   → value of 'col' from n rows BEFORE current
--   LEAD(col, n)  → value of 'col' from n rows AFTER current
--   SUM() OVER()  → running total
--   AVG() OVER()  → moving average
--
-- EXAMPLE:

-- Rank students by GPA within each major:
SELECT
    first_name,
    last_name,
    major,
    gpa,
    RANK()       OVER (PARTITION BY major ORDER BY gpa DESC) AS rank_in_major,
    DENSE_RANK() OVER (ORDER BY gpa DESC)                    AS overall_rank,
    ROW_NUMBER() OVER (ORDER BY gpa DESC)                    AS row_num
FROM students;

-- Running total of enrollments by date:
SELECT
    enrollment_id,
    student_id,
    enrolled_date,
    COUNT(*) OVER (ORDER BY enrolled_date) AS running_count
FROM enrollments;

-- Compare each student's GPA to their major average:
SELECT
    first_name,
    major,
    gpa,
    ROUND(AVG(gpa) OVER (PARTITION BY major), 2) AS major_avg_gpa,
    ROUND(gpa - AVG(gpa) OVER (PARTITION BY major), 2) AS diff_from_avg
FROM students
ORDER BY major, gpa DESC;

-- ==============================================================================
-- DAY 09 — UPDATE & DELETE
-- ==============================================================================
--
-- TEACHER SAYS:
-- UPDATE modifies existing rows. DELETE removes rows.
-- ⚠️  ALWAYS add a WHERE clause! Without it you change/delete EVERY row!
-- ⚠️  ALWAYS run a SELECT first to preview which rows will be affected.
-- Use transactions (BEGIN/COMMIT/ROLLBACK) to safely test changes.
--
-- EXAMPLE:

-- Safe process:
-- Step 1: Preview first
SELECT * FROM students WHERE student_id = 1;

-- Step 2: Then update
UPDATE students
SET gpa = 3.85
WHERE student_id = 1;

-- Update multiple columns:
UPDATE students
SET gpa   = 3.75,
    major = 'Machine Learning'
WHERE student_id = 2;

-- Update with calculation:
UPDATE enrollments
SET grade = grade * 1.05     -- Give 5% curve to all grades
WHERE course_id = 3;

-- Delete a specific row:
DELETE FROM students WHERE student_id = 8;

-- Delete with condition:
DELETE FROM enrollments WHERE grade < 50;

-- TRANSACTION — safe update/delete with rollback option:
BEGIN;
    UPDATE students SET gpa = 0.0 WHERE major = 'Mathematics';
    -- Oops! Let's check:
    SELECT * FROM students WHERE major = 'Mathematics';
ROLLBACK;   -- Undo everything — nothing actually changed!
-- Use COMMIT instead of ROLLBACK to actually save the changes.

-- ==============================================================================
-- DAY 10 — VIEWS, INDEXES & STORED PROCEDURES
-- ==============================================================================
--
-- TEACHER SAYS:
-- VIEW → a saved SELECT query you can use like a table — great for reusable queries
-- INDEX → speeds up lookups on large tables (like a book index)
-- STORED PROCEDURE → reusable block of SQL code (like a function)
--
-- EXAMPLE:

-- VIEW — create a reusable "student report card" view:
CREATE OR REPLACE VIEW student_report AS
SELECT
    s.student_id,
    CONCAT(s.first_name, ' ', s.last_name) AS full_name,
    s.major,
    s.gpa,
    COUNT(e.course_id)      AS courses_enrolled,
    ROUND(AVG(e.grade), 2)  AS avg_grade
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
GROUP BY s.student_id, full_name, s.major, s.gpa;

-- Now use it like a table:
SELECT * FROM student_report ORDER BY avg_grade DESC;
SELECT * FROM student_report WHERE major = 'Data Science';

-- INDEX — speed up queries on frequently searched columns:
CREATE INDEX idx_students_major ON students(major);
CREATE INDEX idx_students_gpa   ON students(gpa);
CREATE INDEX idx_enroll_student ON enrollments(student_id);

-- STORED PROCEDURE — reusable SQL with parameters:
DELIMITER $$
CREATE PROCEDURE get_students_by_major(IN p_major VARCHAR(100))
BEGIN
    SELECT student_id, first_name, last_name, gpa
    FROM students
    WHERE major = p_major
    ORDER BY gpa DESC;
END$$
DELIMITER ;

-- Call the procedure:
CALL get_students_by_major('Data Science');
CALL get_students_by_major('Statistics');
