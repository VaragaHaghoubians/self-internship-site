# ==============================================================================
# DAY 01 — THE COMMAND LINE & NAVIGATION
# ==============================================================================
#
# TEACHER SAYS:
# The TERMINAL (Command Line Interface) lets you control the computer with text.
# As a Data Scientist you MUST be comfortable here — servers have no GUI!
# Linux commands follow this pattern: command [options] [arguments]
# Options usually start with - (short: -l) or -- (long: --all)
#
# KEY NAVIGATION COMMANDS:
#   pwd         → Print Working Directory (where am I?)
#   ls          → List files
#   ls -la      → List ALL files (including hidden) with details
#   cd path     → Change Directory
#   cd ..       → Go UP one directory
#   cd ~        → Go to HOME directory
#   cd -        → Go to PREVIOUS directory
#   clear       → Clear the terminal screen (or Ctrl+L)
#
# EXAMPLE:

pwd                      # Shows: /home/username
ls                       # Lists files in current directory
ls -l                    # Long format: permissions, size, date
ls -la                   # Long + hidden files (those starting with .)
ls -lh                   # Human-readable file sizes (KB, MB, GB)

cd Documents             # Move into Documents folder
cd ..                    # Go back up
cd ~/Desktop             # Go to Desktop (~ = home directory shortcut)
cd /                     # Go to ROOT of the filesystem
pwd                      # Verify where you are

# ==============================================================================
# DAY 02 — FILES & DIRECTORIES
# ==============================================================================
#
# TEACHER SAYS:
# Everything in Linux is a file — text files, programs, devices, network sockets.
# Mastering file operations is fundamental.
#
# KEY COMMANDS:
#   mkdir dir         → Make directory
#   mkdir -p a/b/c    → Make nested directories (-p creates parents)
#   touch file        → Create empty file (or update timestamp)
#   cp src dst        → Copy file
#   cp -r src dst     → Copy directory (recursive)
#   mv src dst        → Move OR rename file/directory
#   rm file           → Remove (delete) file  ⚠️ NO RECYCLE BIN!
#   rm -r dir         → Remove directory recursively  ⚠️ BE CAREFUL!
#   rm -rf dir        → Force remove  ⚠️ VERY DANGEROUS — no confirmation!
#   cat file          → Print file contents
#   less file         → View file page by page (q to quit)
#   head -n 5 file    → First 5 lines
#   tail -n 5 file    → Last 5 lines (great for log files!)
#   wc -l file        → Count lines in file
#
# EXAMPLE:

mkdir my_project                         # Create directory
mkdir -p my_project/data/raw             # Create nested directories at once
touch my_project/README.md               # Create empty file
touch my_project/main.py

cp my_project/README.md backup_README.md # Copy file
mv backup_README.md my_project/backup/   # Move file
mv my_project/main.py my_project/app.py  # Rename file (mv = move/rename)

echo "Hello, Linux!" > hello.txt         # Write text to file (OVERWRITES!)
echo "Second line"  >> hello.txt         # Append to file (>> adds, not overwrites)
cat hello.txt                            # View the file

# View a large file safely:
head -n 20 large_dataset.csv             # First 20 lines
tail -n 20 large_dataset.csv             # Last 20 lines
tail -f log.txt                          # Follow a file in real time (Ctrl+C to stop)
wc -l large_dataset.csv                  # How many rows in CSV?

# ==============================================================================
# DAY 03 — TEXT PROCESSING (grep, sed, awk)
# ==============================================================================
#
# TEACHER SAYS:
# Data Scientists spend a LOT of time processing text/CSV files in the terminal.
# These three tools are essential:
#   grep → search for patterns (filter lines)
#   sed  → stream editor — find and replace text
#   awk  → row/column manipulation — like SQL for text files
#
# EXAMPLE:

# Create a sample CSV:
cat > students.csv << 'EOF'
name,age,major,gpa
Alice,25,Data Science,3.8
Bob,23,Computer Science,3.5
Carol,26,Data Science,3.9
David,24,Statistics,3.7
Eve,22,Mathematics,3.2
Frank,28,Computer Science,2.8
EOF

# grep — search/filter:
grep "Data Science" students.csv         # Lines containing 'Data Science'
grep -i "alice" students.csv             # Case-insensitive search
grep -v "Computer" students.csv          # Lines NOT containing 'Computer'
grep -c "Science" students.csv           # COUNT of matching lines
grep -n "3.8" students.csv               # Show line NUMBERS

# Regex with grep:
grep -E "^[A-C]" students.csv            # Lines starting with A, B, or C
grep -E "3\.[89]" students.csv           # GPAs 3.8 or 3.9

# sed — find and replace:
sed 's/Data Science/DS/g' students.csv   # Replace 'Data Science' with 'DS'
# -i modifies the file IN PLACE (backup with -i.bak):
sed -i.bak 's/Computer Science/CS/g' students.csv

# awk — column operations (fields separated by comma with -F ','):
awk -F',' '{print $1}' students.csv      # Print column 1 (name)
awk -F',' '{print $1, $4}' students.csv  # Print name and GPA
awk -F',' 'NR>1 && $4 > 3.5 {print $1, $4}' students.csv  # Filter by GPA > 3.5
# NR>1 skips the header row, $4 is the 4th column (gpa)

# Count rows:
awk 'END {print NR}' students.csv        # Total rows (including header)
awk 'NR>1 {count++} END {print count}' students.csv  # Data rows only

# ==============================================================================
# DAY 04 — PERMISSIONS & FILE SECURITY
# ==============================================================================
#
# TEACHER SAYS:
# Linux is a multi-user system — permissions control who can read/write/execute.
# Every file has 3 sets of permissions: Owner | Group | Others
# Each set has 3 bits: r(read=4) w(write=2) x(execute=1)
#
# Example: -rwxr-xr--
#   - → regular file (d = directory, l = symlink)
#   rwx → OWNER can read, write, execute
#   r-x → GROUP can read and execute (not write)
#   r-- → OTHERS can only read
#
# chmod changes permissions. Numbers: r=4, w=2, x=1 (add them up)
#   chmod 755 file → rwxr-xr-x (owner full, group+others read+execute)
#   chmod 644 file → rw-r--r-- (owner read+write, others read only)
#   chmod 600 file → rw------- (only owner can read+write — for secrets!)
#
# EXAMPLE:

ls -l hello.txt                          # See current permissions
chmod 755 my_script.sh                   # Make script executable
chmod 600 .env                           # Protect secret file
chmod 644 README.md                      # Standard read-only for others

# View who owns a file:
ls -la                                   # Shows owner and group

# Run a script:
chmod +x my_script.sh                    # Add execute permission
./my_script.sh                           # Run it (the ./ means "current directory")

# Check your username:
whoami
id                                       # Shows uid, gid, groups

# ==============================================================================
# DAY 05 — PROCESSES, JOBS & SYSTEM MONITORING
# ==============================================================================
#
# TEACHER SAYS:
# As a Data Scientist you'll run long Python scripts, Jupyter servers, etc.
# You need to manage these PROCESSES — start, stop, monitor them.
#
# KEY COMMANDS:
#   ps aux         → list ALL running processes
#   top / htop     → live process monitor (like Task Manager)
#   kill PID       → send termination signal to process by ID
#   kill -9 PID    → force kill (SIGKILL) — last resort
#   Ctrl+C         → interrupt (stop) a running foreground process
#   Ctrl+Z         → pause (suspend) a process
#   bg             → resume suspended process in BACKGROUND
#   fg             → bring background process to FOREGROUND
#   nohup cmd &    → run command immune to logout, in background
#   jobs           → list background jobs
#
# DISK & MEMORY:
#   df -h          → disk usage (free/used space)
#   du -sh folder  → size of a specific folder
#   free -h        → RAM usage

ps aux                                   # All processes
ps aux | grep python                     # Find Python processes

top                                      # Live monitor (q to quit)
# htop is better: sudo apt install htop && htop

# Kill a process by name (safer than PID):
pkill python                             # Kill all python processes
killall jupyter                          # Kill all jupyter processes

# Disk and memory:
df -h                                    # Disk space
du -sh ~/Downloads                       # How big is Downloads?
free -h                                  # RAM usage

# Run Python script in background (keep running after you log out):
nohup python my_long_analysis.py > output.log 2>&1 &
# '&' = background, '>' = redirect output, '2>&1' = also capture errors
echo "Process ID: $!"                    # Print the PID of last background job

# ==============================================================================
# DAY 06 — PACKAGE MANAGEMENT & PYTHON ENVIRONMENTS
# ==============================================================================
#
# TEACHER SAYS:
# apt is Ubuntu/Debian's package manager — like an app store for the terminal.
# Always use virtual environments for Python — isolate project dependencies!
#
# WHY VIRTUAL ENVIRONMENTS?
#   Project A needs pandas 1.5. Project B needs pandas 2.0.
#   Without venv: conflict! With venv: each project has its own isolated packages.
#
# EXAMPLE:

# Update package lists (run this before installing anything!):
sudo apt update                          # Update the list of available packages
sudo apt upgrade                         # Actually upgrade installed packages

# Install software:
sudo apt install htop                    # System monitor
sudo apt install tree                    # Show directory tree
sudo apt install python3-pip             # Python package manager
sudo apt install python3-venv            # Virtual environment tool

# Python virtual environments:
python3 -m venv myenv                    # Create virtual environment called 'myenv'
source myenv/bin/activate                # ACTIVATE it (prompt changes to (myenv))
# On Windows Git Bash: source myenv/Scripts/activate

which python                             # Now points to myenv/bin/python
pip install numpy pandas matplotlib     # Installs ONLY in this environment
pip list                                 # List installed packages
pip freeze > requirements.txt           # Save dependencies to file
pip install -r requirements.txt         # Install from requirements file

deactivate                               # Leave the virtual environment

# tree command — see your project structure beautifully:
tree my_project -L 2                     # Show 2 levels deep

# ==============================================================================
# DAY 07 — PIPES, REDIRECTION & SCRIPTS
# ==============================================================================
#
# TEACHER SAYS:
# PIPES (|) chain commands — output of one becomes input of the next.
# REDIRECTION sends output to files instead of screen.
# Writing SHELL SCRIPTS (.sh files) automates repetitive tasks.
#
# REDIRECTION:
#   >    → write stdout to file (OVERWRITES)
#   >>   → append stdout to file
#   2>   → redirect errors (stderr)
#   2>&1 → redirect errors to same place as stdout
#   <    → use file as stdin
#
# EXAMPLE:

# Pipes — chain commands:
ls -la | grep ".py"                      # List only .py files
ps aux | grep python | wc -l            # Count python processes
cat students.csv | sort | uniq           # Sort and remove duplicates
cat students.csv | head -5 | tail -3    # Lines 3, 4, 5

# Redirection:
ls -la > file_list.txt                  # Save directory listing to file
echo "error!" 2> errors.log            # Save errors to log
python script.py > output.log 2>&1     # Save all output + errors

# A real shell script — automate your data science setup:
cat > setup_project.sh << 'SCRIPT'
#!/bin/bash
# Project setup script

PROJECT_NAME=$1                          # First argument from command line

if [ -z "$PROJECT_NAME" ]; then
    echo "Usage: ./setup_project.sh project_name"
    exit 1
fi

echo "Creating project: $PROJECT_NAME"

# Create directory structure:
mkdir -p "$PROJECT_NAME"/{data/{raw,processed},notebooks,src,reports,models}

# Create basic files:
touch "$PROJECT_NAME"/README.md
touch "$PROJECT_NAME"/requirements.txt
touch "$PROJECT_NAME"/.gitignore
touch "$PROJECT_NAME"/src/__init__.py
touch "$PROJECT_NAME"/src/data_loader.py
touch "$PROJECT_NAME"/src/preprocessing.py

# Write .gitignore:
cat > "$PROJECT_NAME"/.gitignore << 'EOF'
__pycache__/
*.pyc
.env
venv/
data/raw/
*.log
EOF

echo "Project '$PROJECT_NAME' created!"
tree "$PROJECT_NAME"
SCRIPT

chmod +x setup_project.sh
./setup_project.sh my_new_ds_project    # Run it!
