# ==============================================================
# WEEK 3: Python Virtual Environments
# FILE: setup_guide.py
#
# TASK: Understand and create a virtual environment for your
#       data science projects.
#
# WHY:  When you work on different projects, each project may
#       need different versions of libraries. A virtual environment
#       keeps each project's packages separate — like separate
#       toolboxes for different jobs.
#
# HOW TO RUN THIS FILE:
#   1. Open your terminal
#   2. Navigate to this folder:  cd week-03-python-environment
#   3. Run: python setup_guide.py
# ==============================================================


# ==============================================================
# PART 1: Check that Python is installed correctly
# ==============================================================

# The 'sys' module is built into Python — no install needed
import sys

# sys.version tells you which version of Python you have
print("=" * 50)
print("Python version installed:")
print(sys.version)
print("=" * 50)

# HINT: You should see something like "3.11.x" or "3.12.x"
# If you see "2.7.x", you have old Python — please reinstall from python.org


# ==============================================================
# PART 2: Check which Python is being used
# ==============================================================

# sys.executable shows the FULL PATH to Python being used
print("\nPython is located at:")
print(sys.executable)

# HINT: If you created a virtual environment and ACTIVATED it,
# this path should contain "ds-env" (the name of your environment)
# Example: C:\Users\varag\lab\projects\ds-env\Scripts\python.exe
#
# If the path does NOT contain "ds-env", it means the virtual
# environment is not activated yet. That's okay for now!


# ==============================================================
# PART 3: How to Create and Use a Virtual Environment
# ==============================================================
# 
# DO NOT RUN THESE IN PYTHON — these are terminal commands!
# Copy them into your terminal one by one.
#
# STEP 1: Navigate to your projects folder
#   cd ~/lab/projects
#
# STEP 2: Create the virtual environment
#   python -m venv ds-env
#   (This creates a folder called "ds-env" with its own Python)
#
# STEP 3: ACTIVATE the environment
#   On Linux/Mac:  source ds-env/bin/activate
#   On Windows:    ds-env\Scripts\activate
#
#   You'll know it worked when you see (ds-env) at the start of your
#   terminal prompt, like this:
#   (ds-env) varag@computer:~/lab/projects$
#
# STEP 4: Install the libraries you need
#   pip install pandas numpy matplotlib jupyter scikit-learn
#
# STEP 5: Verify installation
#   pip list
#   (You should see pandas, numpy, matplotlib, etc. in the list)
#
# STEP 6: When you're done working, deactivate the environment
#   deactivate


# ==============================================================
# PART 4: Try importing the libraries after installing them
# ==============================================================

# After you install the libraries and ACTIVATE the environment,
# come back and uncomment the lines below (remove the # symbol)
# to test that everything is installed correctly.

# HINT: To uncomment a line in VS Code, select it and press Ctrl+/

# ==============================================================
# PART 5: Understanding pip
# ==============================================================

# pip is Python's package manager — it downloads and installs libraries
# Think of it like an app store for Python code

# Useful pip commands (run in terminal, NOT in Python):
#
#   pip install pandas          → install the 'pandas' library
#   pip install pandas==2.0.0   → install a SPECIFIC version
#   pip uninstall pandas        → remove a library
#   pip list                    → see all installed libraries
#   pip freeze > requirements.txt  → save a list of ALL your libraries
#                                    (important for sharing projects!)
#   pip install -r requirements.txt  → install libraries from a saved list

print("\nRead the comments in this file carefully.")
