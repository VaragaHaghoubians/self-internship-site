# ==============================================================
# WEEK 7: APIs & Automation — Fetch Data from the Internet
# FILE: fetch_api_data.py
#
# TASK: Use Python to automatically download data from the internet
#       using a public API, then save it to a CSV file.
#
# WHY:  An API (Application Programming Interface) is a way for
#       programs to talk to each other. Instead of manually
#       downloading files, your Python script can fetch fresh data
#       automatically every day, every hour, or every minute.
#       This is how real data pipelines work in industry.
#
# INSTALL: pip install requests
#
# HOW TO RUN: python fetch_api_data.py
# ==============================================================

import requests    # For making HTTP requests (fetching web data)
import json        # For working with JSON data (built-in to Python)
import pandas as pd
from datetime import datetime   # For timestamps
import os                       # For file operations (built-in)

print("=" * 60)
print("WEEK 7: Fetching Data from APIs")
print("=" * 60)


# ==============================================================
# PART 1: What is an API and what is JSON?
# ==============================================================

# API = Application Programming Interface
# Think of it like a waiter at a restaurant:
#   YOU          = your Python script
#   RESTAURANT   = the data server (e.g., weather service)
#   WAITER       = the API
#   MENU         = available data (what you can request)
#   FOOD         = the data you get back
#
# JSON = JavaScript Object Notation
# It's the most common data format for APIs. It looks like a
# Python dictionary:
# {
#   "temperature": 22.5,
#   "city": "Istanbul",
#   "weather": "Sunny"
# }


# ==============================================================
# PART 2: Fetch a Simple Public API (no key needed)
# ==============================================================

print("\n--- PART 2: Fetching a Public API ---")

# This is a FREE public API for random jokes — no signup needed
# It's perfect for learning how APIs work
API_URL = "https://official-joke-api.appspot.com/jokes/programming/ten"

print(f"Fetching data from: {API_URL}")

# requests.get() sends an HTTP GET request to the URL
# (just like typing a URL in your browser, but from Python code)
response = requests.get(API_URL, timeout=10)

# Check if the request was successful
# Status 200 = OK, Status 404 = Not Found, Status 500 = Server Error
print(f"Response status code: {response.status_code}")

if response.status_code == 200:
    # .json() converts the JSON text into a Python list/dictionary
    jokes = response.json()
    print(f"\nReceived {len(jokes)} jokes!")
    print("\nFirst joke:")
    print("Setup:", jokes[0]["setup"])
    print("Punchline:", jokes[0]["punchline"])
else:
    print("Error: Could not fetch data. Check your internet connection.")
    jokes = []


# ==============================================================
# PART 3: A More Useful API — Open-Meteo (Free Weather Data)
# ==============================================================

print("\n--- PART 3: Fetching Real Weather Data ---")

# Open-Meteo is completely FREE and needs no API key
# We'll get hourly temperature data for Istanbul
# Latitude/Longitude for Istanbul: 41.0082, 28.9784

weather_url = (
    "https://api.open-meteo.com/v1/forecast"
    "?latitude=41.0082"
    "&longitude=28.9784"
    "&hourly=temperature_2m"
    "&forecast_days=3"
    "&timezone=Europe/Istanbul"
)

print(f"Fetching weather data for Istanbul...")

try:
    weather_response = requests.get(weather_url, timeout=10)
    weather_response.raise_for_status()   # Raises an error if status != 200
    weather_data = weather_response.json()

    # Extract the hourly data
    # The API returns a dictionary with nested data
    times = weather_data["hourly"]["time"]           # List of timestamps
    temperatures = weather_data["hourly"]["temperature_2m"]  # List of temps

    print(f"Received {len(times)} hourly temperature readings")
    print("\nFirst 5 readings:")
    for i in range(5):
        print(f"  {times[i]}: {temperatures[i]}°C")

    # ==============================================================
    # PART 4: Save the API Data to a CSV
    # ==============================================================

    print("\n--- PART 4: Saving to CSV ---")

    # Create a DataFrame from the API data
    weather_df = pd.DataFrame({
        "timestamp": times,
        "temperature_celsius": temperatures,
        "fetch_time": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    })

    # Convert the timestamp string to an actual datetime object
    weather_df["timestamp"] = pd.to_datetime(weather_df["timestamp"])

    # Save to CSV with a timestamp in the filename
    filename = f"weather_data_{datetime.now().strftime('%Y%m%d')}.csv"
    weather_df.to_csv(filename, index=False)
    print(f"Saved {len(weather_df)} rows to '{filename}'")

    # Show a quick summary
    print("\nWeather data summary:")
    print(f"  Date range: {weather_df['timestamp'].min()} to {weather_df['timestamp'].max()}")
    print(f"  Min temp: {weather_df['temperature_celsius'].min()}°C")
    print(f"  Max temp: {weather_df['temperature_celsius'].max()}°C")
    print(f"  Avg temp: {weather_df['temperature_celsius'].mean():.1f}°C")

except requests.exceptions.ConnectionError:
    print("No internet connection. Skipping weather fetch.")
except requests.exceptions.Timeout:
    print("Request timed out. The server is too slow right now.")
except Exception as e:
    print(f"An error occurred: {e}")
