# ==============================================================
# PANDAS ADVANCED — Merge, Join, Pivot, Melt, Advanced GroupBy
# FILE: pandas_advanced.py
#
# TASK: Master the pandas operations that separate a beginner
#       from a professional data analyst.
#
# WHY:  Real-world data NEVER comes in one clean table.
#       You constantly combine datasets (merge), reshape them
#       (pivot/melt), and summarize them (groupby with agg).
#       Every DS/DA interview has at least one of these.
#
# PREREQ: pandas_basics.py (you already know basic DataFrame ops)
# INSTALL: pip install pandas numpy matplotlib
# HOW TO RUN: python pandas_advanced.py
# ==============================================================

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

pd.set_option("display.max_columns", 20)
pd.set_option("display.width", 100)

print("=" * 65)
print("  PANDAS ADVANCED — Professional Data Wrangling")
print("=" * 65)


# ==============================================================
# PART 1: MERGE — Combining Tables by a Shared Key
# ==============================================================
print("\n--- PART 1: merge() — SQL-style Joins ---")

# Think of merge() exactly like SQL JOIN.
# You have two tables that share a common column (the key).
# You combine them by matching rows where the key is the same.

orders = pd.DataFrame({
    "order_id":   [1, 2, 3, 4, 5],
    "customer_id":[101, 102, 101, 103, 104],
    "amount":     [250, 180, 310, 95, 420],
    "product":    ["Laptop", "Phone", "Tablet", "Headphones", "Monitor"]
})

customers = pd.DataFrame({
    "customer_id": [101, 102, 103, 105],   # Note: 104 is MISSING, 105 has NO orders
    "name":        ["Alice", "Bob", "Carol", "Diana"],
    "city":        ["New York", "London", "Paris", "Berlin"]
})

print("\nOrders table:")
print(orders)
print("\nCustomers table:")
print(customers)

# INNER JOIN — only rows that have a match in BOTH tables
inner = pd.merge(orders, customers, on="customer_id", how="inner")
print("\nINNER JOIN (only matched rows — customer_id 104 and 105 both dropped):")
print(inner)

# LEFT JOIN — all rows from LEFT table, NaN where no match on right
left = pd.merge(orders, customers, on="customer_id", how="left")
print("\nLEFT JOIN (all orders kept — customer 104 gets NaN for name/city):")
print(left)

# RIGHT JOIN — all rows from RIGHT table
right = pd.merge(orders, customers, on="customer_id", how="right")
print("\nRIGHT JOIN (all customers kept — Diana with no orders gets NaN):")
print(right)

# OUTER JOIN — keep everything from BOTH sides
outer = pd.merge(orders, customers, on="customer_id", how="outer")
print("\nOUTER JOIN (everything from both sides):")
print(outer)

# Merge on columns with DIFFERENT NAMES:
# If the key column has different names in each table:
orders2   = pd.DataFrame({"order_id": [1, 2], "cust_id": [101, 102], "amount": [250, 180]})
customers2 = pd.DataFrame({"customer_id": [101, 102], "name": ["Alice", "Bob"]})
merged = pd.merge(orders2, customers2, left_on="cust_id", right_on="customer_id")
print("\nMerge with different key names (left_on, right_on):")
print(merged)

# ==============================================================
# PART 2: CONCAT — Stack DataFrames Vertically or Horizontally
# ==============================================================
print("\n--- PART 2: concat() — Stacking DataFrames ---")

# concat() is for STACKING rows or columns.
# Use it when you have the SAME columns across multiple DataFrames
# and want to combine them (e.g., data from different months/regions).

jan = pd.DataFrame({"month": ["Jan"]*3, "sales": [100, 200, 150], "region": ["A","B","A"]})
feb = pd.DataFrame({"month": ["Feb"]*3, "sales": [130, 220, 170], "region": ["A","B","C"]})
mar = pd.DataFrame({"month": ["Mar"]*3, "sales": [110, 240, 190], "region": ["A","B","C"]})

# Stack vertically (axis=0, default):
all_data = pd.concat([jan, feb, mar], axis=0, ignore_index=True)
print("Vertically stacked (3 months combined):")
print(all_data)

# Stack horizontally (axis=1) — adds columns side by side:
df_a = pd.DataFrame({"A": [1, 2, 3]})
df_b = pd.DataFrame({"B": [4, 5, 6]})
combined = pd.concat([df_a, df_b], axis=1)
print("\nHorizontally stacked (side by side):")
print(combined)


# ==============================================================
# PART 3: GROUPBY ADVANCED — Beyond Simple mean()
# ==============================================================
print("\n--- PART 3: Advanced GroupBy ---")

np.random.seed(42)
sales_df = pd.DataFrame({
    "region":   np.random.choice(["North", "South", "East", "West"], 100),
    "product":  np.random.choice(["A", "B", "C"], 100),
    "sales":    np.random.randint(100, 1000, 100),
    "units":    np.random.randint(1, 50, 100),
    "month":    np.random.choice(["Jan", "Feb", "Mar"], 100)
})

# Basic groupby (you already know this):
print("Mean sales by region:")
print(sales_df.groupby("region")["sales"].mean().round(2))

# agg() — multiple statistics at once:
summary = sales_df.groupby("region")["sales"].agg(
    total="sum",
    average="mean",
    count="count",
    max_sale="max",
    std_dev="std"
).round(2)
print("\nagg() — multiple stats at once:")
print(summary)

# agg() on MULTIPLE columns:
multi_agg = sales_df.groupby("region").agg({
    "sales": ["sum", "mean", "max"],
    "units": ["sum", "mean"]
}).round(2)
print("\nagg() on multiple columns:")
print(multi_agg)

# Group by MULTIPLE keys:
pivot_summary = sales_df.groupby(["region", "product"])["sales"].sum().reset_index()
print("\nGrouped by region + product (reset_index to keep flat DataFrame):")
print(pivot_summary.head(8))

# transform() — compute group stats but KEEP the original row shape
# Unlike agg() which collapses rows, transform() adds a column to the original df
sales_df["region_avg_sales"] = sales_df.groupby("region")["sales"].transform("mean")
sales_df["pct_of_region_avg"] = (sales_df["sales"] / sales_df["region_avg_sales"] * 100).round(1)
print("\ntransform() — regional avg added as a new column (keeps all 100 rows):")
print(sales_df[["region", "sales", "region_avg_sales", "pct_of_region_avg"]].head(8))

# filter() — keep only groups that meet a condition
# e.g., keep only regions where total sales > 15000
big_regions = sales_df.groupby("region").filter(lambda g: g["sales"].sum() > 15000)
print(f"\nfilter() — regions with total sales > 15000: {big_regions['region'].unique()}")


# ==============================================================
# PART 4: PIVOT TABLE — Summarize Data Like an Excel Pivot
# ==============================================================
print("\n--- PART 4: pivot_table() ---")

# A pivot table lets you restructure your data:
#   rows = one categorical variable
#   columns = another categorical variable
#   values = the number you want to summarize
#   aggfunc = how to summarize (default: mean)

pivot = pd.pivot_table(
    sales_df,
    values="sales",
    index="region",       # rows
    columns="product",    # columns
    aggfunc="sum",        # aggregate function
    fill_value=0          # replace NaN with 0
)
print("Sales by Region (rows) × Product (columns):")
print(pivot)

# Multiple aggregation functions:
pivot2 = pd.pivot_table(
    sales_df,
    values="sales",
    index="region",
    columns="month",
    aggfunc=["sum", "mean"],
    fill_value=0
).round(2)
print("\nSales by Region × Month (both sum and mean):")
print(pivot2)


# ==============================================================
# PART 5: MELT — Wide to Long Format (the opposite of pivot)
# ==============================================================
print("\n--- PART 5: melt() — Wide to Long Format ---")

# WIDE format: one row per subject, multiple columns for measurements
# LONG format: one row per measurement (better for plotting with seaborn)
# melt() converts wide → long

wide = pd.DataFrame({
    "student": ["Alice", "Bob", "Carol"],
    "math":    [88, 75, 92],
    "english": [79, 83, 88],
    "science": [91, 69, 85]
})
print("WIDE format (original):")
print(wide)

# Melt it — each score becomes its own row
long = pd.melt(
    wide,
    id_vars=["student"],          # columns to KEEP as-is (identifier columns)
    value_vars=["math", "english", "science"],  # columns to convert to rows
    var_name="subject",           # name for the new 'variable' column
    value_name="score"            # name for the new 'value' column
)
print("\nLONG format (after melt — 9 rows instead of 3):")
print(long.sort_values("student"))

# WHY long format? Seaborn and many stats tools prefer it:
# sns.boxplot(data=long, x="subject", y="score")  ← works perfectly


# ==============================================================
# PART 6: WINDOW FUNCTIONS — Rolling & Expanding
# ==============================================================
print("\n--- PART 6: Rolling & Expanding Windows ---")

# Rolling = moving average over a FIXED window of rows
# Expanding = grows to include all rows seen so far
# Critical for time series / ops data: 7-day avg, cumulative sum, etc.

daily_sales = pd.DataFrame({
    "day":   range(1, 15),
    "sales": [120, 145, 98, 167, 134, 189, 155, 142, 178, 163, 195, 148, 201, 172]
})

daily_sales["rolling_3day_avg"] = daily_sales["sales"].rolling(window=3).mean().round(1)
daily_sales["rolling_7day_avg"] = daily_sales["sales"].rolling(window=7).mean().round(1)
daily_sales["cumulative_sum"]   = daily_sales["sales"].expanding().sum()
daily_sales["pct_change"]       = daily_sales["sales"].pct_change().mul(100).round(1)

print("Rolling averages and cumulative totals:")
print(daily_sales)

# Plot rolling averages:
plt.figure(figsize=(10, 5))
plt.plot(daily_sales["day"], daily_sales["sales"],            "o-", label="Daily Sales",     alpha=0.6)
plt.plot(daily_sales["day"], daily_sales["rolling_3day_avg"], "s--", label="3-Day Avg",      color="orange")
plt.plot(daily_sales["day"], daily_sales["rolling_7day_avg"], "^-.", label="7-Day Avg",      color="red")
plt.xlabel("Day"); plt.ylabel("Sales")
plt.title("Daily Sales with Rolling Averages")
plt.legend(); plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.savefig("plot_pandas_adv_01_rolling.png", dpi=150)
plt.close()
print("\nSaved: plot_pandas_adv_01_rolling.png")


# ==============================================================
# PART 7: ADVANCED INDEXING — loc, iloc, MultiIndex
# ==============================================================
print("\n--- PART 7: Advanced Indexing ---")

# You know df["column"] and df[condition]. Here are the power moves:

df = pd.DataFrame({
    "name":  ["Alice", "Bob", "Carol", "David", "Eve"],
    "dept":  ["DS", "CS", "DS", "Stats", "CS"],
    "salary":[70000, 65000, 80000, 72000, 68000],
    "years": [3, 5, 7, 2, 4]
})

# loc — label-based: .loc[row_labels, column_labels]
print("loc — rows 1-3, only name and salary:")
print(df.loc[1:3, ["name", "salary"]])

# iloc — position-based: .iloc[row_positions, column_positions]
print("\niloc — first 2 rows, first 3 columns:")
print(df.iloc[:2, :3])

# Boolean mask with loc:
print("\nloc with condition — DS dept with salary > 70000:")
print(df.loc[(df["dept"] == "DS") & (df["salary"] > 70000)])

# Multiple conditions (use & for and, | for or, ~ for not):
print("\nCS dept OR salary > 75000:")
print(df.loc[(df["dept"] == "CS") | (df["salary"] > 75000)])

# at[] / iat[] — single cell access (faster than loc/iloc for single values):
print(f"\nAt row 2, column 'salary': {df.at[2, 'salary']}")
print(f"iat row 0, col 2: {df.iat[0, 2]}")


# ==============================================================
# PART 8: STRING OPERATIONS WITH .str
# ==============================================================
print("\n--- PART 8: String Operations (.str accessor) ---")

# The .str accessor lets you use string methods on an ENTIRE column at once.
# No need to loop!

df_text = pd.DataFrame({
    "raw_name":  ["  alice smith ", "BOB JONES  ", "carol O'Brien", "DAVID LEE"],
    "email":     ["alice@gmail.com", "bob.jones@work.org", "carol@uni.edu", "d.lee@company.co"],
    "phone":     ["555-1234", "555-5678", "invalid", "555-9012"]
})

df_text["clean_name"]   = df_text["raw_name"].str.strip().str.title()
df_text["email_domain"] = df_text["email"].str.split("@").str[1]
df_text["is_gmail"]     = df_text["email"].str.endswith("gmail.com")
df_text["valid_phone"]  = df_text["phone"].str.match(r"^\d{3}-\d{4}$")

print(df_text[["clean_name", "email_domain", "is_gmail", "valid_phone"]])
