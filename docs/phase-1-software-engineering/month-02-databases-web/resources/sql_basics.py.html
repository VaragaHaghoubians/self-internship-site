# ==============================================================
# WEEK 6 (Part 2): SQL — Query Databases Like a Pro
# FILE: sql_basics.py
#
# TASK: Learn SQL by running real queries from Python using
#       SQLite (no server needed — just a .db file).
#
# WHY:  SQL is asked in almost EVERY data job interview.
#       "Write a query to find the top 5 customers by revenue."
#       If you can't answer that, you don't get the job.
#
# HOW IT WORKS:
#   - We use sqlite3 (built into Python — no install needed)
#   - sqlite3 is perfect for learning because it's identical to
#     MySQL/PostgreSQL syntax for the 90% of queries you'll write
#
# HOW TO RUN: python sql_basics.py
# ==============================================================

import sqlite3
import pandas as pd   # pip install pandas

print("=" * 60)
print("WEEK 6 (Part 2): SQL Fundamentals")
print("=" * 60)

# Connect to a database file (creates it if it doesn't exist)
conn   = sqlite3.connect("school.db")
cursor = conn.cursor()

print("\n[DB] Connected to school.db")


# ==============================================================
# PART 1: CREATE TABLES & INSERT DATA
# ==============================================================
print("\n--- PART 1: Create Tables ---")

# Drop tables if they exist (so we start fresh each run)
cursor.executescript("""
    DROP TABLE IF EXISTS enrollments;
    DROP TABLE IF EXISTS students;
    DROP TABLE IF EXISTS courses;
""")

# CREATE TABLE — define the structure
cursor.executescript("""
    CREATE TABLE students (
        student_id   INTEGER PRIMARY KEY,
        name         TEXT    NOT NULL,
        age          INTEGER,
        major        TEXT,
        gpa          REAL
    );

    CREATE TABLE courses (
        course_id    TEXT    PRIMARY KEY,
        course_name  TEXT    NOT NULL,
        credits      INTEGER,
        professor    TEXT
    );

    CREATE TABLE enrollments (
        enrollment_id INTEGER PRIMARY KEY AUTOINCREMENT,
        student_id    INTEGER REFERENCES students(student_id),
        course_id     TEXT    REFERENCES courses(course_id),
        grade         REAL
    );
""")
conn.commit()
print("Tables created: students, courses, enrollments")

# INSERT data
cursor.executemany(
    "INSERT INTO students VALUES (?, ?, ?, ?, ?)",
    [
        (1, "Alice",   22, "Data Science",  3.9),
        (2, "Bob",     24, "CS",            3.2),
        (3, "Carol",   23, "Statistics",    3.7),
        (4, "David",   25, "Data Science",  2.8),
        (5, "Eve",     21, "Math",          3.5),
        (6, "Frank",   22, "CS",            2.5),
        (7, "Grace",   23, "Statistics",    3.8),
        (8, "Hank",    26, "Data Science",  2.3),
    ]
)
cursor.executemany(
    "INSERT INTO courses VALUES (?, ?, ?, ?)",
    [
        ("CS101", "Python Programming",  3, "Dr. Lee"),
        ("CS501", "Machine Learning",    4, "Dr. Smith"),
        ("ST301", "Statistics",          3, "Dr. Jones"),
        ("MA201", "Linear Algebra",      4, "Dr. Brown"),
        ("DS401", "Data Engineering",    3, "Dr. Chen"),
    ]
)
cursor.executemany(
    "INSERT INTO enrollments (student_id, course_id, grade) VALUES (?, ?, ?)",
    [
        (1, "CS101", 95), (1, "CS501", 88), (1, "ST301", 91),
        (2, "CS101", 78), (2, "CS501", 72),
        (3, "ST301", 85), (3, "MA201", 90), (3, "CS501", 87),
        (4, "CS101", 65), (4, "DS401", 70),
        (5, "MA201", 92), (5, "ST301", 89),
        (6, "CS101", 55), (6, "CS501", 61),
        (7, "ST301", 94), (7, "CS501", 90), (7, "DS401", 88),
        (8, "DS401", 60), (8, "CS101", 58),
    ]
)
conn.commit()
print("Sample data inserted.")


# ==============================================================
# HELPER: run a query and print it as a table using pandas
# ==============================================================
def run(label, query):
    print(f"\n--- {label} ---")
    df = pd.read_sql_query(query, conn)
    print(df.to_string(index=False))


# ==============================================================
# PART 2: SELECT — Read Data
# ==============================================================
print("\n" + "=" * 60)
print("PART 2: SELECT Queries")
print("=" * 60)

# SELECT all columns from students
run("All students",
    "SELECT * FROM students")

# SELECT specific columns
run("Names and GPAs only",
    "SELECT name, gpa FROM students")

# WHERE — filter rows
run("Students with GPA above 3.5",
    "SELECT name, major, gpa FROM students WHERE gpa > 3.5")

# Multiple conditions: AND / OR
run("Data Science students with GPA > 3.0",
    "SELECT name, gpa FROM students WHERE major = 'Data Science' AND gpa > 3.0")

# ORDER BY — sort results
run("All students sorted by GPA (highest first)",
    "SELECT name, major, gpa FROM students ORDER BY gpa DESC")

# LIMIT — return only first N rows
run("Top 3 students by GPA",
    "SELECT name, gpa FROM students ORDER BY gpa DESC LIMIT 3")

# LIKE — pattern matching
run("Students whose name starts with A, B or C",
    "SELECT name FROM students WHERE name LIKE 'A%' OR name LIKE 'B%' OR name LIKE 'C%'")


# ==============================================================
# PART 3: AGGREGATE FUNCTIONS
# ==============================================================
print("\n" + "=" * 60)
print("PART 3: Aggregate Functions (COUNT, AVG, MIN, MAX, SUM)")
print("=" * 60)

# COUNT — how many rows
run("Total number of students",
    "SELECT COUNT(*) AS total_students FROM students")

# AVG, MIN, MAX
run("GPA statistics",
    "SELECT ROUND(AVG(gpa),2) AS avg_gpa, MIN(gpa) AS min_gpa, MAX(gpa) AS max_gpa FROM students")

# GROUP BY — aggregate per category
run("Average GPA by major",
    """
    SELECT major,
           COUNT(*)           AS student_count,
           ROUND(AVG(gpa),2)  AS avg_gpa
    FROM   students
    GROUP BY major
    ORDER BY avg_gpa DESC
    """)

# HAVING — filter after grouping (like WHERE but for groups)
run("Majors with average GPA above 3.5",
    """
    SELECT major, ROUND(AVG(gpa),2) AS avg_gpa
    FROM   students
    GROUP BY major
    HAVING avg_gpa > 3.5
    """)


# ==============================================================
# PART 4: JOINS — Combine Tables
# ==============================================================
print("\n" + "=" * 60)
print("PART 4: JOINs — Combining Tables")
print("=" * 60)

# INNER JOIN — only rows that match in BOTH tables
run("Which student is enrolled in which course?",
    """
    SELECT s.name, c.course_name, e.grade
    FROM   enrollments e
    JOIN   students    s ON e.student_id = s.student_id
    JOIN   courses     c ON e.course_id  = c.course_id
    ORDER BY s.name, c.course_name
    """)

# Aggregate across joined tables
run("Average grade per course",
    """
    SELECT c.course_name,
           COUNT(e.student_id)    AS enrolled_students,
           ROUND(AVG(e.grade),1)  AS avg_grade
    FROM   courses c
    JOIN   enrollments e ON c.course_id = e.course_id
    GROUP BY c.course_name
    ORDER BY avg_grade DESC
    """)

# LEFT JOIN — all rows from left table, even with no match on right
run("All students and their enrollment count (0 if not enrolled)",
    """
    SELECT s.name,
           COUNT(e.enrollment_id) AS courses_enrolled
    FROM   students    s
    LEFT JOIN enrollments e ON s.student_id = e.student_id
    GROUP BY s.student_id
    ORDER BY courses_enrolled DESC
    """)


# ==============================================================
# PART 5: SUBQUERIES
# ==============================================================
print("\n" + "=" * 60)
print("PART 5: Subqueries")
print("=" * 60)

# A subquery is a SELECT inside another SELECT
run("Students with above-average GPA",
    """
    SELECT name, gpa
    FROM   students
    WHERE  gpa > (SELECT AVG(gpa) FROM students)
    ORDER BY gpa DESC
    """)

# IN — check if value is in a list from a subquery
run("Students enrolled in Machine Learning",
    """
    SELECT name, gpa
    FROM   students
    WHERE  student_id IN (
        SELECT student_id FROM enrollments WHERE course_id = 'CS501'
    )
    """)


# ==============================================================
# PART 6: CASE WHEN — Conditional Logic
# ==============================================================
print("\n" + "=" * 60)
print("PART 6: CASE WHEN (if/else inside SQL)")
print("=" * 60)

run("Grade letter for each enrollment",
    """
    SELECT s.name,
           c.course_name,
           e.grade,
           CASE
               WHEN e.grade >= 90 THEN 'A'
               WHEN e.grade >= 80 THEN 'B'
               WHEN e.grade >= 70 THEN 'C'
               WHEN e.grade >= 60 THEN 'D'
               ELSE                    'F'
           END AS letter_grade
    FROM   enrollments e
    JOIN   students s ON e.student_id = s.student_id
    JOIN   courses  c ON e.course_id  = c.course_id
    ORDER BY s.name
    """)


# ==============================================================
# PART 7: UPDATE & DELETE
# ==============================================================
print("\n" + "=" * 60)
print("PART 7: UPDATE and DELETE")
print("=" * 60)

# UPDATE — change existing data
cursor.execute("UPDATE students SET gpa = 3.5 WHERE name = 'Hank'")
conn.commit()
run("Hank's updated record",
    "SELECT name, gpa FROM students WHERE name = 'Hank'")

# DELETE — remove rows (ALWAYS use WHERE — or you delete everything!)
cursor.execute("DELETE FROM students WHERE gpa < 2.4 AND name != 'Hank'")
conn.commit()
run("Remaining students (those with very low GPA removed)",
    "SELECT name, gpa FROM students ORDER BY gpa DESC")
