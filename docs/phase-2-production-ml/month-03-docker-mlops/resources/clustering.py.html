# ==============================================================
# CLUSTERING — Unsupervised Learning
# FILE: clustering.py
#
# TASK: Learn to find hidden groups in data WITHOUT labels.
#       Master K-Means and DBSCAN, the two most important
#       clustering algorithms.
#
# WHY:  Clustering is "learning without a teacher."
#       You have data but NO target column (no y).
#       It answers: "Are there natural groups hiding in this data?"
#       Operations use cases:
#         → Customer segmentation (Group buyers by behavior)
#         → Anomaly detection in logistics (which shipments look weird?)
#         → Warehouse zone planning (group similar SKUs together)
#
# PREREQ: numpy_basics.py, pandas_basics.py, linear_regression.py
# INSTALL: pip install scikit-learn matplotlib seaborn pandas numpy
# HOW TO RUN: python clustering.py
# ==============================================================

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

from sklearn.cluster import KMeans, DBSCAN
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import silhouette_score
from sklearn.datasets import make_blobs, make_moons

sns.set_theme(style="whitegrid")
np.random.seed(42)

print("=" * 65)
print("  CLUSTERING — Finding Hidden Groups in Data")
print("=" * 65)


# ==============================================================
# PART 1: THE BIG IDEA — What is Clustering?
# ==============================================================
print("\n--- PART 1: What is Clustering? ---")
print("""
SUPERVISED learning:    You have X (features) AND y (labels/target).
                        The model learns to predict y.

UNSUPERVISED learning:  You ONLY have X. No labels.
                        The model finds structure in X itself.

CLUSTERING = grouping data points so that:
  - Points in the SAME cluster are SIMILAR to each other
  - Points in DIFFERENT clusters are DIFFERENT from each other

The algorithm figures out the groups — you don't tell it what they are.

Real examples:
  → Group customers by purchase frequency + basket size
  → Group delivery routes by density
  → Group machine readings to find normal vs abnormal behavior
""")


# ==============================================================
# PART 2: K-MEANS CLUSTERING
# ==============================================================
print("\n--- PART 2: K-Means Clustering ---")
print("""
How K-Means works (4 steps):
  1. You CHOOSE k (number of clusters)
  2. Place k random "centroids" (cluster centers)
  3. Assign each point to the NEAREST centroid
  4. Move each centroid to the MEAN of its assigned points
  5. Repeat steps 3-4 until centroids stop moving

K-Means limitations:
  - You must choose k in advance
  - Assumes clusters are roughly spherical and equal-sized
  - Sensitive to outliers
  - May converge to local minimum (use n_init > 1)
""")

# Generate clean cluster data to visualize the concept:
X_blob, y_true = make_blobs(n_samples=300, centers=4, cluster_std=0.8, random_state=42)

# Step 1: ALWAYS scale your features before clustering
scaler   = StandardScaler()
X_scaled = scaler.fit_transform(X_blob)

# Step 2: Fit K-Means with k=4
kmeans = KMeans(n_clusters=4, n_init=10, random_state=42)
kmeans.fit(X_scaled)

labels     = kmeans.labels_       # Cluster assignment for each point (0, 1, 2, or 3)
centroids  = kmeans.cluster_centers_

print(f"Cluster assignments (first 20): {labels[:20]}")
print(f"Inertia (within-cluster sum of squares): {kmeans.inertia_:.2f}")
print("  → Lower inertia = tighter clusters (but always lower with more k)")
print(f"Unique clusters found: {np.unique(labels)}")

# How many points in each cluster?
cluster_counts = pd.Series(labels).value_counts().sort_index()
print("\nPoints per cluster:")
print(cluster_counts)

# Visualize clusters:
fig, axes = plt.subplots(1, 2, figsize=(14, 5))

axes[0].scatter(X_scaled[:, 0], X_scaled[:, 1], c=labels, cmap="tab10", alpha=0.7, s=40)
axes[0].scatter(centroids[:, 0], centroids[:, 1], c="black", s=200, marker="X", label="Centroids")
axes[0].set_title("K-Means Result (k=4)")
axes[0].legend()

# True labels for comparison:
axes[1].scatter(X_scaled[:, 0], X_scaled[:, 1], c=y_true, cmap="tab10", alpha=0.7, s=40)
axes[1].set_title("True Labels (for comparison)")

plt.suptitle("K-Means vs True Labels", fontsize=13)
plt.tight_layout()
plt.savefig("plot_cluster_01_kmeans.png", dpi=150)
plt.close()
print("\nSaved: plot_cluster_01_kmeans.png")


# ==============================================================
# PART 3: THE ELBOW METHOD — How to Choose k
# ==============================================================
print("\n--- PART 3: Choosing k — The Elbow Method ---")
print("""
Problem: K-Means requires you to choose k BEFORE running.
Solution: Try multiple values of k and look for the "elbow."

Inertia = total within-cluster sum of squared distances.
  - Always decreases as k increases (more clusters = tighter fit)
  - The "elbow" is where decreasing k stops giving big gains
  - That elbow point is your best k

Also use: Silhouette Score (Part 4)
""")

inertias          = []
silhouette_scores = []
k_range           = range(2, 10)

for k in k_range:
    km = KMeans(n_clusters=k, n_init=10, random_state=42)
    km.fit(X_scaled)
    inertias.append(km.inertia_)
    silhouette_scores.append(silhouette_score(X_scaled, km.labels_))

fig, axes = plt.subplots(1, 2, figsize=(14, 5))

# Elbow plot:
axes[0].plot(k_range, inertias, "bo-", linewidth=2)
axes[0].axvline(x=4, color="red", linestyle="--", label="Elbow at k=4")
axes[0].set_xlabel("Number of Clusters (k)")
axes[0].set_ylabel("Inertia")
axes[0].set_title("Elbow Method — Choose k Where Inertia Bends")
axes[0].legend()

# Silhouette score plot:
axes[1].plot(k_range, silhouette_scores, "go-", linewidth=2)
axes[1].axvline(x=4, color="red", linestyle="--", label="Best k=4")
axes[1].set_xlabel("Number of Clusters (k)")
axes[1].set_ylabel("Silhouette Score")
axes[1].set_title("Silhouette Score — Higher is Better")
axes[1].legend()

plt.tight_layout()
plt.savefig("plot_cluster_02_elbow.png", dpi=150)
plt.close()
print("Saved: plot_cluster_02_elbow.png")

best_k = k_range[np.argmax(silhouette_scores)]
print(f"\nBest k by Silhouette Score: {best_k}")
print("Silhouette Score interpretation:")
print("  0.7 – 1.0  = excellent structure")
print("  0.5 – 0.7  = reasonable structure")
print("  0.25 – 0.5 = weak structure")
print("  < 0.25     = no real structure")


# ==============================================================
# PART 4: SILHOUETTE SCORE — Measure Cluster Quality
# ==============================================================
print("\n--- PART 4: Silhouette Score ---")
print("""
Silhouette score for a single point:
  s(i) = (b - a) / max(a, b)

  a = average distance to other points in the SAME cluster
  b = average distance to points in the NEAREST OTHER cluster

  s close to +1 = well clustered (far from other clusters)
  s close to  0 = on the boundary between two clusters
  s close to -1 = probably in the WRONG cluster

Overall silhouette score = mean of all individual scores.
""")

km_best = KMeans(n_clusters=4, n_init=10, random_state=42)
km_best.fit(X_scaled)

overall_score = silhouette_score(X_scaled, km_best.labels_)
print(f"Overall Silhouette Score (k=4): {overall_score:.3f}")


# ==============================================================
# PART 5: DBSCAN — Density-Based Clustering
# ==============================================================
print("\n--- PART 5: DBSCAN — When Clusters Aren't Round ---")
print("""
DBSCAN (Density-Based Spatial Clustering of Applications with Noise)

How it works:
  - A "core point" has at least min_samples points within radius eps
  - Points within eps of a core point join its cluster
  - Points NOT reachable from any core point are OUTLIERS (label = -1)

DBSCAN advantages over K-Means:
  ✅ You don't need to choose k in advance
  ✅ Can find ANY shape of clusters (not just round blobs)
  ✅ Automatically labels outliers as noise (-1)

DBSCAN disadvantages:
  ❌ Sensitive to eps and min_samples choices
  ❌ Struggles with clusters of very different densities

Parameters:
  eps         = radius of neighborhood (try 0.3 to 1.0, scaled data)
  min_samples = min points to form a core point (try 5 to 20)
""")

# Generate crescent-shaped data that K-Means fails on:
X_moon, _ = make_moons(n_samples=300, noise=0.08, random_state=42)
X_moon_scaled = StandardScaler().fit_transform(X_moon)

# K-Means fails:
km_moon   = KMeans(n_clusters=2, n_init=10, random_state=42)
labels_km = km_moon.fit_predict(X_moon_scaled)

# DBSCAN succeeds:
dbscan      = DBSCAN(eps=0.3, min_samples=5)
labels_db   = dbscan.fit_predict(X_moon_scaled)

n_clusters  = len(set(labels_db)) - (1 if -1 in labels_db else 0)
n_noise     = np.sum(labels_db == -1)

print(f"DBSCAN found: {n_clusters} clusters, {n_noise} noise points (label = -1)")

fig, axes = plt.subplots(1, 2, figsize=(14, 5))

axes[0].scatter(X_moon_scaled[:, 0], X_moon_scaled[:, 1], c=labels_km, cmap="bwr", alpha=0.7, s=40)
axes[0].set_title("K-Means FAILS on crescent shapes")

# Color noise points separately:
colors = np.where(labels_db == -1, "gray", np.where(labels_db == 0, "steelblue", "coral"))
axes[1].scatter(X_moon_scaled[:, 0], X_moon_scaled[:, 1], c=colors, alpha=0.7, s=40)
axes[1].set_title(f"DBSCAN SUCCEEDS — {n_clusters} clusters, {n_noise} noise")

plt.suptitle("K-Means vs DBSCAN on Non-Spherical Data", fontsize=13)
plt.tight_layout()
plt.savefig("plot_cluster_03_dbscan.png", dpi=150)
plt.close()
print("Saved: plot_cluster_03_dbscan.png")


# ==============================================================
# PART 6: REAL BUSINESS EXAMPLE — Customer Segmentation
# ==============================================================
print("\n--- PART 6: Real Example — Customer Segmentation ---")

np.random.seed(2026)
n = 200

customers = pd.DataFrame({
    "customer_id":      range(1, n + 1),
    "orders_per_month": np.concatenate([
        np.random.normal(1.5, 0.4, 60),   # Low-freq buyers
        np.random.normal(4.5, 0.6, 80),   # Mid-freq buyers
        np.random.normal(10, 1.2, 60),    # High-freq buyers
    ]),
    "avg_order_value":  np.concatenate([
        np.random.normal(25, 8, 60),      # Budget buyers
        np.random.normal(80, 15, 80),     # Mid-value buyers
        np.random.normal(200, 30, 60),    # Premium buyers
    ]),
})
customers["orders_per_month"] = customers["orders_per_month"].clip(0.5)
customers["avg_order_value"]  = customers["avg_order_value"].clip(10)

# Scale features:
X_cust = customers[["orders_per_month", "avg_order_value"]].values
X_cust_scaled = StandardScaler().fit_transform(X_cust)

# Run K-Means with k=3:
km_cust = KMeans(n_clusters=3, n_init=10, random_state=42)
customers["segment"] = km_cust.fit_predict(X_cust_scaled)

# Map clusters to meaningful labels based on centroid values:
centroids_orig = scaler.fit_transform(X_cust)   # Refit to get correct inverse
centroid_df    = pd.DataFrame(
    km_cust.cluster_centers_,
    columns=["orders_scaled", "value_scaled"]
)
print("Cluster centroids (scaled):")
print(centroid_df.round(2))

# Business-friendly summary:
segment_summary = customers.groupby("segment").agg(
    customers=("customer_id", "count"),
    avg_orders=("orders_per_month", "mean"),
    avg_value=("avg_order_value", "mean")
).round(2)
print("\nCustomer Segment Summary:")
print(segment_summary)

# Visualize:
plt.figure(figsize=(9, 6))
colors_seg = {0: "steelblue", 1: "coral", 2: "green"}
for seg in [0, 1, 2]:
    mask = customers["segment"] == seg
    plt.scatter(
        customers.loc[mask, "orders_per_month"],
        customers.loc[mask, "avg_order_value"],
        label=f"Segment {seg}  (n={mask.sum()})",
        color=colors_seg[seg],
        alpha=0.7, s=60
    )
plt.xlabel("Orders per Month")
plt.ylabel("Avg Order Value ($)")
plt.title("Customer Segmentation — K-Means (k=3)")
plt.legend()
plt.tight_layout()
plt.savefig("plot_cluster_04_customers.png", dpi=150)
plt.close()
print("\nSaved: plot_cluster_04_customers.png")

print("\nKey insight: segment with highest orders AND value = VIP customers.")
print("Targeted marketing by segment → higher ROI than blasting everyone.")
