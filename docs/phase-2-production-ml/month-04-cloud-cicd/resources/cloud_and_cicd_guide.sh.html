#!/bin/bash
# ==============================================================================
# PHASE 2 — MONTH 4: Cloud Computing & CI/CD
# FILE: cloud_and_cicd_guide.sh
#
# GOAL: Deploy your Dockerized ML model to a real cloud server so that
#       anyone on the internet can call your /predict API endpoint.
#
# THE PROBLEM THIS SOLVES:
#   Your model runs on YOUR laptop. Nobody else can use it.
#   You need to put it on a server that runs 24/7 on the internet.
#   And every time you push a code change, it should redeploy automatically.
#
# WHAT YOU WILL LEARN:
#   Part 1  — Cloud server basics (AWS EC2 / GCP)
#   Part 2  — SSH: how to connect to a remote Linux server
#   Part 3  — Setting up a server (Docker, nginx, firewall)
#   Part 4  — Deploying your Docker container to the cloud
#   Part 5  — GitHub Actions: automate everything on every push
#   Part 6  — Environment secrets and .env files on a server
#   Part 7  — Keeping the server running (systemd / Docker restart policies)
#
# PRE-REQUISITES:
#   - Month 3 complete (you have a working Dockerized FastAPI app)
#   - A GitHub account
#   - A free-tier cloud account (AWS Free Tier or GCP Free Tier)
#
# COST: Free tier is enough for learning. AWS EC2 t2.micro = free for 12 months.
# ==============================================================================


# ==============================================================================
# PART 1 — CLOUD SERVER BASICS
# ==============================================================================
: '
TEACHER SAYS:
A cloud server is just a Linux computer that runs 24/7 in a data centre.
You pay by the hour. AWS, GCP, and Azure are the big three.

For this program, we use AWS EC2 (Elastic Compute Cloud):
  - t2.micro: 1 vCPU, 1 GB RAM — free for 12 months on a new account
  - Ubuntu Server 22.04 LTS (Long-Term Support) is the recommended OS

HOW TO CREATE AN EC2 INSTANCE (one-time setup):
  1. Go to console.aws.amazon.com → EC2 → Launch Instance
  2. Name: "ml-server"
  3. AMI: Ubuntu Server 22.04 LTS (free tier eligible)
  4. Instance type: t2.micro (free tier)
  5. Key pair: Create new → download the .pem file → KEEP IT SAFE
  6. Security group: Add rules:
       - SSH:   port 22  from My IP (for you to connect)
       - HTTP:  port 80  from Anywhere (for the world to reach your API)
       - Custom: port 8000 from Anywhere (FastAPI default port)
  7. Launch!

IMPORTANT: Note your instance'"'"'s "Public IPv4 address" — you need it to connect.
'

echo "=== PART 1: Cloud server is running. Now connect to it. ==="


# ==============================================================================
# PART 2 — SSH: CONNECTING TO YOUR REMOTE SERVER
# ==============================================================================
: '
TEACHER SAYS:
SSH (Secure Shell) is how you control a remote Linux server from your terminal.
It is exactly like opening a terminal on that machine — but over the internet.

The .pem file you downloaded is your "key" — like a password but more secure.
'

# On your LOCAL machine (not the server):

# Fix permissions on the key file (SSH refuses to use keys with open permissions):
# chmod 400 ~/Downloads/your-key.pem

# Connect to the server:
# ssh -i ~/Downloads/your-key.pem ubuntu@YOUR_PUBLIC_IP

# Example:
# ssh -i ~/Downloads/ml-server-key.pem ubuntu@54.123.45.67

echo "Connecting to server..."
echo "Command: ssh -i your-key.pem ubuntu@YOUR_SERVER_IP"
echo ""
echo "Once connected, your terminal prompt changes to:"
echo "  ubuntu@ip-172-31-xx-xx:~\$"
echo "You are now INSIDE the cloud server."


# ==============================================================================
# PART 3 — SETTING UP THE SERVER
# ==============================================================================
: '
TEACHER SAYS:
Run these commands ONCE after your first SSH login to a fresh server.
This installs everything you need to run your Docker container.
'

# --- Run these on the SERVER after SSH login ---

# Update package list and install Docker:
# sudo apt update && sudo apt upgrade -y
# sudo apt install -y docker.io docker-compose git curl
# sudo systemctl enable docker
# sudo systemctl start docker

# Add your user to the docker group (so you don'"'"'t need sudo every time):
# sudo usermod -aG docker $USER
# newgrp docker   # Apply group change immediately

# Verify Docker works:
# docker run hello-world

# Install GitHub CLI (optional but useful):
# curl -fsSL https://cli.github.com/packages/githubcli-archive-keyring.gpg | sudo dd of=/usr/share/keyrings/githubcli-archive-keyring.gpg
# echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/githubcli-archive-keyring.gpg] https://cli.github.com/packages stable main" | sudo tee /etc/apt/sources.list.d/github-cli.list > /dev/null
# sudo apt update && sudo apt install gh -y

echo "Server setup complete."


# ==============================================================================
# PART 4 — DEPLOYING YOUR DOCKER CONTAINER
# ==============================================================================
: '
TEACHER SAYS:
You have two options for getting your Docker image onto the server:
  A) Push to Docker Hub (public registry), pull on server
  B) Use GitHub Actions to build and push automatically (Part 5)

Option A is manual but good to understand first.
'

# --- On your LOCAL machine ---

# Log in to Docker Hub (create a free account at hub.docker.com):
# docker login

# Tag your image with your Docker Hub username:
# docker tag my-ml-app yourusername/my-ml-app:latest

# Push to Docker Hub:
# docker push yourusername/my-ml-app:latest

# --- On the SERVER (via SSH) ---

# Pull the image:
# docker pull yourusername/my-ml-app:latest

# Run the container:
# docker run -d \                          # -d = detached (background)
#   --name ml-api \
#   --restart unless-stopped \             # Auto-restart if server reboots
#   -p 8000:8000 \                         # Map port 8000 host → container
#   -e OPENAI_API_KEY="sk-..." \           # Pass environment variables
#   yourusername/my-ml-app:latest

# Test it works:
# curl http://localhost:8000/health
# curl http://YOUR_PUBLIC_IP:8000/health   # From your laptop!

echo "Container deployed. Test: curl http://YOUR_SERVER_IP:8000/health"


# ==============================================================================
# PART 5 — GITHUB ACTIONS: CI/CD PIPELINE
# ==============================================================================
: '
TEACHER SAYS:
CI/CD = Continuous Integration / Continuous Deployment.
Every time you push to GitHub, GitHub automatically:
  1. Runs your tests (CI)
  2. Builds the Docker image
  3. Pushes it to Docker Hub
  4. SSHes into your server and pulls + restarts the container (CD)

This means: git push → code is live on the internet. That'"'"'s professional deployment.

Create this file in your repo: .github/workflows/deploy.yml
'

cat << 'EOF'
# .github/workflows/deploy.yml
name: Build and Deploy

on:
  push:
    branches: [ main ]   # Trigger only on pushes to main

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Log in to Docker Hub
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKERHUB_USERNAME }}
          password: ${{ secrets.DOCKERHUB_TOKEN }}

      - name: Build and push Docker image
        uses: docker/build-push-action@v5
        with:
          push: true
          tags: ${{ secrets.DOCKERHUB_USERNAME }}/my-ml-app:latest

      - name: Deploy to server via SSH
        uses: appleboy/ssh-action@v1.0.0
        with:
          host: ${{ secrets.SERVER_IP }}
          username: ubuntu
          key: ${{ secrets.SERVER_SSH_KEY }}    # Contents of your .pem file
          script: |
            docker pull ${{ secrets.DOCKERHUB_USERNAME }}/my-ml-app:latest
            docker stop ml-api || true
            docker rm ml-api || true
            docker run -d \
              --name ml-api \
              --restart unless-stopped \
              -p 8000:8000 \
              -e OPENAI_API_KEY="${{ secrets.OPENAI_API_KEY }}" \
              ${{ secrets.DOCKERHUB_USERNAME }}/my-ml-app:latest
EOF

: '
ADD SECRETS in GitHub → Your Repo → Settings → Secrets and Variables → Actions:
  DOCKERHUB_USERNAME   → your Docker Hub username
  DOCKERHUB_TOKEN      → Docker Hub access token (not your password)
  SERVER_IP            → your EC2 public IP
  SERVER_SSH_KEY       → the contents of your .pem key file (cat your-key.pem)
  OPENAI_API_KEY       → (optional) your OpenAI key if your app needs it
'

echo "CI/CD pipeline configured."


# ==============================================================================
# PART 6 — ENVIRONMENT VARIABLES ON THE SERVER
# ==============================================================================
: '
TEACHER SAYS:
NEVER hardcode secrets (API keys, database passwords) in your code or Dockerfile.
Use environment variables. There are two patterns:

Pattern A: Pass via docker run -e (shown in Part 4 and 5 — simple)
Pattern B: Use a .env file on the server (better for many variables)
'

# Create a .env file on the server (do this manually once via SSH):
# cat > /home/ubuntu/ml-api.env << 'ENVEOF'
# OPENAI_API_KEY=sk-...
# DATABASE_URL=postgresql://user:password@localhost:5432/mydb
# SECRET_KEY=some-random-string-here
# ENVEOF

# Then run Docker with the env file:
# docker run -d \
#   --name ml-api \
#   --restart unless-stopped \
#   -p 8000:8000 \
#   --env-file /home/ubuntu/ml-api.env \
#   yourusername/my-ml-app:latest

echo "Environment variables configured."


# ==============================================================================
# PART 7 — MONITORING AND MANAGEMENT COMMANDS
# ==============================================================================
: '
TEACHER SAYS:
Useful commands for managing your running container on the server.
Run all of these on the server via SSH.
'

# View running containers:
# docker ps

# View container logs (real-time):
# docker logs -f ml-api

# View last 50 log lines:
# docker logs --tail 50 ml-api

# Restart the container:
# docker restart ml-api

# Check server resource usage (CPU, RAM, disk):
# htop                # Interactive process viewer (install: sudo apt install htop)
# df -h               # Disk usage
# free -h             # RAM usage

# Check which ports are open:
# sudo ss -tlnp

# View nginx logs (if you use nginx as a reverse proxy):
# sudo tail -f /var/log/nginx/access.log

echo ""
echo "=== MONTH 4 SUMMARY ==="
echo "You can now:"
echo "  ✓ Launch and connect to a cloud Linux server"
echo "  ✓ Install Docker on a server"
echo "  ✓ Deploy a Docker container and expose it to the internet"
echo "  ✓ Set up GitHub Actions to auto-deploy on every git push"
echo "  ✓ Manage secrets safely with environment variables"
echo "  ✓ Monitor and manage running containers"
echo ""
echo "NEXT → Phase 3: Django — build a full web app around your model."
