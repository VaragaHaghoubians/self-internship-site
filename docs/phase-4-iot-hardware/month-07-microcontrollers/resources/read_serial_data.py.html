#
# HARDWARE NEEDED: ELEGOO UNO R3 Kit + DHT11 sensor (in the kit)
#
# INSTALL: pip install pyserial pandas
#
# HOW THIS WORKS:
#   1. The Arduino collects temperature readings every 2 seconds
#   2. It sends those readings through the USB cable as text
#   3. This Python script reads that text, parses the numbers,
#      adds a timestamp, and saves it to a CSV file
#
# HOW TO RUN:
#   1. First, upload the Arduino code (see ARDUINO CODE section below)
#   2. Note which USB port your Arduino is on (e.g., COM3 on Windows,
#      /dev/ttyUSB0 on Linux)
#   3. Update the PORT variable below
#   4. Run: python read_serial_data.py
# ==============================================================


# ==============================================================
# ARDUINO CODE (upload this to your Arduino FIRST)
# ==============================================================

# Open the Arduino IDE, create a new sketch, and paste this code:

ARDUINO_CODE = """
// ============================================================
// Arduino Code: Read DHT11 and send data over USB Serial
// Upload this to your ELEGOO Arduino UNO board
// ============================================================

#include <DHT.h>            // You need to install the DHT library
                            // In Arduino IDE: Tools > Manage Libraries
                            // Search "DHT sensor library" by Adafruit, install it

#define DHTPIN 2            // DHT11 data pin connected to Arduino pin 2
#define DHTTYPE DHT11       // We're using the DHT11 model

DHT dht(DHTPIN, DHTTYPE);  // Create DHT sensor object

void setup() {
    Serial.begin(9600);     // Start communication at 9600 baud rate
    dht.begin();            // Initialize the sensor
    Serial.println("Arduino ready. Starting temperature readings...");
}

void loop() {
    float humidity    = dht.readHumidity();       // Read humidity %
    float temperature = dht.readTemperature();    // Read temperature in Celsius

    // Check if reading was successful (sensor can sometimes fail)
    if (isnan(humidity) || isnan(temperature)) {
        Serial.println("ERROR: Failed to read from DHT sensor!");
    } else {
        // Send data as: "TEMP:22.5,HUM:55.0"
        // This format is easy for Python to parse
        Serial.print("TEMP:");
        Serial.print(temperature);
        Serial.print(",HUM:");
        Serial.println(humidity);
    }

    delay(2000);   // Wait 2 seconds before next reading
}
"""

  DHT11 Pin 1 (VCC/+)  → Arduino 5V pin
  DHT11 Pin 2 (DATA)   → Arduino Digital Pin 2
  DHT11 Pin 3 (NC)     → Not connected (skip this one)
  DHT11 Pin 4 (GND/-)  → Arduino GND pin

  If your DHT11 comes on a small breakout board (3-pin version),

# ==============================================================
# PYTHON SCRIPT: Read Serial Data from Arduino
# ==============================================================

import serial          # For USB serial communication (pip install pyserial)
import pandas as pd
from datetime import datetime
import time
import csv
import os

# ----------------------------------------------------------
# CONFIGURATION — UPDATE THESE VALUES FOR YOUR COMPUTER
# ----------------------------------------------------------

# Find your Arduino's USB port:
# Windows: Open Device Manager → Ports (COM & LPT) → look for "Arduino" → e.g., COM3
# Linux: Run `ls /dev/tty*` before and after plugging in Arduino → the new one is yours
# Mac: Run `ls /dev/cu.*` in terminal

PORT = "COM3"          # CHANGE THIS: e.g., "COM3" on Windows, "/dev/ttyUSB0" on Linux
BAUD_RATE = 9600       # Must match the Serial.begin(9600) in your Arduino code
DURATION_SECONDS = 60  # How long to collect data (60 seconds = 30 readings at 2s interval)
OUTPUT_FILE = "arduino_temperature_log.csv"


def parse_arduino_line(line):
    """
    Parse a line like "TEMP:22.5,HUM:55.0" into a dictionary.
    Returns None if the line format is unexpected.
    """
    try:
        # Split by comma: ["TEMP:22.5", "HUM:55.0"]
        parts = line.strip().split(",")
        if len(parts) != 2:
            return None

        # Split each part by colon and take the value after it
        temp = float(parts[0].split(":")[1])   # "TEMP:22.5" → 22.5
        hum = float(parts[1].split(":")[1])    # "HUM:55.0" → 55.0

        return {"temperature": temp, "humidity": hum}
    except (ValueError, IndexError):
        return None


def collect_data_from_arduino():
    """Connect to Arduino and collect sensor readings."""

    print(f"\nConnecting to Arduino on port {PORT}...")

    try:
        # Open the serial connection to Arduino
        ser = serial.Serial(PORT, BAUD_RATE, timeout=1)
        time.sleep(2)   # Give Arduino 2 seconds to reset after connecting
        print("Connected! Starting data collection...")
        print(f"Collecting for {DURATION_SECONDS} seconds. Press Ctrl+C to stop early.\n")

        readings = []
        start_time = time.time()

        while time.time() - start_time < DURATION_SECONDS:
            # Read one line from the serial port
            if ser.in_waiting > 0:
                raw_line = ser.readline().decode("utf-8", errors="ignore")
                parsed = parse_arduino_line(raw_line)

                if parsed:
                    # Add a timestamp to each reading
                    parsed["timestamp"] = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
                    readings.append(parsed)
                    print(f"  [{parsed['timestamp']}] Temp: {parsed['temperature']}°C | Humidity: {parsed['humidity']}%")
                else:
                    # Print raw messages (like "Arduino ready..." status messages)
                    raw_stripped = raw_line.strip()
                    if raw_stripped:
                        print(f"  Arduino says: {raw_stripped}")

        ser.close()
        print(f"\nDone! Collected {len(readings)} readings.")
        return readings

    except serial.SerialException as e:
        print(f"\nCould not connect to Arduino on {PORT}")
        print(f"Error: {e}")
        print("\nTROUBLESHOOTING:")
        print("  1. Is the Arduino plugged in via USB?")
        print("  2. Did you update the PORT variable above?")
        print("  3. On Linux/Mac, run: ls /dev/tty* to find the port")
        print("  4. On Windows, check Device Manager → Ports")
        print("\nRUNNING IN DEMO MODE with fake data instead...\n")
        return generate_demo_data()


def generate_demo_data():
    """
    Generate fake sensor data for practice when no Arduino is connected.
    Remove this function and the fallback once your hardware is set up.
    """
    import random

    print("Generating 30 fake sensor readings for practice...")
    readings = []
    base_time = datetime.now()

    for i in range(30):
        # Simulate a temperature that fluctuates slightly around 22°C
        temp = round(22.0 + random.uniform(-2.0, 2.0), 1)
        hum = round(55.0 + random.uniform(-5.0, 5.0), 1)
        timestamp = base_time.replace(second=base_time.second + (i * 2))

        readings.append({
            "timestamp": timestamp.strftime("%Y-%m-%d %H:%M:%S"),
            "temperature": temp,
            "humidity": hum
        })

    return readings


def save_to_csv(readings, filename):
    """Save the list of readings to a CSV file."""
    if not readings:
        print("No data to save.")
        return

    df = pd.DataFrame(readings)
    df.to_csv(filename, index=False)
    print(f"\nSaved {len(readings)} readings to '{filename}'")

    print("\nData summary:")
    print(f"  Temperature: avg={df['temperature'].mean():.1f}°C, "
          f"min={df['temperature'].min()}°C, max={df['temperature'].max()}°C")
    print(f"  Humidity:    avg={df['humidity'].mean():.1f}%, "
          f"min={df['humidity'].min()}%, max={df['humidity'].max()}%")


# ==============================================================
# MAIN — Run the data collection
# ==============================================================

if __name__ == "__main__":
    readings = collect_data_from_arduino()
    save_to_csv(readings, OUTPUT_FILE)

    print(f"\n✅ Your data is in '{OUTPUT_FILE}'")
    print("This CSV becomes your Portfolio Project 1 dataset!")
    print("\nNext step: Open month-02-data-science/week-05-pandas-numpy/pandas_basics.py")
    print("and load this CSV to analyze your own sensor data!")
