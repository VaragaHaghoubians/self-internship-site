# ==============================================================
# WEEK 14: Flask API Server
# FILE: flask_server.py
#
# TASK: Build a Python web server that receives sensor data from
#       the ESP32 board via HTTP POST requests.
#
# WHY:  Your ESP32 can't talk to Python code directly.
#       Instead, ESP32 sends data over WiFi to a web server URL.
#       Flask lets you build that web server in Python with very
#       few lines of code.
#
# HOW IT WORKS:
#   1. Your laptop runs this Flask server
#   2. ESP32 sends a POST request to http://YOUR_IP:5000/upload_data
#   3. Flask receives the request, reads the JSON body, saves to DB
#   4. Flask returns a "success" response to ESP32
#
# INSTALL: pip install flask
#
# HOW TO RUN:
#   python flask_server.py
#   Then test it with curl in another terminal window:
#   curl -X POST http://localhost:5000/upload_data \
#        -H "Content-Type: application/json" \
#        -d '{"temperature": 22.5, "humidity": 55.0}'
# ==============================================================

from flask import Flask, request, jsonify
import sqlite3
from datetime import datetime
import os

# ==============================================================
# APP SETUP
# ==============================================================

# Flask(__name__) creates the web application
# __name__ tells Flask which file it's in (for finding templates)
app = Flask(__name__)

# Name of our database file
DATABASE = "sensor_data.db"


# ==============================================================
# DATABASE SETUP
# ==============================================================

def init_database():
    """
    Create the database and sensor_readings table if they don't exist.
    This runs once when the server starts.
    """
    conn = sqlite3.connect(DATABASE)
    cursor = conn.cursor()

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS sensor_readings (
            id          INTEGER PRIMARY KEY AUTOINCREMENT,
            temperature REAL NOT NULL,
            humidity    REAL,
            timestamp   TEXT NOT NULL,
            device_id   TEXT DEFAULT 'esp32_01'
        )
    """)
    # AUTOINCREMENT = id number automatically increases with each new row

    conn.commit()
    conn.close()
    print(f"Database initialized: {DATABASE}")


def save_reading(temperature, humidity, device_id="esp32_01"):
    """
    Save one sensor reading to the database.
    Returns the ID of the newly inserted row.
    """
    conn = sqlite3.connect(DATABASE)
    cursor = conn.cursor()

    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    cursor.execute("""
        INSERT INTO sensor_readings (temperature, humidity, timestamp, device_id)
        VALUES (?, ?, ?, ?)
    """, (temperature, humidity, timestamp, device_id))
    # SECURITY NOTE: We use ? placeholders, NEVER string formatting in SQL
    # String formatting would allow SQL injection attacks!

    new_id = cursor.lastrowid    # Get the ID of the row we just inserted
    conn.commit()
    conn.close()

    return new_id, timestamp


def get_all_readings(limit=100):
    """
    Retrieve the most recent sensor readings from the database.
    Returns a list of dictionaries.
    """
    conn = sqlite3.connect(DATABASE)
    conn.row_factory = sqlite3.Row   # Makes rows behave like dictionaries
    cursor = conn.cursor()

    cursor.execute("""
        SELECT id, temperature, humidity, timestamp, device_id
        FROM sensor_readings
        ORDER BY id DESC
        LIMIT ?
    """, (limit,))

    rows = [dict(row) for row in cursor.fetchall()]
    conn.close()
    return rows

# API ENDPOINTS (Routes)
# ==============================================================

@app.route("/")
def home():
    """
    The home page — just confirms the server is running.
    Visit http://localhost:5000 in your browser to see this.
    """
    return jsonify({
        "status": "running",
        "message": "Smart Home Flask Server is online!",
        "endpoints": {
            "POST /upload_data": "Send sensor data here",
            "GET /data": "View all stored readings",
            "GET /latest": "View the most recent reading",
            "GET /dashboard": "View the live chart (HTML page)"
        }
    })


@app.route("/upload_data", methods=["POST"])
def upload_data():
    """
    MAIN ENDPOINT: Receives sensor data from ESP32.

    Expected JSON body:
    {
        "temperature": 22.5,
        "humidity": 55.0,
        "device_id": "esp32_01"   (optional)
    }

    Returns JSON response confirming success or describing the error.
    """
    # Get the JSON data from the request body
    data = request.get_json()

    # Validate that we received JSON
    if not data:
        return jsonify({"error": "No JSON data received"}), 400
        # 400 = HTTP "Bad Request"

    # Validate required fields
    if "temperature" not in data:
        return jsonify({"error": "Missing 'temperature' field"}), 400

    # Extract values
    try:
        temperature = float(data["temperature"])
        humidity = float(data.get("humidity", 0))      # Default to 0 if not provided
        device_id = str(data.get("device_id", "esp32_01"))
    except (ValueError, TypeError) as e:
        return jsonify({"error": f"Invalid data format: {str(e)}"}), 400

    # Validate range (sanity check)
    if not (-50 <= temperature <= 100):
        return jsonify({"error": "Temperature out of valid range (-50 to 100°C)"}), 400

    # Save to database
    new_id, timestamp = save_reading(temperature, humidity, device_id)

    print(f"[{timestamp}] Received: Temp={temperature}°C, Humidity={humidity}%, Device={device_id}")

    # Return success response
    return jsonify({
        "status": "success",
        "id": new_id,
        "timestamp": timestamp,
        "received": {
            "temperature": temperature,
            "humidity": humidity,
            "device_id": device_id
        }
    }), 201
    # 201 = HTTP "Created" (we created a new database record)


@app.route("/data", methods=["GET"])
def get_data():
    """Return all stored readings as JSON."""
    limit = request.args.get("limit", 100, type=int)   # e.g., /data?limit=50
    readings = get_all_readings(limit)
    return jsonify({
        "count": len(readings),
        "readings": readings
    })


@app.route("/latest", methods=["GET"])
def get_latest():
    """Return only the most recent reading."""
    readings = get_all_readings(limit=1)
    if readings:
        return jsonify(readings[0])
    return jsonify({"error": "No data yet"}), 404

        <title>Smart Home Dashboard</title>

            .stat { display: inline-block; margin: 10px; padding: 15px 25px;
                    background: #4299e1; color: white; border-radius: 8px; text-align: center; }
            .stat h3 { margin: 0; font-size: 2em; }
            .stat p { margin: 5px 0 0 0; }
        </style>
    </head>
    <body>
        <h1>🌡️ Smart Home Temperature Dashboard</h1>
        <div id="stats"></div>
        <div class="chart-container">
            <canvas id="tempChart" width="400" height="200"></canvas>
        </div>
        <p style="color: #718096; margin-top: 10px;">Auto-refreshes every 10 seconds</p>

        <script>
            let chart = null;

            async function fetchAndUpdate() {
                const res = await fetch('/data?limit=50');
                const json = await res.json();
                const readings = json.readings.reverse();  // Oldest first

                const labels = readings.map(r => r.timestamp.slice(11, 19)); // Time only
                const temps = readings.map(r => r.temperature);
                const humids = readings.map(r => r.humidity);

                // Update stats
                if (readings.length > 0) {
                    const latest = readings[readings.length - 1];
                    const avgTemp = (temps.reduce((a,b) => a+b, 0) / temps.length).toFixed(1);
                    document.getElementById('stats').innerHTML = `
                        <div class="stat"><h3>${latest.temperature}°C</h3><p>Current Temp</p></div>
                        <div class="stat"><h3>${latest.humidity}%</h3><p>Current Humidity</p></div>
                        <div class="stat"><h3>${avgTemp}°C</h3><p>Average Temp</p></div>
                        <div class="stat"><h3>${readings.length}</h3><p>Total Readings</p></div>
                    `;
                }

                if (chart) {
                    chart.data.labels = labels;
                    chart.data.datasets[0].data = temps;
                    chart.data.datasets[1].data = humids;
                    chart.update();
                } else {
                    const ctx = document.getElementById('tempChart').getContext('2d');
                    chart = new Chart(ctx, {
                        type: 'line',
                        data: {
                            labels: labels,
                            datasets: [
                                { label: 'Temperature (°C)', data: temps, 
                                  borderColor: '#e53e3e', backgroundColor: 'rgba(229,62,62,0.1)',
                                  tension: 0.4, fill: true },
                                { label: 'Humidity (%)', data: humids,
                                  borderColor: '#4299e1', backgroundColor: 'rgba(66,153,225,0.1)',
                                  tension: 0.4, fill: true }
                            ]
                        },
                        options: {
                            responsive: true,
                            plugins: { title: { display: true, text: 'Live Sensor Data' }},
                            scales: { y: { beginAtZero: false } }
                        }
                    });
                }
            }

            fetchAndUpdate();
            setInterval(fetchAndUpdate, 10000);  // Refresh every 10 seconds
        </script>
    </body>
    </html>
    """
    return html
