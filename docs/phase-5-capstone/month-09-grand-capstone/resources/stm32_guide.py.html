# ==============================================================
# SECTION 4: Core STM32 Concepts (different from Arduino/ESP32)
# ==============================================================

print("""
─────────────────────────────────────────────────────────────
KEY CONCEPTS UNIQUE TO STM32
─────────────────────────────────────────────────────────────

1. GPIO PORTS (A, B, C, D...)
   STM32 organises pins into ports (groups of 16 pins each).
   GPIOA → pins PA0–PA15
   GPIOB → pins PB0–PB15
   On Arduino you just say pin 5. On STM32 you say GPIOA, GPIO_PIN_5.

2. CLOCK CONFIGURATION
   Every peripheral needs its clock enabled before use.
   Forgetting this is the #1 beginner mistake on STM32!
   CubeIDE does this automatically.
   Manual: __HAL_RCC_GPIOA_CLK_ENABLE();

3. HAL (Hardware Abstraction Layer)
   Instead of writing raw register values, you call HAL functions:
     HAL_GPIO_WritePin(...)    ← like digitalWrite()
     HAL_GPIO_ReadPin(...)     ← like digitalRead()
     HAL_Delay(ms)             ← like delay()
     HAL_UART_Transmit(...)    ← like Serial.print()
     HAL_ADC_Start(...)        ← like analogRead() (but more steps)

4. INTERRUPTS
   STM32 has more powerful interrupt handling than Arduino.
   HAL_GPIO_EXTI_Callback() fires when a pin changes state.
   No need for constant polling — more efficient for real-time systems.

5. REAL-TIME OPERATING SYSTEM (FreeRTOS)
   STM32CubeIDE can generate FreeRTOS boilerplate.
   FreeRTOS lets you run multiple "tasks" concurrently:
     Task 1: read sensor every 100ms
     Task 2: check button every 10ms
     Task 3: send data every 5 seconds
   This is impossible with a simple Arduino loop.
   FreeRTOS is used in professional embedded products everywhere.

6. DMA (Direct Memory Access)
   Transfer data between memory and peripherals WITHOUT using the CPU.
   Example: read 1000 ADC samples into a buffer while the CPU does something else.
   Essential for real-time signal processing and high-speed logging.
""")

print("""
─────────────────────────────────────────────────────────────
PROGRAM 1: Blink LED (Hello World)
─────────────────────────────────────────────────────────────
// Paste this in Arduino IDE (PATH B) targeting Nucleo-F103RB
// The built-in LED is on PA5 (exposed as LED_BUILTIN)

void setup() {
    pinMode(LED_BUILTIN, OUTPUT);
    Serial.begin(115200);
    Serial.println("STM32 is alive!");
}

void loop() {
    digitalWrite(LED_BUILTIN, HIGH);
    delay(500);
    digitalWrite(LED_BUILTIN, LOW);
    delay(500);
}

// WHAT TO VERIFY:
// - The green LED on the Nucleo board blinks once per second
// - Serial Monitor shows "STM32 is alive!"

─────────────────────────────────────────────────────────────
PROGRAM 2: Read Analog Sensor (Temperature / Potentiometer)
─────────────────────────────────────────────────────────────
// STM32 has 12-bit ADC (0-4095) vs Arduino's 10-bit (0-1023)
// This gives 4x finer resolution for sensor readings

void setup() {
    Serial.begin(115200);
    analogReadResolution(12);   // Enable 12-bit mode (STM32 specific)
}

void loop() {
    int raw = analogRead(A0);             // 0–4095
    float voltage = raw * (3.3 / 4095.0); // STM32 uses 3.3V (not 5V!)
    
    // LM35 temperature sensor: 10mV per °C
    float temp_c = (voltage * 1000.0) / 10.0;
    
    Serial.print("Raw: ");   Serial.print(raw);
    Serial.print(" | V: ");  Serial.print(voltage, 3);
    Serial.print(" | T: ");  Serial.print(temp_c, 1);
    Serial.println("°C");
    
    delay(500);
}

// KEY DIFFERENCE FROM ARDUINO:
// Arduino UNO → 5V reference, 10-bit → 1024 steps of 4.9mV each
// STM32       → 3.3V reference, 12-bit → 4096 steps of 0.8mV each
// STM32 is 6x more precise for voltage measurements!
// NEVER connect 5V signals directly to STM32 pins — it will damage the chip.

─────────────────────────────────────────────────────────────
PROGRAM 3: UART Serial Communication (talk to Python)
─────────────────────────────────────────────────────────────
// Send structured data to Python — same concept as your week-08 Arduino code

void setup() {
    Serial.begin(115200);    // Higher baud rate than Arduino's 9600
}

void loop() {
    int raw_temp = analogRead(A0);
    int raw_hum  = analogRead(A1);
    
    float voltage_t = raw_temp * (3.3 / 4095.0);
    float temp_c    = (voltage_t * 1000.0) / 10.0;   // LM35 formula
    
    // Send as CSV: timestamp_ms,temperature,raw_adc
    Serial.print(millis());    // ms since boot
    Serial.print(",");
    Serial.print(temp_c, 2);
    Serial.print(",");
    Serial.println(raw_temp);
    
    delay(200);   // 5 readings per second
}

// Python side (same as your read_serial_data.py from week-08):
// import serial
// ser = serial.Serial('COM3', 115200, timeout=2)
// line = ser.readline().decode('utf-8').strip()
// ms, temp, raw = line.split(',')

─────────────────────────────────────────────────────────────
PROGRAM 4: Interrupt-Driven Button (no polling!)
─────────────────────────────────────────────────────────────
// This is where STM32 exceeds Arduino in real-world reliability.
// Instead of constantly checking if(buttonPressed) in loop(),
// the hardware calls our function the INSTANT the button is pressed.

volatile bool button_pressed = false;   // volatile: modified by interrupt

void setup() {
    Serial.begin(115200);
    pinMode(LED_BUILTIN, OUTPUT);
    pinMode(PC13, INPUT_PULLUP);              // Nucleo user button is on PC13
    attachInterrupt(digitalPinToInterrupt(PC13), onButtonPress, FALLING);
    Serial.println("Press the blue USER button on the Nucleo board");
}

void loop() {
    if (button_pressed) {
        button_pressed = false;
        digitalWrite(LED_BUILTIN, !digitalRead(LED_BUILTIN));  // toggle LED
        Serial.println("Button pressed — LED toggled");
    }
    // CPU is free to do other work here while waiting for the interrupt
}

void onButtonPress() {
    button_pressed = true;    // just set a flag — keep ISR short!
}

// RULE: Interrupt Service Routines (ISR) must be SHORT.
// Never do Serial.print(), delay(), or heavy logic inside an ISR.
// Just set a flag and handle it in loop().
""")

─────────────────────────────────────────────────────────────
FULL PROJECT: Industrial Sensor Logger
─────────────────────────────────────────────────────────────

CONCEPT:
  A production-grade data acquisition system that reads multiple
  sensors at high speed, detects anomalies in real time, and
  logs structured data to a Python dashboard over UART.

  This is the same architecture used in SCADA systems,
  machine monitoring, and the ai_ie_portfolio projects.

HARDWARE NEEDED:
  - STM32 Nucleo-F103RB (~$12)
  - LM35 temperature sensor (~$1) — OR use a potentiometer to simulate
  - 10kΩ resistor × 2
  - Breadboard + jumper wires (reuse from your ELEGOO kit)

WHAT THE SYSTEM DOES:
  STM32 side (C++ in Arduino IDE / CubeIDE):
    - Reads temperature from LM35 every 200ms (5 Hz)
    - Computes rolling average over last 10 readings (on-device smoothing)
    - Detects if temperature exceeds threshold → sets alarm flag
    - Sends CSV line over UART every 200ms:
      timestamp_ms, raw_adc, temperature, rolling_avg, alarm (0 or 1)
  
  Python side (build on top of your week-08 read_serial_data.py):
    - Reads the UART stream
    - Appends each row to a pandas DataFrame
    - Every 60 seconds: saves to CSV + checks for anomalies
    - Optional: POST the alarm flag to your Flask server (week-14)
      so the live dashboard shows the alert

─────────────────────────────────────────────────────────────
STM32 CODE (paste into Arduino IDE, Nucleo-F103RB target)
─────────────────────────────────────────────────────────────

const int TEMP_PIN       = A0;
const int LED_PIN        = LED_BUILTIN;
const int WINDOW_SIZE    = 10;
const float ALARM_TEMP   = 35.0;     // °C — change to suit your sensor

float readings[WINDOW_SIZE] = {0};
int   read_idx              = 0;
bool  window_full           = false;

void setup() {
    Serial.begin(115200);
    analogReadResolution(12);          // 12-bit ADC
    pinMode(LED_PIN, OUTPUT);
    Serial.println("ts_ms,raw,temp_c,avg_c,alarm");  // CSV header
}

void loop() {
    int   raw       = analogRead(TEMP_PIN);
    float voltage   = raw * (3.3f / 4095.0f);
    float temp_c    = voltage * 100.0f;              // LM35: 10mV/°C → ×100

    // Rolling average
    readings[read_idx] = temp_c;
    read_idx = (read_idx + 1) % WINDOW_SIZE;
    if (read_idx == 0) window_full = true;

    int   count = window_full ? WINDOW_SIZE : read_idx;
    float sum   = 0;
    for (int i = 0; i < count; i++) sum += readings[i];
    float avg_c = sum / count;

    // Alarm
    bool alarm = (temp_c > ALARM_TEMP);
    digitalWrite(LED_PIN, alarm ? HIGH : LOW);

    // Output CSV
    Serial.print(millis());  Serial.print(",");
    Serial.print(raw);       Serial.print(",");
    Serial.print(temp_c, 2); Serial.print(",");
    Serial.print(avg_c, 2);  Serial.print(",");
    Serial.println(alarm ? 1 : 0);

    delay(200);
}

# pip install pyserial pandas matplotlib requests

import serial
import pandas as pd
import matplotlib.pyplot as plt
from datetime import datetime
import time

PORT      = "COM3"       # Change to your port (COMx on Windows)
BAUD_RATE = 115200
LOG_FILE  = "stm32_sensor_log.csv"
FLASK_URL = "http://localhost:5000/upload_data"   # your week-14 server

columns = ["ts_ms", "raw", "temp_c", "avg_c", "alarm"]
records = []

print(f"Connecting to {PORT} at {BAUD_RATE} baud...")
ser = serial.Serial(PORT, BAUD_RATE, timeout=2)
time.sleep(2)   # wait for STM32 to reset after connection
ser.readline()  # discard header line

print("Logging — Ctrl+C to stop")
try:
    while True:
        line = ser.readline().decode("utf-8", errors="ignore").strip()
        if not line or "," not in line:
            continue
        parts = line.split(",")
        if len(parts) != 5:
            continue
        try:
            row = [float(p) for p in parts]
        except ValueError:
            continue   # skip malformed lines
        
        records.append(row)
        ts, raw, temp, avg, alarm = row
        status = "⚠️ ALARM" if alarm else "OK"
        print(f"  {datetime.now():%H:%M:%S}  Temp: {temp:.1f}°C  Avg: {avg:.1f}°C  {status}")
        
        # Every 50 readings: save to CSV and plot
        if len(records) % 50 == 0:
            df = pd.DataFrame(records, columns=columns)
            df.to_csv(LOG_FILE, index=False)
            print(f"  → Saved {len(df)} rows to {LOG_FILE}")
            
            plt.figure(figsize=(12, 4))
            plt.plot(df["ts_ms"] / 1000, df["temp_c"],  label="Raw temp",     alpha=0.5)
            plt.plot(df["ts_ms"] / 1000, df["avg_c"],   label="Rolling avg",  linewidth=2)
            alarm_rows = df[df["alarm"] == 1]
            if not alarm_rows.empty:
                plt.scatter(alarm_rows["ts_ms"] / 1000, alarm_rows["temp_c"],
                            color="red", zorder=5, label="ALARM")
            plt.xlabel("Time (seconds since boot)")
            plt.ylabel("Temperature (°C)")
            plt.title("STM32 Live Temperature Log")
            plt.legend()
            plt.grid(True, alpha=0.3)
            plt.tight_layout()
            plt.savefig("stm32_live_chart.png", dpi=80)
            plt.close()

except KeyboardInterrupt:
    print("\\nStopped.")
    df = pd.DataFrame(records, columns=columns)
    df.to_csv(LOG_FILE, index=False)
    print(f"Final save: {len(df)} rows → {LOG_FILE}")
    ser.close()
