# ==============================================================================
# PHASE 6 — MONTH 11: RAG — Retrieval-Augmented Generation
# FILE: rag_pipeline.py
#
# GOAL: Teach your LLM to answer questions about YOUR documents —
#       documents it has never seen during training.
#
# THE PROBLEM THIS SOLVES:
#   You ask GPT-4: "What does our company policy say about remote work?"
#   GPT-4 was not trained on your company's policy document.
#   It either makes something up (hallucinates) or says "I don't know."
#
#   RAG fixes this by:
#   1. Converting your documents into searchable vectors
#   2. Finding the most relevant chunks when a question is asked
#   3. Injecting those chunks into the LLM's context window
#   4. The LLM then answers using REAL content from your documents
#
# WHY THIS MATTERS:
#   RAG is how 90% of real-world enterprise AI chatbots work.
#   It's cheaper and more reliable than fine-tuning for most use cases.
#
# PRE-REQUISITES:
#   - Month 10 complete (you understand tokens, embeddings, LLM API calls)
#   - You can build and run a FastAPI endpoint
#   - You understand SQL and databases (Month 2 & 5)
#
# WHAT YOU WILL LEARN:
#   Part 1  — The RAG pipeline explained step by step
#   Part 2  — Chunking documents
#   Part 3  — Generating embeddings
#   Part 4  — Storing vectors in ChromaDB (local vector database)
#   Part 5  — Retrieving relevant chunks for a query
#   Part 6  — The full RAG chain (retrieve → inject → generate)
#   Part 7  — LangChain: the RAG framework
#   Part 8  — Building a "Chat with your PDF" FastAPI app
#
# INSTALL:
#   pip install chromadb openai langchain langchain-openai langchain-community

# PART 1 — THE RAG PIPELINE EXPLAINED
# ==============================================================================

print("=" * 65)
print("PART 1: THE RAG PIPELINE")
print("=" * 65)
print("""
The two phases of a RAG system:

PHASE A — INDEXING (done once, or whenever documents change):
  ┌─────────────────────────────────────────────────────────────┐
  │  Your Documents (PDF, Word, web pages, database records)    │
  │         ↓                                                   │
  │  Chunking  → split into overlapping text chunks (~500 tokens│
  │         ↓                                                   │
  │  Embedding → convert each chunk to a vector (list of floats)│
  │         ↓                                                   │
  │  Vector Database → store vectors + original text            │
  └─────────────────────────────────────────────────────────────┘

PHASE B — QUERYING (done for every user question):
  ┌─────────────────────────────────────────────────────────────┐
  │  User Question: "What is the remote work policy?"           │
  │         ↓                                                   │
  │  Embed the question (same embedding model as indexing!)     │
  │         ↓                                                   │
  │  Vector Similarity Search → find top-K most similar chunks  │
  │         ↓                                                   │
  │  Build prompt: [System] + [Retrieved chunks] + [Question]   │
  │         ↓                                                   │
  │  LLM generates answer grounded in your actual documents     │
  └─────────────────────────────────────────────────────────────┘

KEY INSIGHT:
  The embedding model turns text into a point in high-dimensional space.
  Similar text → nearby points → high similarity score.
  This is why "remote work" finds "work from home policy" — they are
  semantically close, even though the words are different.
""")


# ==============================================================================
# PART 2 — CHUNKING DOCUMENTS
# ==============================================================================
#
# TEACHER SAYS:
# Documents are too long to fit in a context window, so we split them into
# overlapping "chunks." Overlap ensures that a sentence on the boundary
# of two chunks is not split in a way that loses its meaning.
#
# chunk_size    = how many characters per chunk
# chunk_overlap = how many characters from the previous chunk are repeated
#                 at the start of the next (to preserve context at boundaries)

print("\n" + "=" * 65)
print("PART 2: CHUNKING")
print("=" * 65)

sample_document = """
Machine Learning Engineering at Acme Corp — Employee Handbook

Section 3: Remote Work Policy
Employees are permitted to work remotely up to 3 days per week, 
subject to manager approval. Core hours of 10am to 3pm local time 
must be maintained during remote days. All remote work must be 
conducted from a secure, private network.

Section 4: Equipment Policy
The company provides a laptop for all full-time employees. 
Personal use of company equipment is permitted for incidental 
tasks but must not interfere with work responsibilities. 
Any equipment faults must be reported within 24 hours.

Section 5: Data Security
All source code must be committed to the company GitHub repository.
No customer data may be stored on personal devices. VPN must be 
active when accessing internal systems from outside the office.
All passwords must be stored in the company-approved password manager.
""".strip()

def simple_chunker(text: str, chunk_size: int = 300, overlap: int = 50) -> list[dict]:
    """Split text into overlapping chunks with metadata."""
    chunks = []
    start = 0
    chunk_id = 0
    while start < len(text):
        end = start + chunk_size
        chunk_text = text[start:end]
        # Try to end at a sentence boundary:
        if end < len(text):
            last_period = chunk_text.rfind('.')
            if last_period > chunk_size * 0.5:    # Only if period is in the latter half
                chunk_text = chunk_text[:last_period + 1]
                end = start + last_period + 1
        chunks.append({
            "id": chunk_id,
            "text": chunk_text.strip(),
            "start_char": start,
            "end_char": end,
        })
        start = end - overlap  # Overlap: step back 'overlap' chars for next chunk
        chunk_id += 1
    return chunks

chunks = simple_chunker(sample_document, chunk_size=250, overlap=50)
print(f"Document split into {len(chunks)} chunks:")
for chunk in chunks:
    print(f"\n  Chunk {chunk['id']} (chars {chunk['start_char']}–{chunk['end_char']}):")
    print(f"  {chunk['text'][:100]}...")

# ==============================================================================
# PART 3 — GENERATING EMBEDDINGS
# ==============================================================================
#
# TEACHER SAYS:
# An embedding model converts text into a vector (list of floats).
# We use the SAME embedding model for both indexing AND querying.
# If you use a different model, the vectors won't be comparable.
#
# Popular embedding models:
#   text-embedding-3-small (OpenAI) → 1536 dimensions, costs ~$0.02/1M tokens
#   all-MiniLM-L6-v2 (Hugging Face) → 384 dimensions, FREE, runs locally

print("\n" + "=" * 65)
print("PART 3: GENERATING EMBEDDINGS")
print("=" * 65)

try:
    from sentence_transformers import SentenceTransformer
    import numpy as np

    print("Loading local embedding model (free, no API key needed)...")
    embed_model = SentenceTransformer('all-MiniLM-L6-v2')

    # Embed all our chunks:
    chunk_texts = [c["text"] for c in chunks]
    embeddings = embed_model.encode(chunk_texts, show_progress_bar=False)

    print(f"Embedded {len(embeddings)} chunks")
    print(f"Each embedding = {len(embeddings[0])} dimensions (floats)")
    print(f"First 5 values of chunk 0: {embeddings[0][:5].round(4)}")

    # Similarity demo:
    query = "Can I work from home?"
    query_embedding = embed_model.encode([query])[0]

    # Cosine similarity: 1.0 = identical, 0.0 = unrelated, -1.0 = opposite
    from numpy.linalg import norm
    similarities = [
        float(np.dot(query_embedding, emb) / (norm(query_embedding) * norm(emb)))
        for emb in embeddings
    ]
    ranked = sorted(enumerate(similarities), key=lambda x: x[1], reverse=True)
    print(f"\nQuery: '{query}'")
    print("Most relevant chunks:")
    for idx, score in ranked[:2]:
        print(f"  Score {score:.3f}: {chunk_texts[idx][:80]}...")

except ImportError:
    print("Run: pip install sentence-transformers")

# ==============================================================================
# PART 4 — CHROMADB: YOUR LOCAL VECTOR DATABASE
# ==============================================================================
#
# TEACHER SAYS:
# Computing embeddings every time a question is asked is slow and expensive.
# A vector database stores your pre-computed embeddings and runs fast
# nearest-neighbour searches (finding the most similar chunks).
#
# ChromaDB = a lightweight vector database that runs locally.
# In production you might use: pgvector (PostgreSQL), Pinecone, Weaviate, Qdrant.

print("\n" + "=" * 65)
print("PART 4: CHROMADB VECTOR DATABASE")
print("=" * 65)

try:
    import chromadb
    from chromadb.utils import embedding_functions

    # Create a local ChromaDB instance (stores on disk in ./chroma_db/):
    chroma_client = chromadb.Client()  # In-memory for this demo

    # Use a built-in embedding function (no extra code needed):
    ef = embedding_functions.SentenceTransformerEmbeddingFunction(
        model_name="all-MiniLM-L6-v2"
    )

    # Create a collection (like a table in SQL):
    collection = chroma_client.get_or_create_collection(
        name="company_handbook",
        embedding_function=ef,
        metadata={"description": "Acme Corp Employee Handbook"}
    )

    # Add documents (ChromaDB embeds them automatically):
    collection.add(
        documents=chunk_texts,
        ids=[f"chunk_{i}" for i in range(len(chunk_texts))],
        metadatas=[{"source": "handbook.pdf", "chunk_id": i} for i in range(len(chunk_texts))]
    )

    print(f"Collection '{collection.name}' has {collection.count()} documents")

    # Query the collection:
    query_text = "What are the rules for working from home?"
    results = collection.query(query_texts=[query_text], n_results=2)

    print(f"\nQuery: '{query_text}'")
    print("Top 2 retrieved chunks:")
    for i, (doc, dist) in enumerate(zip(
        results['documents'][0],
        results['distances'][0]
    )):
        print(f"\n  Result {i+1} (distance={dist:.4f}):")
        print(f"  {doc[:150]}...")

except ImportError:
    print("Run: pip install chromadb")
except Exception as e:
    print(f"ChromaDB error: {e}")

# ==============================================================================
# PART 5 — THE FULL RAG CHAIN (retrieve → inject → generate)
# ==============================================================================
#
# TEACHER SAYS:
# Now we combine everything: retrieve the relevant chunks from the vector DB,
# inject them into the prompt, and let the LLM answer.
# This is the core RAG loop that powers almost all enterprise AI chatbots.

print("\n" + "=" * 65)
print("PART 5: THE FULL RAG CHAIN")
print("=" * 65)

import os
try:
    from openai import OpenAI
    from dotenv import load_dotenv

    load_dotenv()
    api_key = os.getenv("OPENAI_API_KEY")

    if not api_key:
        print("Set OPENAI_API_KEY in your .env file to run this section.")
    else:
        client = OpenAI(api_key=api_key)

        def rag_answer(question: str, collection, n_chunks: int = 2) -> dict:
            """Full RAG pipeline: retrieve relevant chunks, then generate answer."""

            # Step 1: Retrieve relevant chunks from the vector database
            results = collection.query(query_texts=[question], n_results=n_chunks)
            retrieved_chunks = results['documents'][0]

            # Step 2: Build the prompt with retrieved context
            context = "\n\n---\n\n".join(retrieved_chunks)
            prompt = f"""You are a helpful assistant. Answer the question using ONLY 
the context provided below. If the answer is not in the context, 
say "I don't have that information."

CONTEXT:
{context}

QUESTION: {question}

ANSWER:"""

            # Step 3: Generate answer using the LLM
            response = client.chat.completions.create(
                model="gpt-4o-mini",
                temperature=0,  # Deterministic for factual Q&A
                messages=[{"role": "user", "content": prompt}]
            )

            return {
                "question": question,
                "answer": response.choices[0].message.content,
                "sources": retrieved_chunks,
                "tokens_used": response.usage.total_tokens
            }

        # Test questions:
        test_questions = [
            "How many days per week can I work remotely?",
            "What should I do if my laptop breaks?",
            "What are the data security rules?",
            "Can I store customer data on my personal phone?",
        ]

        for q in test_questions:
            result = rag_answer(q, collection)
            print(f"\nQ: {q}")
            print(f"A: {result['answer']}")
            print(f"   (tokens: {result['tokens_used']})")

except ImportError:
    print("Run: pip install openai python-dotenv")
except Exception as e:
    print(f"Error: {e}")


# ==============================================================================
# PART 6 — LANGCHAIN: THE RAG FRAMEWORK
# ==============================================================================
#
# TEACHER SAYS:
# LangChain is a framework that handles the RAG pipeline for you —
# loading documents, chunking, embedding, storing, and retrieving.
# Instead of writing all the Part 2–5 code yourself, LangChain gives
# you pre-built components you can chain together.
#
# Core concepts:
#   Document Loaders  → read PDFs, Word docs, web pages, databases
#   Text Splitters    → chunk the documents
#   Embedding Models  → convert text to vectors
#   Vector Stores     → store and search vectors
#   Retrievers        → search interface over a vector store
#   Chains            → combine retriever + LLM into a pipeline

print("\n" + "=" * 65)
print("PART 6: LANGCHAIN RAG PIPELINE")
print("=" * 65)

LANGCHAIN_EXAMPLE = '''
# langchain_rag.py — Minimal LangChain RAG pipeline

from langchain_community.document_loaders import PyPDFLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain.chains import RetrievalQA
from dotenv import load_dotenv

load_dotenv()

# Step 1: Load your PDF
loader = PyPDFLoader("your_document.pdf")
documents = loader.load()
print(f"Loaded {len(documents)} pages")

# Step 2: Split into chunks
splitter = RecursiveCharacterTextSplitter(
    chunk_size=500,
    chunk_overlap=50
)
chunks = splitter.split_documents(documents)
print(f"Split into {len(chunks)} chunks")

# Step 3: Embed and store in ChromaDB
vectorstore = Chroma.from_documents(
    documents=chunks,
    embedding=OpenAIEmbeddings(),
    persist_directory="./chroma_db"    # Saves to disk
)
vectorstore.persist()

# Step 4: Create the RAG chain
retriever = vectorstore.as_retriever(search_kwargs={"k": 3})  # Return top 3 chunks
llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

qa_chain = RetrievalQA.from_chain_type(
    llm=llm,
    retriever=retriever,
    return_source_documents=True       # Include retrieved chunks in response
)

# Step 5: Ask questions
result = qa_chain.invoke({"query": "What is the remote work policy?"})
print(result["result"])
print(f"Based on {len(result['source_documents'])} source chunks")
'''
print(LANGCHAIN_EXAMPLE)

CHAT_PDF_APP = '''
# chat_pdf_api.py — "Chat with your PDF" endpoint
from fastapi import FastAPI, UploadFile, File, HTTPException
from pydantic import BaseModel
import chromadb
from chromadb.utils import embedding_functions
from openai import OpenAI
from dotenv import load_dotenv
import os, tempfile

load_dotenv()
app = FastAPI(title="Chat with your PDF")
openai_client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

ef = embedding_functions.SentenceTransformerEmbeddingFunction(
    model_name="all-MiniLM-L6-v2"
)
chroma = chromadb.PersistentClient(path="./chroma_db")

class QuestionRequest(BaseModel):
    collection_name: str
    question: str

@app.post("/upload/{collection_name}")
async def upload_pdf(collection_name: str, file: UploadFile = File(...)):
    """Upload a PDF and index it into ChromaDB."""
    if not file.filename.endswith(".pdf"):
        raise HTTPException(status_code=400, detail="Only PDF files supported")

    # Save uploaded file temporarily
    with tempfile.NamedTemporaryFile(delete=False, suffix=".pdf") as tmp:
        content = await file.read()
        tmp.write(content)
        tmp_path = tmp.name

    # Load, chunk, and embed (using PyPDF + LangChain splitter)
    from langchain_community.document_loaders import PyPDFLoader
    from langchain.text_splitter import RecursiveCharacterTextSplitter

    loader = PyPDFLoader(tmp_path)
    pages = loader.load()
    splitter = RecursiveCharacterTextSplitter(chunk_size=500, chunk_overlap=50)
    chunks = splitter.split_documents(pages)

    # Store in ChromaDB
    collection = chroma.get_or_create_collection(
        name=collection_name, embedding_function=ef
    )
    collection.add(
        documents=[c.page_content for c in chunks],
        ids=[f"chunk_{i}" for i in range(len(chunks))]
    )
    os.unlink(tmp_path)
    return {"message": f"Indexed {len(chunks)} chunks from {file.filename}"}

@app.post("/ask")
async def ask_question(request: QuestionRequest):
    """Ask a question about an indexed PDF."""
    try:
        collection = chroma.get_collection(
            name=request.collection_name, embedding_function=ef
        )
    except Exception:
        raise HTTPException(status_code=404, detail="Collection not found. Upload a PDF first.")

    results = collection.query(query_texts=[request.question], n_results=3)
    context = "\\n\\n---\\n\\n".join(results["documents"][0])

    response = openai_client.chat.completions.create(
        model="gpt-4o-mini",
        temperature=0,
        messages=[
            {"role": "system", "content":
             "Answer using ONLY the context. Say 'I don\\'t know' if not in context."},
            {"role": "user", "content": f"Context:\\n{context}\\n\\nQuestion: {request.question}"}
        ]
    )
    return {
        "answer": response.choices[0].message.content,
        "sources_used": len(results["documents"][0])
    }
'''
print(CHAT_PDF_APP)
