# GOAL: Build an AI agent that can USE TOOLS — an LLM that doesn't just
#       answer questions, but takes real actions in the world.
#
# THE DIFFERENCE:
#   LLM:   You ask it a question → it generates text back.
#   Agent: You give it a GOAL → it decides which tools to use,
#          calls them in sequence, observes results, and keeps going
#          until the goal is achieved. It can loop, retry, and adapt.
#
# REAL-WORLD EXAMPLES:
#   • "Research the top 5 competitors of our company and write a report"
#     → Agent: searches the web, reads pages, writes to a file
#   • "Find all customers who haven't ordered in 90 days and email them"
#     → Agent: queries your database, composes emails, sends them
#   • "Monitor the ESP32 sensor and alert me if temperature exceeds 30°C"

KEY INSIGHT:

  The LLM only decides WHICH tool to call and with WHAT arguments.
  Your Python code executes the tool and feeds the result back to the LLM.

import os
import json
import requests
from datetime import datetime

# ── Tool 1: Web search ──────────────────────────────────────────────────────
def search_web(query: str) -> str:
    """
    Search the web for current information.
    Use this tool when you need up-to-date facts, news, or information
    that may not be in your training data.
    
    Args:
        query: The search query string.
    
    Returns:
        A string containing the top search results.
    """
    try:
        from duckduckgo_search import DDGS
        with DDGS() as ddgs:
            results = list(ddgs.text(query, max_results=3))
        if not results:
            return "No results found."
        return "\n".join([f"- {r['title']}: {r['body'][:200]}" for r in results])
    except ImportError:
        return "[SIMULATED] Search result: Found relevant information about: " + query
    except Exception as e:
        return f"Search error: {e}"


# ── Tool 2: Python code executor ────────────────────────────────────────────
def execute_python(code: str) -> str:
    """
    Execute a Python code snippet and return the output.
    Use this tool for mathematical calculations, data analysis, or
    generating plots. The code runs in a restricted environment.
    
    Args:
        code: A valid Python code string to execute.
    
    Returns:
        The stdout output of the code, or an error message.
    """
    import io
    import contextlib
    output = io.StringIO()
    try:
        with contextlib.redirect_stdout(output):
            exec(code, {"__builtins__": {
                # Only allow safe builtins:
                "print": print, "range": range, "len": len,
                "sum": sum, "min": min, "max": max, "sorted": sorted,
                "enumerate": enumerate, "zip": zip, "list": list,
                "dict": dict, "str": str, "int": int, "float": float,
                "__import__": __import__
            }})
        return output.getvalue() or "Code executed successfully (no output)."
    except Exception as e:
        return f"Error: {type(e).__name__}: {e}"


# ── Tool 3: Get current time/date ───────────────────────────────────────────
def get_current_datetime() -> str:
    """
    Get the current date and time.
    Use this tool when you need to know today's date or current time,
    for example when answering questions about 'today', 'this week', or
    when calculating time differences.
    
    Returns:
        A string with the current date and time in ISO format.
    """
    return datetime.now().strftime("%Y-%m-%d %H:%M:%S")

def search_web(query: str) -> str:

    from duckduckgo_search import DDGS

# PART 6 — LANGGRAPH: STATEFUL AGENTS
# ==============================================================================
#
# TEACHER SAYS:
# LangChain agents are a linear loop. LangGraph is for COMPLEX agents that:
# - Need conditional branching (if this result, do X, else do Y)
# - Run multiple steps in parallel
# - Have human-in-the-loop approval steps
# - Need persistent state across multiple conversation turns
#
# LangGraph models the agent as a GRAPH where:
#   NODES = functions (LLM calls, tool calls, conditions)
#   EDGES = connections between nodes (sequential or conditional)
#   STATE = a dictionary shared across all nodes

print("\n" + "=" * 65)
print("PART 6: LANGGRAPH STATEFUL AGENT")
print("=" * 65)

LANGGRAPH_CODE = '''
# langgraph_agent.py — Stateful agent with conditional branching
from langgraph.graph import StateGraph, END
from langgraph.prebuilt import ToolNode
from langchain_openai import ChatOpenAI
from langchain.tools import tool
from typing import TypedDict, Annotated
import operator

# Define the state that flows through the graph:
class AgentState(TypedDict):
    messages: Annotated[list, operator.add]   # All conversation messages
    step_count: int

# Define tools:
@tool
def get_weather(city: str) -> str:
    """Get current weather for a city."""
    return f"Weather in {city}: 18°C, partly cloudy"

@tool
def send_alert(message: str) -> str:
    """Send an alert notification."""
    print(f"ALERT SENT: {message}")
    return "Alert sent successfully"

# Create nodes:
llm = ChatOpenAI(model="gpt-4o-mini")
tools = [get_weather, send_alert]
llm_with_tools = llm.bind_tools(tools)

def call_llm(state: AgentState) -> AgentState:
    """Node: call the LLM."""
    response = llm_with_tools.invoke(state["messages"])
    return {"messages": [response], "step_count": state["step_count"] + 1}

def should_continue(state: AgentState) -> str:
    """Edge condition: should we call a tool or are we done?"""
    last_message = state["messages"][-1]
    if hasattr(last_message, "tool_calls") and last_message.tool_calls:
        return "use_tools"
    return END

# Build the graph:
graph = StateGraph(AgentState)
graph.add_node("llm", call_llm)
graph.add_node("tools", ToolNode(tools))
graph.set_entry_point("llm")
graph.add_conditional_edges("llm", should_continue)
graph.add_edge("tools", "llm")  # After tool call, go back to LLM

app = graph.compile()

# Run:
result = app.invoke({
    "messages": [{"role": "user", "content": "Get the weather in Tokyo and alert me if above 25°C"}],
    "step_count": 0
})
print(result["messages"][-1].content)
'''
print(LANGGRAPH_CODE)

  │ the web  │  │ analysis in  │  │ writes the   │         │
  │          │  │ Python       │  │ final report │         │
  └──────────┘  └──────────────┘  └──────────────┘         │
       ↘              ↓               ↙                    │
  ┌─────────────────────────────────────────────────────────┤
  │  Final report returned to user                          │
  └─────────────────────────────────────────────────────────┘

Why does this work?
  - Each agent has a focused system prompt and limited tools
  - Smaller context = cheaper API calls, fewer errors
  - Agents can run in PARALLEL (the research + analysis agents
    can both run at the same time)
  - Easy to debug: if the report is wrong, you know which
    agent produced the bad output

LangGraph handles this with conditional edges and parallel execution.

An unguarded agent in production will:
  ❌ Loop forever (cost you hundreds of dollars)
  ❌ Call tools with dangerous inputs
  ❌ Hallucinate tool arguments
  ❌ Leak private data into search queries

GUARDRAILS TO IMPLEMENT:

1. MAX STEPS LIMIT
   Always cap the agent loop. Never let it run indefinitely.
   max_steps = 15 is a reasonable starting point.

2. COST TRACKING
   import tiktoken
   enc = tiktoken.get_encoding("cl100k_base")
   token_count = len(enc.encode(text))
   # At $0.15/1M tokens (gpt-4o-mini), 10,000 tokens = $0.0015

3. TOOL INPUT VALIDATION
   Before executing ANY tool call, validate the arguments:
   - SQL queries: check they start with SELECT
   - Python code: run in a sandbox (Docker container)
   - Web searches: filter out internal/private search terms

4. HUMAN IN THE LOOP
   For destructive actions (sending emails, deleting records),
   pause the agent and ask for human confirmation:
   LangGraph supports "interrupt" nodes for exactly this.

5. TIMEOUTS
   Wrap every tool call in a timeout:
   from concurrent.futures import ThreadPoolExecutor, TimeoutError
   with ThreadPoolExecutor() as executor:
       future = executor.submit(tool_function, **args)
       result = future.result(timeout=30)  # 30 second max

6. OUTPUT FILTERING
   Check the final answer before returning it to the user.
   Use a second LLM call as a "safety reviewer":
   "Does this response contain any private data or harmful content?"
""")
